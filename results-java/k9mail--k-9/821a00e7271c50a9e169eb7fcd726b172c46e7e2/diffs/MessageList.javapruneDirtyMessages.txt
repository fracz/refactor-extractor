public void pruneDirtyMessages() {
    synchronized (mAdapter.messages) {
        Iterator<MessageInfoHolder> iter = mAdapter.messages.iterator();
        while (iter.hasNext()) {
            MessageInfoHolder holder = iter.next();
            if (holder.dirty) {
                if (holder.selected) {
                    mSelectedCount--;
                    toggleBatchButtons();
                }
                mAdapter.removeMessage(holder);
            }
        }
    }
