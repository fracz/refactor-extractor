@Override
public boolean exists() throws MessagingException {
    Cursor cursor = null;
    try {
        cursor = mDb.rawQuery("SELECT id FROM folders " + "where folders.name = ?", new String[] { this.getName() });
        if (cursor.moveToFirst()) {
            int folderId = cursor.getInt(0);
            return (folderId > 0);
        } else {
            return false;
        }
    } finally {
        if (cursor != null) {
            cursor.close();
        }
    }
