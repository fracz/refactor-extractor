/**
 * An implementation of {@link Map#equals}.
 */
static boolean equalsImpl(Map<?, ?> map, Object object) {
    if (map == object) {
        return true;
    }
    if (object instanceof Map) {
        Map<?, ?> o = (Map<?, ?>) object;
        return map.entrySet().equals(o.entrySet());
    }
    return false;
