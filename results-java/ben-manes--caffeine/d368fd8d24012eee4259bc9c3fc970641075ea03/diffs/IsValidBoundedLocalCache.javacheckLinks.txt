private void checkLinks(BoundedLocalCache<K, V> cache, ImmutableList<LinkedDeque<Node<K, V>>> deques, DescriptionBuilder desc) {
    int size = 0;
    long weightedSize = 0;
    Set<Node<K, V>> seen = Sets.newIdentityHashSet();
    for (LinkedDeque<Node<K, V>> deque : deques) {
        size += deque.size();
        weightedSize += scanLinks(cache, seen, deque, desc);
    }
    desc.expectThat(() -> "deque size " + deques, size, is(cache.size()));
    Supplier<String> errorMsg = () -> String.format("Size != list length; pending=%s, additional: %s", cache.writeQueue().size(), Sets.difference(seen, ImmutableSet.copyOf(cache.data.values())));
    desc.expectThat(errorMsg, cache.size(), is(seen.size()));
    final long weighted = weightedSize;
    if (cache.evicts()) {
        Supplier<String> error = () -> String.format("WeightedSize != link weights [%d vs %d] {%d vs %d}", cache.adjustedWeightedSize(), weighted, seen.size(), cache.size());
        desc.expectThat("non-negative weight", weightedSize, is(greaterThanOrEqualTo(0L)));
        desc.expectThat(error, cache.adjustedWeightedSize(), is(weightedSize));
    }
