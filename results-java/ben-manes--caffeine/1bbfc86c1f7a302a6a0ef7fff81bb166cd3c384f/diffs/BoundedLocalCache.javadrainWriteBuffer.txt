/**
 * Drains the read buffer up to an amortized threshold.
 */
@GuardedBy("evictionLock")
void drainWriteBuffer() {
    if (!buffersWrites()) {
        return;
    }
    for (int i = 0; i < WRITE_BUFFER_DRAIN_THRESHOLD; i++) {
        final Runnable task = writeQueue().poll();
        if (task == null) {
            break;
        }
        task.run();
    }
