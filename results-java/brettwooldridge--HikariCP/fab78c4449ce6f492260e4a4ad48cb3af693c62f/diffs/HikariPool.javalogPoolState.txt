/**
 * Log the current pool state at debug level.
 *
 * @param prefix an optional prefix to prepend the log message
 */
public final void logPoolState(String... prefix) {
    if (LOGGER.isDebugEnabled()) {
        LOGGER.debug("{}pool {} stats (total={}, active={}, idle={}, waiting={})", (prefix.length > 0 ? prefix[0] : ""), config.getPoolName(), getTotalConnections(), getActiveConnections(), getIdleConnections(), getThreadsAwaitingConnection());
    }
