/**
 * Get max height of labels.
 */
protected int getLabelHeight() {
    if (mLabelHeight == -1) {
        int result = 0;
        for (int i = 0; i < mChartView.data.get(0).size(); i++) {
            result = mChartView.style.getTextHeightBounds(mChartView.data.get(0).getLabel(i));
            if (result != 0)
                break;
        }
        mLabelHeight = result;
    }
    return mLabelHeight;
