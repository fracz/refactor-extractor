 * Given an index in points, it will make sure the the returned index is
 * within the array.
 */
private int si(ArrayList<ChartEntry> points, int i) {
    if (i > points.size() - 1)
        return points.size() - 1;
    else if (i < 0)
        return 0;
    return i;
