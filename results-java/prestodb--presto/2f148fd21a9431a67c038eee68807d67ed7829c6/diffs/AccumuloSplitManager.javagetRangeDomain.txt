        for (ColumnDomain<ColumnHandle> cd : constraint.getColumnDomains().get()) {
            AccumuloColumnHandle col = checkType(cd.getColumn(), AccumuloColumnHandle.class, "column handle");

                return Optional.of(cd.getDomain());
