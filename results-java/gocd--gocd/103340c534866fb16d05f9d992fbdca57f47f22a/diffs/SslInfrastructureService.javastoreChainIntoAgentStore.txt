    try {
        keyStoreManager.storeCertificate(CHAIN_ALIAS, AGENT_CERTIFICATE_FILE, AGENT_STORE_PASSWORD, keyEntry);

    } catch (Exception e) {
        throw bomb("Couldn't save agent key into store", e);
    }
