@Override
public boolean send(Message msg) {
    if (LOGGER.isDebugEnabled()) {
        LOGGER.debug(sessionName() + " send message: " + msg);
    }
    try {
        this.session.getRemote().sendString(Message.encode(msg));
        return true;
    } catch (IOException e) {
        onError(e);
        return false;
    }
