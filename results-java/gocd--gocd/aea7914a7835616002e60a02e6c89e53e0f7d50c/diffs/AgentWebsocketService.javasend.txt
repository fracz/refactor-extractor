public boolean send(Message message) {
    if (LOGGER.isDebugEnabled()) {
        LOGGER.debug(sessionName() + " send message: " + message);
    }
    try {
        this.session.getRemote().sendString(Message.encode(message));
        return true;
    } catch (IOException e) {
        onError(e);
        return false;
    }
