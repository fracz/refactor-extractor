    try {
        if (sslInfrastructureService.isRegistered()) {

            agentRuntimeInfo.refreshUsableSpace();
            instruction = server.ping(agentRuntimeInfo);

        LOG.error("Error occurred when agent tried to ping server: ", e);
    }
