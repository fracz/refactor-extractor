private void flushCacheIfRequired(MappedStatement ms) {
    Cache cache = ms.getCache();
    if (cache != null) {
        if (ms.isFlushCacheRequired()) {
            // issue #524. Disable using cached data for this session
            if (!autoCommit)
                dirty = true;
            tcm.clear(cache);
        }
    }
