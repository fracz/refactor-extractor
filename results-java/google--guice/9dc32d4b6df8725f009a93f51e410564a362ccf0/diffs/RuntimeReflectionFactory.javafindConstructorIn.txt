private <T> Constructor<T> findConstructorIn(Class<T> implementation) {
    Constructor<T> found = null;
    @SuppressWarnings("unchecked") Constructor<T>[] constructors = (Constructor<T>[]) implementation.getDeclaredConstructors();
    for (Constructor<T> constructor : constructors) {
        Inject inject = constructor.getAnnotation(Inject.class);
        if (inject != null) {
            if (inject.optional()) {
                errorHandler.handle(StackTraceElements.forMember(constructor), ErrorMessage.optionalConstructor());
            }
            if (found != null) {
                errorHandler.handle(StackTraceElements.forMember(found), ErrorMessage.tooManyConstructors());
                return invalidConstructor();
            }
            found = constructor;
        }
    }
    if (found != null) {
        return found;
    }
    // instead.
    try {
        return implementation.getDeclaredConstructor();
    } catch (NoSuchMethodException e) {
        errorHandler.handle(StackTraceElements.forMember(implementation.getDeclaredConstructors()[0]), ErrorMessage.missingConstructor(implementation));
        return invalidConstructor();
    }
