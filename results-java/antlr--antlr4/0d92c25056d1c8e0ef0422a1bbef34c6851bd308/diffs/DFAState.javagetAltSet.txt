/**
 * Get the set of all alts mentioned by all ATN configurations in this
 * DFA state.
 */
public Set<Integer> getAltSet() {
    // TODO (sam): what to do when configs==null?
    Set<Integer> alts = new HashSet<Integer>();
    for (ATNConfig c : configs) {
        alts.add(c.alt);
    }
    if (alts.isEmpty())
        return null;
    return alts;
