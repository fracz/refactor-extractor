public void remove(String id) throws NotFoundException {
    lock.writeLock().lock();
    try {
        User myUser = null;
        for (int i = 0, size = users.size(); i < size && myUser == null; i++) {
            if (users.get(i).getId().equals(id)) {
                myUser = users.get(i);
            }
        }
        if (myUser == null) {

    } finally {
        lock.writeLock().unlock();
    }
