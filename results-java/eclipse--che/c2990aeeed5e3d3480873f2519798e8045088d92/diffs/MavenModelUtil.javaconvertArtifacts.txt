private static List<MavenArtifact> convertArtifacts(Collection<Artifact> artifacts, Map<Artifact, MavenArtifact> convertedArtifacts, File localRepository) {
    ArrayList<MavenArtifact> result = new ArrayList<>();
    if (artifacts != null) {
        result.addAll(artifacts.stream().map(artifact -> convertArtifact(artifact, convertedArtifacts, localRepository)).collect(Collectors.toList()));
    }
    return result;
