private void fetchBranch(GitConnection gitConnection, String remote, String branch, DtoFactory dtoFactory) throws UnauthorizedException, GitException {
    final List<String> refSpecs = Collections.singletonList(String.format("refs/heads/%1$s:refs/remotes/origin/%1$s", branch));
    try {
        fetchRefSpecs(gitConnection, remote, refSpecs, dtoFactory);
    } catch (GitException e) {
        LOG.warn("Git exception on branch fetch", e);
        throw new GitException(String.format("Unable to fetch remote branch %s. Make sure it exists and can be accessed.", branch), e);
    }
