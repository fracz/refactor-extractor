/**
 * Flips the current texture region.
 */
public void flip(boolean x, boolean y) {
    float[] vertices = this.vertices;
    if (x) {
        float u = vertices[U1];
        float u2 = vertices[U3];
        vertices[U1] = u2;
        vertices[U2] = u2;
        vertices[U3] = u;
        vertices[U4] = u;
    }
    if (y) {
        float v = vertices[V1];
        float v2 = vertices[V3];
        vertices[V1] = v2;
        vertices[V2] = v;
        vertices[V3] = v;
        vertices[V4] = v2;
    }
