public void flush() {
    OutputStream out = null;
    try {
        out = new BufferedOutputStream(Gdx.files.external(LwjglPreferences.this.name).write(false));
        properties.storeToXML(out, null);

    } finally {
        if (out != null)
            try {
                out.close();
            } catch (Exception e) {
            }
    }
