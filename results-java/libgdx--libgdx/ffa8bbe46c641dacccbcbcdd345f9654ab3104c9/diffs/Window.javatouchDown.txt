@Override
public boolean touchDown(float x, float y, int pointer) {
    if (pointer != 0)
        return false;
    if (parent.getActors().size() > 1)
        parent.swapActor(this, parent.getActors().get(parent.getActors().size() - 1));
    if (titleBounds.contains(x, y)) {
        if (isMovable)
            move = true;
        initial.set(x, y);
    } else
        super.touchDown(x, y, pointer);
    return true;
