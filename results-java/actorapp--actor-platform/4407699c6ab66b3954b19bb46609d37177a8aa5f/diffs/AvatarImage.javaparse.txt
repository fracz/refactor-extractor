@Override
public void parse(BserValues values) throws IOException {
    // Is Wrapper layout
    if (values.getBool(5, false)) {
        // Parse wrapper layout
        super.parse(values);
    } else {
        // Convert old layout
        ObsoleteAvatarImage obsoleteAvatarImage = new ObsoleteAvatarImage(values);
        setWrapped(new im.actor.core.api.AvatarImage(new FileLocation(obsoleteAvatarImage.getFileReference().getFileId(), obsoleteAvatarImage.getFileReference().getAccessHash()), obsoleteAvatarImage.getWidth(), obsoleteAvatarImage.getHeight(), obsoleteAvatarImage.getFileReference().getFileSize()));
    }
