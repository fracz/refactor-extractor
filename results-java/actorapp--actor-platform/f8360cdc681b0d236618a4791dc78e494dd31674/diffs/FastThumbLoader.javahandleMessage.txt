@Override
public void handleMessage(Message msg) {
    synchronized (LOCKER) {
        if (!isActive) {
            return;
        }
        if (msg.what == currentRequest) {
            preview.setSrc((android.graphics.Bitmap) msg.obj);
            isActive = false;
        } else {
            return;
        }
    }
