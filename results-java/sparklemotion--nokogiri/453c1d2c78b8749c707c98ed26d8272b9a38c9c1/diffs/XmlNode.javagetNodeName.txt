    String str = null;

        str = node.getNodeName();
        if (str == null) {
            str = "";
        }
        if (str.equals("#document")) {
            str = "document";
        } else if (str.equals("#text")) {
            str = "text";
        } else {
            str = NokogiriHelpers.getLocalPart(str);
        }
        if (str == null)
            str = "";
        this.name = context.getRuntime().newString(str);
    }
    return (RubyString) this.name;
