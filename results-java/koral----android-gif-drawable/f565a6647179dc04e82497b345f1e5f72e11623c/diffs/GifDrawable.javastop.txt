/**
 * Stops the animation. Does nothing if GIF is not animated.
 * This method is thread-safe.
 */
@Override
public void stop() {
    mIsRunning = false;
    unscheduleSelf(mInvalidateTask);
    mExecutor.execute(new SafeRunnable() {

        @Override
        public void doWork() {
            mNativeInfoHandle.saveRemainder();
        }
    });
