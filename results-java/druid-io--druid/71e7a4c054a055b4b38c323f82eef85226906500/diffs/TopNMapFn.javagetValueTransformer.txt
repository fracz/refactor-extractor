public static Function<Object, Object> getValueTransformer(ValueType outputType) {
    switch(outputType) {
        case STRING:
            return STRING_TRANSFORMER;
        case LONG:
            return LONG_TRANSFORMER;
        case FLOAT:
            return FLOAT_TRANSFORMER;
        default:
            throw new IAE("invalid type: %s", outputType);
