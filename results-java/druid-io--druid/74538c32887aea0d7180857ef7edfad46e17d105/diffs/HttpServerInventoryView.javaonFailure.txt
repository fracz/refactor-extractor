@Override
public void onFailure(Throwable t) {
    try {
        String logMsg = StringUtils.nonStrictFormat("failed to fetch segment list from server [%s]. Return code [%s], Reason: [%s]", druidServer.getName(), responseHandler.status, responseHandler.description);
        if (hasUnstabilityTimeoutPassed()) {
            if (t != null) {
                log.error(t, logMsg);
            } else {
                log.error(logMsg);
            }
        } else {
            log.info("Temporary Failure. %s", logMsg);
            if (t != null) {
                log.debug(t, logMsg);
            } else {
                log.debug(logMsg);
            }
        }
        // sleep for a bit so that retry does not happen immediately.
        try {
            Thread.sleep(5000);
        } catch (InterruptedException ex) {
            Thread.currentThread().interrupt();
        }
    } finally {
        queue.add(druidServer.getName());
    }
