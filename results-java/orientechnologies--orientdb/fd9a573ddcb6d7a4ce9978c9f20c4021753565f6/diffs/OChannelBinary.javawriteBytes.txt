/**
 * Send byte arrays split in chunks of maxChunkSize bytes
 */
public OChannelBinary writeBytes(final byte[] iContent) throws IOException {
    if (iContent == null) {
        out.writeInt(-1);
    } else {
        out.writeInt(iContent.length);
        out.write(iContent);
    }
    return this;
