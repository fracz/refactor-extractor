public boolean shutdown() {
    try {
        boolean res = deinit();
        if (!OGlobalConfiguration.SERVER_BACKWARD_COMPATIBILITY.getValueAsBoolean() && databases != null) {
            databases.close();
            databases = null;
        }
        return res;
    } finally {
        startupLatch = null;
        if (shutdownLatch != null) {
            shutdownLatch.countDown();
            shutdownLatch = null;
        }
    }
