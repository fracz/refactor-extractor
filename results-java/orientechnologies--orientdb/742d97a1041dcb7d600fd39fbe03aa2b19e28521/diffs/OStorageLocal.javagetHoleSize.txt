/**
 * Returns the total size used by holes
 *
 * @throws IOException
 */
public long getHoleSize() {
    lock.acquireSharedLock();
    try {
        final List<ODataHoleInfo> holes = getHolesList();
        long size = 0;
        for (ODataHoleInfo h : holes) {
            if (h.dataOffset > -1 && h.size > 0)
                size += h.size;
        }
        return size;
    } finally {
        lock.releaseSharedLock();
    }
