@Override
public OSequence onSequenceUpdated(ODocument iDocument) {
    String name = OSequence.getSequenceName(iDocument);
    if (name == null) {
        return null;

    if (sequence == null) {
        return null;
    }
    sequence.onUpdate(iDocument);
    return sequence;
