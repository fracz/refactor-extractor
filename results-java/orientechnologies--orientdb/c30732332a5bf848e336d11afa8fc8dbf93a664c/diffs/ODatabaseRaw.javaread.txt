public ORawBuffer read(final ORecordId iRid, final String iFetchPlan) {
    if (!iRid.isValid())
        return null;
    OFetchHelper.checkFetchPlanValid(iFetchPlan);
    try {
        return storage.readRecord(databaseOwner, iRid, iFetchPlan, null);
    } catch (Throwable t) {
        if (iRid.isTemporary())
            throw new ODatabaseException("Error on retrieving record using temporary RecordId: " + iRid, t);
        else
            throw new ODatabaseException("Error on retrieving record " + iRid + " (cluster: " + storage.getPhysicalClusterNameById(iRid.clusterId) + ")", t);
    }
