        final int maxChildDepth = directory.getMaxLeftChildDepth(parentPath.nodeIndex);
        if (childDepth > maxChildDepth)

        if (childDepth + 1 > maxChildDepth)
            directory.setMaxRightChildDepth(parentPath.nodeIndex, (byte) childDepth);
    }
