    if (iNetwork == null)
        return;
    try {
        iNetwork.flush();

    }
    if (debug)
        System.out.println("<- req: " + getSessionId());
    iNetwork.getLockWrite().unlock();
    synchronized (networkPool) {
        networkPool.notifyAll();
    }
