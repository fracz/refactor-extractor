public long getSize() {
    long size = 0;
    lock.acquireSharedLock();
    try {
        for (ODataSegmentMemory d : dataSegments) if (d != null)
            size += d.getSize();
        for (OClusterMemory c : clusters) if (c != null)
            size += c.getSize();
    } finally {
        lock.releaseSharedLock();
    }
    return size;
