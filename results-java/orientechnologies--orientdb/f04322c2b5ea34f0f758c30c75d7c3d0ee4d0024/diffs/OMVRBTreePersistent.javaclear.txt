@Override
public void clear() {
    final long timer = PROFILER.startChrono();
    try {
        recordsToCommit.clear();
        entryPoints.clear();
        cache.clear();
        if (root != null) {
            try {
                ((OMVRBTreeEntryPersistent<K, V>) root).delete();
            } catch (Exception e) {
                dataProvider = new OMVRBTreeMapProvider<K, V>(null, dataProvider.getClusterName(), ((OMVRBTreeMapProvider<K, V>) dataProvider).getKeySerializer(), ((OMVRBTreeMapProvider<K, V>) dataProvider).getValueSerializer());
            }
            super.clear();
            markDirty();
        }
    } finally {
        PROFILER.stopChrono(PROFILER.getProcessMetric("mvrbtree.clear"), "Clear a MVRBTree", timer);
    }
