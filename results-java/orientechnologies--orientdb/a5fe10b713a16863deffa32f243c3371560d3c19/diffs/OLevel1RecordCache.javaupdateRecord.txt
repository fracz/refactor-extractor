 */

    if (!isEnabled() || record.getIdentity().getClusterId() == excludedCluster || !record.getIdentity().isValid())
        return;
    if (underlying.get(record.getIdentity()) != record)
        underlying.put(record);
    secondary.get().updateRecord(record);
