 * @see #isSubClassOf(OClass)
 */
public boolean isSuperClassOf(final OClass iClass) {
    OClass cls = iClass;
    while (cls != null) {
        if (cls.equals(this))
            return true;
        cls = cls.getSuperClass();
    }
    return false;
