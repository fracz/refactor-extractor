protected ORecordInternal<?> popRecord(final ORID iRID) {
    if (maxSize == 0)
        // PRECONDITIONS
        return null;
    acquireExclusiveLock();
    try {
        final ORecordInternal<?> record = entries.remove(iRID);
        return record;
    } finally {
        releaseExclusiveLock();
    }
