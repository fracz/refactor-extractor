    // TRANSFORMERS
    String name;
    transformers = new ArrayList<OTransformer>();
    if (iTransformers != null) {
        for (ODocument t : iTransformers) {
            name = t.fieldNames()[0];
            final OTransformer tr = factory.getTransformer(name);
            transformers.add(tr);
            configureComponent(tr, (ODocument) t.field(name), iContext);
        }
    }
