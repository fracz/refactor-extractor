/**
 * Flushes all the buffers of the passed file.
 *
 * @param iFile
 * file to flush on disk.
 */
public void flushFile(final OFileMMap iFile) {
    lock.readLock().lock();
    try {
        final List<OMMapBufferEntry> entries = bufferPoolPerFile.get(iFile);
        if (entries != null)
            for (OMMapBufferEntry entry : entries) entry.flush();
    } finally {
        lock.readLock().unlock();
    }
