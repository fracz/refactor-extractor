public int dictionarySize(final ODatabaseRecord<?> iDatabase) {
    checkConnection();
    do {
        boolean locked = lock.acquireExclusiveLock();
        try {
            final int reqId;
            try {
                reqId = beginRequest(OChannelBinaryProtocol.REQUEST_DICTIONARY_SIZE);
            } finally {
                endRequest();
            }
            try {
                beginResponse(reqId);
                return network.readInt();
            } finally {
                endResponse();
            }
        } catch (Exception e) {
            handleException("Error on getting size of database's dictionary", e);
        } finally {
            lock.releaseExclusiveLock(locked);
        }
    } while (true);
