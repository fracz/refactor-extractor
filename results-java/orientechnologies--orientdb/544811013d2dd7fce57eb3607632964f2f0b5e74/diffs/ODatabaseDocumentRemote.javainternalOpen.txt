public void internalOpen(String user, String password, OrientDBConfig config) {
    boolean failure = true;
    setupThreadOwner();
    try {
        storage.open(user, password, properties);
        status = STATUS.OPEN;
        initAtFirstOpen(user, password);
        // WAKE UP LISTENERS
        callOnOpenListeners();
        failure = false;
    } catch (OException e) {
        close();
        throw e;
    } catch (Exception e) {
        close();
        throw OException.wrapException(new ODatabaseException("Cannot open database url=" + getURL()), e);
    } finally {
        if (failure)
            owner.set(null);
    }
