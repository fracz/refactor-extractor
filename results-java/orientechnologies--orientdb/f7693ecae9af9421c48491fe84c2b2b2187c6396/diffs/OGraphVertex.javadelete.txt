@Override
public void delete() {
    // DELETE ALL THE IN-OUT EDGES FROM RAM
    if (inEdges != null && inEdges.get() != null)
        inEdges.get().clear();
    inEdges = null;
    if (outEdges != null && outEdges.get() != null)
        outEdges.get().clear();
    outEdges = null;
    List<ODocument> docs = (List<ODocument>) document.field(OGraphDatabase.VERTEX_FIELD_IN_EDGES);
    if (docs != null)
        while (!docs.isEmpty()) OGraphEdge.delete(database, docs.get(0));
    docs = (List<ODocument>) document.field(OGraphDatabase.VERTEX_FIELD_OUT_EDGES);
    if (docs != null)
        while (!docs.isEmpty()) OGraphEdge.delete(database, docs.get(0));
    database.unregisterPojo(this, document);
    document.delete();
    document = null;
