public static void main(String[] iArgs) throws InstantiationException, IllegalAccessException {
    OProfiler.getInstance().startRecording();
    database = new ODatabaseGraphTx(System.getProperty("url")).open("admin", "admin");
    database.declareIntent(new OIntentMassiveRead());
    database.begin(TXTYPE.NOTX);
    long time = System.currentTimeMillis();
    OGraphVertexOutIterator iterator = readSubNodes(database.getRoot("HighDensityGraph"));
    System.out.println("Read of the entire graph with depth=" + iterator.getMaxDeepLevel() + ". Total " + iterator.getCount() + " nodes in " + ((System.currentTimeMillis() - time) / 1000f) + " sec.");
    System.out.println("Repeating the same operation but with hot cache");
    time = System.currentTimeMillis();
    iterator = readSubNodes(database.getRoot("HighDensityGraph"));
    System.out.println("Read using the cache of the entire graph with depth=" + iterator.getMaxDeepLevel() + ". Total " + iterator.getCount() + " nodes in " + ((System.currentTimeMillis() - time) / 1000f) + " sec.");
    database.close();
