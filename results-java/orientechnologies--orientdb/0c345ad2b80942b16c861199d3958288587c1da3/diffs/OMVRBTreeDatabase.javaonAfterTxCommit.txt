    if (cache.keySet().size() == 0)
        return;

    final Set<ORID> keys = new HashSet<ORID>(cache.keySet());
    OMVRBTreeEntryDatabase<K, V> entry;
    for (ORID rid : keys) {
        if (rid.getClusterPosition() < -1) {
            // FIX IT IN CACHE
            entry = (OMVRBTreeEntryDatabase<K, V>) cache.get(rid);

            cache.remove(rid);
        }
    }
