    for (ODocument rec : database.browseCluster("Profile")) {
        Assert.assertFalse(rec.isDirty());
        List<?> followers = rec.field("followers");
        if (followers != null && followers.size() > 0) {

            Assert.assertTrue(rec.isDirty());
            break;

    database.close();
