@Override
public boolean lockRecord(final ORID iRecord, final String iNodeName) {
    final boolean locked = lockManager.putIfAbsent(iRecord, iNodeName) == null;
    if (ODistributedServerLog.isDebugEnabled())
        if (locked)
            ODistributedServerLog.debug(this, getLocalNodeName(), null, DIRECTION.NONE, "Distributed transaction: locked record %s in database '%s' owned by server '%s'", iRecord, databaseName, iNodeName);
        else
            ODistributedServerLog.debug(this, getLocalNodeName(), null, DIRECTION.NONE, "Distributed transaction: cannot lock record %s in database '%s' owned by server '%s'", iRecord, databaseName, iNodeName);
    return locked;
