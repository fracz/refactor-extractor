public void close() {
    if (status != STATUS.OPEN)
        return;
    if (currentIntent != null) {
        currentIntent.end(this);
        currentIntent = null;

    callOnCloseListeners();
    resetListeners();
    if (storage != null)
        storage.close();
    storage = null;
    status = STATUS.CLOSED;
