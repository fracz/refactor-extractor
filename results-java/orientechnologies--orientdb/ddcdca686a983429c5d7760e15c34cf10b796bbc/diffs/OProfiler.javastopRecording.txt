public void stopRecording() {
    if (recordingFrom == -1)
        return;
    acquireExclusiveLock();
    try {
        OLogManager.instance().config(this, "Profiler has stopped recording metrics");
        timer.cancel();
        lastSnapshot.clear();
        realTime.clear();
        recordingFrom = -1;
    } finally {
        releaseExclusiveLock();
    }
