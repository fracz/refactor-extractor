@Override
public long countRecords() {
    checkConnection();
    do {
        try {
            try {
                beginRequest(OChannelBinaryProtocol.REQUEST_DB_COUNTRECORDS);
            } finally {
                endRequest();
            }
            try {
                beginResponse();
                return network.readLong();
            } finally {
                endResponse();
            }
        } catch (Exception e) {
            handleException("Error on read database record count", e);
        }
    } while (true);
