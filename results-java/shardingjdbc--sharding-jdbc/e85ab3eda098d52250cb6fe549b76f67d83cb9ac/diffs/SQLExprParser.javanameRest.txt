public SQLName nameRest(SQLName name) {
    if (getLexer().equalToken(Token.DOT)) {
        getLexer().nextToken();
        if (getLexer().equalToken(Token.KEY)) {
            name = new SQLPropertyExpr(name, "KEY");
            getLexer().nextToken();
            return name;
        }
        if (!getLexer().equalToken(Token.LITERAL_ALIAS) && !getLexer().equalToken(Token.IDENTIFIER) && (!getLexer().containsToken())) {
            throw new ParserException("error, " + getLexer().getToken());
        }
        if (getLexer().equalToken(Token.LITERAL_ALIAS)) {
            name = new SQLPropertyExpr(name, '"' + getLexer().getLiterals() + '"');
        } else {
            name = new SQLPropertyExpr(name, getLexer().getLiterals());
        }
        getLexer().nextToken();
        name = nameRest(name);
    }
    return name;
