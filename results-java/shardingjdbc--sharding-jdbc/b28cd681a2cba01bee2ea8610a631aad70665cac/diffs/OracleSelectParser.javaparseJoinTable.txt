protected void parseJoinTable() {
    if (getLexer().equalToken(Token.HINT)) {
        getLexer().nextToken();
    }
    if (parseJoinType()) {
        parseTableSource();

            getLexer().nextToken();

            getLexer().skipParentheses();

        parseJoinTable();
    }
