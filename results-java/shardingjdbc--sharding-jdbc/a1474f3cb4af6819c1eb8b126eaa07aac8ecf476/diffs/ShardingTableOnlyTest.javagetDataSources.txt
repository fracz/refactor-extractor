@Override
protected final Map<DatabaseType, ShardingDataSource> getDataSources() throws SQLException {
    if (!shardingDataSources.isEmpty()) {
        return shardingDataSources;
    }
    Map<DatabaseType, Map<String, DataSource>> dataSourceMap = createDataSourceMap();
    for (Map.Entry<DatabaseType, Map<String, DataSource>> each : dataSourceMap.entrySet()) {
        DataSourceRule dataSourceRule = new DataSourceRule(each.getValue());
        TableRule orderTableRule = TableRule.builder("t_order").actualTables(Arrays.asList("t_order_0", "t_order_1", "t_order_2", "t_order_3", "t_order_4", "t_order_5", "t_order_6", "t_order_7", "t_order_8", "t_order_9")).dataSourceRule(dataSourceRule).build();
        TableRule orderItemTableRule = TableRule.builder("t_order_item").actualTables(Arrays.asList("t_order_item_0", "t_order_item_1", "t_order_item_2", "t_order_item_3", "t_order_item_4", "t_order_item_5", "t_order_item_6", "t_order_item_7", "t_order_item_8", "t_order_item_9")).dataSourceRule(dataSourceRule).generateKeyColumn("item_id").build();
        ShardingRule shardingRule = ShardingRule.builder(dataSourceRule).tableRules(orderTableRule, orderItemTableRule).bindingTableRules(new BindingTableRule(Arrays.asList(orderTableRule, orderItemTableRule))).databaseShardingStrategy(new NoneShardingStrategy()).tableShardingStrategy(new StandardShardingStrategy("order_id", new PreciseModuloTableShardingAlgorithm(), new RangeModuloTableShardingAlgorithm())).build();
        shardingDataSources.put(each.getKey(), new ShardingDataSource(shardingRule));
    }
    return shardingDataSources;
