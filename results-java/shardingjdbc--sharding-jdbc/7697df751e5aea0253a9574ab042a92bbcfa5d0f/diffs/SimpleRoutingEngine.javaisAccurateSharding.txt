private boolean isAccurateSharding(final List<ShardingValue<?>> shardingValues) {
    if (shardingValues.isEmpty()) {
        return false;
    }
    for (ShardingValue<?> each : shardingValues) {
        if (ShardingValue.ShardingValueType.RANGE == each.getType()) {
            return false;
        }
    }
    return true;
