/**
 * Loads the data of the file specified.
 */
public String loadData() {
    try {
        BufferedReader br = new BufferedReader(new FileReader(new File(this.fileName)));
        String text = "";
        String line = "";
        while ((line = br.readLine()) != null) {
            text += line + "\n";
        }
        this.loaded = true;
        br.close();
        return text;
    } catch (Exception e) {
        e.printStackTrace();
    }
    return null;
