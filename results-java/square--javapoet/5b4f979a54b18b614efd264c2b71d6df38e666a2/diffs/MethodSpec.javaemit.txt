void emit(CodeWriter codeWriter, ClassName enclosing, ImmutableSet<Modifier> implicitModifiers) {
    codeWriter.emitAnnotations(annotations, false);
    codeWriter.emitModifiers(modifiers, implicitModifiers);
    if (!typeVariables.isEmpty()) {
        codeWriter.emitTypeVariables(typeVariables);
        codeWriter.emit(" ");
    }
    if (name == Name.CONSTRUCTOR) {
        codeWriter.emit("$L(", enclosing.simpleName());
    } else {
        codeWriter.emit("$T $L(", returnType, name);
    }
    boolean firstParameter = true;
    for (ParameterSpec parameterSpec : parameters) {
        if (!firstParameter)
            codeWriter.emit(", ");
        parameterSpec.emit(codeWriter);
        firstParameter = false;
    }
    if (hasModifier(Modifier.ABSTRACT)) {
        codeWriter.emit(");\n");
        return;
    }
    codeWriter.emit(")");
    if (!exceptions.isEmpty()) {
        codeWriter.emit(" throws");
        boolean firstException = true;
        for (ClassName exception : exceptions) {
            if (!firstException)
                codeWriter.emit(",");
            codeWriter.emit(" $T", exception);
            firstException = false;
        }
    }
    codeWriter.emit(" {\n");
    codeWriter.indent();
    for (Snippet snippet : snippets) {
        codeWriter.emit(snippet);
    }
    codeWriter.unindent();
    codeWriter.emit("}\n");
