@Override
public void close(byte[] data) {
    // 获取prepare stmt id
    long pstmtId = ByteUtil.readUB4(data, 5);
    LOGGER.debug("close prepare stmt, stmtId = " + pstmtId);
    PreparedStatement pstmt = pstmtForId.remove(pstmtId);
    if (pstmt != null) {
        pstmtForSql.remove(pstmt.getStatement());
    }
