private int startTaskUnchecked() {
    if (isUsing()) {
        if (isRunning()) {
            throw new IllegalStateException(FileDownloadUtils.formatString("This task is running %d, if you" + " want to start the same task, please create a new one by" + " FileDownloader.create", getId()));
        } else {
            throw new IllegalStateException("This task is dirty to restart, If you want to " + "reuse this task, please invoke #reuse method manually and retry to " + "restart again.");
        }
    }
    if (!isAttached()) {
        setAttachKey(mListener.hashCode());
    }
    mHunter.intoLaunchPool();
    return getId();
