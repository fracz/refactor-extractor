                    mCurrentProfile = mProfiles.get(i);

                    mProfileFirst = mProfiles.get(i);

                    mProfileSecond = mProfiles.get(i);
                } else if (setCount == 3 && (overwrite || mProfileThird == null)) {

                    break;

                setCount++;
