        }

        @Override

            if (drawerItem != null) {
                if (drawerItem instanceof Nameable) {
                    Toast.makeText(MultiDrawerActivity.this, MultiDrawerActivity.this.getString(((Nameable) drawerItem).getNameRes()), Toast.LENGTH_SHORT).show();
