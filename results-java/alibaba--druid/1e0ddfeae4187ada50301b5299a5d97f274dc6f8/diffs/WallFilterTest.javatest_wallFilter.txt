        while (rs.next()) {
        }
        rs.close();
        stmt.close();
        conn.close();
    }
    Assert.assertEquals(10, wallFilter.getProvider().getTableStat("t").getFetchRowCount());

        stmt.execute("DELETE from t where FID = 0");
        stmt.close();
        conn.close();
    }
