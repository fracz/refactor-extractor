public static long fnv_64_lower_normalized(String name) {
    if (name == null) {
        return 0;
    }
    boolean quote = false;
    int len = name.length();
    if (len > 2) {
        char c0 = name.charAt(0);
        char c1 = name.charAt(len - 1);
        if ((c0 == '`' && c1 == '`') || (c0 == '"' && c1 == '"') || (c0 == '[' && c1 == ']')) {
            quote = true;
        }
    }
    if (quote) {
        return FNVUtils.fnv_64_lower(name, 1, len - 1);
    } else {
        return FNVUtils.fnv_64_lower(name);
    }
