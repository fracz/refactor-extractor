public SQLSelectItem parseSelectItem() {
    SQLExpr expr;
    boolean connectByRoot = false;
    if (lexer.token() == Token.IDENTIFIER) {
        String ident = lexer.stringVal();
        lexer.nextTokenComma();
        if ("CONNECT_BY_ROOT".equalsIgnoreCase(ident)) {
            connectByRoot = true;
            expr = new SQLIdentifierExpr(lexer.stringVal());
            lexer.nextToken();
        } else if ("DATE".equalsIgnoreCase(ident) && lexer.token() == Token.LITERAL_CHARS && (JdbcConstants.ORACLE.equals(getDbType()) || JdbcConstants.POSTGRESQL.equals(getDbType()))) {
            String literal = lexer.stringVal();
            lexer.nextToken();
            SQLDateExpr dateExpr = new SQLDateExpr();
            dateExpr.setLiteral(literal);
            expr = dateExpr;
        } else {
            expr = new SQLIdentifierExpr(ident);
        }
        if (lexer.token() != Token.COMMA) {
            expr = this.primaryRest(expr);
            expr = this.exprRest(expr);
        }
    } else if (lexer.token() == Token.DO || lexer.token() == Token.JOIN) {
        expr = this.name();
    } else {
        expr = expr();
    }
    final String alias;
    switch(lexer.token()) {
        case FULL:
            alias = lexer.stringVal();
            lexer.nextToken();
            break;
        default:
            alias = as();
            break;
    }
    return new SQLSelectItem(expr, alias, connectByRoot);
