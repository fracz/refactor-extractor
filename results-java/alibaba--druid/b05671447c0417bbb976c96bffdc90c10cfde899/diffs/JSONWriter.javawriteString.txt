public void writeString(String text) {
    if (text == null) {
        writeNull();
        return;
    }
    write('"');
    for (int i = 0; i < text.length(); ++i) {
        char c = text.charAt(i);
        if (c == '"') {
            write("\\\"");
        } else if (c == '\n') {
            write("\\n");
        } else if (c == '\r') {
            write("\\r");
        } else if (c == '\\') {
            write("\\\\");
        } else if (c == '\t') {
            write("\\t");
        } else {
            write(c);
        }
    }
    write('"');
