        if (lexer.token() == Token.IN) {
            lexer.nextToken();
            stmt.setIn(this.exprParser.expr());

            lexer.nextToken();
            stmt.setFrom(this.exprParser.expr());

            if (lexer.token() == (Token.COMMA)) {

                limit.setRowCount(exprParser.expr());
