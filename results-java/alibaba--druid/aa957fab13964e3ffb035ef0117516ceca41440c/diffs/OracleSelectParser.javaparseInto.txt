        SQLExpr expr = expr();
        if (lexer.token() != Token.COMMA) {
            x.setInto(expr);
            return;

        SQLListExpr list = new SQLListExpr();
        list.addItem(expr);

            lexer.nextToken();
            list.addItem(expr());

        x.setInto(list);
    }
