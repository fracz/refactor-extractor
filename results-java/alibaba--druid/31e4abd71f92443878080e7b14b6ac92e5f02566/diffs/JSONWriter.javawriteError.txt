public void writeError(Throwable error) {
    if (error == null) {
        write("null");
        return;
    }
    write("{\"Class\":");
    writeString(error.getClass().getName());
    write(",\"Message\":");
    writeString(error.getMessage());
    write(",\"StackTrace\":");
    writeString(IOUtils.getStackTrace(error));
    write('}');
