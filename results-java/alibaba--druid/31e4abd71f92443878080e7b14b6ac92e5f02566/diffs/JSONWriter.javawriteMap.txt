public void writeMap(Map<String, Object> map) {
    if (map == null) {
        write("null");
        return;
    }
    int entryIndex = 0;
    write('{');
    for (Map.Entry<String, Object> entry : map.entrySet()) {
        if (entryIndex != 0) {
            write(',');
        }
        writeString(entry.getKey());
        write(':');
        writeObject(entry.getValue());
        entryIndex++;
    }
    write('}');
