private void printSQLPartitions(List<SQLPartition> partitions) {
    int partitionsSize = partitions.size();
    if (partitionsSize > 0) {
        print0(" (");
        incrementIndent();
        for (int i = 0; i < partitionsSize; ++i) {
            println();
            partitions.get(i).accept(this);
            if (i != partitionsSize - 1) {
                print0(", ");
            }
        }
        decrementIndent();
        println();
        print(')');
    }
