            item.setAfterColumn(this.exprParser.name());
        } else if (identifierEquals("FIRST")) {
            lexer.nextToken();
            if (lexer.token() == Token.IDENTIFIER) {
                item.setFirstColumn(this.exprParser.name());

                item.setFirst(true);

        if (parenFlag && lexer.token() == Token.COMMA) {
            lexer.nextToken();
            continue;
        }
        break;

    stmt.getItems().add(item);

        accept(Token.RPAREN);
