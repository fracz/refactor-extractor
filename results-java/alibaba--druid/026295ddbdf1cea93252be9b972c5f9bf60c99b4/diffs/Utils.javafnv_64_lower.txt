public static long fnv_64_lower(String key) {
    long hashCode = 0xcbf29ce484222325L;
    for (int i = 0; i < key.length(); ++i) {
        char ch = key.charAt(i);
        if (ch == '_' || ch == '-') {
            continue;
        }
        if (ch >= 'A' && ch <= 'Z') {
            ch = (char) (ch + 32);
        }
        hashCode ^= ch;
        hashCode *= 0x100000001b3L;
    }
    return hashCode;
