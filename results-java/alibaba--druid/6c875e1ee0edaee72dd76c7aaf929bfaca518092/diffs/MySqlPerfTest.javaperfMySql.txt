long perfMySql(String sql) {
    long startMillis = System.currentTimeMillis();
    for (int i = 0; i < 1000 * 1000; ++i) {
        execMySql(sql);
    }
    long millis = System.currentTimeMillis() - startMillis;
    System.out.println("MySql\t" + millis);
    return millis;
