            if (lexer.token() != Token.COMMA) {
                break;

            lexer.nextToken();

        if (lexer.token() == Token.INTO) {
            lexer.nextToken();
            output.setInto(new SQLExprTableSource(this.name()));
            if (lexer.token() == (Token.LPAREN)) {
                lexer.nextToken();

        }
