public SQLStatement parseDropIndex() {
    accept(Token.INDEX);
    SQLDropIndexStatement stmt = new SQLDropIndexStatement(getDbType());
    stmt.setIndexName(this.exprParser.name());
    if (lexer.token() == Token.ON) {
        lexer.nextToken();
        stmt.setTableName(this.exprParser.name());
    }
    return stmt;
