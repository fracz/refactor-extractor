public SQLLimit parseLimit() {

        SQLExpr temp = this.expr();
        if (lexer.token == (Token.COMMA)) {
            limit.setOffset(temp);

            limit.setRowCount(this.expr());

            limit.setRowCount(temp);
            lexer.nextToken();
            limit.setOffset(this.expr());
        } else {
            limit.setRowCount(temp);
        }
        return limit;
