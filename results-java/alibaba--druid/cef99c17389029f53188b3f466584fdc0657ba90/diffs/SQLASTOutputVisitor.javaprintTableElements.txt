protected void printTableElements(List<SQLTableElement> tableElementList) {
    int size = tableElementList.size();
    if (size > 0) {
        print0(" (");
        this.indentCount++;
        println();
        for (int i = 0; i < size; ++i) {
            if (i != 0) {
                print0(",");
                println();
            }
            tableElementList.get(i).accept(this);
        }
        this.indentCount--;
        println();
        print(')');
    }
