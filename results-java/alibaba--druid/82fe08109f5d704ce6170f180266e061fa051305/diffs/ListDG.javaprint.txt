/*
     * 打印矩阵队列图
     */
public void print() {
    System.out.printf("== List Graph:\n");
    for (int i = 0; i < mVexs.size(); i++) {
        System.out.printf("%d(%c): ", i, mVexs.get(i).data);
        ENode node = mVexs.get(i).firstEdge;
        while (node != null) {
            System.out.printf("%d(%c) ", node.ivex, mVexs.get(node.ivex).data);
            node = node.nextEdge;
        }
        System.out.printf("\n");
    }
