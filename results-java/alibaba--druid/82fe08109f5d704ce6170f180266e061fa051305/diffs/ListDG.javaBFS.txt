/*
     * 广度优先搜索（类似于树的层次遍历）
     */
public void BFS() {
    int head = 0;
    int rear = 0;
    // 辅组队列
    int[] queue = new int[mVexs.size()];
    // 顶点访问标记
    boolean[] visited = new boolean[mVexs.size()];
    for (int i = 0; i < mVexs.size(); i++) visited[i] = false;
    System.out.printf("== BFS: ");
    for (int i = 0; i < mVexs.size(); i++) {
        if (!visited[i]) {
            visited[i] = true;
            System.out.printf("%c ", mVexs.get(i).data);
            // 入队列
            queue[rear++] = i;
        }
        while (head != rear) {
            // 出队列
            int j = queue[head++];
            ENode node = mVexs.get(j).firstEdge;
            while (node != null) {
                int k = node.ivex;
                if (!visited[k]) {
                    visited[k] = true;
                    System.out.printf("%c ", mVexs.get(k).data);
                    queue[rear++] = k;
                }
                node = node.nextEdge;
            }
        }
    }
    System.out.printf("\n");
