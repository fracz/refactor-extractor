            boolean quote = name.charAt(0) == '`' && name.charAt(name.length() - 1) == '`';
            boolean isNumber = true;
            int end = name.length();

                end--;
            }

                    isNumber = false;
                    break;
                }

                int start = quote ? 1 : 0;
