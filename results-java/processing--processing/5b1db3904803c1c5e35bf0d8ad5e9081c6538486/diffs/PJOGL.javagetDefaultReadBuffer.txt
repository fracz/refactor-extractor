@Override
protected int getDefaultReadBuffer() {
    if (fboLayerInUse) {
        return COLOR_ATTACHMENT0;
    } else if (capabilities.isFBO()) {
        return GL.GL_COLOR_ATTACHMENT0;
    } else if (capabilities.getDoubleBuffered()) {
        return GL.GL_BACK;
    } else {
        return GL.GL_FRONT;
    }
