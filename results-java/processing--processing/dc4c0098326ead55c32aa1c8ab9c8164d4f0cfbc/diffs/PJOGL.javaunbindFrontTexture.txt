        if (textureIsBound(TEXTURE_2D, frontTexAttach.getName())) {
            // that might be bound instead of this one.
            if (!texturingIsEnabled(TEXTURE_2D)) {
                enableTexturing(TEXTURE_2D);

                disableTexturing(TEXTURE_2D);
            } else {
                bindTexture(TEXTURE_2D, 0);
            }
        }
