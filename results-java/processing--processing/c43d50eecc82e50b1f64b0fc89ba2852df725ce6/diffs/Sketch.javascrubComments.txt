public static String scrubComments(String what) {
    char[] p = what.toCharArray();
    int index = 0;
    while (index < p.length) {
        // for any double slash comments, ignore until the end of the line
        if ((p[index] == '/') && (index < p.length - 1) && (p[index + 1] == '/')) {
            p[index++] = ' ';
            p[index++] = ' ';
            while ((index < p.length) && (p[index] != '\n')) {
                p[index++] = ' ';
            }
        // check to see if this is the start of a new multiline comment.
        // if it is, then make sure it's actually terminated somewhere.
        } else if ((p[index] == '/') && (index < p.length - 1) && (p[index + 1] == '*')) {
            p[index++] = ' ';
            p[index++] = ' ';
            boolean endOfRainbow = false;
            while (index < p.length - 1) {
                if ((p[index] == '*') && (p[index + 1] == '/')) {
                    p[index++] = ' ';
                    p[index++] = ' ';
                    endOfRainbow = true;
                    break;
                } else {
                    index++;
                }
            }
            if (!endOfRainbow) {
                throw new RuntimeException("Missing the */ from the end of a " + "/* comment */");
            }
        } else {
            // any old character, move along
            index++;
        }
    }
    return new String(p);
