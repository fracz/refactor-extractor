private static void tryStartLocalServerAndClient(String neoDbPath, boolean readOnly, String[] args) throws Exception {
    final LocalNeoShellServer server = new LocalNeoShellServer(neoDbPath, readOnly);
    Runtime.getRuntime().addShutdownHook(new Thread() {

        @Override
        public void run() {
            server.shutdown();
        }
    });
    System.out.println("NOTE: Using local neo service at '" + neoDbPath + "'");
    ShellClient client = new SameJvmClient(server);
    setSessionVariablesFromArgs(client, args);
    client.grabPrompt();
    server.shutdown();
