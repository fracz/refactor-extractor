        int id = 0;

        @Override
        protected R fetchNextOrNull() {
            scan: while (id <= highId && id >= 0) {
                if (!continueScanning) {
                    return null;

                R record = getRecord(store, id++);
                for (Predicate<? super R> filter : filters) {

                return record;

            return null;
        }
    };
