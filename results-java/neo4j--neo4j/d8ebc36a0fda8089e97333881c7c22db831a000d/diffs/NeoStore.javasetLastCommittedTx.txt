    long current = getLastCommittedTx();
    if ((current + 1) != txId && !isInRecoveryMode()) {
        throw new InvalidRecordException("Could not set tx commit id[" + txId + "] since the current one is[" + current + "]");
    }
    setRecord(3, txId);
    lastCommittedTx.set(txId);
