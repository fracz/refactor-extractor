private void addNodeIntoGraph() {
    Transaction tx = db.graph.beginTx();
    Node referenceNode = db.graph.getReferenceNode();
    Node myNewNode = db.graph.createNode();
    myNewNode.setProperty("id", UUID.randomUUID().toString());

            return "knows_about";
        }
    });
