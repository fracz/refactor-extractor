@Override
protected void forceAfterAppend(long ticket) throws IOException {
    LockSupport.unpark(forceThread);
    // Stay a while and listen... while:
    while (// the forcer hasn't yet caught up with me
    (ticket > forceTicket.get() || // OR I've wrapped around Long.MAX_VALUE
    !haveSameSign(ticket, forceTicket.get())) && // AND this appender hasn't yet been shut down
    !shutDown && // AND the forcer is of good health
    forceThread.checkHealth()) {
        // 0,1 ms
        LockSupport.parkNanos(100_000);
    }
