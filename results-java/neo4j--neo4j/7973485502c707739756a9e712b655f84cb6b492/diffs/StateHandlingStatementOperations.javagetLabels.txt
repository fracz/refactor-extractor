private PrimitiveIntCollection getLabels(NodeCursor node) {
    PrimitiveIntStack labelIds = new PrimitiveIntStack();
    try (LabelCursor labels = node.labels()) {
        while (labels.next()) {
            labelIds.push(labels.getLabel());
        }
    }
    return labelIds;
