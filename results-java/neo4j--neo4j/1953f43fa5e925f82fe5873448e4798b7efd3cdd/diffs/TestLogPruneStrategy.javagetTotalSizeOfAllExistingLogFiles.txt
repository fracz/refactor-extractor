public int getTotalSizeOfAllExistingLogFiles() {
    int size = 0;
    for (long version = getHighestLogVersion() - 1; version >= 0; version--) {
        File file = getFileName(version);
        if (file.exists())
            size += file.length();
        else
            break;
    }
    return size;
