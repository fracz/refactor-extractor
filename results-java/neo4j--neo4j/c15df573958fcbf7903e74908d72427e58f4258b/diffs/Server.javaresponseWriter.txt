private Runnable responseWriter(final RequestType<M> type, final Channel channel, final SlaveContext context, final ChunkingChannelBuffer targetBuffer, final ByteBuffer targetByteBuffer, final Response<R> response) {
    return new Runnable() {

        @SuppressWarnings("unchecked")
        public void run() {
            try {
                type.getObjectSerializer().write(response.response(), targetBuffer);
                writeTransactionStreams(response.transactions(), targetBuffer, targetByteBuffer);
                targetBuffer.done();
                responseWritten(type, channel, context);
            } catch (IOException e) {
                e.printStackTrace();
                throw new RuntimeException(e);
            } catch (RuntimeException e) {
                e.printStackTrace();
                throw e;
            }
        }
    };
