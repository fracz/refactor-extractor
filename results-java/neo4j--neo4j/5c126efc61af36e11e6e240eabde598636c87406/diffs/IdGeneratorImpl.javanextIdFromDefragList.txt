private long nextIdFromDefragList() {
    if (aggressiveReuse) {
        Long id = releasedIdList.poll();
        if (id != null) {
            defraggedIdCount--;
            return id.longValue();
        }
    }
    if (!idsReadFromFile.isEmpty() || canReadMoreIdBatches()) {
        if (idsReadFromFile.isEmpty()) {
            readIdBatch();
        }
        long id = idsReadFromFile.removeFirst();
        defraggedIdCount--;
        return id;
    }
    return -1;
