private static boolean encodeLatin1(String string, PropertyRecord target) {
    // see doEncode
    Bits bits = newBits();
    for (int i = 0; i < string.length(); i++) {
        if (i > 0)
            bits.shiftLeft(8);
        char c = string.charAt(i);
        if (c < 0 || c >= 256)
            return false;
        bits.or(c, 0xFF);
    }
    applyOnRecord(target, 10, string.length(), bits.getLongs());
    return true;
