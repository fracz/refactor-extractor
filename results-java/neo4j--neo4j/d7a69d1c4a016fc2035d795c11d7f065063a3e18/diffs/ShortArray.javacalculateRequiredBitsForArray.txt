public int calculateRequiredBitsForArray(Object array) {
    int arrayLength = Array.getLength(array);
    if (arrayLength == 0) {
        return 0;
    }
    int highest = 1;
    for (int i = 0; i < arrayLength; i++) {
        Object value = Array.get(array, i);
        highest = Math.max(highest, getRequiredBits(value));
    }
    return highest;
