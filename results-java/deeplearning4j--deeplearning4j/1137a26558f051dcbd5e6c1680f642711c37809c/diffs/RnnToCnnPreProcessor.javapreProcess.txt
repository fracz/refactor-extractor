        in2d = input.tensorAlongDimension(0, 1, 2);

        // Edge case: time series length = 1
        in2d = input.tensorAlongDimension(0, 1, 0);

        // Permute, so we get correct order after reshaping
        INDArray permuted = input.permute(0, 2, 1);
