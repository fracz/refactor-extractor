 * @return the negative log likelihood of the model
 */
public double negativeLogLikelihood() {
    MatrixUtil.complainAboutMissMatchedMatrices(input, labels);
    DoubleMatrix z = predict(input);
    // weight decay
    if (this.useRegularization) {
        double reg = (2 / l2) * MatrixFunctions.pow(this.W, 2).sum();
        return -labels.mul(log(z)).add(oneMinus(labels).mul(log(oneMinus(z)))).columnSums().mean() + reg;
    }
    return -labels.mul(log(z)).add(oneMinus(labels).mul(log(oneMinus(z)))).columnSums().mean();
