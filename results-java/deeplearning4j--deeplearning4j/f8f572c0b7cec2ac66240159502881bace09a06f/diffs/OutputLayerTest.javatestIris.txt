public void testIris() {
    RandomGenerator gen = new MersenneTwister(123);
    NeuralNetConfiguration conf = new NeuralNetConfiguration.Builder().weightInit(WeightInit.ZERO).lossFunction(LossFunctions.LossFunction.MCXENT).constrainGradientToUnitNorm(true).activationFunction(Activations.softMaxRows()).iterations(10).dist(Distributions.normal(gen, 1e-1)).rng(gen).regularization(true).l2(2e-4f).momentum(0.9f).learningRate(1e-1f).nIn(4).nOut(3).build();
    OutputLayer l = new OutputLayer.Builder().configure(conf).build();
    DataSetIterator iter = new IrisDataSetIterator(150, 150);
    DataSet next = iter.next(150);
    next.normalizeZeroMeanZeroUnitVariance();
    l.fit(next);
    int[] predictions = l.predict(next.getFeatureMatrix());
    Evaluation eval = new Evaluation();
    INDArray output = l.output(next.getFeatureMatrix());
    eval.eval(next.getLabels(), output);
    log.info("Score " + eval.stats());
