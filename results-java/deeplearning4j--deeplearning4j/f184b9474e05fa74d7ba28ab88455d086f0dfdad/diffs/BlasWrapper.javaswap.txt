/**
 * Compute x <-> y (swap two matrices)
 */
public NDArray swap(NDArray x, NDArray y) {
    // NativeBlas.dswap(x.length(), x.data(), 0, 1, y.data(), 0, 1);
    JavaBlas.rswap(x.length(), x.data(), 0, 1, y.data(), 0, 1);
    return y;
