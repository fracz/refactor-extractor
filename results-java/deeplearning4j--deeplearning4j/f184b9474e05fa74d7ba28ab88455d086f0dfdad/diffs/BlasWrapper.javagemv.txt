/**
 * Compute y <- alpha*op(a)*x + beta * y (general matrix vector
 * multiplication)
 */
public NDArray gemv(double alpha, NDArray a, NDArray x, double beta, NDArray y) {
    if (false) {
        NativeBlas.dgemv('N', a.rows(), a.columns(), alpha, a.data(), 0, a.rows(), x.data(), 0, 1, beta, y.data(), 0, 1);
    } else {
        if (beta == 0.0) {
            for (int j = 0; j < a.columns(); j++) {
                double xj = x.get(j);
                if (xj != 0.0) {
                    for (int i = 0; i < a.rows(); i++) {
                        y.putScalar(i, y.get(i) + a.get(i, j) * xj);
                    }
                }
            }
        } else {
            for (int j = 0; j < a.columns(); j++) {
                double byj = beta * y.data()[j];
                double xj = x.get(j);
                for (int i = 0; i < a.rows(); i++) {
                    y.putScalar(j, a.get(i, j) * xj + byj);
                }
            }
        }
    }
    return y;
