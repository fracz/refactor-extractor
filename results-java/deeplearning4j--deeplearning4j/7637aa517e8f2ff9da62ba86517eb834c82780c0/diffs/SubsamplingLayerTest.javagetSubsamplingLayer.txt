private Layer getSubsamplingLayer(SubsamplingLayer.PoolingType pooling) {
    NeuralNetConfiguration conf = new NeuralNetConfiguration.Builder().gradientNormalization(GradientNormalization.RenormalizeL2PerLayer).seed(123).layer(new SubsamplingLayer.Builder(pooling, new int[] { 2, 2 }).activation(Activation.RELU).build()).build();
    return conf.getLayer().instantiate(conf, null, 0, null, true);
