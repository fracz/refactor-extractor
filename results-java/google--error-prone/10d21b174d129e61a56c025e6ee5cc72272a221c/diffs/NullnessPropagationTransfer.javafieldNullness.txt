private static Nullness fieldNullness(ClassAndField accessed) {
    if (accessed == null) {
        return NULLABLE;
    }
    if (accessed.field.equals("class")) {
        return NONNULL;
    }
    if (accessed.isEnumConstant) {
        return NONNULL;
    }
    if (accessed.isPrimitive) {
        return NONNULL;
    }
    if (accessed.hasNonNullConstantValue) {
        return NONNULL;
    }
    return NULLABLE;
