    TypeI type = types.get(index);
    if (index >= minArgs) {
        if (maxArgs < Integer.MAX_VALUE || index < types.size() - 1) {
            return restrictByUndefined(type).toNonNullAnnotationString() + "=";
        }
        return "..." + restrictByUndefined(type).toNonNullAnnotationString();
    }
