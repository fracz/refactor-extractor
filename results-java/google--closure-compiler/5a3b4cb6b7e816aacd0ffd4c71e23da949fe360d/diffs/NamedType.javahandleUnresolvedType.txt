// Warns about this type being unresolved iff it's not a forward-declared
// type name.
private void handleUnresolvedType(ErrorReporter t, boolean ignoreForwardReferencedTypes) {
    if (registry.isLastGeneration()) {
        boolean beForgiving = forgiving || (ignoreForwardReferencedTypes && registry.isForwardDeclaredType(reference));
        if (!beForgiving && registry.isLastGeneration()) {
            t.warning("Unknown type " + reference, sourceName, lineno, null, charno);
        } else {
            setReferencedType(registry.getNativeObjectType(JSTypeNative.CHECKED_UNKNOWN_TYPE));
        }
        setResolvedTypeInternal(getReferencedType());
    } else {
        setResolvedTypeInternal(this);
    }
