void removeRef(Ref ref) {
    if (ref == declaration || (refs != null && refs.remove(ref))) {
        if (ref == declaration) {
            declaration = null;
            if (refs != null) {
                for (Ref maybeNewDecl : refs) {
                    if (maybeNewDecl.type == Ref.Type.SET_FROM_GLOBAL) {
                        declaration = maybeNewDecl;
                        refs.remove(declaration);
                        break;
                    }
                }
            }
        }
        switch(ref.type) {
            case SET_FROM_GLOBAL:
                globalSets--;
                break;
            case SET_FROM_LOCAL:
                localSets--;
                break;
            case PROTOTYPE_GET:
            case DIRECT_GET:
                totalGets--;
                break;
            case ALIASING_GET:
                aliasingGets--;
                totalGets--;
                break;
            case CALL_GET:
                callGets--;
                totalGets--;
                break;
            default:
                throw new IllegalStateException();
        }
    }
