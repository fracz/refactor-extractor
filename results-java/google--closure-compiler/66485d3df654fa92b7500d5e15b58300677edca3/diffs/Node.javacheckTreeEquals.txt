 */
public String checkTreeEquals(Node node2) {
    Pair<Node, Node> diff = checkTreeEqualsImpl(node2);
    if (diff != null) {
        return "Node tree inequality:" + "\nTree1:\n" + toStringTree() + "\n\nTree2:\n" + node2.toStringTree();
