private FlowScope restrictParameter(Node parameter, JSType type, FlowScope blindScope, Function<Pair<JSType, Boolean>, JSType> restriction, boolean outcome) {
    // restricting
    type = restriction.apply(Pair.of(type, outcome));
    // changing the scope
    if (type != null) {
        FlowScope informed = blindScope.createChildFlowScope();
        declareNameInScope(informed, parameter, type);
        return informed;
    } else {
        return blindScope;
    }
