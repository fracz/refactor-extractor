public void testFoldLiteralConstructors() {
    // Can fold
    fold("x = new Array", "x = []");
    fold("x = new Array()", "x = []");
    fold("x = new Object", "x = ({})");
    fold("x = new Object()", "x = ({})");
    // Cannot fold, there are arguments
    fold("x = new Array(7)", "x = new Array(7)");
    // Cannot fold, the constructor being used is actually a local function
    fold("x = " + "(function(){function Object(){this.x=4};return new Object();})();", "x = (function(){function Object(){this.x=4}return new Object})()");
