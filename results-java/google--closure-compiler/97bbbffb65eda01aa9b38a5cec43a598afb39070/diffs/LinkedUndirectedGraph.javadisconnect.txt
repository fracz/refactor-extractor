    node = getUndirectedGraphNode(destValue);
    if (node == null) {
        throw new IllegalArgumentException(destValue + " does not exist in graph");
    }
    LinkedUndirectedGraphNode<N, E> dest = (LinkedUndirectedGraphNode<N, E>) node;
    for (UndiGraphEdge<N, E> edge : getUndirectedGraphEdges(srcValue, destValue)) {
