    if (type instanceof UnionType) {
        type = type.restrictByNotNullOrUndefined();

            for (JSType alt : ((UnionType) type).getAlternates()) {
                if (isInvalidatingType(alt)) {
                    return true;
                }
            }
            return false;
        }
