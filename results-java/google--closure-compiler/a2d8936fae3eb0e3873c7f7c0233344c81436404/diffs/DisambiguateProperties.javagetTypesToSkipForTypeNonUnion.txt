private Set<JSType> getTypesToSkipForTypeNonUnion(JSType type) {
    Set<JSType> types = Sets.newHashSet();
    JSType skipType = type;
    while (skipType != null) {
        types.add(skipType);
        if (skipType instanceof ObjectType) {
            skipType = ((ObjectType) skipType).getImplicitPrototype();
        } else {
            break;
        }
    }
    return types;
