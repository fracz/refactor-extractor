// A special-case of meet
static NominalType pickSubclass(NominalType c1, NominalType c2) {
    if (c1 == null) {
        return c2;
    } else if (c2 == null) {
        return c1;
    }
    if (c1.isSubtypeOf(c2)) {
        return c1;
    }
    return c2.isSubtypeOf(c1) ? c2 : null;
