/**
 * Mark names that are constants by convention.
 */
private void annotateConstants(Node n, Node parent) {
    Preconditions.checkState(n.getType() == Token.NAME || n.getType() == Token.STRING);
    // or an OBJECTLIT key.
    if (n.getType() != Token.STRING || parent.getType() == Token.OBJECTLIT || parent.getType() == Token.GETPROP) {
        if (NodeUtil.isConstantByConvention(convention, n, parent)) {
            n.putBooleanProp(Node.IS_CONSTANT_NAME, true);
        }
    }
