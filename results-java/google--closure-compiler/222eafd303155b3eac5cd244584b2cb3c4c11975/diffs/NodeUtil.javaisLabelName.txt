/**
 * @return Whether the node is a label name.
 */
static boolean isLabelName(Node n) {
    if (n != null && n.getType() == Token.NAME) {
        Node parent = n.getParent();
        switch(parent.getType()) {
            case Token.LABEL:
            case Token.BREAK:
            case Token.CONTINUE:
                if (n == parent.getFirstChild()) {
                    return true;
                }
        }
    }
    return false;
