private Node maybeReplaceNode(Node lhs, Node rhs) {
    if (lhs.isEquivalentTo(rhs)) {
        return lhs;
    }
    Node parent = lhs.getParent();
    parent.replaceChild(lhs, rhs);
    reportCodeChange();
    return rhs;
