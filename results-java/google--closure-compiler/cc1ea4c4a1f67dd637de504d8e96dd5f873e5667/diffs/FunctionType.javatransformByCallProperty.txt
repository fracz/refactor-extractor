        return QMARK_FUNCTION;
    }

    builder.addReqFormal(fromReceiverToFirstFormal());
    for (JSType type : this.requiredFormals) {
        builder.addReqFormal(type);
    }
    for (JSType type : this.optionalFormals) {
        builder.addOptFormal(type);
