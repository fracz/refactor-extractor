public JSType substituteGenerics(Map<String, JSType> concreteTypes) {
    if (isTop() || isUnknown() || getObjs().isEmpty() && getTypeVar() == null || concreteTypes.isEmpty()) {
        return this;
    }
    ImmutableSet.Builder<ObjectType> builder = ImmutableSet.builder();
    for (ObjectType obj : getObjs()) {
        builder.add(obj.substituteGenerics(concreteTypes));
    }
    JSType current = makeType(getMask() & ~TYPEVAR_MASK, builder.build(), null, getEnums());
    if (hasTypeVariable()) {
        current = JSType.join(current, concreteTypes.containsKey(getTypeVar()) ? concreteTypes.get(getTypeVar()) : fromTypeVar(getTypeVar()));
    }
    return current;
