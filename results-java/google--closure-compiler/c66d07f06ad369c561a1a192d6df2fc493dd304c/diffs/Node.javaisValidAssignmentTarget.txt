public boolean isValidAssignmentTarget() {
    switch(getType()) {
        // the cryptic case from cl/41958159.
        case Token.CAST:
        case Token.NAME:
        case Token.GETPROP:
        case Token.GETELEM:
        case Token.ARRAY_PATTERN:
        case Token.OBJECT_PATTERN:
            return true;
        default:
            return false;
    }
