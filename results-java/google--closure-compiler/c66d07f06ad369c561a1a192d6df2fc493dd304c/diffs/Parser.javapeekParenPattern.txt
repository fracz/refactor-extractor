private boolean peekParenPattern(PatternKind kind, EnumSet<TokenType> follow) {
    if (!peekParenPatternStart()) {
        return false;
    }
    Parser p = createLookaheadParser();
    p.parsePattern(kind);
    return !p.errorReporter.hadError() && follow.contains(p.peekType());
