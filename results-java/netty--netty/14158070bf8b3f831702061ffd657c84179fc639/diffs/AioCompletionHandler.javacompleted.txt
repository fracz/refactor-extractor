                completed0(result, channel);

                STACK_DEPTH.set(d);
            }
        } else {
            // schedule it with a special runnable to make sure we keep the right

            loop.execute(new AioEventLoop.RecursionBreakingRunnable() {

                }
            });
        }
    } else {

            }
        });
    }
