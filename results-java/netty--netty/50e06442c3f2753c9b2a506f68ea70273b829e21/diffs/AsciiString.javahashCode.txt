@Override
public int hashCode() {
    int hash = this.hash;
    final byte[] value = this.value;
    if (hash != 0 || value.length == 0) {
        return hash;
    }
    for (byte b : value) {
        hash = hash * 31 ^ b & 31;
    }
    return this.hash = hash;
