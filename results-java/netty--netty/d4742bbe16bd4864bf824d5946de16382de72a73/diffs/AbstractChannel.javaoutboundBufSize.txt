private int outboundBufSize() {
    final int bufSize;
    final ChannelHandlerContext ctx = directOutboundContext();
    if (ctx.hasOutboundByteBuffer()) {
        bufSize = ctx.outboundByteBuffer().readableBytes();
    } else {
        bufSize = ctx.outboundMessageBuffer().size();
    }
    return bufSize;
