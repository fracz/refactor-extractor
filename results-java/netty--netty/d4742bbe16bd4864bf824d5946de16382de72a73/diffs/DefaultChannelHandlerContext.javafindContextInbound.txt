private DefaultChannelHandlerContext findContextInbound() {
    DefaultChannelHandlerContext ctx = this;
    do {
        ctx = ctx.next;
    } while (ctx != null && !(ctx.handler() instanceof ChannelStateHandler));
    return ctx;
