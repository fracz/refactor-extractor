@Override
public String toString() {
    StringBuilder buf = new StringBuilder();
    buf.append(name());
    buf.append('=');
    buf.append(value());
    if (domain() != null) {
        buf.append(", domain=");
        buf.append(domain());
    }
    if (path() != null) {
        buf.append(", path=");
        buf.append(path());
    }
    if (comment() != null) {
        buf.append(", comment=");
        buf.append(comment());
    }
    if (maxAge() >= 0) {
        buf.append(", maxAge=");
        buf.append(maxAge());
        buf.append('s');
    }
    if (isSecure()) {
        buf.append(", secure");
    }
    if (isHttpOnly()) {
        buf.append(", HTTPOnly");
    }
    return buf.toString();
