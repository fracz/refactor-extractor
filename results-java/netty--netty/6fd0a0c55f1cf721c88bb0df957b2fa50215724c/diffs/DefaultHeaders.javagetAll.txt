    int h = hashCodeGenerator.generateHashCode(name);
    int i = index(h);
    HeaderEntry e = entries.get(i);
    while (e != null) {
        if (e.hash == h && keyComparator.compare(e.name, name) == 0) {
            values.add(e.value);
        }
        e = e.next;

    return values;
