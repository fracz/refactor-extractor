public void run() throws Exception {
    System.err.println("Proxying *:" + localPort + " to " + remoteHost + ':' + remotePort + " ...");
    // Configure the bootstrap.
    ServerBootstrap b = new ServerBootstrap();
    try {
        b.group(new NioEventLoopGroup(), new NioEventLoopGroup()).channel(new NioServerSocketChannel()).localAddress(localPort).childHandler(new HexDumpProxyInitializer(remoteHost, remotePort));
        b.bind().sync().channel().closeFuture().sync();
    } finally {
        b.shutdown();
    }
