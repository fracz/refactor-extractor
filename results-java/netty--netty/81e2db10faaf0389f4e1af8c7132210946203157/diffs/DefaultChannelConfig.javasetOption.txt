public <T> boolean setOption(ChannelOption<T> option, T value) {
    validate(option, value);
    if (option == CONNECT_TIMEOUT_MILLIS) {
        setConnectTimeoutMillis((Integer) value);

        setWriteSpinCount((Integer) value);
    } else {
        return false;
    }
    return true;
