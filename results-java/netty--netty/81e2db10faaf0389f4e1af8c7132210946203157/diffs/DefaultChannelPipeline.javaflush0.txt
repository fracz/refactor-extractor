    try {
        ctx.flushBridge();
        ((ChannelOperationHandler) ctx.handler()).flush(ctx, future);
    } catch (Throwable t) {
        notifyHandlerException(t);
    } finally {
        if (ctx.outByteBuf != null) {
            ByteBuf buf = ctx.outByteBuf;
            if (!buf.readable()) {
                buf.discardReadBytes();

        }
    }
