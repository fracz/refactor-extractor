static <T> List<Keyframe<T>> parseKeyframes(JSONArray json, LottieComposition composition, float scale, AnimatableValue<T, ?> animatableValue) {
    int length = json.length();
    if (length == 0) {
        return Collections.emptyList();
    }
    List<Keyframe<T>> keyframes = new ArrayList<>();
    for (int i = 0; i < length; i++) {
        keyframes.add(Keyframe.Factory.newInstance(json.optJSONObject(i), composition, scale, animatableValue));
    }
    setEndFrames(keyframes);
    return keyframes;
