public void doVerb(MessageIn<ReadCommand> message, String id) {
    if (StorageService.instance.isBootstrapMode()) {
        throw new RuntimeException("Cannot service reads while bootstrapping!");
    }
    try {
        ReadCommand command = message.payload;
        Table table = Table.open(command.table);
        Row row = command.getRow(table);
        MessageOut<ReadResponse> reply = new MessageOut<ReadResponse>(MessagingService.Verb.REQUEST_RESPONSE, getResponse(command, row), ReadResponse.serializer);
        if (logger.isDebugEnabled())
            logger.debug(String.format("Read key %s; sending response to %s@%s", ByteBufferUtil.bytesToHex(command.key), id, message.from));
        MessagingService.instance().sendReply(reply, id, message.from);
    } catch (IOException ex) {
        throw new RuntimeException(ex);
    }
