        assertEquals(ByteBufferUtil.bytes(0), row.getCell(cfs.metadata.getColumnDefinition(new ColumnIdentifier("c", false))).value());

        for (int i = 0; i < 2; i++) {
            row = Util.getOnlyRow(Util.cmd(cfs, "0").columns("c").includeRow(i).build());
