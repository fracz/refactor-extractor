private List<Column> thriftifyColumnsAsColumns(Collection<Cell> cells, long now) {
    List<Column> thriftColumns = new ArrayList<Column>(cells.size());
    for (Cell cell : cells) {
        if (!cell.isLive(now))
            continue;
        thriftColumns.add(thriftifySubColumn(cell));
    }
    return thriftColumns;
