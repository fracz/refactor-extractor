static Descriptor rename(Descriptor tmpdesc, Set<Component> components) {
    Descriptor newdesc = tmpdesc.asTemporary(false);
    try {
        for (Component component : components) FBUtilities.renameWithConfirm(tmpdesc.filenameFor(component), newdesc.filenameFor(component));
    } catch (IOException e) {
        throw new IOError(e);
    }
    return newdesc;
