static ColumnFamily resolveSuperset(List<ColumnFamily> versions) {
    assert versions.size() > 0;
    ColumnFamily resolved = null;
    for (ColumnFamily cf : versions) {
        if (cf != null) {
            resolved = cf.cloneMe();
            break;
        }
    }
    if (resolved == null)
        return null;
    for (ColumnFamily cf : versions) {
        resolved.resolve(cf);
    }
    return resolved;
