/**
 * Read a sequence of entries asynchronously.
 *
 * @param firstEntry
 * id of first entry of sequence
 * @param lastEntry
 * id of last entry of sequence
 * @param cb
 * object implementing read callback interface
 * @param ctx
 * control object
 */
public void asyncReadEntries(long firstEntry, long lastEntry, ReadCallback cb, Object ctx) throws InterruptedException {
    // Little sanity check
    if (firstEntry < 0 || lastEntry > lastAddConfirmed || firstEntry > lastEntry) {
        cb.readComplete(BKException.Code.ReadException, this, null, ctx);
        return;
    }
    new PendingReadOp(this, firstEntry, lastEntry, cb, ctx).initiate();
    opCounterSem.acquire();
