/**
 * Add a shortcut, or update one with the same ID, with taking over existing flags.
 *
 * It checks the max number of dynamic shortcuts.
 */
public void addDynamicShortcut(@NonNull ShortcutService s, @NonNull ShortcutInfo newShortcut) {
    onShortcutPublish(s);
    newShortcut.addFlags(ShortcutInfo.FLAG_DYNAMIC);
    final ShortcutInfo oldShortcut = mShortcuts.get(newShortcut.getId());
    final boolean wasPinned;
    final int newDynamicCount;
    if (oldShortcut == null) {
        wasPinned = false;
        // adding a dynamic shortcut.
        newDynamicCount = mDynamicShortcutCount + 1;
    } else {
        wasPinned = oldShortcut.isPinned();
        if (oldShortcut.isDynamic()) {
            // not adding a dynamic shortcut.
            newDynamicCount = mDynamicShortcutCount;
        } else {
            // adding a dynamic shortcut.
            newDynamicCount = mDynamicShortcutCount + 1;
        }
    }
    // Make sure there's still room.
    s.enforceMaxDynamicShortcuts(newDynamicCount);
    // Okay, make it dynamic and add.
    if (wasPinned) {
        newShortcut.addFlags(ShortcutInfo.FLAG_PINNED);
    }
    addShortcut(s, newShortcut);
    mDynamicShortcutCount = newDynamicCount;
