public String getUserData(Account account, String key) throws RemoteException {
    SQLiteDatabase db = mOpenHelper.getReadableDatabase();
    db.beginTransaction();
    try {
        long accountId = getAccountId(db, account);
        if (accountId < 0) {
            return null;
        }
        Cursor cursor = db.query(TABLE_EXTRAS, new String[] { EXTRAS_VALUE }, EXTRAS_ACCOUNTS_ID + "=" + accountId + " AND " + EXTRAS_KEY + "=?", new String[] { key }, null, null, null);
        try {
            if (cursor.moveToNext()) {
                return cursor.getString(0);
            }
            return null;
        } finally {
            cursor.close();
        }
    } finally {
        db.setTransactionSuccessful();
        db.endTransaction();
    }
