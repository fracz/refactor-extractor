@Override
public boolean canReverse() {
    if (!mReversible) {
        return false;
    }
    // Loop to make sure all the Nodes can reverse.
    for (Node node : mNodes) {
        if (!node.animation.canReverse() || node.animation.getStartDelay() > 0) {
            return false;
        }
    }
    return true;
