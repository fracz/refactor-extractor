@Override
public void pause() {
    boolean previouslyPaused = mPaused;
    super.pause();
    if (!previouslyPaused && mPaused) {
        if (mDelayAnim != null) {
            mDelayAnim.pause();
        } else {
            for (Node node : mNodes) {
                node.animation.pause();
            }
        }
    }
