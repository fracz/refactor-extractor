void sendIntent(PendingIntent pendingIntent, Intent intent) {
    try {
        mWakeLock.acquire();
        pendingIntent.send(mContext, 0, intent, this, null, permission.ACCESS_FINE_LOCATION);
    } catch (PendingIntent.CanceledException e) {
        removeFence(pendingIntent);
        mWakeLock.release();
    }
