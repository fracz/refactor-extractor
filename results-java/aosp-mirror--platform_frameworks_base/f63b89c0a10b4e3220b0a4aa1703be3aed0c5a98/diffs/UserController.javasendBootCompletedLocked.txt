void sendBootCompletedLocked(IIntentReceiver resultTo) {
    for (int i = 0; i < mStartedUsers.size(); i++) {
        UserState uss = mStartedUsers.valueAt(i);
        if (uss.mState == UserState.STATE_BOOTING) {
            uss.mState = UserState.STATE_RUNNING;
            final int userId = mStartedUsers.keyAt(i);
            Intent intent = new Intent(Intent.ACTION_BOOT_COMPLETED, null);
            intent.putExtra(Intent.EXTRA_USER_HANDLE, userId);
            intent.addFlags(Intent.FLAG_RECEIVER_NO_ABORT);
            mService.broadcastIntentLocked(null, null, intent, null, resultTo, 0, null, null, new String[] { android.Manifest.permission.RECEIVE_BOOT_COMPLETED }, AppOpsManager.OP_NONE, null, true, false, ActivityManagerService.MY_PID, Process.SYSTEM_UID, userId);
        }
    }
