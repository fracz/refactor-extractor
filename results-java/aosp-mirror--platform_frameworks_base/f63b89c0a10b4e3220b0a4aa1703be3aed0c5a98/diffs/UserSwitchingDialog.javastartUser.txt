void startUser() {
    synchronized (this) {
        if (!mStartedUser) {
            mService.startUserInForeground(mUserId, this);
            mStartedUser = true;
            final View decorView = getWindow().getDecorView();
            if (decorView != null) {
                decorView.getViewTreeObserver().removeOnWindowShownListener(this);
            }
            mHandler.removeMessages(MSG_START_USER);
        }
    }
