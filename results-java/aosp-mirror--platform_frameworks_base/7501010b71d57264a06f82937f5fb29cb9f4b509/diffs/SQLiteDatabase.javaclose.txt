/**
 * Close the database.
 */
public void close() {
    if (!isOpen()) {
        // already closed
        return;
    }
    lock();
    try {
        closeClosable();
        // finalize ALL statements queued up so far
        closePendingStatements();
        // close this database instance - regardless of its reference count value
        dbclose();
        if (mConnectionPool != null) {
            mConnectionPool.close();
        }
    } finally {
        unlock();
    }
