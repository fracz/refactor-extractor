/**
 * Clears all existing bindings. Unset bindings are treated as NULL.
 */
public void clearBindings() {
    mDatabase.verifyDbIsOpen();
    acquireReference();
    try {
        native_clear_bindings();
    } finally {
        releaseReference();
    }
