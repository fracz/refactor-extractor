boolean removeAppToken(AppWindowToken wtoken) {
    mAppTokens.remove(wtoken);
    if (mAppTokens.size() == 0) {
        EventLog.writeEvent(com.android.server.EventLogTags.WM_TASK_REMOVED, taskId, "removeAppToken: last token");
        return true;
    }
    return false;
