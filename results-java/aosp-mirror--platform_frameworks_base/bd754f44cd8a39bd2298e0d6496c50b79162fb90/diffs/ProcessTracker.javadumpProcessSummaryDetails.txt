static void dumpProcessSummaryDetails(PrintWriter pw, ProcessState proc, String prefix, String label, int[] screenStates, int[] memStates, int[] procStates, long now, boolean full) {
    ProcessDataCollection totals = new ProcessDataCollection(screenStates, memStates, procStates);
    computeProcessData(proc, totals, now);
    if (totals.totalTime != 0 || totals.numPss != 0) {
        if (prefix != null) {
            pw.print(prefix);
        }
        if (label != null) {
            pw.print(label);
        }
        totals.print(pw, full);
        if (prefix != null) {
            pw.println();
        }
    }
