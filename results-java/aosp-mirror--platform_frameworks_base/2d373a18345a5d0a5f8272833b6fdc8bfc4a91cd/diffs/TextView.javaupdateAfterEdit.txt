void updateAfterEdit() {
    invalidate();
    int curs = getSelectionStart();
    if (curs >= 0 || (mGravity & Gravity.VERTICAL_GRAVITY_MASK) == Gravity.BOTTOM) {
        registerForPreDraw();
    }
    if (curs >= 0) {
        mHighlightPathBogus = true;
        if (mEditor != null)
            getEditor().makeBlink();
        bringPointIntoView(curs);
    }
    checkForResize();
