private static boolean isCanonicalFilePath(String path) {
    if (path.contains("..") || path.contains("//")) {
        if (RefactoredBackupManagerService.MORE_DEBUG) {
            Slog.w(RefactoredBackupManagerService.TAG, "Dropping invalid path " + path);
        }
        return false;
    }
    return true;
