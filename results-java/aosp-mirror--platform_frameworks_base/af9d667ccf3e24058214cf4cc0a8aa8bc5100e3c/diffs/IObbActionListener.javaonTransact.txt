    switch(code) {
        case INTERFACE_TRANSACTION:
            {
                reply.writeString(DESCRIPTOR);
                return true;
            }
        case TRANSACTION_onObbResult:
            {
                data.enforceInterface(DESCRIPTOR);
                String filename;
                filename = data.readString();
                String status;
                status = data.readString();

                return true;

    }
