static int native_setPathEffect(int native_object, int effect) {
    // get the delegate from the native int.
    Paint_Delegate delegate = sManager.getDelegate(native_object);
    if (delegate == null) {
        assert false;
        return effect;
    }
    return delegate.mPathEffect = effect;
