public void testNameDoCoMo() {
    mVerifier.initForExportTest(VCardConfig.VCARD_TYPE_DOCOMO);
    ContactEntry entry = mVerifier.addInputEntry();
    entry.addContentValues(StructuredName.CONTENT_ITEM_TYPE).put(StructuredName.FAMILY_NAME, "\u3075\u308B\u3069").put(StructuredName.GIVEN_NAME, "\u3091\u308A\u304B").put(StructuredName.MIDDLE_NAME, "B").put(StructuredName.PREFIX, "Dr.").put(StructuredName.SUFFIX, "Ph.D");
    final String fullName = "Dr. \u3075\u308B\u3069 B \u3091\u308A\u304B Ph.D";
