private void registerSensors(int[] sensors) {
    for (int sensorType : sensors) {
        Sensor s = mSensorManager.getDefaultSensor(sensorType);
        if (s != null) {
            mSensorManager.registerListener(this, s, SensorManager.SENSOR_DELAY_GAME);
        }
    }
