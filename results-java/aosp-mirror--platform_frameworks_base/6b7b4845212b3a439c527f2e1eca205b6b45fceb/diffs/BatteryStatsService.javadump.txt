@Override
protected void dump(FileDescriptor fd, PrintWriter pw, String[] args) {
    synchronized (mStats) {
        boolean isCheckin = false;
        if (args != null) {
            for (String arg : args) {
                if ("--checkin".equals(arg)) {
                    isCheckin = true;
                    break;
                }
            }
        }
        if (isCheckin)
            mStats.dumpCheckinLocked(pw, args);
        else
            mStats.dumpLocked(pw);
    }
