@Override
public void onTaskLaunching(boolean isTaskInStackBounds) {
    mTaskLaunched = true;
    // Fade out the scrim
    RecentsConfiguration config = RecentsConfiguration.getInstance();
    if (!isTaskInStackBounds && config.hasNavBarScrim()) {
        mNavBarScrimView.animate().alpha(0f).setStartDelay(0).setDuration(config.taskBarExitAnimDuration).setInterpolator(config.fastOutSlowInInterpolator).withLayer().start();
    }
