@Override
public WindowInsets onApplyWindowInsets(WindowInsets insets) {
    if (Console.Enabled) {
        Console.log(Constants.Log.UI.MeasureAndLayout, "[RecentsView|fitSystemWindows]", "insets: " + insets, Console.AnsiGreen);
    }
    // Update the configuration with the latest system insets and trigger a relayout
    RecentsConfiguration config = RecentsConfiguration.getInstance();
    config.updateSystemInsets(insets.getSystemWindowInsets());
    requestLayout();
    return insets.consumeSystemWindowInsets(false, false, false, true);
