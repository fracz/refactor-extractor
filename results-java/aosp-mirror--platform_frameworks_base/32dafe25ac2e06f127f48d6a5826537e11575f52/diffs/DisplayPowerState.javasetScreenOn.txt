/**
 * Sets whether the screen is on or off.
 */
public void setScreenOn(boolean on) {
    if (mScreenOn != on) {
        if (DEBUG) {
            Slog.d(TAG, "setScreenOn: on=" + on);
        }
        mScreenOn = on;
        invalidate(DIRTY_SCREEN_ON);
    }
