@Override
public PackageInstaller getPackageInstaller() {
    synchronized (mLock) {
        if (mInstaller == null) {
            try {
                mInstaller = new PackageInstaller(this, mPM.getPackageInstaller(), mContext.getPackageName(), mContext.getUserId());
            } catch (RemoteException e) {
                throw e.rethrowAsRuntimeException();
            }
        }
        return mInstaller;
    }
