private void invalidateAuthTokenLocked(UserAccounts accounts, String accountType, String authToken) {
    if (authToken == null || accountType == null) {
        return;
    }
    Cursor cursor = accounts.accountsDb.findAuthtokenForAllAccounts(accountType, authToken);
    try {
        while (cursor.moveToNext()) {
            String authTokenId = cursor.getString(0);
            String accountName = cursor.getString(1);
            String authTokenType = cursor.getString(2);
            accounts.accountsDb.deleteAuthToken(authTokenId);
            writeAuthTokenIntoCacheLocked(accounts, new Account(accountName, accountType), authTokenType, null);
        }
    } finally {
        cursor.close();
    }
