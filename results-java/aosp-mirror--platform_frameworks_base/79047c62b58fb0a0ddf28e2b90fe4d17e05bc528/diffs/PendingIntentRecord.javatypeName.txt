            return "broadcastIntent";
        case ActivityManager.INTENT_SENDER_SERVICE:
            return "startService";
        case ActivityManager.INTENT_SENDER_ACTIVITY_RESULT:
            return "activityResult";
    }
    return Integer.toString(type);
