    final Session session;
    synchronized (mLock) {
        session = mSessions.get(resultCode);
        if (session == null) {
            Slog.w(TAG, "no server callback for id " + resultCode);
            return;

    }
    mService.autoFill(structure, session.mServerCallback, session.mExtras, flags);
