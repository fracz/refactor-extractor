/**
 * Creates a DozeMachine with its parts for {@code dozeService}.
 */
public DozeMachine assembleMachine(DozeService dozeService) {
    Context context = dozeService;
    SensorManager sensorManager = context.getSystemService(SensorManager.class);
    PowerManager powerManager = context.getSystemService(PowerManager.class);
    DozeHost host = getHost(dozeService);
    AmbientDisplayConfiguration config = new AmbientDisplayConfiguration(context);
    DozeParameters params = new DozeParameters(context);
    Handler handler = new Handler();
    DozeFactory.WakeLock wakeLock = new DozeFactory.WakeLock(powerManager.newWakeLock(PowerManager.PARTIAL_WAKE_LOCK, "Doze"));
    DozeMachine machine = new DozeMachine(dozeService, params, wakeLock);
    machine.setParts(new DozeMachine.Part[] { createDozeTriggers(context, sensorManager, host, config, params, handler, wakeLock, machine), createDozeUi(context, host, wakeLock, machine) });
    return machine;
