private DozeMachine.Part createDozeUi(Context context, DozeHost host, WakeLock wakeLock, DozeMachine machine) {
    if (mDozePlugin != null) {
        DozeProvider.DozeUi dozeUi = mDozePlugin.provideDozeUi(context, pluginMachine(context, machine, host), wakeLock);
        return (oldState, newState) -> {
            dozeUi.transitionTo(pluginState(oldState), pluginState(newState));
        };
    } else {
        return new DozeUi(context, machine, wakeLock, host);
    }
