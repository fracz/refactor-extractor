public static String transportNamesOf(int[] types) {
    if (types == null || types.length == 0) {
        return "";
    }
    StringBuilder transports = new StringBuilder();
    for (int t : types) {
        transports.append("|").append(transportNameOf(t));
    }
    return transports.substring(1);
