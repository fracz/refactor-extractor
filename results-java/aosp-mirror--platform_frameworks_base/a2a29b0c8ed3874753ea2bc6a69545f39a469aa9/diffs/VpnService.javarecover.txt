void recover(VpnServiceBinder context) {
    mContext = context;
    mNotification = new NotificationHelper();
    if (VpnState.CONNECTED.equals(mState)) {
        Log.i("VpnService", "     recovered: " + mProfile.getName());
        new Thread(new Runnable() {

            public void run() {
                enterConnectivityLoop();
            }
        }).start();
    }
