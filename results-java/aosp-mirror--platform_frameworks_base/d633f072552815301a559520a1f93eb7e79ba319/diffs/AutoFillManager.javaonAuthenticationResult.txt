/**

public void onAuthenticationResult(Intent data) {
    if (data == null) {
        return;
    }
    Parcelable result = data.getParcelableExtra(EXTRA_AUTHENTICATION_RESULT);
    Bundle responseData = new Bundle();
    responseData.putParcelable(EXTRA_AUTHENTICATION_RESULT, result);
    try {
        mService.setAuthenticationResult(responseData, mContext.getActivityToken(), mContext.getUserId());
    } catch (RemoteException e) {
        Log.e(TAG, "Error delivering authentication result", e);
    }
