private void handleKeyguardDone(boolean wakeup) {
    if (DEBUG)
        Log.d(TAG, "handleKeyguardDone");
    handleHide();
    if (wakeup) {
        mPM.userActivity(SystemClock.uptimeMillis(), true);
    }
    mWakeLock.release();
    mContext.sendBroadcast(mUserPresentIntent);
