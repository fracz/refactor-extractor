/**
 * Clears the graph.
 */
void clear() {
    final SparseArray<Node> nodes = mNodes;
    final int count = nodes.size();
    for (int i = 0; i < count; i++) {
        nodes.valueAt(i).release();
    }
    nodes.clear();
    mNakedRoots.clear();
    mRoots.clear();
