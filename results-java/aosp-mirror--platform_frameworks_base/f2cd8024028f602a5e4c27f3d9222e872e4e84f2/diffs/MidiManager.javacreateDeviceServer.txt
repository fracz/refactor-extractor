public MidiDeviceServer createDeviceServer(MidiReceiver[] inputPortReceivers, int numOutputPorts, Bundle properties, int type, MidiDeviceServer.Callback callback) {
    try {
        MidiDeviceServer server = new MidiDeviceServer(mService, inputPortReceivers, numOutputPorts, callback);
        MidiDeviceInfo deviceInfo = mService.registerDeviceServer(server.getBinderInterface(), inputPortReceivers.length, numOutputPorts, properties, type);
        if (deviceInfo == null) {
            Log.e(TAG, "registerVirtualDevice failed");
            return null;
        }
        server.setDeviceInfo(deviceInfo);
        return server;
    } catch (RemoteException e) {
        Log.e(TAG, "RemoteException in createVirtualDevice");
        return null;
    }
