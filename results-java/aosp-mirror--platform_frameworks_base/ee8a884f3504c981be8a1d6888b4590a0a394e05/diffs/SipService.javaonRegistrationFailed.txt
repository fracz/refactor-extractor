@Override
public void onRegistrationFailed(ISipSession session, int errorCode, String message) {
    if (DEBUG)
        Log.d(TAG, "onRegistrationFailed(): " + session + ": " + SipErrorCode.toString(errorCode) + ": " + message);
    synchronized (SipService.this) {
        if (notCurrentSession(session))
            return;
        if (errorCode == SipErrorCode.INVALID_CREDENTIALS) {
            if (DEBUG)
                Log.d(TAG, "   pause auto-registration");
            stop();
        } else {
            onError();
        }
        mErrorCode = errorCode;
        mErrorMessage = message;
        mProxy.onRegistrationFailed(session, errorCode, message);
    }
