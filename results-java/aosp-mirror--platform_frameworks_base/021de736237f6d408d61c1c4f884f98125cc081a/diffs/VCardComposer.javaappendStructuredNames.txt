private void appendStructuredNames(final StringBuilder builder, final Map<String, List<ContentValues>> contentValuesListMap) {
    final List<ContentValues> contentValuesList = contentValuesListMap.get(StructuredName.CONTENT_ITEM_TYPE);
    if (contentValuesList != null && contentValuesList.size() > 0) {
        appendStructuredNamesInternal(builder, contentValuesList);
    } else if (mIsDoCoMo) {
        appendVCardLine(builder, VCARD_PROPERTY_NAME, "");
    } else if (mIsV30) {
        // vCard 3.0 requires "N" and "FN" properties.
        appendVCardLine(builder, VCARD_PROPERTY_NAME, "");
        appendVCardLine(builder, VCARD_PROPERTY_FULL_NAME, "");
    }
