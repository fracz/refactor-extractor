public void findAccessibilityNodeInfoByAccessibilityIdUiThread(Message message) {
    final int prefetchFlags = message.arg1;
    SomeArgs args = (SomeArgs) message.obj;
    final int accessibilityViewId = args.argi1;
    final int virtualDescendantId = args.argi2;
    final int interactionId = args.argi3;
    final IAccessibilityInteractionConnectionCallback callback = (IAccessibilityInteractionConnectionCallback) args.arg1;
    mPool.release(args);
    List<AccessibilityNodeInfo> infos = mTempAccessibilityNodeInfoList;
    infos.clear();
    try {
        View target = null;
        if (accessibilityViewId == AccessibilityNodeInfo.UNDEFINED) {
            target = ViewRootImpl.this.mView;
        } else {
            target = findViewByAccessibilityId(accessibilityViewId);
        }
        if (target != null && target.getVisibility() == View.VISIBLE) {
            getAccessibilityNodePrefetcher().prefetchAccessibilityNodeInfos(target, virtualDescendantId, prefetchFlags, infos);
        }
    } finally {
        try {
            callback.setFindAccessibilityNodeInfosResult(infos, interactionId);
            infos.clear();
        } catch (RemoteException re) {
        /* ignore - the other side will time out */
        }
    }
