private void addBluetoothUsage() {
    long btOnTimeMs = mStats.getBluetoothOnTime(mBatteryRealtime, mStatsType) / 1000;
    double btPower = btOnTimeMs * mPowerProfile.getAveragePower(PowerProfile.POWER_BLUETOOTH_ON) / (60 * 60 * 1000);
    if (DEBUG && btPower != 0) {
        Log.d(TAG, "Bluetooth: time=" + btOnTimeMs + " power=" + makemAh(btPower));
    }
    int btPingCount = mStats.getBluetoothPingCount();
    double pingPower = (btPingCount * mPowerProfile.getAveragePower(PowerProfile.POWER_BLUETOOTH_AT_CMD)) / (60 * 60 * 1000);
    if (DEBUG && pingPower != 0) {
        Log.d(TAG, "Bluetooth ping: count=" + btPingCount + " power=" + makemAh(pingPower));
    }
    btPower += pingPower;
    BatterySipper bs = addEntry(BatterySipper.DrainType.BLUETOOTH, btOnTimeMs, btPower + mBluetoothPower);
    aggregateSippers(bs, mBluetoothSippers, "Bluetooth");
