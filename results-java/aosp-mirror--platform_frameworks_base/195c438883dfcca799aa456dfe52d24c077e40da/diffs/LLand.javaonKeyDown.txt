@Override
public boolean onKeyDown(int keyCode, KeyEvent ev) {
    if (DEBUG)
        L("keyDown: %d", keyCode);
    switch(keyCode) {
        case KeyEvent.KEYCODE_DPAD_CENTER:
        case KeyEvent.KEYCODE_DPAD_UP:
        case KeyEvent.KEYCODE_SPACE:
        case KeyEvent.KEYCODE_ENTER:
        case KeyEvent.KEYCODE_BUTTON_A:
            poke();
            return true;
    }
    return false;
