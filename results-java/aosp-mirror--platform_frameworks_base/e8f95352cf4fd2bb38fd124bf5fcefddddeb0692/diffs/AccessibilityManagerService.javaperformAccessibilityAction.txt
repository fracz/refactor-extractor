public boolean performAccessibilityAction(int accessibilityWindowId, int accessibilityViewId, int action) {
    IAccessibilityInteractionConnection connection = null;
    synchronized (mLock) {
        final boolean permissionGranted = mSecurityPolicy.canPerformActionLocked(this, accessibilityWindowId, action);
        if (permissionGranted) {
            connection = mWindowIdToInteractionConnectionMap.get(accessibilityWindowId);
        }
    }
    if (connection == null) {
        if (DEBUG) {
            Slog.e(LOG_TAG, "No interaction connection to window: " + accessibilityWindowId);
        }
        return false;
    }
    final long identityToken = Binder.clearCallingIdentity();
    try {
        final int interactionId = mInteractionIdCounter.getAndIncrement();
        connection.performAccessibilityAction(accessibilityViewId, action, interactionId, mCallback);
        return mCallback.getPerformAccessibilityActionResult(interactionId);
    } catch (RemoteException re) {
        if (DEBUG) {
            Slog.e(LOG_TAG, "Error requesting node with accessibilityViewId: " + accessibilityViewId);
        }
    } finally {
        Binder.restoreCallingIdentity(identityToken);
    }
    return false;
