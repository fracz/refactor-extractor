private void updateAirplaneMode(boolean force) {
    boolean airplaneMode = (Settings.Global.getInt(mContext.getContentResolver(), Settings.Global.AIRPLANE_MODE_ON, 0) == 1);
    if (airplaneMode != mAirplaneMode || force) {
