@Override
public void clearWallpaper() throws IOException {
    try {
        Resources resources = getResources();
        /* Set the wallpaper to the default values */
        ParcelFileDescriptor fd = getWallpaperService().setWallpaper("res:" + resources.getResourceName(com.android.internal.R.drawable.default_wallpaper));
        if (fd != null) {
            FileOutputStream fos = null;
            try {
                fos = new ParcelFileDescriptor.AutoCloseOutputStream(fd);
                setWallpaper(resources.openRawResource(com.android.internal.R.drawable.default_wallpaper), fos);
            } finally {
                if (fos != null) {
                    fos.close();
                }
            }
        }
    } catch (RemoteException e) {
    }
