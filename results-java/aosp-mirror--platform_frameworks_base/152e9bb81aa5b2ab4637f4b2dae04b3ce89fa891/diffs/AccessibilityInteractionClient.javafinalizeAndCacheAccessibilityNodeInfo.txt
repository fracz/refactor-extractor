/**
 * Finalize an {@link AccessibilityNodeInfo} before passing it to the client.
 *
 * @param info The info.
 * @param connectionId The id of the connection to the system.
 * @param windowScale The source window compatibility scale.
 */
private void finalizeAndCacheAccessibilityNodeInfo(AccessibilityNodeInfo info, int connectionId, float windowScale) {
    if (info != null) {
        applyCompatibilityScaleIfNeeded(info, windowScale);
        info.setConnectionId(connectionId);
        info.setSealed(true);
        sAccessibilityNodeInfoCache.add(info);
    }
