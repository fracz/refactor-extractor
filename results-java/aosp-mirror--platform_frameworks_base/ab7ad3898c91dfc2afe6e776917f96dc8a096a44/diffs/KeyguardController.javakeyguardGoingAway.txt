/**
 * Called when Keyguard is going away.
 *
 * @param flags See {@link android.view.WindowManagerPolicy#KEYGUARD_GOING_AWAY_FLAG_TO_SHADE}
 * etc.
 */
void keyguardGoingAway(int flags) {
    if (mKeyguardShowing) {
        mKeyguardGoingAway = true;
        mWindowManager.prepareAppTransition(TRANSIT_KEYGUARD_GOING_AWAY, false, /* alwaysKeepCurrent */
        convertTransitFlags(flags), false);
        mWindowManager.keyguardGoingAway(flags);
        mService.updateSleepIfNeededLocked();
        // Some stack visibility might change (e.g. docked stack)
        mStackSupervisor.ensureActivitiesVisibleLocked(null, 0, !PRESERVE_WINDOWS);
        mWindowManager.executeAppTransition();
        mService.applyVrModeIfNeededLocked(mStackSupervisor.getResumedActivityLocked(), true);
    }
