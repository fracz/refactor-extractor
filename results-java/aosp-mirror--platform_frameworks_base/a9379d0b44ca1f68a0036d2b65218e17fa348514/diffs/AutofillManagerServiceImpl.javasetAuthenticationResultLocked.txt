void setAuthenticationResultLocked(Bundle data, int sessionId, int uid) {
    if (!isEnabled()) {
        return;
    }
    final Session session = mSessions.get(sessionId);
    if (session != null && uid == session.uid) {
        session.setAuthenticationResultLocked(data);
    }
