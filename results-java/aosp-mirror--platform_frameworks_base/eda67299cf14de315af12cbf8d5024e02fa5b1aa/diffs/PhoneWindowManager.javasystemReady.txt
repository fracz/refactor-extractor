public void systemReady() {
    if (!mHeadless) {
        mKeyguardDelegate = new KeyguardServiceDelegate(mContext, null);
        mKeyguardDelegate.onSystemReady();
    }
    synchronized (mLock) {
        updateOrientationListenerLp();
        mSystemReady = true;
        mHandler.post(new Runnable() {

            public void run() {
                updateSettings();
            }
        });
    }
