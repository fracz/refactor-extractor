/**
 * Return an instance back to be reused.
 * <p>
 * <b>Note: You must not touch the object after calling this function.</b>
 */
public void recycle() {
    if (mIsInPool) {
        return;
    }
    clear();
    synchronized (mPoolLock) {
        if (sPoolSize <= MAX_POOL_SIZE) {
            mNext = sPool;
            sPool = this;
            mIsInPool = true;
            sPoolSize++;
        }
    }
