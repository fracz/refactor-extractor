@LargeTest
public void testObjectContract() throws Exception {
    final long startTimeMillis = SystemClock.uptimeMillis();
    try {
        // find a view and make sure it is not focused
        AccessibilityNodeInfo button = AccessibilityInteractionClient.getInstance().findAccessibilityNodeInfoByViewIdInActiveWindow(sConnectionId, R.id.button5);
        AccessibilityNodeInfo parent = button.getParent();
        final int childCount = parent.getChildCount();
        for (int i = 0; i < childCount; i++) {
            AccessibilityNodeInfo child = parent.getChild(i);
            assertNotNull(child);
            if (child.equals(button)) {
                assertEquals("Equal objects must have same hasCode.", button.hashCode(), child.hashCode());
                return;
            }
        }
        fail("Parent's children do not have the info whose parent is the parent.");
    } finally {
        if (DEBUG) {
            final long elapsedTimeMillis = SystemClock.uptimeMillis() - startTimeMillis;
            Log.i(LOG_TAG, "testObjectContract: " + elapsedTimeMillis + "ms");
        }
    }
