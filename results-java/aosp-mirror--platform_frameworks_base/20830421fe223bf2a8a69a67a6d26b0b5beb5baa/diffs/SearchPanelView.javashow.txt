public void show(final boolean show, boolean animate) {
    if (animate) {
        if (mShowing != show) {
            mShowing = show;
        // TODO: start animating ring
        }
    } else {
        mShowing = show;
        onAnimationEnd(null);
    }
    postDelayed(new Runnable() {

        public void run() {
            setVisibility(show ? View.VISIBLE : View.INVISIBLE);
            if (show) {
                setFocusable(true);
                setFocusableInTouchMode(true);
                requestFocus();
            }
        }
    }, show ? 0 : 100);
