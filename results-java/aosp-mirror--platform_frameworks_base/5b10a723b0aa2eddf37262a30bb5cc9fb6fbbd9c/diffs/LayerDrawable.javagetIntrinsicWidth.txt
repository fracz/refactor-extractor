@Override
public int getIntrinsicWidth() {
    int width = -1;
    final ChildDrawable[] array = mLayerState.mChildren;
    final int N = mLayerState.mNum;
    int padL = 0, padR = 0;
    for (int i = 0; i < N; i++) {
        final ChildDrawable r = array[i];
        int w = r.mDrawable.getIntrinsicWidth() + r.mInsetL + r.mInsetR + padL + padR;
        if (w > width) {
            width = w;
        }
        padL += mPaddingL[i];
        padR += mPaddingR[i];
    }
    return width;
