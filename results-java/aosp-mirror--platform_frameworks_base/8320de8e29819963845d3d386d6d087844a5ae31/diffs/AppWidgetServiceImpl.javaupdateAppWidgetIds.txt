public void updateAppWidgetIds(int[] appWidgetIds, RemoteViews views) {
    if (appWidgetIds == null) {
        return;
    }
    int bitmapMemoryUsage = 0;
    if (views != null) {
        bitmapMemoryUsage = views.estimateMemoryUsage();
    }
    if (bitmapMemoryUsage > mMaxWidgetBitmapMemory) {
        throw new IllegalArgumentException("RemoteViews for widget update exceeds maximum" + " bitmap memory usage (used: " + bitmapMemoryUsage + ", max: " + mMaxWidgetBitmapMemory + ") The total memory cannot exceed that required to" + " fill the device's screen once.");
    }
    if (appWidgetIds.length == 0) {
        return;
    }
    final int N = appWidgetIds.length;
    synchronized (mAppWidgetIds) {
        ensureStateLoadedLocked();
        for (int i = 0; i < N; i++) {
            AppWidgetId id = lookupAppWidgetIdLocked(appWidgetIds[i]);
            updateAppWidgetInstanceLocked(id, views);
        }
    }
