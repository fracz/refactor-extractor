private int findViewLocked(View view, boolean required) {
    synchronized (this) {
        final int count = mViews != null ? mViews.length : 0;
        for (int i = 0; i < count; i++) {
            if (mViews[i] == view) {
                return i;
            }
        }
        if (required) {
            throw new IllegalArgumentException("View not attached to window manager");
        }
        return -1;
    }
