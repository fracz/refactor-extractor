public void setDataOnRoamingEnabled(boolean enabled) {
    if (getDataOnRoamingEnabled() != enabled) {
        Settings.Secure.putInt(phone.getContext().getContentResolver(), Settings.Secure.DATA_ROAMING, enabled ? 1 : 0);
    }
    Message roamingMsg = phone.getServiceState().getRoaming() ? obtainMessage(EVENT_ROAMING_ON) : obtainMessage(EVENT_ROAMING_OFF);
    sendMessage(roamingMsg);
