// Report the name of the currently active transport
public String getCurrentTransport() {
    mContext.enforceCallingOrSelfPermission(android.Manifest.permission.BACKUP, "getCurrentTransport");
    String currentTransport = mTransportManager.getCurrentTransportName();
    if (MORE_DEBUG)
        Slog.v(TAG, "... getCurrentTransport() returning " + currentTransport);
    return currentTransport;
