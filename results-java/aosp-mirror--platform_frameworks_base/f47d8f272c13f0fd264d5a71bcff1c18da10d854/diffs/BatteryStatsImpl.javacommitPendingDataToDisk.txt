public void commitPendingDataToDisk() {
    Parcel next;
    synchronized (this) {
        next = mPendingWrite;
        mPendingWrite = null;
        mWriteLock.lock();
    }
    try {
        FileOutputStream stream = new FileOutputStream(mFile.chooseForWrite());
        stream.write(next.marshall());
        stream.flush();
        stream.close();
        mFile.commit();
    } catch (IOException e) {
        Slog.w("BatteryStats", "Error writing battery statistics", e);
        mFile.rollback();
    } finally {
        next.recycle();
        mWriteLock.unlock();
    }
