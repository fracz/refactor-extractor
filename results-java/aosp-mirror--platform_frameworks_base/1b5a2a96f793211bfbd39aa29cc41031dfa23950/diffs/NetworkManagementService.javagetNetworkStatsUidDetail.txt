public NetworkStats getNetworkStatsUidDetail(int uid) {
    if (Binder.getCallingUid() != uid) {
        mContext.enforceCallingOrSelfPermission(android.Manifest.permission.ACCESS_NETWORK_STATE, "NetworkManagementService");
