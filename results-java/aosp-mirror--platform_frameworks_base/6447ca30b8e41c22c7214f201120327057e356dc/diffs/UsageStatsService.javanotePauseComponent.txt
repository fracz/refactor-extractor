public void notePauseComponent(ComponentName componentName) {
    enforceCallingPermission();
    String pkgName;
    if ((componentName == null) || ((pkgName = componentName.getPackageName()) == null)) {
        return;
    }
    if ((mResumedPkg == null) || (!pkgName.equalsIgnoreCase(mResumedPkg))) {
        Log.w(TAG, "Something wrong here, Didn't expect " + pkgName + " to be paused");
        return;
    }
    if (localLOGV)
        Log.i(TAG, "paused component:" + pkgName);
    synchronized (mStatsLock) {
        PkgUsageStatsExtended pus = mStats.get(pkgName);
        if (pus == null) {
            // Weird some error here
            Log.w(TAG, "No package stats for pkg:" + pkgName);
            return;
        }
        pus.updatePause();
    }
    // Persist data to file
    writeStatsToFile();
