// Tells the FaceLock service to start displaying its UI and perform recognition
private void startUi(IBinder windowToken, int x, int y, int w, int h) {
    synchronized (mServiceRunningLock) {
        if (!mServiceRunning) {
            if (DEBUG)
                Log.d(TAG, "Starting FaceLock");
            try {
                mService.startUi(windowToken, x, y, w, h, mLockPatternUtils.isBiometricWeakLivelinessEnabled());
            } catch (RemoteException e) {
                Log.e(TAG, "Caught exception starting FaceLock: " + e.toString());
                return;
            }
            mServiceRunning = true;
        } else {
            if (DEBUG)
                Log.w(TAG, "startUi() attempted while running");
        }
    }
