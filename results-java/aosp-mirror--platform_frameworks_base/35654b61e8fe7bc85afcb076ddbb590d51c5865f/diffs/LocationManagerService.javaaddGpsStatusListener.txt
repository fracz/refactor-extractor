@Override
public boolean addGpsStatusListener(IGpsStatusListener listener, String packageName) {
    if (mGpsStatusProvider == null) {
        return false;
    }
    checkResolutionLevelIsSufficientForProviderUse(getCallerAllowedResolutionLevel(), LocationManager.GPS_PROVIDER);
    final int uid = Binder.getCallingUid();
    final long ident = Binder.clearCallingIdentity();
    try {
        if (mAppOps.noteOp(AppOpsManager.OP_LOCATION, uid, packageName) != AppOpsManager.MODE_ALLOWED) {
            return false;
        }
    } finally {
        Binder.restoreCallingIdentity(ident);
    }
    try {
        mGpsStatusProvider.addGpsStatusListener(listener);
    } catch (RemoteException e) {
        Slog.e(TAG, "mGpsStatusProvider.addGpsStatusListener failed", e);
        return false;
    }
    return true;
