            if (perm.readActivities.size() == 0 && (perm.globalModeFlags & Intent.FLAG_GRANT_READ_URI_PERMISSION) == 0) {
                perm.modeFlags &= ~Intent.FLAG_GRANT_READ_URI_PERMISSION;
                service.removeUriPermissionIfNeededLocked(perm);
            }

        }
