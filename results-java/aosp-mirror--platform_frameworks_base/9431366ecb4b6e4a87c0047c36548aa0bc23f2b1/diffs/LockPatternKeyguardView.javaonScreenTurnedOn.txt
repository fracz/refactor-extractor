@Override
public void onScreenTurnedOn() {
    if (DEBUG)
        Log.d(TAG, "screen on");
    boolean runFaceLock = false;
    // Make sure to start facelock iff the screen is both on and focused
    synchronized (mFaceLockStartupLock) {
        mScreenOn = true;
        runFaceLock = mWindowFocused;
    }
    show();
    restoreWidgetState();
    if (runFaceLock)
        activateFaceLockIfAble();
