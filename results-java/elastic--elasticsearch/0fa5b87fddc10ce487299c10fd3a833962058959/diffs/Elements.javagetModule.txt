/**
 * Returns the module composed of {@code elements}.
 */
public static Module getModule(final Iterable<? extends Element> elements) {
    return new Module() {

        public void configure(Binder binder) {
            for (Element element : elements) {
                element.applyTo(binder);
            }
        }
    };
