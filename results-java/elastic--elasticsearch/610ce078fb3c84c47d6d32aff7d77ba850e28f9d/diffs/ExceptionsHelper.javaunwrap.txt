public static <T extends Throwable> T unwrap(Throwable t, Class<T> clazz) {
    if (t != null) {
        do {
            if (clazz.isInstance(t)) {
                return clazz.cast(t);
            }
        } while ((t = t.getCause()) != null);
    }
    return null;
