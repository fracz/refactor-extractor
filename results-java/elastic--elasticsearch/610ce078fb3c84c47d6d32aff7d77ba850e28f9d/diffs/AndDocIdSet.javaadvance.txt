    int i = 1;
    while (i < size) {
        if (i != skip) {
            dcit = iterators[i];
            int docid = dcit.advance(target);
            if (docid > target) {
                target = docid;
                if (i != 0) {
                    skip = i;
                    i = 0;
                    continue;
                } else {
                    skip = 0;
                }
            }
        }
        i++;
    }
