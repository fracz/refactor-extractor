    OrdinalsBuilder builder = new OrdinalsBuilder(maxDoc);
    for (int i = 0; i < maxOrds; i++) {
        builder.nextOrdinal();

        builder.addDoc(1);

            builder.addDoc(4);
        }
        if (i == 1) {
            builder.addDoc(5);
        }
        if (i < 10) {
            builder.addDoc(6);
        }
    }
