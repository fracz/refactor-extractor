public void testCopyToFieldsNonExistingInnerObjectParsing() throws Exception {
    String mapping = jsonBuilder().startObject().startObject("type1").startObject("properties").startObject("copy_test").field("type", "string").field("copy_to", "very.inner.field").endObject().endObject().endObject().endObject().string();
    DocumentMapper docMapper = createIndex("test").mapperService().documentMapperParser().parse(mapping);
    BytesReference json = jsonBuilder().startObject().field("copy_test", "foo").endObject().bytes();
    try {
        docMapper.parse("test", "type1", "1", json).rootDoc();
        fail();
    } catch (MapperParsingException ex) {
        assertThat(ex.getMessage(), startsWith("attempt to copy value to non-existing object"));
    }
