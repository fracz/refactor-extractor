@Test
public void testAddQueryWithNoMapping() throws Exception {
    client().admin().indices().prepareCreate("test").get();
    ensureGreen();
    try {
        client().prepareIndex("test", PercolatorService.TYPE_NAME).setSource(jsonBuilder().startObject().field("query", termQuery("field1", "value")).endObject()).get();
        fail();
    } catch (PercolatorException e) {
        assertThat(e.getRootCause(), instanceOf(QueryShardException.class));
    }

        client().prepareIndex("test", PercolatorService.TYPE_NAME).setSource(jsonBuilder().startObject().field("query", rangeQuery("field1").from(0).to(1)).endObject()).get();
        fail();

        assertThat(e.getRootCause(), instanceOf(QueryShardException.class));
    }
