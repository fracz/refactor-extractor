@Test
public void testPutRepositoryWithBlocks() {
    logger.info("-->  registering a repository is blocked when the cluster is read only");
    try {
        setClusterReadOnly(true);
        assertBlocked(client().admin().cluster().preparePutRepository("test-repo-blocks").setType("fs").setVerify(false).setSettings(Settings.settingsBuilder().put("location", randomRepoPath())), MetaData.CLUSTER_READ_ONLY_BLOCK);
    } finally {
        setClusterReadOnly(false);
    }
    logger.info("-->  registering a repository is allowed when the cluster is not read only");
    assertAcked(client().admin().cluster().preparePutRepository("test-repo-blocks").setType("fs").setVerify(false).setSettings(Settings.settingsBuilder().put("location", randomRepoPath())));
