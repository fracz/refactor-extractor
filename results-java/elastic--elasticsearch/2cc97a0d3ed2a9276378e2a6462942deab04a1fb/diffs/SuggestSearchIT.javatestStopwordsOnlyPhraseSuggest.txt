@Test
public void testStopwordsOnlyPhraseSuggest() throws IOException {
    assertAcked(prepareCreate("test").addMapping("typ1", "body", "type=string,analyzer=stopwd").setSettings(settingsBuilder().put("index.analysis.analyzer.stopwd.tokenizer", "whitespace").putArray("index.analysis.analyzer.stopwd.filter", "stop")));
    ensureGreen();
    index("test", "typ1", "1", "body", "this is a test");
    refresh();
    Suggest searchSuggest = searchSuggest("a an the", phraseSuggestion("simple_phrase").field("body").gramSize(1).addCandidateGenerator(PhraseSuggestionBuilder.candidateGenerator("body").minWordLength(1).suggestMode("always")).size(1));
    assertSuggestionSize(searchSuggest, 0, 0, "simple_phrase");
