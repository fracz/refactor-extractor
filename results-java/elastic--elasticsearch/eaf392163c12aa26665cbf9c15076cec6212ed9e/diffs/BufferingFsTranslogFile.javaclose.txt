@Override
public void close(boolean delete) {
    if (!closed.compareAndSet(false, true)) {
        return;
    }
    try {
        if (!delete) {
            try {
                sync();
            } catch (Exception e) {
                throw new TranslogException(shardId, "failed to sync on close", e);
            }
        }
    } finally {
        raf.decreaseRefCount(delete);
    }
