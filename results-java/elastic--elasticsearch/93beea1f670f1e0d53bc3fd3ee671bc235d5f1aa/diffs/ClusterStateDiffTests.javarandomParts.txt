                randomPart.put(builder, randomPart.randomChange(parts.get(part)));
            }
        }

    for (int i = 0; i < additionalPartCount; i++) {
        String name = randomName(prefix);
        randomPart.put(builder, randomPart.randomCreate(name));
    }
    return builder.build();
