private ClusterBlock randomGlobalBlock() {
    switch(randomInt(2)) {
        case 0:
            return DiscoverySettings.NO_MASTER_BLOCK_ALL;
        case 1:
            return DiscoverySettings.NO_MASTER_BLOCK_WRITES;
        default:
            return GatewayService.STATE_NOT_RECOVERED_BLOCK;
    }
