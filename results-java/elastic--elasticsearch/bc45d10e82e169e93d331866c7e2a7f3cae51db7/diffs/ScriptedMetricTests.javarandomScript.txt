private Script randomScript(String script) {
    if (randomBoolean()) {
        return new Script(script);
    } else {
        ScriptType type = randomFrom(ScriptType.values());
        return new Script(type, type == ScriptType.STORED ? null : randomFrom("my_lang", Script.DEFAULT_SCRIPT_LANG), script, Collections.emptyMap());
    }
