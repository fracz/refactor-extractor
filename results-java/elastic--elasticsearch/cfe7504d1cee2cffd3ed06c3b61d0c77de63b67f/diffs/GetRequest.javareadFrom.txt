public void readFrom(StreamInput in) throws IOException {
    super.readFrom(in);
    type = in.readUTF();
    id = in.readUTF();
    if (in.readBoolean()) {
        routing = in.readUTF();
    }
    if (in.readBoolean()) {
        preference = in.readUTF();
    }
    refresh = in.readBoolean();
    int size = in.readInt();
    if (size >= 0) {
        fields = new String[size];
        for (int i = 0; i < size; i++) {
            fields[i] = in.readUTF();
        }
    }
    byte realtime = in.readByte();
    if (realtime == 0) {
        this.realtime = false;
    } else if (realtime == 1) {
        this.realtime = true;
    }
