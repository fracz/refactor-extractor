        expected += "\n# header" + i + ": value";
    }

    if (random().nextBoolean()) {

        httpResponse.setEntity(new StringEntity(responseBody, StandardCharsets.UTF_8));
        expected += "\n# {";
        expected += "\n#   \"field\": \"value\"";
