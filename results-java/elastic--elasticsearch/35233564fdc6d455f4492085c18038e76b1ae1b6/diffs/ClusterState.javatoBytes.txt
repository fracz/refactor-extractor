    CachedStreamOutput.Entry cachedEntry = CachedStreamOutput.popEntry();

        BytesStreamOutput os = cachedEntry.bytes();

        return os.copiedByteArray();

        CachedStreamOutput.pushEntry(cachedEntry);
    }
