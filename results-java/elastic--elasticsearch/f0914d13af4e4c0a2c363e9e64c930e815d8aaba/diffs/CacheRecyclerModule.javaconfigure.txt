    if ("none".equalsIgnoreCase(type)) {
        bind(CacheRecycler.class).to(NoneCacheRecycler.class).asEagerSingleton();

        bind(CacheRecycler.class).to(DefaultCacheRecycler.class).asEagerSingleton();
    }
