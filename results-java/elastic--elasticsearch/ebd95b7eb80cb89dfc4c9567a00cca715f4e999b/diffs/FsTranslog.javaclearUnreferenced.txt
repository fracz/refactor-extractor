@Override
public void clearUnreferenced() {
    synchronized (mutex) {
        File[] files = location.listFiles();
        if (files != null) {
            for (File file : files) {
                if (file.getName().equals("translog-" + id)) {
                    continue;
                }
                try {
                    file.delete();
                } catch (Exception e) {
                // ignore
                }
            }
        }
    }
