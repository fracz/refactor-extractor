    bytesSinceLastRateLimit += bytes;
    if (bytesSinceLastRateLimit >= rateLimiter.getMinPauseCheckBytes()) {
        long pause = rateLimiter.pause(bytesSinceLastRateLimit);
        bytesSinceLastRateLimit = 0;
        if (pause > 0) {
            listener.onPause(pause);
        }
    }
