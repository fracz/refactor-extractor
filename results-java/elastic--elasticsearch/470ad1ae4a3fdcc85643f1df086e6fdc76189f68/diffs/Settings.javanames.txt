public Set<String> names() {
    Set<String> names = new HashSet<>();
    for (String key : settings.keySet()) {
        int i = key.indexOf(".");
        if (i < 0) {
            names.add(key);
        } else {
            names.add(key.substring(0, i));
        }
    }
