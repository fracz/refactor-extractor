        response.writeTo(out);
    }
    out.writeVInt(nodes.length);
    for (DiscoveryNode node : nodes) {
        node.writeTo(out);
    }
    if (out.getVersion().onOrAfter(V_5_1_0_UNRELEASED)) {
        out.writeVInt(indicesAndFilters.size());

        }
    }
