/**
 * Returns whether one of the parents is a {@link BucketsAggregator}.
 */
public static boolean descendsFromBucketAggregator(Aggregator parent) {
    while (parent != null) {
        if (parent instanceof BucketsAggregator) {
            return true;
        }
        parent = parent.parent;
    }
    return false;
