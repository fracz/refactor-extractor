public static String toCamelCase(String value) {
    StringBuilder sb = new StringBuilder();
    char[] values = value.toCharArray();
    for (int i = 0; i < values.length; i++) {
        if (values[i] == '_') {
            sb.append(Character.toUpperCase(values[++i]));
        } else {
            sb.append(values[i]);
        }
    }
    return sb.toString();
