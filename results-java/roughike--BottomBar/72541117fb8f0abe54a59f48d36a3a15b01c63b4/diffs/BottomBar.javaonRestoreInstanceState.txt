private void onRestoreInstanceState(Bundle savedInstanceState) {
    if (savedInstanceState != null) {
        mCurrentTabPosition = savedInstanceState.getInt(STATE_CURRENT_SELECTED_TAB, -1);
        mBadgeStateMap = (HashMap<Integer, Boolean>) savedInstanceState.getSerializable(STATE_BADGE_STATES_BUNDLE);
        if (mCurrentTabPosition == -1) {
            mCurrentTabPosition = 0;
            Log.e("BottomBar", "You must override the Activity's onSave" + "InstanceState(Bundle outState) and call BottomBar.onSaveInstanc" + "eState(outState) there to restore the state properly.");
        }
        mIsComingFromRestoredState = true;
        mShouldUpdateFragmentInitially = true;
    }
