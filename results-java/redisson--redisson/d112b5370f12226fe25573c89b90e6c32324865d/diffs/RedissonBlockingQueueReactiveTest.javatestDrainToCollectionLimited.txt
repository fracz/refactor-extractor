    sync(queue1.put(1));
    sync(queue1.put(2L));
    sync(queue1.put("e"));
    ArrayList<Object> dst = new ArrayList<Object>();
    sync(queue1.drainTo(dst, 2));
    MatcherAssert.assertThat(dst, Matchers.<Object>contains(1, 2L));
    Assert.assertEquals(1, sync(queue1.size()).intValue());
