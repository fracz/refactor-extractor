public void testAddOffer() {
    RQueueReactive<Integer> queue = redisson.getQueue("queue");
    sync(queue.add(1));
    sync(queue.offer(2));
    sync(queue.add(3));
    sync(queue.offer(4));
    MatcherAssert.assertThat(sync(queue), Matchers.contains(1, 2, 3, 4));
    Assert.assertEquals((Integer) 1, sync(queue.poll()));
