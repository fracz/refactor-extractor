@Override
public boolean containsKey(Object key) {
    if (key == null) {
        throw new NullPointerException();
    }
    CachedValue<K, V> entry = map.get(key);
    if (entry == null) {
        return false;
    }
    if (entry.isExpired()) {
        if (map.remove(key, entry)) {
            onValueRemove(entry);
            return false;
        }
        return containsKey(key);
    }
    return true;
