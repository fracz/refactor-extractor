private byte[] encode(Object task) {
    // erase RedissonClient field to avoid its serialization
    Field[] fields = task.getClass().getDeclaredFields();
    for (Field field : fields) {
        if (RedissonClient.class.isAssignableFrom(field.getType()) && field.isAnnotationPresent(RInject.class)) {
            field.setAccessible(true);
            try {
                field.set(task, null);
            } catch (IllegalAccessException e) {
                throw new IllegalStateException(e);
            }
        }
    }
    try {
        return codec.getValueEncoder().encode(task);
    } catch (IOException e) {
        throw new IllegalArgumentException(e);
    }
