public void setCaptureQuality(int cameraType, String captureQuality) {
    Camera camera = _cameras.get(cameraType);
    if (null == camera) {
        return;
    }
    Camera.Parameters parameters = camera.getParameters();
    Camera.Size pictureSize = null;
    switch(captureQuality) {
        case "low":
            // select the lowest res
            pictureSize = getSmallestPictureSize(cameraType);
            break;
        case "medium":
            List<Camera.Size> sizes = parameters.getSupportedPictureSizes();
            pictureSize = sizes.get(sizes.size() / 2);
            break;
        case "high":
            // select the highest res
            pictureSize = getBestPictureSize(cameraType, Integer.MAX_VALUE, Integer.MAX_VALUE);
            break;
    }
    if (pictureSize != null) {
        parameters.setPictureSize(pictureSize.width, pictureSize.height);
        camera.setParameters(parameters);
    }
