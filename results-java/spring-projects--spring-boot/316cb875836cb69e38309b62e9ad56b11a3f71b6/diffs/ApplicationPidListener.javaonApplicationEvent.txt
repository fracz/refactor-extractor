@Override
public void onApplicationEvent(ApplicationStartedEvent event) {
    if (pidFileCreated.get()) {
        return;
    }
    String applicationPid;
    try {
        applicationPid = SystemUtils.getApplicationPid();
    } catch (IllegalStateException ignore) {
        return;
    }
    if (pidFileCreated.compareAndSet(false, true)) {
        File file = new File(this.pidFileName);
        FileOutputStream fileOutputStream = null;
        try {
            File parent = file.getParentFile();
            if (parent != null) {
                parent.mkdirs();
            }
            fileOutputStream = new FileOutputStream(file);
            fileOutputStream.write(applicationPid.getBytes());
        } catch (FileNotFoundException e) {
            logger.warn(String.format("Cannot create pid file %s !", this.pidFileName));
        } catch (Exception e) {
            logger.warn(String.format("Cannot write to pid file %s!", this.pidFileName));
        } finally {
            if (fileOutputStream != null) {
                try {
                    fileOutputStream.close();
                } catch (IOException e) {
                    logger.warn(String.format("Cannot close pid file %s!", this.pidFileName));
                }
            }
        }
    }
