@Override
public void run() {
    try {
        while (this.acceptConnections) {
            SocketChannel socket = this.serverSocketChannel.accept();
            try {
                handleConnection(socket);
            } finally {
                socket.close();
            }
        }
    } catch (Exception ex) {
        logger.trace("Unexpected exception from tunnel client", ex);
    }
