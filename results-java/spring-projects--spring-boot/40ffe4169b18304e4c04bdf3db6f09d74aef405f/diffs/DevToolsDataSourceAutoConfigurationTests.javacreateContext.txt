private ConfigurableApplicationContext createContext(String driver, Class<?>... classes) {
    AnnotationConfigApplicationContext context = new AnnotationConfigApplicationContext();
    context.register(classes);
    context.register(DevToolsDataSourceAutoConfiguration.class);
    EnvironmentTestUtils.addEnvironment(context, "spring.datasource.driver-class-name:" + driver);
    context.refresh();
    return context;
