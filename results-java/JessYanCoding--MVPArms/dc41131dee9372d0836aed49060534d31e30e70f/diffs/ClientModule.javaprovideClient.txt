/**
 * 提供OkhttpClient
 *
 * @param okHttpClient
 * @return
 */
@Singleton
@Provides
OkHttpClient provideClient(OkHttpClient.Builder okHttpClient, Cache cache, Interceptor intercept, List<Interceptor> interceptors) {
    OkHttpClient.Builder builder = okHttpClient.connectTimeout(TIME_OUT, TimeUnit.SECONDS).readTimeout(TIME_OUT, TimeUnit.SECONDS).cache(// 设置缓存
    cache).addNetworkInterceptor(intercept);
    if (interceptors != null && interceptors.size() > 0) {
        // 如果外部提供了interceptor的数组则遍历添�
        for (Interceptor interceptor : interceptors) {
            builder.addInterceptor(interceptor);
        }
    }
    return builder.build();
