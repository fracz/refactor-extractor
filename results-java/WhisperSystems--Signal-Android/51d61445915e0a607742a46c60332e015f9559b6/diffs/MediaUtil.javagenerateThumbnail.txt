@Nullable
public static ThumbnailData generateThumbnail(Context context, MasterSecret masterSecret, String contentType, Uri uri) throws BitmapDecodingException {
    long startMillis = System.currentTimeMillis();
    ThumbnailData data = null;
    if (ContentType.isImageType(contentType)) {
        data = new ThumbnailData(generateImageThumbnail(context, masterSecret, uri));
    }
    if (data != null) {
        Log.w(TAG, String.format("generated thumbnail for part, %dx%d (%.3f:1) in %dms", data.getBitmap().getWidth(), data.getBitmap().getHeight(), data.getAspectRatio(), System.currentTimeMillis() - startMillis));
    }
    return data;
