private boolean isChallenge(Context context, Intent intent) {
    String messageBody = getSmsMessageBodyFromIntent(intent);
    if (messageBody == null)
        return false;
    if (messageBody.matches(".*Your TextSecure verification code: [0-9]{3,4}-[0-9]{3,4}") && TextSecurePreferences.isVerifying(context)) {
        return true;
    }
    return false;
