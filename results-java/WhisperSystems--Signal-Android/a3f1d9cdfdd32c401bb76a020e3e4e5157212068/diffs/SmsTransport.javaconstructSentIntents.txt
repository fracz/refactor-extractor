    ArrayList<PendingIntent> sentIntents = new ArrayList<PendingIntent>(messages.size());
    for (String ignored : messages) {
        sentIntents.add(PendingIntent.getBroadcast(context, 0, constructSentIntent(context, messageId, type, secure, false), 0));
