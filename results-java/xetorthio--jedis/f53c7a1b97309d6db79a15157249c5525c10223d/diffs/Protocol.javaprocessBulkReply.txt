private Object processBulkReply(InputStream is) throws IOException {
    String ret = null;
    int len = Integer.parseInt(readLine(is));
    if (len == -1) {
        return null;
    }
    byte[] read = new byte[len];
    is.read(read);
    // read 2 more bytes for the command delimiter
    is.read();
    is.read();
    ret = new String(read);
    return ret;
