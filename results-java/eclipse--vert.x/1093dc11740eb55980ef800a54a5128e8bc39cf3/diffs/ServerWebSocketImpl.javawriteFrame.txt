public ServerWebSocket writeFrame(WebSocketFrame frame) {
    synchronized (conn) {
        if (connectRunnable != null) {
            if (rejected) {
                throw new IllegalStateException("Cannot write to websocket, it has been rejected");

                connect();

        return super.writeFrame(frame);
    }
