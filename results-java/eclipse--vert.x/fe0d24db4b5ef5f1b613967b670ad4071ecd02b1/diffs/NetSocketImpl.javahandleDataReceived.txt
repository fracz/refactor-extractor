public synchronized void handleDataReceived(Buffer data) {
    checkContext();
    if (paused) {
        if (pendingData == null) {
            pendingData = data.copy();
        } else {
            pendingData.appendBuffer(data);
        }
        return;
    }
    if (pendingData != null) {
        data = pendingData.appendBuffer(data);
        pendingData = null;
    }
    reportBytesRead(data.length());
    if (dataHandler != null) {
        dataHandler.handle(data);
    }
