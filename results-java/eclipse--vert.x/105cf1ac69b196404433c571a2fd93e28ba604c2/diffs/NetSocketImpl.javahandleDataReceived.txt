synchronized void handleDataReceived(Buffer data) {
    checkContext();
    if (paused) {
        if (pendingData == null) {
            pendingData = new ArrayDeque<>();
        }
        pendingData.add(data);
        return;
    }
    if (metrics.isEnabled()) {
        metrics.bytesRead(remoteAddress(), data.length());
    }
    if (dataHandler != null) {
        dataHandler.handle(data);
    }
