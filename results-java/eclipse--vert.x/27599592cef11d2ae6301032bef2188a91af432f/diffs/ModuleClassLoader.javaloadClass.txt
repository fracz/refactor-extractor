@Override
protected synchronized Class<?> loadClass(String name, boolean resolve) throws ClassNotFoundException {
    Class<?> c;
    try {
        c = platformClassLoader.loadClass(name);
    } catch (ClassNotFoundException e) {
        c = doLoadClass(name);
        if (c == null) {
            throw new ClassNotFoundException(name);
        }
    }
    if (resolve) {
        resolveClass(c);
    }
    return c;
