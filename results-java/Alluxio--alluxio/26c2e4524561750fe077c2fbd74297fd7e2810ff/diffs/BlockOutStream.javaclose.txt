@Override
public void close() throws IOException {
    if (!mClosed) {
        if (mBuffer.position() > 0) {
            appendCurrentBuffer(mBuffer.array(), 0, mBuffer.position());
        }
        mCloser.close();
        if (mLocalFilePath != null) {
            mTachyonFS.cacheBlock(mBlockId);
        }
        mClosed = true;
    }
