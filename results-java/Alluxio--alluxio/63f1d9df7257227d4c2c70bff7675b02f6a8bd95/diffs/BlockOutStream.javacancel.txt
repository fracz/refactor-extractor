@Override
public void cancel() throws IOException {
    if (mClosed) {
        return;
    }
    Exception exception = null;
    try {
        mOutStream.cancel();
    } catch (Exception e) {
        exception = e;
    }
    try {
        mBlockWorkerClient.cancelBlock(mBlockId);
    } catch (Exception e) {
        exception = e;
    }
    if (exception == null) {
        mClosed = true;
        return;
    }
    CommonUtils.closeQuietly(mCloser);
    mClosed = true;
    throw AlluxioStatusException.from(exception);
