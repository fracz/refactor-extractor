    if (INACTIVE) {
        return;
    }
    try {
        DOS.writeLong(++mTransactionId);
        DOS.writeByte(OP_CREATE_FILE);
        DOS.writeBoolean(recursive);
        Utils.writeString(path, DOS);
        DOS.writeBoolean(directory);

        DOS.writeLong(creationTimeMs);

        CommonUtils.runtimeException(e);
