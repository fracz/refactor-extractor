/**
 * Resets the handler to its initial state.
 */
private void reset() {
    if (mRequest != null) {
        mRequest.close();
        mRequest = null;
    }
    mLock.lock();
    try {
        mPacketReaderActive = false;
        mPosToQueue = -1;
        mPosToWrite = -1;
    } finally {
        mLock.unlock();
    }
