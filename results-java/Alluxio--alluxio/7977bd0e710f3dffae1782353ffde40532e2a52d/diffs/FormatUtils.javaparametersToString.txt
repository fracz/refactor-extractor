public static String parametersToString(Object... objs) {
    StringBuilder sb = new StringBuilder("(");
    for (int k = 0; k < objs.length; k++) {
        if (k != 0) {
            sb.append(", ");
        }
        sb.append(objs[k].toString());
    }
    sb.append(")");
    return sb.toString();
