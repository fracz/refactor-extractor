@Test
public void testStatus() throws Exception {
    MessageFramer framer = new MessageFramer(TRANSPORT_FRAME_SIZE);
    byte[] unframedStream = Bytes.concat(new byte[] { GrpcFramingUtil.STATUS_FRAME }, // Len is 2 bytes
    new byte[] { 0, 0, 0, 2 }, // Internal==13
    new byte[] { 0, 13 });
    CapturingSink sink = new CapturingSink();
    for (int i = 0; i < 1000; i++) {
        framer.writeStatus(new Status(Transport.Code.INTERNAL), (i % 17 == 11), sink);
        if ((i + 1) % 13 == 0) {
            framer.flush(sink);
        }
    }
    framer.flush(sink);
    assertEquals(sink.deframedStream.length, unframedStream.length * 1000);
    for (int i = 0; i < 1000; i++) {
        assertArrayEquals(unframedStream, Arrays.copyOfRange(sink.deframedStream, i * unframedStream.length, (i + 1) * unframedStream.length));
    }
