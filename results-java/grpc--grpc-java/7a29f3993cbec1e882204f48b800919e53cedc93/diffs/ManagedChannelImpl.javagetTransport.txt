@Override
public ListenableFuture<ClientTransport> getTransport(final EquivalentAddressGroup addressGroup) {
    Preconditions.checkNotNull(addressGroup, "addressGroup");
    TransportSet ts;
    synchronized (lock) {
        if (shutdown) {
            return NULL_VALUE_TRANSPORT_FUTURE;
        }
        ts = transports.get(addressGroup);
        if (ts == null) {
            ts = new TransportSet(addressGroup, authority(), loadBalancer, backoffPolicyProvider, transportFactory, scheduledExecutor, new TransportSet.Callback() {

                @Override
                public void onTerminated() {
                    synchronized (lock) {
                        transports.remove(addressGroup);
                        if (shutdown && transports.isEmpty()) {
                            if (terminated) {
                                log.warning("transportTerminated called after already terminated");
                            }
                            terminated = true;
                            lock.notifyAll();
                            onChannelTerminated();
                        }
                    }
                }
            });
            transports.put(addressGroup, ts);
        }
    }
    return ts.obtainActiveTransport();
