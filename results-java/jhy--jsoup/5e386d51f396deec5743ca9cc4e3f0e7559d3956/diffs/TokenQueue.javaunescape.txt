/**
 * Unescaped a \ escaped string.
 * @param in backslash escaped string
 * @return unescaped string
 */
public static String unescape(String in) {
    StringBuilder out = new StringBuilder();
    char last = 0;
    for (char c : in.toCharArray()) {
        if (c == ESC) {
            if (last != 0 && last == ESC)
                out.append(c);
        } else
            out.append(c);
        last = c;
    }
    return out.toString();
