private void forgeAndOpenDocument(String xmlValue) {
    try {
        DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
        factory.setNamespaceAware(true);
        factory.setIgnoringComments(true);
        factory.setIgnoringElementContentWhitespace(true);
        DocumentBuilder documentBuilder = factory.newDocumentBuilder();
        Document document = documentBuilder.parse(new ByteArrayInputStream(xmlValue.getBytes()));
        parser = new XmlResourceParserImpl(document, "file", TestUtil.testResources().getPackageName(), TEST_PACKAGE, resourceLoader);
        // Navigate to the root element
        parseUntilNext(XmlResourceParser.START_TAG);
    } catch (Exception parsingException) {
        // exception for convenience.
        throw new RuntimeException("Cannot forge a Document from an invalid XML", parsingException);
    }
