private void reportCounter(String name, Counter counter) {
    final String group = group(name);
    try {
        announce(prefix(name, "count"), group, counter.getCount(), "");
    } catch (GangliaException e) {
        LOGGER.warn("Unable to report counter {}", name, e);
    }
