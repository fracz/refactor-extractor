private AstyanaxContext.Builder getContextBuilder(Configuration config) {
    final int maxConnsPerHost = config.getInt(MAX_CONNECTIONS_PER_HOST_KEY, MAX_CONNECTIONS_PER_HOST_DEFAULT);
    final ConnectionPoolType poolType = ConnectionPoolType.valueOf(config.getString(CONNECTION_POOL_TYPE_KEY, CONNECTION_POOL_TYPE_DEFAULT));
    final NodeDiscoveryType discType = NodeDiscoveryType.valueOf(config.getString(NODE_DISCOVERY_TYPE_KEY, NODE_DISCOVERY_TYPE_DEFAULT));
    AstyanaxContext.Builder builder = new AstyanaxContext.Builder().forCluster(clusterName).forKeyspace(ksName).withAstyanaxConfiguration(new AstyanaxConfigurationImpl().setConnectionPoolType(poolType).setDiscoveryType(discType)).withConnectionPoolConfiguration(new ConnectionPoolConfigurationImpl("TitanConnectionPool").setPort(config.getInt(PORT_KEY, PORT_DEFAULT)).setMaxConnsPerHost(maxConnsPerHost).setRetryBackoffStrategy(// TODO configuration
    new FixedRetryBackoffStrategy(1000, 5000)).setSocketTimeout(config.getInt(THRIFT_TIMEOUT_KEY, THRIFT_TIMEOUT_DEFAULT)).setConnectTimeout(config.getInt(THRIFT_TIMEOUT_KEY, THRIFT_TIMEOUT_DEFAULT)).setSeeds(// "127.0.0.1:9160" or "127.0.0.1"
    config.getString(HOSTNAME_KEY, HOSTNAME_DEFAULT))).withConnectionPoolMonitor(new CountingConnectionPoolMonitor());
    return builder;
