@Override
public boolean equals(Object obj) {
    if (this == obj)
        return true;
    if (obj == null)
        return false;
    if (getClass() != obj.getClass())
        return false;
    LockClaim other = (LockClaim) obj;
    return other.cf.equals(this.cf) && other.key.equals(this.key) && other.column.equals(this.column);
