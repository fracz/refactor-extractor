    graph.makeLabel("knows").sortKey(name).sortOrder(Order.DESC).make();
    TitanVertex previous = null;
    for (int i = 0; i < numV; i++) {
        TitanVertex v = graph.addVertex(null);
        v.setProperty("name", strs[i % strs.length]);
