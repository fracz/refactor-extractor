public static ByteBuffer stringToByteBuffer(String s) {
    byte[] b;
    try {
        b = s.getBytes("UTF-8");
    } catch (UnsupportedEncodingException e) {
        throw new RuntimeException(e);
    }
    ByteBuffer bb = ByteBuffer.allocate(b.length);
    bb.put(b);
    bb.flip();
    return bb;
