private void deleteIndexEntry(TitanProperty prop, StoreMutator mutator) {
    TitanKey pt = prop.getPropertyKey();
    assert pt.isSimple();
    if (pt.hasIndex()) {
        if (pt.isUnique()) {
            mutator.mutateIndex(getIndexKey(prop.getAttribute()), null, Lists.newArrayList(getKeyedIndexColumn(prop.getPropertyKey())));
        } else {
            mutator.mutateIndex(getIndexKey(prop.getAttribute()), null, Lists.newArrayList(getIndexColumn(prop.getPropertyKey(), prop.getID())));
        }
    }
