private void mutate(Map<ByteBuffer, Mutation> mutations, ByteBuffer key, List<Entry> additions, List<ByteBuffer> deletions) {
    if ((additions == null || additions.isEmpty()) && (deletions == null || deletions.isEmpty()))
        return;
    Mutation m = new Mutation(additions, deletions);
    Mutation existingM = mutations.get(key);
    if (existingM != null) {
        existingM.merge(m);
    } else {
        mutations.put(key, m);
    }
    if (additions != null)
        numMutations += additions.size();
    if (deletions != null)
        numMutations += deletions.size();
    if (numMutations >= bufferSize) {
        flushInternal();
    }
