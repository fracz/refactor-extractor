@Override
public synchronized void shutdown() throws TitanException {
    if (!isOpen)
        return;
    try {
        super.shutdown();
        idAssigner.close();
        backend.close();
        queryCache.close();
    } catch (StorageException e) {
        throw new TitanException("Could not close storage backend", e);
    } finally {
        isOpen = false;
    }
