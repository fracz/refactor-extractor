@Override
public <T> LocalLockMediator<T> get(String namespace) {
    Preconditions.checkNotNull(namespace);
    @SuppressWarnings("unchecked") LocalLockMediator<T> m = (LocalLockMediator<T>) mediators.get(namespace);
    if (null == m) {
        m = new LocalLockMediator<T>(namespace);
        @SuppressWarnings("unchecked") LocalLockMediator<T> old = (LocalLockMediator<T>) mediators.putIfAbsent(namespace, m);
        if (null != old)
            m = old;
        else
            log.debug("Local lock mediator instantiated for namespace {}", namespace);
    }
    return m;
