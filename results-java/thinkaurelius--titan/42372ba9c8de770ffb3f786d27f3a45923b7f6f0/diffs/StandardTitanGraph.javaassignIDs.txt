private void assignIDs(Collection<InternalRelation> addedEdges, InternalTitanTransaction tx) {
    for (InternalRelation edge : addedEdges) {
        if (edge.isRemoved())
            continue;
        if (edge.hasID())
            continue;
        for (int i = 0; i < edge.getArity(); i++) {
            InternalTitanVertex node = edge.getVertex(i);
            if (!node.hasID()) {
                assert node.isNew();
                assignID(node);
            }
        }
        assert !edge.hasID();
        assignID(edge);
    }
