private <O> O valueInternal(RelationType type) {
    if (type == null) {
        return null;
    } else if (type.isPropertyKey()) {
        return valueOrNull((PropertyKey) type);
    } else {
        assert type.isEdgeLabel();
        Object val = it().getValueDirect(type);
        if (val == null)
            return null;
        else if (val instanceof Number)
            return (O) tx().getInternalVertex(((Number) val).longValue());
        else if (val instanceof TitanVertex)
            return (O) val;
        else
            throw new IllegalStateException("Invalid object found instead of vertex: " + val);
    }
