private static long readUnsignedBackward(ReadBuffer in) {
    int position = in.getPosition();
    long value = 0;
    int b;
    do {
        b = in.getByte(position);
        value = value << 7;
        value = value | (b & BIT_MASK);
        position--;
    } while (b >= 0);
    in.movePosition(position - in.getPosition());
    return value;
