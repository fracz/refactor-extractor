@Override
public synchronized void commit() throws StorageException {
    if (tx == null)
        return;
    db.setSessionId(tx.getSessionId());
    int retries = 3;
    try {
        int i = 0;
        while (true) {
            try {
                tx.commit();
                break;
            } catch (RollbackException ex) {
                if (i++ >= retries) {
                    throw ex;
                }
            }
        }
    } catch (PersistitException ex) {
        throw new PermanentStorageException(ex.toString());
    } finally {
        tx.end();
        isOpen = false;
    }
