@Override
public List<String> completion(String s, int i) {
    final List<String> suggestions = new ArrayList<>();
    if (StringUtils.isEmpty(s)) {
        suggestions.addAll(COMMANDS);
    } else {
        for (String cmd : COMMANDS) {
            if (cmd.toLowerCase().contains(s)) {
                suggestions.add(cmd);
            }
        }
    }
    return suggestions;
