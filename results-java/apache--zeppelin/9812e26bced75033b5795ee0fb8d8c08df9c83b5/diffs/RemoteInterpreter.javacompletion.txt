public List<InterpreterCompletion> completion(final String buf, final int cursor, final InterpreterContext interpreterContext) {
    if (!isOpened) {
        LOGGER.warn("completion is called when RemoterInterpreter is not opened for " + className);
        return new ArrayList<>();
    }
    RemoteInterpreterProcess interpreterProcess = getOrCreateInterpreterProcess();
    return interpreterProcess.callRemoteFunction(new RemoteInterpreterProcess.RemoteFunction<List<InterpreterCompletion>>() {

        @Override
        public List<InterpreterCompletion> call(Client client) throws Exception {
            return client.completion(sessionId, className, buf, cursor, convert(interpreterContext));
        }
    });
