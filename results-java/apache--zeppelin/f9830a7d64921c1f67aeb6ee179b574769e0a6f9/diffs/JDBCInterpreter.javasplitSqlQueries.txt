    ArrayList<String> queries = new ArrayList<>();
    StringBuilder query = new StringBuilder();
    Character character;
    Boolean antiSlash = false;
    Boolean quoteString = false;
    Boolean doubleQuoteString = false;

            antiSlash = true;
        }

            if (antiSlash) {
                antiSlash = false;
            } else if (quoteString) {
                quoteString = false;

                quoteString = true;
            }

            query.append(character);
