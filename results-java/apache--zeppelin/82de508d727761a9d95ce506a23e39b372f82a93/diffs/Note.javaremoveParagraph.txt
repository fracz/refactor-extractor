/**
 * Remove paragraph by id.
 *
 * @param paragraphId
 * @return
 */
public Paragraph removeParagraph(String paragraphId) {
    synchronized (paragraphs) {
        for (int i = 0; i < paragraphs.size(); i++) {
            Paragraph p = paragraphs.get(i);
            if (p.getId().equals(paragraphId)) {
                paragraphs.remove(i);
                return p;
            }
        }
    }
    return null;
