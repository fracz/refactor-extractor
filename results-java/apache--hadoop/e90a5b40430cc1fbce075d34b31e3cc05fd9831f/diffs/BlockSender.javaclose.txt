/**
 * close opened files.
 */
public void close() throws IOException {
    IOException ioe = null;
    // close checksum file
    if (checksumIn != null) {
        try {
            checksumIn.close();
        } catch (IOException e) {
            ioe = e;
        }
        checksumIn = null;
    }
    // close data file
    if (blockIn != null) {
        try {
            blockIn.close();
        } catch (IOException e) {
            ioe = e;
        }
        blockIn = null;
    }
    // throw IOException if there is any
    if (ioe != null) {
        throw ioe;
    }
