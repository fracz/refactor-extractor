protected synchronized void closeImpl() throws IOException {
    if (isClosed()) {
        IOException e = streamer.getLastException().getAndSet(null);
        if (e == null)
            return;
        else
            throw e;
    }
    try {
        // flush from all upper layers
        flushBuffer();
        if (currentPacket != null) {
            streamer.waitAndQueuePacket(currentPacket);
            currentPacket = null;
        }
        if (streamer.getBytesCurBlock() != 0) {
            // send an empty packet to mark the end of the block
            currentPacket = createPacket(0, 0, streamer.getBytesCurBlock(), streamer.getAndIncCurrentSeqno(), true);
            currentPacket.setSyncBlock(shouldSyncBlock);
        }
        // flush all data to Datanodes
        flushInternal();
        // get last block before destroying the streamer
        ExtendedBlock lastBlock = streamer.getBlock();
        closeThreads(false);
        TraceScope scope = Trace.startSpan("completeFile", Sampler.NEVER);
        try {
            completeFile(lastBlock);
        } finally {
            scope.close();
        }
        dfsClient.endFileLease(fileId);
    } catch (ClosedChannelException e) {
    } finally {
        setClosed();
    }
