    fc.mkdir(testDir, FsPermission.getDefault(), true);
    Assert.assertTrue(fc.exists(testDir));
    createFile(getTestRootPath("test/hadoop/file"));
    Path testSubDir = getTestRootPath("test/hadoop/file/subdir");
