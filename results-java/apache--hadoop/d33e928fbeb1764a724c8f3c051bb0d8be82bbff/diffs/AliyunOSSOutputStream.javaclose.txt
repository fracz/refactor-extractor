@Override
public synchronized void close() throws IOException {
    if (closed) {
        return;
    }
    closed = true;
    if (backupStream != null) {
        backupStream.close();
    }
    long dataLen = tmpFile.length();
    try {
        if (dataLen <= partSizeThreshold) {
            uploadObject();
        } else {
            multipartUploadObject();
        }
    } finally {
        if (!tmpFile.delete()) {
            LOG.warn("Can not delete file: " + tmpFile);
        }
    }
