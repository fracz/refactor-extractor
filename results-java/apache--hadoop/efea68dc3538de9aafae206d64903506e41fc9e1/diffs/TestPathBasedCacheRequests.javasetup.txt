public void setup() throws Exception {
    conf = new HdfsConfiguration();
    // set low limits here for testing purposes
    conf.setInt(DFSConfigKeys.DFS_NAMENODE_LIST_CACHE_POOLS_NUM_RESPONSES, 2);
    conf.setInt(DFSConfigKeys.DFS_NAMENODE_LIST_CACHE_DIRECTIVES_NUM_RESPONSES, 2);
    cluster = new MiniDFSCluster.Builder(conf).numDataNodes(1).build();
    cluster.waitActive();
    dfs = cluster.getFileSystem();
    proto = cluster.getNameNodeRpc();
    prevCacheManipulator = NativeIO.POSIX.cacheManipulator;
    // Save the current CacheManipulator and replace it at the end of the test
    // Stub out mlock calls to avoid failing when not enough memory is lockable
    // by the operating system.
    NativeIO.POSIX.cacheManipulator = new CacheManipulator() {

        @Override
        public void mlock(String identifier, ByteBuffer mmap, long length) throws IOException {
            LOG.info("mlocking " + identifier);
        }
    };
