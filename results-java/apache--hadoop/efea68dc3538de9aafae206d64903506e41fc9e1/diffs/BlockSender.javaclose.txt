/**
 * close opened files.
 */
@Override
public void close() throws IOException {
    if (blockInFd != null && ((dropCacheBehindAllReads) || (dropCacheBehindLargeReads && isLongRead()))) {
        try {
            NativeIO.POSIX.posixFadviseIfPossible(block.getBlockName(), blockInFd, lastCacheDropOffset, offset - lastCacheDropOffset, NativeIO.POSIX.POSIX_FADV_DONTNEED);
        } catch (Exception e) {
            LOG.warn("Unable to drop cache on file close", e);
        }
    }
    if (curReadahead != null) {
        curReadahead.cancel();
    }
    IOException ioe = null;
    if (checksumIn != null) {
        try {
            // close checksum file
            checksumIn.close();
        } catch (IOException e) {
            ioe = e;
        }
        checksumIn = null;
    }
    if (blockIn != null) {
        try {
            // close data file
            blockIn.close();
        } catch (IOException e) {
            ioe = e;
        }
        blockIn = null;
        blockInFd = null;
    }
    // throw IOException if there is any
    if (ioe != null) {
        throw ioe;
    }
