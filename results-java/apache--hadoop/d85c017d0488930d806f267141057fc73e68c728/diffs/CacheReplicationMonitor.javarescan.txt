private void rescan() {
    scannedDirectives = 0;
    scannedBlocks = 0;
    namesystem.writeLock();
    try {
        resetStatistics();
        rescanCacheDirectives();
        rescanCachedBlockMap();
        blockManager.getDatanodeManager().resetLastCachingDirectiveSentTime();
    } finally {
        namesystem.writeUnlock();
    }
