@Override
public void handleMessage(Message msg) {
    Request request = (Request) msg.obj;
    if (request.future.isCancelled()) {
        return;
    }
    switch(msg.what) {
        case REQUEST_COMPLETE:
            request.complete();
            break;
        case REQUEST_RETRY:
            request.picasso.retry(request);
            break;
        default:
            throw new IllegalArgumentException(String.format("Unknown handler message received! %d", msg.what));
    }
