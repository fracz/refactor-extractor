@Override
void complete() {
    if (result == null) {
        throw new AssertionError(String.format("Attempted to complete request with no result!\n%s", this));
    }
    Target target = this.target.get();
    if (target != null) {
        picasso.targetsToRequests.remove(target);
        target.onSuccess(result);
    }
