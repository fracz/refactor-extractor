@Test
public void post() throws Exception {
    server.enqueue(new MockResponse().setBody("abc"));
    server.play();
    Request request = new Request.Builder().url(server.getUrl("/")).post(RequestBody.create(MediaType.parse("text/plain"), "def")).build();
    executeSynchronously(request).assertCode(200).assertBody("abc");
    RecordedRequest recordedRequest = server.takeRequest();
    assertEquals("POST", recordedRequest.getMethod());
    assertEquals("def", recordedRequest.getUtf8Body());
    assertEquals("3", recordedRequest.getHeader("Content-Length"));
    assertEquals("text/plain; charset=utf-8", recordedRequest.getHeader("Content-Type"));
