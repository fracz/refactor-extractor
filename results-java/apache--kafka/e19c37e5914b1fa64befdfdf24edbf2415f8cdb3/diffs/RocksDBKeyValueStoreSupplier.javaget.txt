    }
    if (cached) {
        return new CachingKeyValueStore<>(new MeteredKeyValueStore<>(rocks, METRICS_SCOPE, time), keySerde, valueSerde);
    } else {
        // logged
        return new MeteredKeyValueStore<>(new ChangeLoggingKeyValueStore<>(rocks, keySerde, valueSerde), METRICS_SCOPE, time);
    }
