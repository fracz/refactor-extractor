private void maybeEvict(final String namespace) {
    while (sizeBytes() > maxCacheSizeBytes) {
        final NamedCache cache = getOrCreateCache(namespace);
        // namespaced cache.
        if (cache.size() == 0) {
            return;
        }
        log.trace("Thread {} evicting cache {}", name, namespace);
        cache.evict();
        numEvicts++;
    }
