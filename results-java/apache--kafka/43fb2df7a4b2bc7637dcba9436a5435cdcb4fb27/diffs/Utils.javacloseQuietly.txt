/**
 * Closes {@code closeable} and if an exception is thrown, it is logged at the WARN level.
 */
public static void closeQuietly(Closeable closeable, String name) {
    if (closeable != null) {
        try {
            closeable.close();
        } catch (Throwable t) {
            log.warn("Failed to close " + name, t);
        }
    }
