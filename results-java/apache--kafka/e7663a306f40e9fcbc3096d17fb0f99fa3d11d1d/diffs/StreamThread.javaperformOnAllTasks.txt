private void performOnAllTasks(final AbstractTaskAction action, final String exceptionMessage, final boolean throwExceptions) {
    final List<AbstractTask> allTasks = new ArrayList<AbstractTask>(activeTasks.values());
    allTasks.addAll(standbyTasks.values());
    for (final AbstractTask task : allTasks) {
        try {
            action.apply(task);
        } catch (KafkaException e) {
            log.error(String.format("stream-thread [%s] Failed to %s for %s %s: ", StreamThread.this.getName(), exceptionMessage, task.getClass().getSimpleName(), task.id()), e);
            if (throwExceptions) {
                throw e;
            }
        }
    }
