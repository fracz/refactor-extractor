@Override
public void initialize(GraylogServer server, Map<String, String> config) throws InitializerConfigurationException {
    Core srv = (Core) server;
    if (srv.getConfiguration().isEnableLibratoSystemMetrics()) {
        // Enable librato metrics reporter.
        LibratoReporter.enable(LibratoReporter.builder(srv.getConfiguration().getLibratoMetricsAPIUser(), srv.getConfiguration().getLibratoMetricsAPIToken(), srv.getServerId() + "-" + srv.getConfiguration().getLibratoMetricsPrefix()), srv.getConfiguration().getLibratoMetricsInterval(), TimeUnit.SECONDS);
    }
    configureScheduler((Core) server, new LibratoMetricsWriterThread(srv), LibratoMetricsWriterThread.INITIAL_DELAY, srv.getConfiguration().getLibratoMetricsInterval());
