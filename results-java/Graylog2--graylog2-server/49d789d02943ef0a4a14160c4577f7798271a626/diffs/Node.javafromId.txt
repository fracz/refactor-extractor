public static Node fromId(String id) {
    NodeSummaryResponse response = null;
    try {
        response = Api.get(Configuration.getServerRestUris().get(0), "/cluster/nodes/" + id, NodeSummaryResponse.class);
    } catch (IOException e) {
        return null;
    } catch (APIException e) {
        return null;
    }
    return new Node(response);
