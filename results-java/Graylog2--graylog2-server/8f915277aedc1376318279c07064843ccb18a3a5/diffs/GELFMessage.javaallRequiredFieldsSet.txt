 * must be set: _version, _short_message, _host
 *
 * @return boolean
 */
public boolean allRequiredFieldsSet() {
    if (this.getVersion() == null || this.getShortMessage() == null || this.getHost() == null || this.getVersion().length() == 0 || this.getShortMessage().length() == 0 || this.getHost().length() == 0) {
        return false;
    }
    return true;
