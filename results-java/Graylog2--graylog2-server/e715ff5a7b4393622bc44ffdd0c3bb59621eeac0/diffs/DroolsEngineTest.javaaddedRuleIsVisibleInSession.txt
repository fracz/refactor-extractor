public void addedRuleIsVisibleInSession() {
    final DroolsEngine engine = new DroolsEngine();
    String rule1 = "package org.graylog2.rules\n" + "import org.graylog2.plugin.Message\n" + "global org.slf4j.Logger log \n\n" + "declare Message\n" + "    @role( event )\n" + "end\n" + "\n" + "rule \"filter out all messages\"\n" + "when\n" + "    $m : Message( filterOut == false )\n" + "then\n" + "    modify($m) { setFilterOut(true) };\n" + "    log.info(\"filtering out message from \" + $m.getSource());\n" + "end\n";
    String rule2 = "package org.graylog2.rules\n" + "import org.graylog2.plugin.Message\n" + "global org.slf4j.Logger log \n\n" + "declare Message\n" + "    @role( event )\n" + "end\n" + "\n" + "rule \"print filtered out message source\"\n" + "when\n" + "    $m : Message( filterOut == true )\n" + "then\n" + "    log.info(\"message from \" + $m.getSource() + \" filtered out\");\n" + "end\n";
