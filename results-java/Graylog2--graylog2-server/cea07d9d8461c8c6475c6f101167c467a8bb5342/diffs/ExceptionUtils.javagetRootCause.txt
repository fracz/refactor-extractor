public static Throwable getRootCause(Throwable t) {
    Throwable rootCause = t;
    Throwable cause = rootCause.getCause();
    while (cause != null && cause != rootCause) {
        rootCause = cause;
        cause = cause.getCause();
    }
    return rootCause;
