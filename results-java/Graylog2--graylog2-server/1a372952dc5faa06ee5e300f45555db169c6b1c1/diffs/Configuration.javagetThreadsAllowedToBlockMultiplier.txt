public static int getThreadsAllowedToBlockMultiplier(Properties config) {
    String val = config.getProperty("mongodb_threads_allowed_to_block_multiplier");
    if (val != null) {
        int res = Integer.parseInt(val);
        if (res > 0) {
            return res;
        }
    }
    // Default value.
    return 5;
