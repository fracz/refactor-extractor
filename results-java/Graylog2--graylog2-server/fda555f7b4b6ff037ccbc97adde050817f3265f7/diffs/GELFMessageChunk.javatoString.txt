@Override
public String toString() {
    StringBuilder sb = new StringBuilder();
    try {
        sb.append("GELF chunk:\n");
        sb.append("\tID: ");
        sb.append(this.id);
        sb.append("\tSequence: ");
        // +1 for readability: 1/2 not 0/2
        sb.append(this.sequenceNumber + 1);
        sb.append("/");
        sb.append(this.sequenceCount);
        sb.append("\tArrival: ");
        sb.append(this.arrival);
        sb.append("\tData size: ");
        sb.append(this.payload.length);
    } catch (Exception e) {
        sb.append("UNKNOWN/EMPTY/INVALID");
    }
    return sb.toString();
