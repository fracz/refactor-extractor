/**
 * Set the size of the drawable.
 *
 * @param size The size in pixels (px).
 * @return The current IconExtDrawable for chaining.
 */
public IconicsDrawable sizePx(@Dimension(unit = PX) int size) {
    this.mSizeX = size;
    this.mSizeY = size;
    setBounds(0, 0, size, size);
    invalidateSelf();
    return this;
