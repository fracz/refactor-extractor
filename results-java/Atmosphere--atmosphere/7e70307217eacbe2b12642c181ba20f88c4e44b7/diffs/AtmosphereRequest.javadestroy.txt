public void destroy() {
    if (!b.destroyable)
        return;
    b.localAttributes.clear();
    if (bis != null) {
        try {
            bis.close();
        } catch (IOException e) {
        }
    }
    if (br != null) {
        try {
            br.close();
        } catch (IOException e) {
        }
    }
    headers.clear();
    queryStrings.clear();
    // Help GC
    if (b.request != null) {
        b.request.removeAttribute(FrameworkConfig.ATMOSPHERE_RESOURCE);
        b.request = null;
    }
