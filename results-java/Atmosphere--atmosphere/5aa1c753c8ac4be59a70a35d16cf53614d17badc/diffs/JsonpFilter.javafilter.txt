public BroadcastAction filter(Object o) {
    if (o instanceof String) {
        String m = (String) o;
        String name = m;
        String message = "";
        if (m.indexOf("__") > 0) {
            name = m.substring(0, m.indexOf("__"));
            message = m.substring(m.indexOf("__") + 2);
        }
        return new BroadcastAction(BEGIN_SCRIPT_TAG + "window.parent.app.update({ name: \"" + name + "\", message: \"" + message + "\" });\n" + END_SCRIPT_TAG);
    } else {
        return new BroadcastAction(o);
    }
