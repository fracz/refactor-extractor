        @Override
        public void call(BangumiRecommend bangumiRecommend) {
            banners = bangumiRecommend.getBanners();
            recommends = bangumiRecommend.getRecommends();
        // finishTask();
        }
    }, new Action1<Throwable>() {

        @Override
        public void call(Throwable throwable) {

        }
