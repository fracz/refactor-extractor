/**
 * Tokenizes a two-char operator.
 * @return true if it tokenized an operator
 */
private boolean tokenizeTwoChars() {
    if (pos + 2 >= buffer.length) {
        return false;
    }
    char c1 = buffer[pos];
    char c2 = buffer[pos + 1];
    TokenKind tok = null;
    if (c2 == '=') {
        tok = EQUAL_TOKENS.get(c1);
    }
    if (tok == null) {
        return false;
    } else {
        addToken(new Token(tok, pos, pos + 2));
        return true;
    }
