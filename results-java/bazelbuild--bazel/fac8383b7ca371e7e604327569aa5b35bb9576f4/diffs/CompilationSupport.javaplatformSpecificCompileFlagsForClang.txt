private static List<String> platformSpecificCompileFlagsForClang(AppleConfiguration configuration) {
    switch(configuration.getPlatform(PlatformType.IOS)) {
        case IOS_DEVICE:
            return ImmutableList.of();
        case IOS_SIMULATOR:
            return IOS_SIMULATOR_COMPILE_FLAGS;
        default:
            throw new AssertionError();
    }
