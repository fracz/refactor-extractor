/**
 * Returns a particular configured target.
 *
 * <p>Used only for testing.
 */
@VisibleForTesting
@Nullable
public ConfiguredTarget getConfiguredTargetForTesting(EventHandler eventHandler, Label label, BuildConfiguration configuration) {
    if (memoizingEvaluator.getExistingValueForTesting(PrecomputedValue.WORKSPACE_STATUS_KEY.getKeyForTesting()) == null) {
        injectWorkspaceStatusData();
    }
    return Iterables.getFirst(getConfiguredTargets(eventHandler, configuration, ImmutableList.of(new Dependency(label, configuration)), true), null);
