private static void passRemoteOutErr(RemoteActionCache cache, ActionResult result, FileOutErr outErr) throws IOException {
    try {
        ImmutableList<byte[]> streams = cache.downloadBlobs(ImmutableList.of(result.getStdoutDigest(), result.getStderrDigest()));
        byte[] stdout = streams.get(0);
        byte[] stderr = streams.get(1);
        // Don't round-trip through String - write to the output streams directly.
        if (stdout.length != 0) {
            outErr.getOutputStream().write(stdout);
            outErr.getOutputStream().flush();
        }
        if (stderr.length != 0) {
            outErr.getErrorStream().write(stderr);
            outErr.getErrorStream().flush();
        }
    } catch (CacheNotFoundException e) {
    // Ignoring.
    }
