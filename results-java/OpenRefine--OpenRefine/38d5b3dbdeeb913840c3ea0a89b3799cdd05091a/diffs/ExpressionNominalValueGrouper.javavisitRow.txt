    Object value = _evaluable.evaluate(bindings);
    if (value != null) {
        if (value.getClass().isArray()) {
            Object[] a = (Object[]) value;
            for (Object v : a) {
                processValue(v, rowIndex);
            }

            for (Object v : ExpressionUtils.toObjectCollection(value)) {

    }
