public boolean deleteProject(Project project) {
    if (_projectsMetadata.containsKey(project.id)) {
        _projects.remove(project.id);
        _projectsMetadata.remove(project.id);
        File file = new File(_dir, project.id + ".project");
        if (file.exists()) {
            file.delete();
        }
        save();
        return true;
    } else {
        return false;
    }
