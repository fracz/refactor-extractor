private void RunTest(String testString) {
    try {
        inputStream = new ByteArrayInputStream(testString.getBytes("UTF-8"));
    } catch (UnsupportedEncodingException e1) {
        Assert.fail();
    }
    try {
        SUT.read(inputStream, project, options);
    } catch (Exception e) {
        Assert.fail();
    }
    try {
        inputStream.close();
    } catch (IOException e) {
        Assert.fail();
    }
