@Override
public void destroy() {
    logger.trace("> destroy");
    // cancel automatic periodic saving and force a complete save.
    if (_timer != null) {
        _timer.cancel();
        _timer = null;
    }
    if (ProjectManager.singleton != null) {
        // complete save
        ProjectManager.singleton.save(true);
        ProjectManager.singleton = null;
    }
    super.destroy();
    logger.trace("< destroy");
