public void apply(Project project) {
    if (_change == null) {
        loadChange();
    }
    synchronized (project) {
        _change.apply(project);
        try {
            saveChange();
        } catch (Exception e) {
            e.printStackTrace();
            _change.revert(project);
            throw new RuntimeException("Failed to apply change", e);
        }
    }
