        @Override
        protected ComponentLayout onCreateLayout(ComponentContext c) {
            return Column.create(c).child(TestDrawableComponent.create(c).measuredWidth(width).measuredHeight(height)).build();
        }
    };
    InternalNode node = LayoutState.createAndMeasureTreeForComponent(c, component, SizeSpec.makeSizeSpec(width, SizeSpec.AT_MOST), SizeSpec.makeSizeSpec(height, SizeSpec.AT_MOST));
