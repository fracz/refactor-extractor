private void exportURLs(SiteNode node, BufferedWriter writer, boolean html) {
    if (node == null) {
        return;
    }
    try {
        if (node.getHistoryReference() != null && (node.getHistoryReference().getHistoryType() == HistoryReference.TYPE_MANUAL || node.getHistoryReference().getHistoryType() == HistoryReference.TYPE_SPIDER) && node.getHistoryReference().getHttpMessage() != null && node.getHistoryReference().getHttpMessage().getRequestHeader() != null && node.getHistoryReference().getHttpMessage().getRequestHeader().getURI() != null) {
            writer.write(node.getHistoryReference().getHttpMessage().getRequestHeader().getMethod());
            writer.write("\t");
            if (html) {
                writer.write("<a href=\"");
                writer.write(node.getHistoryReference().getHttpMessage().getRequestHeader().getURI().toString());
                writer.write("\">");
                writer.write(node.getHistoryReference().getHttpMessage().getRequestHeader().getURI().toString());
                writer.write("</a><br>");
            } else {
                writer.write(node.getHistoryReference().getHttpMessage().getRequestHeader().getURI().toString());
            }
            writer.write(CRLF);
        }
    } catch (HttpMalformedHeaderException e) {
        log.warn(e.getStackTrace(), e);
    } catch (IOException e) {
        log.warn(e.getStackTrace(), e);
    } catch (SQLException e) {
        log.warn(e.getStackTrace(), e);
    }
    @SuppressWarnings("unchecked") Enumeration<SiteNode> en = node.children();
    while (en.hasMoreElements()) {
        exportURLs(en.nextElement(), writer, html);
    }
