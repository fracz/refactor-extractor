@Override
public void sessionChanged(final Session session) {
    if (EventQueue.isDispatchThread()) {
        sessionChangedEventHandler(session);
    } else {
        try {
            EventQueue.invokeAndWait(new Runnable() {

                @Override
                public void run() {
                    sessionChangedEventHandler(session);
                }
            });
        } catch (Exception e) {
            logger.error(e.getMessage(), e);
        }
    }
