private JPanel getWindowPanel() {
    if (panelWindow == null) {
        panelWindow = new JPanel();
        panelWindow.setLayout(new BorderLayout());
        panelWindow.add(requestResponsePanel);

        panelWindow.add(getFooterStatusBar(), BorderLayout.SOUTH);
    }
    return panelWindow;
