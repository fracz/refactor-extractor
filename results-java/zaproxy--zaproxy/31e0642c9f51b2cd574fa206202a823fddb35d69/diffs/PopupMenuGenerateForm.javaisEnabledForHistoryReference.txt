@Override
public boolean isEnabledForHistoryReference(HistoryReference href) {
    if (API.getInstance().isEnabled() && DesktopUtils.canOpenUrlInBrowser()) {
        try {
            if (href.getHttpMessage().getRequestHeader().getMethod().equals(HttpRequestHeader.POST) && href.getHttpMessage().getRequestBody().length() > 0) {
                return true;
            }
        } catch (Exception e) {
        // Ignore - this is 'just' for a right click menu
        }
    }
    return false;
