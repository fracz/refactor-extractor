/**
 * Draws the MarkerView on the given position on the screen with the given
 * Canvas object.
 *
 * @param canvas
 * @param posx
 * @param posy
 */
public void draw(Canvas canvas, float posx, float posy) {
    // take offsets into consideration
    posx += mXOffset;
    posy += mYOffset;
    canvas.translate(posx, posy);
    draw(canvas);
    canvas.translate(-posx, -posy);
