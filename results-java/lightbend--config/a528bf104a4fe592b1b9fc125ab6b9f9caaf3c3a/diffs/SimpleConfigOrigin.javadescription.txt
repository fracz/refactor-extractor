@Override
public String description() {
    // parsing "file: line" syntax would hit the ":" in the URL.
    if (lineNumber < 0) {
        return description;
    } else if (endLineNumber == lineNumber) {
        return description + ": " + lineNumber;
    } else {
        return description + ": " + lineNumber + "-" + endLineNumber;
    }
