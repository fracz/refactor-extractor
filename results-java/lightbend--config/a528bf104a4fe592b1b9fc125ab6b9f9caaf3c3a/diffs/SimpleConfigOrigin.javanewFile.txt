static SimpleConfigOrigin newFile(String filename) {
    String url;
    try {
        url = (new File(filename)).toURI().toURL().toExternalForm();
    } catch (MalformedURLException e) {
        url = null;
    }
    return new SimpleConfigOrigin(filename, -1, -1, OriginType.FILE, url, null);
