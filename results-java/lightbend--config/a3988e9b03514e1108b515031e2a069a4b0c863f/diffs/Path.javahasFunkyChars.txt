// this doesn't have a very precise meaning, just to reduce
// noise from quotes in the rendered path
static boolean hasFunkyChars(String s) {
    for (int i = 0; i < s.length(); ++i) {
        char c = s.charAt(i);
        if (Character.isLetterOrDigit(c) || c == ' ')
            continue;
        else
            return true;
    }
    return false;
