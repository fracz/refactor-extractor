/**
 * Install Material Design components
 */
private void installMaterialComponents() {
    MTConfig mtConfig = MTConfig.getInstance();
    this.isMaterialDesign = mtConfig.getIsMaterialDesign();
    if (mtConfig.getIsMaterialDesign()) {
        replaceButtons();
        replaceTextFields();
        replaceProgressBar();
        replaceTree();
        replaceTableHeaders();
    }
