@Override
public void generate(final HttpServletRequest request, final HttpServletResponse response, final OAuthRegisteredService registeredService, final Service service, final AccessToken accessTokenId, final RefreshToken refreshTokenId, final long timeout) {
    if (registeredService.isJsonFormat()) {
        response.setContentType(MediaType.APPLICATION_JSON_VALUE);
        final JsonFactory jsonFactory = new JsonFactory(new ObjectMapper().findAndRegisterModules());
        try (JsonGenerator jsonGenerator = jsonFactory.createGenerator(response.getWriter())) {
            jsonGenerator.writeStartObject();
            generateJsonInternal(request, response, jsonGenerator, accessTokenId, refreshTokenId, timeout, service, registeredService);
            jsonGenerator.writeEndObject();
        } catch (final Exception e) {
            throw new IllegalArgumentException(e);
        }
    } else {
        generateTextInternal(request, response, accessTokenId, refreshTokenId, timeout);
    }
