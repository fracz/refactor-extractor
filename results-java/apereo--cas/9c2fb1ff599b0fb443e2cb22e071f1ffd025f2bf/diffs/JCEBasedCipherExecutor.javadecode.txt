        final SecretKeySpec key = new SecretKeySpec(this.encryptionSecretKey.getBytes(UTF8_ENCODING), this.secretKeyAlgorithm);
        cipher.init(Cipher.DECRYPT_MODE, key, new IvParameterSpec(this.encryptionSeed.getBytes(UTF8_ENCODING)));
