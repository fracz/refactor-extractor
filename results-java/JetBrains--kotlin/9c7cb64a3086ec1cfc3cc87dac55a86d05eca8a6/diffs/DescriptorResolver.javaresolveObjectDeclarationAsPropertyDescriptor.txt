@NotNull
public PropertyDescriptor resolveObjectDeclarationAsPropertyDescriptor(@NotNull DeclarationDescriptor containingDeclaration, @NotNull JetClassOrObject objectDeclaration, @NotNull ClassDescriptor classDescriptor, BindingTrace trace) {
    JetModifierList modifierList = objectDeclaration.getModifierList();
    PropertyDescriptor propertyDescriptor = new PropertyDescriptor(containingDeclaration, annotationResolver.getResolvedAnnotations(modifierList, trace), Modality.FINAL, ModifiersChecker.resolveVisibilityFromModifiers(objectDeclaration), false, JetPsiUtil.safeName(objectDeclaration.getName()), CallableMemberDescriptor.Kind.DECLARATION);
    propertyDescriptor.setType(classDescriptor.getDefaultType(), Collections.<TypeParameterDescriptor>emptyList(), getExpectedThisObjectIfNeeded(containingDeclaration), NO_RECEIVER_PARAMETER);
    propertyDescriptor.initialize(createDefaultGetter(propertyDescriptor), null);
    trace.record(BindingContext.OBJECT_DECLARATION_CLASS, propertyDescriptor, classDescriptor);
    JetObjectDeclarationName nameAsDeclaration = objectDeclaration.getNameAsDeclaration();
    if (nameAsDeclaration != null) {
        trace.record(BindingContext.OBJECT_DECLARATION, nameAsDeclaration, propertyDescriptor);
    }
    return propertyDescriptor;
