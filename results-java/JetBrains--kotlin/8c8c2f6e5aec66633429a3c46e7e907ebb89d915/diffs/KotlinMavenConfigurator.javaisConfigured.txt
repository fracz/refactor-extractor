@Override
public boolean isConfigured(@NotNull Module module) {
    if (isKotlinModule(module)) {
        return true;
    }
    PsiFile pomFile = findModulePomFile(module);
    if (pomFile == null)
        return false;
    String text = pomFile.getText();
    return text.contains("<artifactId>" + MAVEN_PLUGIN_ID + "</artifactId>") && text.contains("<artifactId>" + getLibraryId() + "</artifactId>");
