private void declareTupleIntrinsicAccessors(@NotNull ClassifierDescriptor tupleDescriptor, int tupleSize) {
    for (int elementIndex = 0; elementIndex < tupleSize; ++elementIndex) {
        String accessorName = "_" + (elementIndex + 1);
        PropertyDescriptor propertyDescriptor = getPropertyByName(tupleDescriptor.getDefaultType().getMemberScope(), accessorName);
        functionIntrinsics.put(propertyDescriptor.getGetter(), new TupleAccessIntrinsic(elementIndex));
    }
