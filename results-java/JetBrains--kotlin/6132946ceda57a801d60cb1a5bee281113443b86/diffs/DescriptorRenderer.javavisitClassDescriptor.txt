@Override
public Void visitClassDescriptor(ClassDescriptor descriptor, StringBuilder builder) {
    String keyword;
    switch(descriptor.getKind()) {
        case TRAIT:
            keyword = "trait";
            break;
        case ENUM_CLASS:
            keyword = "enum class";
            break;
        case OBJECT:
            keyword = "object";
            break;
        case ANNOTATION_CLASS:
            keyword = "annotation class";
            break;
        default:
            keyword = "class";
    }
    renderClassDescriptor(descriptor, builder, keyword);
    return null;
