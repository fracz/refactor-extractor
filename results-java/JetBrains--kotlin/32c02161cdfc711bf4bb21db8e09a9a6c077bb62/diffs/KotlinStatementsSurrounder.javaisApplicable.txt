@Override
public boolean isApplicable(@NotNull PsiElement[] elements) {
    if (elements.length == 0) {
        return false;
    }
    if (elements.length == 1 || !(elements[0] instanceof KtExpression) && !isApplicableWhenUsedAsExpression()) {
        KtElement expression = (KtElement) elements[0];
        Boolean usedAsExpression = ResolutionUtils.analyze(expression, BodyResolveMode.PARTIAL).get(BindingContext.USED_AS_EXPRESSION, expression);
        if (usedAsExpression != null && usedAsExpression) {
            return false;
        }
    }
    return true;
