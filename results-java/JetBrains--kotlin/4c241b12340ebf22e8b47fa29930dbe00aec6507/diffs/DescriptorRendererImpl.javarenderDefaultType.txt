private String renderDefaultType(JetType type) {
    StringBuilder sb = new StringBuilder();
    sb.append(renderTypeName(type.getConstructor()));
    if (!type.getArguments().isEmpty()) {
        sb.append("<");
        appendTypeProjections(sb, type.getArguments());
        sb.append(">");
    }
    if (type.isNullable()) {
        sb.append("?");
    }
    return sb.toString();
