@Override
public void visitFormalTypeParameter(final String name, TypeInfoVariance variance) {
    if (!hasFormals) {
        hasFormals = true;
        buf.append('<');
    }
    switch(variance) {
        case OUT:
            buf.append("out ");
            break;
        case IN:
            buf.append("in ");
            break;
        case INVARIANT:
            break;
        default:
            throw new IllegalStateException();
    }
    buf.append(name);
    buf.append(':');
