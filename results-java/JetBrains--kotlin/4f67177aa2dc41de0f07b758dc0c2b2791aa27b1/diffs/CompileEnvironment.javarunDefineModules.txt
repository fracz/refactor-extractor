private List<Module> runDefineModules(String moduleFile, ClassFileFactory factory) {
    GeneratedClassLoader loader = myStdlib != null ? new GeneratedClassLoader(factory, new URLClassLoader(new URL[] { myStdlib }, AllModules.class.getClassLoader())) : new GeneratedClassLoader(factory, CompileEnvironment.class.getClassLoader());
    try {
        Class namespaceClass = loader.loadClass(JvmAbi.PACKAGE_CLASS);
        final Method method = namespaceClass.getDeclaredMethod("project");
        if (method == null) {
            throw new CompileEnvironmentException("Module script " + moduleFile + " must define project() function");
        }
        method.setAccessible(true);
        method.invoke(null);
        ArrayList<Module> modules = AllModules.modules.get();
        ArrayList<Module> answer = new ArrayList<Module>(modules);
        modules.clear();
        return answer;
    } catch (Exception e) {
        throw new ModuleExecutionException(e);
    } finally {
        loader.dispose();
    }
