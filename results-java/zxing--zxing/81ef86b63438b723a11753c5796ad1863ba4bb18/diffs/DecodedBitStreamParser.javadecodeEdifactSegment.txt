/**
 * See ISO 16022:2006, 5.2.8 and Annex C Table C.3
 */
private static int decodeEdifactSegment(BitSource bits, StringBuffer result) {
    boolean unlatch = false;
    do {
        // If there is only two or less bytes left then it will be encoded as ASCII
        if (bits.available() <= 16) {
            return ASCII_ENCODE;
        }
        for (int i = 0; i < 4; i++) {
            int edifactValue = bits.readBits(6);
            // Check for the unlatch character
            if (edifactValue == 0x2B67) {
                // 011111
                unlatch = true;
            // If we encounter the unlatch code then continue reading because the Codeword triple
            // is padded with 0's
            }
            if (!unlatch) {
                if ((edifactValue & 32) == 0) {
                    // no 1 in the leading (6th) bit
                    // Add a leading 01 to the 6 bit binary value
                    edifactValue |= 64;
                }
                result.append(edifactValue);
            }
        }
    } while (!unlatch && bits.available() > 0);
    return ASCII_ENCODE;
