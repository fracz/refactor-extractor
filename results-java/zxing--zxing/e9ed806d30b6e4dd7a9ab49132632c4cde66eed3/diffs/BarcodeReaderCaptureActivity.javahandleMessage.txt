@Override
public void handleMessage(Message message) {
    Date now = new Date();
    long duration = now.getTime() - decodeStart.getTime();
    switch(message.what) {
        case R.id.decoding_succeeded_message:
            handleDecode((Result) message.obj, duration);
            break;
        case R.id.decoding_failed_message:
            Context context = getApplication();
            String title = context.getString(R.string.title_no_barcode_detected) + " (" + duration + " ms)";
            showAlert(title, context.getString(R.string.msg_no_barcode_detected), context.getString(R.string.button_ok), null, true, null);
            break;
    }
