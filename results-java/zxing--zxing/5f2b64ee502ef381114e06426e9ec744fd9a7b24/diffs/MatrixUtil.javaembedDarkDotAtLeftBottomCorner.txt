private static void embedDarkDotAtLeftBottomCorner(ByteMatrix matrix) throws WriterException {
    if (matrix.get(matrix.height() - 8, 8) == 0) {
        throw new WriterException();
    }
    matrix.set(matrix.height() - 8, 8, 1);
