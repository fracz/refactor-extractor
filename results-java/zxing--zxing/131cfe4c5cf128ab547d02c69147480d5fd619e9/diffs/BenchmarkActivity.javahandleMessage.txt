public void handleMessage(Message message) {
    switch(message.what) {
        case R.id.benchmark_done:
            handleBenchmarkDone(message);
            mBenchmarkThread = null;
            mRunBenchmarkButton.setEnabled(true);
            mTextView.setText(R.string.benchmark_help);
            break;
        default:
            break;
    }
