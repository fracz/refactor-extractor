private void initConstraint() {
    String constraint = "";
    if (isPrimaryKey()) {
        constraint += "PRIMARY KEY";
        if (pkAsc) {
            constraint += " ASC";
        }
        if (pkDesc) {
            constraint += " DESC";
        }
    }
    if (notNull) {
        constraint += " NOT NULL";
    }
    if (unique) {
        constraint += " UNIQUE";
    }
    constraint = constraint.trim();
    if (constraint.length() > 0) {
        setConstraints(constraint);
    }
