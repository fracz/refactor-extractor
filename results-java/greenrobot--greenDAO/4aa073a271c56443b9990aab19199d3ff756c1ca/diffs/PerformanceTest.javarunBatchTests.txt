protected void runBatchTests(List<T> list) {
    long start;
    long time;
    start = System.currentTimeMillis();

    time = System.currentTimeMillis() - start;
    Log.d("DAO", "Inserted (batch) " + list.size() + " entities in " + time + "ms");
    list = null;
    System.gc();
    start = System.currentTimeMillis();

    System.gc();
    start = System.currentTimeMillis();
