@Override
public boolean equals(Object obj) {
    if (this == obj)
        return true;
    if (obj == null)
        return false;
    if (getClass() != obj.getClass())
        return false;
    CalDavEvent other = (CalDavEvent) obj;
    if (calendarId == null) {
        if (other.calendarId != null)
            return false;
    } else if (!calendarId.equals(other.calendarId))
        return false;
    if (id == null) {
        if (other.id != null)
            return false;
    } else if (!id.equals(other.id))
        return false;
    return true;
