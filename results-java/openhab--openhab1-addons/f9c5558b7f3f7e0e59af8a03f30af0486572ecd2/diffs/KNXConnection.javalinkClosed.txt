    if (!e.isUserRequest()) {
        logger.warn("KNX link has been lost (reason: {} on object {}) - reconnecting...", e.getReason(), e.getSource().toString());
        connect();
    }
    if (!link.isOpen()) {
        logger.error("KNX link has been lost!");
