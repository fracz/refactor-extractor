private void writeXmlDeclaration(Writer writer, String encoding) {
    try {
        writer.write("<?xml version=\"1.0\"");
        if (encoding != null) {
            writer.write(" encoding=\"");
            writer.write(encoding);
            writer.write("\"");
        }
        writer.write("?>");
        writer.write(TextUtil.LINE_SEPARATOR);
    } catch (IOException e) {
        throw UncheckedException.asUncheckedException(e);
    }
