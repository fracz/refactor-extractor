private void addProjects(ProjectInternal parent, ProjectDescriptor parentProjectDescriptor, StartParameter startParameter, Map<String, String> externalProjectProperties) {
    for (ProjectDescriptor childProjectDescriptor : parentProjectDescriptor.getChildren()) {
        ProjectInternal childProject = (ProjectInternal) parent.addChildProject(childProjectDescriptor.getName(), childProjectDescriptor.getDir());
        addPropertiesToProject(startParameter.getGradleUserHomeDir(), externalProjectProperties, childProject);
        addProjects(childProject, childProjectDescriptor, startParameter, externalProjectProperties);
    }
