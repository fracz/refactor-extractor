private ModuleDescriptor findProject(DependencyDescriptor descriptor) {
    String projectPathValue = descriptor.getAttribute(DependencyDescriptorFactory.PROJECT_PATH_KEY);
    if (projectPathValue == null) {
        return null;
    }
    Project project = gradle.getRootProject().project(projectPathValue);
    DefaultModule module = new DefaultModule(project.getGroup().toString(), project.getName(), project.getVersion().toString(), project.getStatus().toString());
    return moduleDescriptorConverter.convertForPublish(project.getConfigurations().getAll(), false, module, IvyContext.getContext().getIvy().getSettings());
