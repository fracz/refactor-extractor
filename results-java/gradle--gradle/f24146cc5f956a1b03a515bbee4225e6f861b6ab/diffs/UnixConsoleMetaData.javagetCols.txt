public int getCols() {
    final String columns = System.getenv("COLUMNS");
    if (columns != null) {
        try {
            return Integer.parseInt(columns);
        } catch (NumberFormatException ex) {
            LOGGER.debug("Cannot parse COLUMNS environment variable to get console width. Value: '{}'", columns);
        }
    }
    return 0;
