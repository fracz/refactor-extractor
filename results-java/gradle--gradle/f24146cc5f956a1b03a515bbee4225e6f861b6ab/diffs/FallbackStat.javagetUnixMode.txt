public int getUnixMode(File f) throws IOException {
    if (f.isDirectory()) {
        return FileSystem.DEFAULT_DIR_MODE;
    } else if (f.exists()) {
        return FileSystem.DEFAULT_FILE_MODE;
    } else {
        throw new FileNotFoundException(String.format("File '%s' not found.", f));
    }
