public ExecHandleState waitForFinish() {
    if (!stateEquals(ExecHandleState.STARTED))
        throw new IllegalStateException("not in started state!");
    execHandleRunLock.lock();
    try {
        shutdownThreadPool();
    } finally {
        execHandleRunLock.unlock();
    }
    return getState();
