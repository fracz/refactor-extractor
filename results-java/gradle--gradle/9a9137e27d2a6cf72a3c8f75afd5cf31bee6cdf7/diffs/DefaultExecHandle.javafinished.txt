    if (exitCode != 0) {
        setEndStateInfo(ExecHandleState.SUCCEEDED, exitCode, new RuntimeException("exitCode != 0!"));
        new Thread(new FailedNotifier(this, listeners)).start();

        new Thread(new FinishedNotifier(this, listeners)).start();
