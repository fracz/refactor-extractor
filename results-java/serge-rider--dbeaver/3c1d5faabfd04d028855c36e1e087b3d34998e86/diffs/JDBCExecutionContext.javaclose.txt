@Override
public void close() {
    // while UI may invoke callbacks to operate with connection
    synchronized (this) {
        // }
        if (connection != null) {
            try {
                connection.close();
            } catch (Throwable ex) {
                log.error(ex);
            }
            QMUtils.getDefaultHandler().handleContextClose(this);
            connection = null;
        }
    }
    // Remove self from context list
    this.dataSource.allContexts.remove(this);
