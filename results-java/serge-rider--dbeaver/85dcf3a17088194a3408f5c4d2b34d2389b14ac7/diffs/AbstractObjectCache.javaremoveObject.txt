@Override
public void removeObject(@NotNull OBJECT object) {
    synchronized (this) {
        if (this.objectList != null) {
            detectCaseSensitivity(object);
            this.objectList.remove(object);
            if (this.objectMap != null) {
                this.objectMap.remove(getObjectName(object));
            }
        }
    }
