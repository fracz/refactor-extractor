public static String getFullTypeName(@NotNull DBSTypedObject typedObject) {
    String typeName = typedObject.getTypeName();
    switch(typedObject.getDataKind()) {
        case STRING:
        case LOB:
            return typeName + "(" + typedObject.getMaxLength() + ")";
        default:
            return typeName;
    }
