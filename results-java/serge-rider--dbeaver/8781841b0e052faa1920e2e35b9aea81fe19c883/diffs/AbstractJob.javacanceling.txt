@Override
protected void canceling() {
    // Run canceling job
    if (!blockCanceled) {
        Job cancelJob = new // $NON-NLS-1$
        Job(// $NON-NLS-1$
        "Cancel block") {

            @Override
            protected IStatus run(IProgressMonitor monitor) {
                if (!finished && !blockCanceled) {
                    DBRBlockingObject block = progressMonitor.getActiveBlock();
                    if (block != null) {
                        try {
                            block.cancelBlock();
                        } catch (DBException e) {
                            // $NON-NLS-1$
                            log.error("Can't interrupt operation " + block, e);
                        }
                        blockCanceled = true;
                    }
                }
                return Status.OK_STATUS;
            }
        };
        // Cancel it in three seconds
        cancelJob.schedule(cancelTimeout);
    }
