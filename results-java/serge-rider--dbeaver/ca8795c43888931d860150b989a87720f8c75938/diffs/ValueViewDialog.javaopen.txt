public final int open() {
    try {
        int result = super.open();
        if (result == IDialogConstants.OK_ID) {
            getValueController().updateValue(editedValue);
        }
        return result;
    } finally {
        dialogCount--;

            this.valueController.unregisterEditor(this);
            this.valueController = null;
        }
    }
