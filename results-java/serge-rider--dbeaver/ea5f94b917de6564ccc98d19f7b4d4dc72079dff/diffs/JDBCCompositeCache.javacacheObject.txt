@Override
public void cacheObject(OBJECT object) {
    super.cacheObject(object);
    synchronized (objectCache) {
        List<OBJECT> objects = objectCache.get(getParent(object));
        if (!CommonUtils.isEmpty(objects)) {
            objects.add(object);
        }
    }
