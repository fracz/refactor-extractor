@Override
public JDBCResultSet executeQuery() throws SQLException {
    this.beforeExecute();
    try {
        return new JDBCResultSetImpl(this, getOriginal().executeQuery());
    } catch (Throwable e) {
        throw super.handleExecuteError(e);
    } finally {
        super.afterExecute();
    }
