@Override
public void close() {
    DBCExecutionPurpose purpose = getPurpose();
    if (purpose == DBCExecutionPurpose.USER || purpose == DBCExecutionPurpose.USER_SCRIPT) {
        // Check for warnings
        try {
            final Connection connection = getConnection();
            if (connection != null) {
                JDBCUtils.reportWarnings(this, connection.getWarnings());
                connection.clearWarnings();
            }
        } catch (Throwable e) {
            log.debug("Could not check for connection warnings", e);
        }
    }
    if (isolatedConnection != null) {
        try {
            isolatedConnection.getConnection().close();
        } catch (SQLException e) {
            log.error("Error closing isolated connection", e);
        }
        isolatedConnection = null;
    } else {
    // do nothing
    // closing of context doesn't close real connection
    }
    super.close();
