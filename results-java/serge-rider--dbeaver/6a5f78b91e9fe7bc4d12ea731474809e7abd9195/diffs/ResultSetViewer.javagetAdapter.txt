@Override
public Object getAdapter(Class adapter) {
    if (adapter == IPropertySheetPage.class) {
        PropertyPageStandard page = new PropertyPageStandard();
        page.setPropertySourceProvider(new IPropertySourceProvider() {

            @Override
            public IPropertySource getPropertySource(Object object) {
                if (object instanceof GridPos) {
                    final GridPos cell = translateGridPos((GridPos) object);
                    boolean noCellSelected = (cell.row < 0 || model.getRowCount() <= cell.row || cell.col < 0 || cell.col >= model.getVisualColumnCount());
                    if (!noCellSelected) {
                        final ResultSetValueController valueController = new ResultSetValueController(cell, DBDValueController.EditType.NONE, null);
                        PropertyCollector props = new PropertyCollector(valueController.getAttribute(), false);
                        props.collectProperties();
                        valueController.getValueHandler().fillProperties(props, valueController);
                        return props;
                    }
                }
                return null;
            }
        });
        return page;
    }
    return null;
