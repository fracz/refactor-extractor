        return resourceImage;

    if (resource == null) {

    switch(resource.getType()) {
        case IResource.FOLDER:
            return resource.isLinked() ? DBIcon.TREE_FOLDER_LINK.getImage() : DBIcon.TREE_FOLDER.getImage();
