public Image getNodeIconDefault() {
    Image image = getNodeIcon();
    if (image == null) {
        if (this.allowsChildren()) {
            return DBIcon.TREE_FOLDER.getImage();
        } else {
            return DBIcon.TREE_PAGE.getImage();
        }
    } else {
        return image;
    }
