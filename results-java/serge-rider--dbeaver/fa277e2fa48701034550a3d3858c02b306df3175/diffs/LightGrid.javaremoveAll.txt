/**
 * Removes all of the items from the receiver.
 */
public void removeAll() {
    checkWidget();
    deselectAll();
    currentVisibleItems = 0;
    focusItem = -1;
    focusColumn = null;
    topIndex = -1;
    bottomIndex = -1;
    List<GridColumn> columnsCopy = new ArrayList<GridColumn>(columns);
    for (GridColumn column : columnsCopy) {
        column.dispose();
    }
    redraw();
