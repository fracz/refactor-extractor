private void setBinaryContent(String stringValue) {
    byte[] bytes;
    try {
        bytes = stringValue.getBytes(ContentUtils.getDefaultBinaryFileEncoding(getValueController().getDataSource()));
    } catch (UnsupportedEncodingException e) {
        log.error(e);
        bytes = stringValue.getBytes();
    }
    hexEditControl.setContent(bytes);
