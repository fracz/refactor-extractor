        if (table instanceof DBPHiddenObject && ((DBPHiddenObject) table).isHidden() || (table instanceof DBSTable && ((DBSTable) table).isView())) {
            // Skip hidden tables
            continue;

        addDiagramEntity(monitor, table);

    }
