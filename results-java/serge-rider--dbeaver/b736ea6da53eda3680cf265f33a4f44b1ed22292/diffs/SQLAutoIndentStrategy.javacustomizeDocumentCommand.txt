@Override
public void customizeDocumentCommand(IDocument d, DocumentCommand c) {
    if (!c.doit) {
        return;
    }
    clearCachedValues();
    if (c.length == 0 && c.text != null && isLineDelimiter(d, c.text)) {
        smartIndentAfterNewLine(d, c);
    }
