@Override
public Object getValueAt(int rowIndex, int columnIndex) {
    Device obj = devices.get(rowIndex);
    Object value = null;
    switch(columnIndex) {
        case COLUMN_DEVICE:
            value = obj.toString();
            break;
        case COLUMN_STATE:
            value = obj.isConnected() ? "connected" : "disconnected";
            break;
        case COLUMN_ACTION:
            break;
        default:
            return null;
    }
    return value;
