@Override
public String getColumnName(int column) {
    String value;
    switch(column) {
        case COLUMN_DEVICE:
            value = "Device";
            break;
        case COLUMN_STATE:
            value = "State";
            break;
        case COLUMN_ACTION:
            value = "Action";
            break;
        default:
            return null;
    }
    return value;
