@Override
public void onComplete() {
    if (done) {
        return;
    }
    done = true;
    try {
        parent.onComplete.run();
    } catch (Throwable ex) {
        Exceptions.throwIfFatal(ex);
        actual.onError(ex);
        return;
    }
    actual.onComplete();

        parent.onAfterTerminated.run();
    } catch (Throwable ex) {
        Exceptions.throwIfFatal(ex);
        RxJavaPlugins.onError(ex);
    }
