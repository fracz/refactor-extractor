    return s -> {
        Publisher<? extends U> other;
        try {
            other = mapper.apply(value);
        } catch (Throwable e) {
            s.onSubscribe(EmptySubscription.INSTANCE);
            s.onError(e);
            return;
        }
        other.subscribe(s);
    };
