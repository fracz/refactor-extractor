public final <U> Observable<U> ofType(final Class<U> clazz) {
    Objects.requireNonNull(clazz, "clazz is null");
    return filter(new Predicate<T>() {

        @Override
        public boolean test(T c) {
            return clazz.isInstance(c);
        }
    }).cast(clazz);
