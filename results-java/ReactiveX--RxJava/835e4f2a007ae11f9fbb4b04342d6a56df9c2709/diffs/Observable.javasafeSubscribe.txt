@SchedulerSupport(SchedulerKind.NONE)
public final void safeSubscribe(Subscriber<? super T> s) {
    Objects.requireNonNull(s, "s is null");
    if (s instanceof SafeSubscriber) {
        subscribeActual(s);
    } else {
        subscribeActual(new SafeSubscriber<T>(s));
    }
