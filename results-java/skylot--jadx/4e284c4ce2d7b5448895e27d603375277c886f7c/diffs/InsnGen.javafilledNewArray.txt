private void filledNewArray(InsnNode insn, CodeWriter code) throws CodegenException {
    int c = insn.getArgsCount();
    code.add("new ").add(useType(insn.getResult().getType()));
    code.add("{");
    for (int i = 0; i < c; i++) {
        code.add(arg(insn, i));
        if (i + 1 < c)
            code.add(", ");
    }
    code.add("}");
