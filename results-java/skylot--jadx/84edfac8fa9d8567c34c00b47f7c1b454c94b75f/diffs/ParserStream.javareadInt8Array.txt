public byte[] readInt8Array(int count) throws IOException {
    if (count == 0) {
        return new byte[0];
    }
    readPos += count;
    byte[] arr = new byte[count];
    int pos = input.read(arr, 0, count);
    while (pos < count) {
        int read = input.read(arr, pos, count - pos);
        if (read == -1) {
            throw new IOException("No data, can't read " + count + " bytes");
        }
        pos += read;
    }
    return arr;
