public static void shrinkMethod(MethodNode mth) {
    if (mth.isNoCode() || mth.getAttributes().contains(AttributeFlag.DONT_SHRINK)) {
        return;
    }
    for (BlockNode block : mth.getBasicBlocks()) {
        shrinkBlock(mth, block);
    }
