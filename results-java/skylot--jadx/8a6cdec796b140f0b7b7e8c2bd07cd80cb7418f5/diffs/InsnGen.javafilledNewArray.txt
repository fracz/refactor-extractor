private void filledNewArray(InsnNode insn, CodeWriter code) throws CodegenException {
    int c = insn.getArgsCount();
    code.add("new ");
    useType(code, insn.getResult().getType());
    code.add('{');
    for (int i = 0; i < c; i++) {
        addArg(code, insn.getArg(i), false);
        if (i + 1 < c) {
            code.add(", ");
        }
    }
    code.add('}');
