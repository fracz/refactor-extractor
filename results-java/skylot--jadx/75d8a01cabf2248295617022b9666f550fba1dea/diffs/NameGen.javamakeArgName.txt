private String makeArgName(RegisterArg arg) {
    String name = arg.getName();
    if (fallback) {
        String base = "r" + arg.getRegNum();
        if (name != null) {
            return base + "_" + name;
        }
        return base;
    }
    String varName;
    if (name != null) {
        if ("this".equals(name)) {
            return name;
        }
        varName = name;
    } else {
        varName = makeNameForType(arg.getType());
    }
    if (NameMapper.isReserved(varName)) {
        return varName + "R";
    }
    return varName;
