public static ClassInfo fromDex(DexNode dex, int clsIndex) {
    if (clsIndex == DexNode.NO_INDEX) {
        return null;
    }
    ArgType type = dex.getType(clsIndex);
    if (type.isArray()) {
        type = ArgType.OBJECT;
    }
    return fromType(dex, type);
