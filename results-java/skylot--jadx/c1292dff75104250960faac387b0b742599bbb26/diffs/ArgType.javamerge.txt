@Nullable
public static ArgType merge(ArgType a, ArgType b) {
    if (a == null || b == null) {
        return null;
    }
    if (a.equals(b)) {
        return a;
    }
    ArgType res = mergeInternal(a, b);
    if (res == null) {
        // swap
        res = mergeInternal(b, a);
    }
    return res;
