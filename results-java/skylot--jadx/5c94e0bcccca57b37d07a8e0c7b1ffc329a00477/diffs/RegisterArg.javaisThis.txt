@Override
public boolean isThis() {
    if (isRegister()) {
        String name = getTypedVar().getName();
        if ("this".equals(name)) {
            return true;
        }
        // maybe it was moved from 'this' register
        InsnNode ai = getAssignInsn();
        if (ai != null && ai.getType() == InsnType.MOVE) {
            InsnArg arg = ai.getArg(0);
            if (arg != this && arg.isThis()) {
                return true;
            }
        }
    }
    return false;
