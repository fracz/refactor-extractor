/**
 * Returns the matrix inverse. For pseudo inverse, use QRDecomposition.
 */
public double[][] inverse() {
    int m = LU.nrows();
    int n = LU.ncols();
    if (m != n)
        throw new IllegalArgumentException(String.format("Matrix is not square: %d x %d", m, n));
    double[][] I = Math.eye(n);
    solve(I);
    return I;
