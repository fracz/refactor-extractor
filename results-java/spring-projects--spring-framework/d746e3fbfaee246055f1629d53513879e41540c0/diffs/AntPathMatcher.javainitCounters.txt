            this.uriVars++;
            pos++;
        } else if (this.pattern.charAt(pos) == '*') {
            if (pos + 1 < this.pattern.length() && this.pattern.charAt(pos + 1) == '*') {
                this.doubleWildcards++;
                pos += 2;

                this.singleWildcards++;
                pos++;
            }
        } else {
            pos++;
        }
    }
