    pathMatcher.setCachePatterns(true);
    match();
    assertTrue(pathMatcher.stringMatcherCache.size() > 20);
    for (int i = 0; i < 65536; i++) {

    assertTrue(pathMatcher.stringMatcherCache.size() > 65536);
