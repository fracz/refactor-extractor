@Test
public void headersCondition() {
    PathMatcher pathMatcher = new AntPathMatcher();
    MockHttpServletRequest request = new MockHttpServletRequest("GET", "/foo");
    request.addHeader("foo", "bar");
    String lookupPath = new UrlPathHelper().getLookupPathForRequest(request);
    RequestMappingInfo key = new RequestMappingInfo(singleton("/foo"), null, null, RequestConditionFactory.parseHeaders("foo=bar"), null, null);
    RequestMappingInfo match = key.getMatchingRequestMapping(lookupPath, request, pathMatcher);
    assertNotNull(match);
    key = new RequestMappingInfo(singleton("/foo"), null, null, RequestConditionFactory.parseHeaders("foo!=bar"), null, null);
    match = key.getMatchingRequestMapping(lookupPath, request, pathMatcher);
    assertNull(match);
