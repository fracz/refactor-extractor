public void encode() {
    byte[] fooBytes = "foo".getBytes(StandardCharsets.UTF_8);
    byte[] barBytes = "bar".getBytes(StandardCharsets.UTF_8);
    Flux<byte[]> source = Flux.just(fooBytes, barBytes);
    Flux<DataBuffer> output = this.encoder.encode(source, this.bufferFactory, ResolvableType.forClassWithGenerics(Publisher.class, ByteBuffer.class), null, Collections.emptyMap());
    TestSubscriber.subscribe(output).assertValuesWith(b -> {
        byte[] buf = new byte[3];
        b.read(buf);
        assertArrayEquals(fooBytes, buf);
    }, b -> {
        byte[] buf = new byte[3];
        b.read(buf);
        assertArrayEquals(barBytes, buf);
    });
