public void onTimeout(AsyncEvent event) throws IOException {
    if (this.timeoutHandler == null) {
        getResponse().sendError(HttpStatus.SERVICE_UNAVAILABLE.value());
    } else {
        this.timeoutHandler.run();
    }
    completeInternal();
