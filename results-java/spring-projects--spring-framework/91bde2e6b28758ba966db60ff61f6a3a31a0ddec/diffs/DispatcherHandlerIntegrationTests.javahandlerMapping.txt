@Bean
public HandlerMapping handlerMapping(RoutingFunction<?> routingFunction, ApplicationContext applicationContext) {
    return Router.toHandlerMapping(routingFunction, new Router.Configuration() {

        @Override
        public Supplier<Stream<HttpMessageReader<?>>> messageReaders() {
            return () -> getMessageReaders().stream();
        }

        @Override
        public Supplier<Stream<HttpMessageWriter<?>>> messageWriters() {
            return () -> getMessageWriters().stream();
        }

        @Override
        public Supplier<Stream<ViewResolver>> viewResolvers() {
            return () -> Collections.<ViewResolver>emptySet().stream();
        }
    });
