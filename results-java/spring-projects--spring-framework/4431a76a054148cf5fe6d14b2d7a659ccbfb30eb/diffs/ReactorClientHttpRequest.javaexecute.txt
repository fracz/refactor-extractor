@Override
public Mono<ClientHttpResponse> execute() {
    return this.httpClient.request(new io.netty.handler.codec.http.HttpMethod(httpMethod.toString()), uri.toString(), channel -> {
        // see https://github.com/reactor/reactor-io/pull/8
        if (body == null) {
            channel.removeTransferEncodingChunked();
        }
        return applyBeforeCommit().after(() -> {
            getHeaders().entrySet().stream().forEach(e -> channel.headers().set(e.getKey(), e.getValue()));
            getCookies().values().stream().flatMap(Collection::stream).forEach(cookie -> channel.addCookie(cookie.getName(), new NettyCookie(cookie)));
            return Mono.empty();
        }).after(() -> {
            if (body != null) {
                return channel.send(body);
            } else {
                return channel.sendHeaders();
            }
        });
    }).map(httpChannel -> new ReactorClientHttpResponse(httpChannel, allocator));
