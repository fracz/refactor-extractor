public List<byte[]> getParts() throws Exception {
    List<byte[]> parts = new ArrayList<byte[]>();
    for (MimePart part : mimeParts) {
        ByteArrayOutputStream bos = new ByteArrayOutputStream();
        try {
            part.writeTo(bos);
            parts.add(bos.toByteArray());
        } finally {
            bos.close();
        }
    }
    return parts;
