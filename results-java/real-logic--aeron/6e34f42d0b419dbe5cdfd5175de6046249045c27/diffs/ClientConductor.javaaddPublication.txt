    verifyActive();
    Publication publication = activePublications.get(channel, streamId);
    if (null == publication) {
        awaitResponse(driverProxy.addPublication(channel, streamId), channel);
        publication = activePublications.get(channel, streamId);
    }
    publication.incRef();
    return publication;
