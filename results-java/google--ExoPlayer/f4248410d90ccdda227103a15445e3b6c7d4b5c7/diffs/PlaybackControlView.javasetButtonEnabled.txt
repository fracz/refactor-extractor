private void setButtonEnabled(View button, boolean enabled) {
    button.setEnabled(enabled);
    if (Util.SDK_INT >= 11) {
        button.setAlpha(enabled ? 1f : 0.3f);
        button.setVisibility(VISIBLE);
    } else {
        button.setVisibility(enabled ? VISIBLE : INVISIBLE);
    }
