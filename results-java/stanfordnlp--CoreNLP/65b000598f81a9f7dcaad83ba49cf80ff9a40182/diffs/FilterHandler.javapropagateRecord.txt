@Override
public boolean propagateRecord(Record record) {
    for (LogFilter filter : filters) {
        boolean match = filter.matches(record);
        if (match && disjunctiveMode) {
            return true;
        }
        if (!match && !disjunctiveMode) {
            return false;
        }
    }
    return !disjunctiveMode;
