private void loadCountriesLists(String file) {
    try {
        BufferedReader reader = IOUtils.readerFromString(file);
        while (reader.ready()) {
            String line = reader.readLine();
            countries.add(line.split("\t")[1].toLowerCase());
        }
        reader.close();
    } catch (IOException e) {
        throw new RuntimeIOException(e);
    }
