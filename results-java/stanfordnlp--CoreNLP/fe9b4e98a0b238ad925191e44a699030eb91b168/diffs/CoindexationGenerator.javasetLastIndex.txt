public void setLastIndex(Tree t) {
    lastIndex = 0;
    for (Tree node : t) {
        String value = node.label().value();
        if (value != null) {
            Matcher m = coindexationPattern.matcher(value);
            if (m.find()) {
                int thisIndex = Integer.parseInt(m.group(1));
                lastIndex = Math.max(thisIndex, lastIndex);
            }
        }
    }
