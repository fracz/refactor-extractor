@SuppressWarnings({ "UnnecessaryBoxing" })
public static Character getCharacter(char c) {
    Character cC = CharacterHolder.canonicalCharacters[c];
    if (cC == null) {
        cC = Character.valueOf(c);
        CharacterHolder.canonicalCharacters[c] = cC;
    }
    return cC;
