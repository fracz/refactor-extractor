private void countRoot(Tree tree) {
    Integer gold = RNNCoreAnnotations.getGoldClass(tree);
    Integer predicted = RNNCoreAnnotations.getPredictedClass(tree);
    if (gold >= 0) {
        if (gold.equals(predicted)) {
            rootLabelsCorrect++;
        } else {
            rootLabelsIncorrect++;
        }
        rootLabelConfusion[gold][predicted]++;
    }
