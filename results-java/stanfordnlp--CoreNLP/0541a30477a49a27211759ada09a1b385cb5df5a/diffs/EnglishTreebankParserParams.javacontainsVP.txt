private boolean containsVP(Tree t) {
    String cat = tlp.basicCategory(t.label().value());
    if (cat.equals("VP")) {
        return true;
    } else {
        for (Tree kid : t.children()) {
            if (containsVP(kid)) {
                return true;
            }
        }
        return false;
    }
