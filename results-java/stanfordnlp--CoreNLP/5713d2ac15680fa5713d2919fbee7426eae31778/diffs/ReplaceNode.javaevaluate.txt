    if (oldNode == t) {
        if (children.length > 2) {

    }
    Tree parent = oldNode.parent(t);
    int i = parent.objectIndexOf(oldNode);
    parent.removeChild(i);
    for (int j = 1; j < children.length; ++j) {

        parent.insertDtr(newNode.deepCopy(), i + j - 1);
    }
    return t;
