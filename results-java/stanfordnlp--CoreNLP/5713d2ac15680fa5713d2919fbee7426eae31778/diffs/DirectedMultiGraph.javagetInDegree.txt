@Override
public int getInDegree(V vertex) {
    if (!containsVertex(vertex)) {
        throw new IllegalArgumentException("Unknown vertex " + vertex);
    }
    int result = 0;
    Map<V, List<E>> incoming = incomingEdges.get(vertex);
    for (List<E> edges : incoming.values()) {
        result += edges.size();
    }
    return result;
