 * @param v
 */
@Override
public boolean addVertex(V v) {
    if (outgoingEdges.containsKey(v))
        return false;
    outgoingEdges.put(v, Generics.<V, List<E>>newHashMap());
    incomingEdges.put(v, Generics.<V, List<E>>newHashMap());
    return true;
