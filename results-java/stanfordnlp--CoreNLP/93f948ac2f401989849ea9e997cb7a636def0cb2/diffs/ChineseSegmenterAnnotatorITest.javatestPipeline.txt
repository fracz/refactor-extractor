@Test
public void testPipeline() {
    testOne("你马上回来北京吗？", new String[] { "你", "马上", "回来", "北京", "吗", "？" }, new int[] { 0, 1, 3, 5, 7, 8 }, new int[] { 1, 3, 5, 7, 8, 9 });

    testOne("<post id=\"something\" anything>这是一个测试</post>", new String[] { "<post id=\"something\" anything>", "这", "是", "一", "个", "测试", "</post>" }, new int[] { 0, 30, 31, 32, 33, 34, 36 }, new int[] { 30, 31, 32, 33, 34, 36, 43 });
    // KBP corpus examples, containing newlines and spaces
    // The segmenter should be able to keep spaces within the xml tags, but skip those out of xml tags

    testOne("这里有一个图片。<img src=\"http://bbsfile.ifeng.com/bbsfile/images/smilies/default/lol.gif\"/>  希望你们都能看看。", new String[] { "这里", "有", "一", "个", "图片", "。", "<img src=\"http://bbsfile.ifeng.com/bbsfile/images/smilies/default/lol.gif\"/>", "希望", "你们", "都", "能", "看看", "。" }, new int[] { 0, 2, 3, 4, 5, 7, 8, 86, 88, 90, 91, 92, 94 }, new int[] { 2, 3, 4, 5, 7, 8, 84, 88, 90, 91, 92, 94, 95 });

    testOne("买点咖啡提点精神\uD83D\uDE0A", new String[] { "买", "点", "咖啡", "提", "点", "精神", "\uD83D\uDE0A" }, new int[] { 0, 1, 2, 4, 5, 6, 8 }, new int[] { 1, 2, 4, 5, 6, 8, 10 });
