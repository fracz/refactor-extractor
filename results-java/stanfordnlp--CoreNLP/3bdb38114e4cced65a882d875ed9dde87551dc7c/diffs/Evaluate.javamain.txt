    String modelPath = null;
    String treePath = null;
    boolean filterUnknown = false;
    List<String> remainingArgs = Generics.newArrayList();
    for (int argIndex = 0; argIndex < args.length; ) {

            treePath = args[argIndex + 1];
            argIndex += 2;

            argIndex++;
