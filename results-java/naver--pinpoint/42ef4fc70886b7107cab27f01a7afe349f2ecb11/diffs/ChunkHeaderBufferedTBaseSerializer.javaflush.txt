public void flush() {
    synchronized (out) {
        if (flushHandler != null && out.size() > Header.HEADER_SIZE) {
            flushHandler.handle(out.toByteArray(), 0, out.size());
        }
        out.reset();
        writeChunkHeader = false;
    }
