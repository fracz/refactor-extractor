private void closeOrFlush() {
    if (this.closer == null) {
        return;
    }
    if (this.closer.await()) {
        // flush.
        this.trace.flush();
        if (isDebug) {
            logger.debug("Flush trace={}, closer={}", this, this.closer);
        }
    } else {
        // close.
        this.trace.close();
        if (isDebug) {
            logger.debug("Close trace={}. closer={}", this, this.closer);
        }
    }
    this.closer = null;
