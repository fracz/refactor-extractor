public void close() {
    final JSONLexer lexer = getLexer();
    try {
        if (isEnabled(Feature.AutoCloseSource)) {
            if (lexer.token() != JSONToken.EOF) {
                throw new JSONException("not close json text, token : " + JSONToken.name(lexer.token()));
            }
        }
    } finally {
        lexer.close();
    }
