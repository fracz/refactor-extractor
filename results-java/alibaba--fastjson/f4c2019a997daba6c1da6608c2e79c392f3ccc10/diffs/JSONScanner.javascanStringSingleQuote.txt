                }
                System.arraycopy(buf, np + 1, sbuf, 0, sp);
            }
            ch = buf[++bp];

                case '"':
                    putChar('"');
                    break;

                    putChar('\\');
                    break;
                case '/':
                    putChar('/');
                    break;
                case '\'':
