    List<List<Span>> result = new LinkedList<>();
    try {
        reader.beginArray();
        while (reader.hasNext()) {
            reader.beginArray();
            // cause we don't know how long it will be
            List<Span> trace = new LinkedList<>();
            while (reader.hasNext()) {
                trace.add(SPAN_ADAPTER.fromJson(reader));
            }
            reader.endArray();
            result.add(trace);
        }
        reader.endArray();
        return result;
