@VisibleForTesting
void writeDependencyLinks(List<DependencyLink> links, long timestampMillis) {
    long midnight = Util.midnightUTC(timestampMillis);
    BulkRequestBuilder request = storage.client().prepareBulk();
    for (DependencyLink link : links) {
        request.add(storage.client().prepareIndex(storage.indexNameFormatter.indexNameForTimestamp(midnight), ElasticsearchConstants.DEPENDENCY_LINK).setId(// Unique constraint
        link.parent + "|" + link.child).setSource("parent", link.parent, "child", link.child, "callCount", link.callCount));
    }
    request.execute().actionGet();
    storage.client().admin().indices().flush(new FlushRequest()).actionGet();
