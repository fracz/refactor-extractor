@Bean
SpanStore spanStore() {
    SpanStore result;
    if (this.datasource != null && this.server.getStore().getType() == Type.mysql) {
        result = new JDBCSpanStore(this.datasource, new Settings().withRenderSchema(false), this.listener);
    } else {
        result = new InMemorySpanStore();
    }
    return result;
