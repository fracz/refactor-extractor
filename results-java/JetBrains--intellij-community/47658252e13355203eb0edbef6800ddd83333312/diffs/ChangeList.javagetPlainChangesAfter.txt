private List<Change> getPlainChangesAfter(Change target) {
    List<Change> result = new ArrayList<Change>();
    for (Change changeSet : Reversed.list(myChanges)) {
        if (changeSet == target)
            return result;
        for (Change c : Reversed.list(changeSet.getChanges())) {
            if (c == target)
                return result;
            result.add(c);
        }
    }
    return result;
