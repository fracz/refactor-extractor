@NotNull
public static File getTempDirectory() {
    File dir = new File(FileUtil.getTempDirectory());
    if (SystemInfo.isWindows && dir.getPath().contains("~")) {
        try {
            dir = dir.getCanonicalFile();
        } catch (IOException ignored) {
        }
    }
    return dir;
