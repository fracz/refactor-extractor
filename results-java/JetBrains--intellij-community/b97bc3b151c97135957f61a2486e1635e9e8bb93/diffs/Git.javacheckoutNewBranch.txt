/**
 * {@code git checkout -b &lt;branchName&gt;}
 */
public static GitCommandResult checkoutNewBranch(@NotNull GitRepository repository, @NotNull String branchName, @Nullable GitLineHandlerListener listener) {
    final GitLineHandler h = new GitLineHandler(repository.getProject(), repository.getRoot(), GitCommand.CHECKOUT);
    h.setSilent(false);
    h.addParameters("-b");
    h.addParameters(branchName);
    if (listener != null) {
        h.addLineListener(listener);
    }
    return run(h);
