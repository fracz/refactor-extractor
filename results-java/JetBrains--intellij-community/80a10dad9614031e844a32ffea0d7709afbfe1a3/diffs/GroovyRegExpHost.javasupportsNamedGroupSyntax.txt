@Override
public boolean supportsNamedGroupSyntax(RegExpGroup group) {
    if (group.isNamedGroup()) {
        final String version = getGroovyVersion(group);
        return version != null && version.compareTo(GroovyConfigUtils.GROOVY2_0) >= 0;
    }
    return false;
