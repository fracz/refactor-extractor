@Override
public Boolean isConvertible(@NotNull PsiType lType, @NotNull PsiType rType, @NotNull GroovyPsiElement context) {
    if (isMethodCallConversion(context))
        return null;
    if (!GroovyConfigUtils.getInstance().isVersionAtLeast(context, GroovyConfigUtils.GROOVY1_8))
        return null;
    if (!(InheritanceUtil.isInheritor(rType, JAVA_LANG_CHAR_SEQUENCE) || InheritanceUtil.isInheritor(rType, GROOVY_LANG_GSTRING))) {
        return null;
    }
    if (lType == PsiType.BOOLEAN || TypesUtil.resolvesTo(lType, JAVA_LANG_BOOLEAN))
        return true;
    if (TypesUtil.resolvesTo(lType, JAVA_LANG_CLASS))
        return true;
    if (isEnum(lType))
        return true;
    return null;
