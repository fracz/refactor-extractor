    int modHashCode = hashCode % NUM_ENTRIES;
    final char[] kwd = myTable[modHashCode];
    if (kwd == null)
        return false;
    if (bufferArray != null) {
        for (int j = 0; j < kwd.length; j++) {
            if (bufferArray[j + offset] != kwd[j])
                return false;
        }
    } else {

        }
    }
