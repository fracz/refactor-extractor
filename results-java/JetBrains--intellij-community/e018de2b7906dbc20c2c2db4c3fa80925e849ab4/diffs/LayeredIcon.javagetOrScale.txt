    if (myScale == 1f) {
        return myIcons[i];
    }
    if (myScaledIcons == null) {
        myScaledIcons = new Icon[myIcons.length];
    }
    Icon icon = myScaledIcons[i];
    if (icon == null && myIcons[i] != null) {
        icon = myIcons[i];
        if (icon instanceof ScalableIcon) {
            icon = myScaledIcons[i] = ((ScalableIcon) icon).scale(myScale);
