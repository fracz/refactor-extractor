@Override
public int getElementProblemCount(RefJavaElement refElement) {
    if (refElement instanceof RefParameter)
        return 0;
    if (refElement.isEntry() || !((RefElementImpl) refElement).isSuspicious() || refElement.isSyntheticJSP())
        return 0;
    if (!(refElement instanceof RefMethod || refElement instanceof RefClass || refElement instanceof RefField))
        return 0;
    if (!myTool.getContext().isToCheckMember(refElement, myTool))
        return 0;
    if (refElement instanceof RefField) {
        RefField refField = (RefField) refElement;
        if (refField.isUsedForReading() && !refField.isUsedForWriting())
            return 1;
        if (refField.isUsedForWriting() && !refField.isUsedForReading())
            return 1;
    }
    if (refElement instanceof RefClass && ((RefClass) refElement).isAnonymous())
        return 0;
    return -1;
