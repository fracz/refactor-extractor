@Override
public void visitMethod(PsiMethod method) {
    try {
        // noinspection HardCodedStringLiteral
        String methodName = method.isConstructor() ? "<init>" : method.getName();
        PsiClass containingClass = method.getContainingClass();
        if (containingClass != null && containingClass.equals(myCompiledClass) && methodName.equals(myMethodName) && JVMNameUtil.getJVMSignature(method).getName(myDebugProcess).equals(myMethodSignature)) {
            myCompiledMethod = method;
        }
    } catch (EvaluateException e) {
        LOG.debug(e);
    }
