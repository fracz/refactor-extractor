private static int getInterestLevel(final ThreadState state) {
    if (state.isEmptyStackTrace())
        return -10;
    if (isKnownJdkThread(state))
        return -5;
    if (state.isSleeping()) {
        return -2;
    }
    if (state.isSocketOperation()) {
        return -1;
    }
    return state.getStackTrace().split("\n").length;
