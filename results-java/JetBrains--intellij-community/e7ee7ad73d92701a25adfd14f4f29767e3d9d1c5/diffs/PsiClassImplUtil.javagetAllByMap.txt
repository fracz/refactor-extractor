    final List<Pair<T, PsiSubstitutor>> pairs = allMap.get(ALL);
    if (pairs == null) {
        LOG.error("pairs should be already computed. Wrong allMap: " + allMap);

    final List<T> ret = new ArrayList<T>(pairs.size());

    return ret.toArray(emptyArrayByType(type));
