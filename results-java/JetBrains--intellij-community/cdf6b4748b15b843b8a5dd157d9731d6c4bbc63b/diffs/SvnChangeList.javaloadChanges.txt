private void loadChanges() {
    myChanges = new ArrayList<Change>();
    for (Object o : myLogEntry.getChangedPaths().values()) {
        SVNLogEntryPath entry = (SVNLogEntryPath) o;
        SvnRepositoryContentRevision beforeRevision = (entry.getType() == 'A') ? null : new SvnRepositoryContentRevision(myRepository, entry, myLogEntry.getRevision() - 1);
        SvnRepositoryContentRevision afterRevision = (entry.getType() == 'D') ? null : new SvnRepositoryContentRevision(myRepository, entry, myLogEntry.getRevision());
        myChanges.add(new Change(beforeRevision, afterRevision));
    }
