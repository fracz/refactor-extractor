        if (rhs != null) {
            rhs.accept(this);
        }
        if (constant == CONSTANT_FALSE) {
            this.constant = CONSTANT_FALSE;
        } else if (constant == NOT_CONSTANT) {
            this.constant = NOT_CONSTANT;
        }

            definitelyAssigned = da;
