public void installRenderer(final CheckboxTreeCellRendererBase cellRenderer) {
    setCellRenderer(cellRenderer);
    new ClickListener() {

        @Override
        public boolean onClick(@NotNull MouseEvent e, int clickCount) {
            int row = getRowForLocation(e.getX(), e.getY());
            if (row < 0)
                return false;
            final Object o = getPathForRow(row).getLastPathComponent();
            if (!(o instanceof CheckedTreeNode))
                return false;
            Rectangle rowBounds = getRowBounds(row);
            cellRenderer.setBounds(rowBounds);
            Rectangle checkBounds = cellRenderer.myCheckbox.getBounds();
            checkBounds.setLocation(rowBounds.getLocation());
            if (checkBounds.height == 0)
                checkBounds.height = checkBounds.width = rowBounds.height;
            final CheckedTreeNode node = (CheckedTreeNode) o;
            if (checkBounds.contains(e.getPoint())) {
                if (node.isEnabled()) {
                    toggleNode(node);
                    setSelectionRow(row);
                    return true;
                }
            } else if (clickCount > 1) {
                onDoubleClick(node);
                return true;
            }
            return false;
        }
    }.installOn(this);
