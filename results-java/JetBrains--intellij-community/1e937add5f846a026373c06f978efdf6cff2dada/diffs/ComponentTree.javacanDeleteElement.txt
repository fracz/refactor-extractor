        if (suppressions != null) {
            return true;
        }
        DeleteProvider baseProvider = (DeleteProvider) myEditor.getData(DataConstants.DELETE_ELEMENT_PROVIDER);
        if (baseProvider != null) {
            return baseProvider.canDeleteElement(dataContext);
        }
    }
