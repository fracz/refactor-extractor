private void doPaint(Graphics g) {
    final JBTabsPosition pos = myTabs.getTabsPosition();
    int dX = 0, dXs = 0, dY = 0, dYs = 0;
    int selected = 1, plain = 2;
    switch(pos) {
        case bottom:
            dY = -plain;
            dYs = -selected;
            break;
        case left:
            dX = plain;
            dXs = selected;
            break;
        case right:
            dX = -plain;
            dXs = -selected;
            break;
        case top:
            dY = plain;
            dYs = selected;
            break;
    }
    if (myTabs.getSelectedInfo() != myInfo) {
        g.translate(dX, dY);
    } else {
        g.translate(dXs, dYs);
    }
    super.paint(g);
    if (myTabs.getSelectedInfo() != myInfo) {
        g.translate(dX, -dY);
    } else {
        g.translate(dX, dY);
    }
