@Override
public void visitUnaryExpression(GrUnaryExpression expression) {
    final GrExpression operand = expression.getOperand();
    if (operand != null) {
        final boolean negation = expression.getOperationTokenType() == GroovyTokenTypes.mLNOT;
        if (negation) {
            myNegate = !myNegate;
        }
        operand.accept(this);
        if (negation) {
            myNegate = !myNegate;
        }
    }
