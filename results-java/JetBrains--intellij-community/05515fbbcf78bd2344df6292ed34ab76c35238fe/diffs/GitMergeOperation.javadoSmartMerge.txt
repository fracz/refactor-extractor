private boolean doSmartMerge(@NotNull final Collection<GitRepository> repositories) {
    final AtomicBoolean success = new AtomicBoolean();
    myPreservingProcess = new GitPreservingProcess(myProject, myGit, repositories, "merge", myBranchToMerge, getIndicator(), new Runnable() {

        @Override
        public void run() {
            success.set(doMerge(repositories));
        }
    });

            return myConflictedRepositories.isEmpty();

    return success.get();
