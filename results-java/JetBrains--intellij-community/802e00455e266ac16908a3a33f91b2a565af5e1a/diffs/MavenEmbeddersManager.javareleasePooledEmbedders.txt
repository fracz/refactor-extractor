private synchronized void releasePooledEmbedders() {
    for (Object eachKey : myPools.keySet()) {
        LinkedList<MavenEmbedderWrapper> eachPool = myPools.get(eachKey);
        // collected
        if (eachPool == null)
            continue;
        for (MavenEmbedderWrapper each : eachPool) {
            each.release();
        }
        eachPool.clear();
    }
    myPools.clear();
