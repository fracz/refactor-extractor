protected void doCleanUp() {
    Runnable cleanup = new Runnable() {

        public void run() {
            if (!isReleaseRequested()) {
                cleanUpNow();
            }
        }
    };
    if (isPassthroughMode()) {
        cleanup.run();
    } else {
        UIUtil.invokeLaterIfNeeded(cleanup);
    }
