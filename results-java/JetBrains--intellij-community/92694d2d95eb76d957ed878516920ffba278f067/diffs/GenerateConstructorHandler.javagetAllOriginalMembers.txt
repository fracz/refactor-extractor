protected Object[] getAllOriginalMembers(PsiClass aClass) {
    PsiField[] fields = aClass.getFields();
    ArrayList<PsiElement> array = new ArrayList<PsiElement>();
    final PsiSearchHelper searchHelper = aClass.getManager().getSearchHelper();
    for (PsiField field : fields) {
        if (field.hasModifierProperty(PsiModifier.STATIC))
            continue;
        if (field.hasModifierProperty(PsiModifier.FINAL) && field.getInitializer() != null)
            continue;
        if (searchHelper.isFieldBoundToForm(field))
            continue;
        array.add(field);
    }
    return array.toArray(new PsiElement[array.size()]);
