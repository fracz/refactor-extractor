private void processOneLine(String line) {
    int indentSize = myIndentHelper.getIndent(line, false);
    if (StringUtil.isEmptyOrSpaces(line)) {
        doProcessLine("\n");
    } else if (indentSize == 0 && indentSize < myCurrentIndentSize && !shouldIndent(line)) {
        doProcessLine("\n");
        doProcessLine(line);
    } else {
        doProcessLine(line);
    }
