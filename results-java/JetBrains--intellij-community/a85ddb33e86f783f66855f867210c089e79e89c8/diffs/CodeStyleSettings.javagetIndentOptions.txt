public IndentOptions getIndentOptions(FileType fileType) {
    if (USE_SAME_INDENTS || fileType == null || fileType == StdFileTypes.JAVA)
        return JAVA_INDENT_OPTIONS;
    final IndentOptions indentOptions = ourAdditionalIndentOptions.get(fileType);
    if (indentOptions != null)
        return indentOptions;
    if (fileType == StdFileTypes.JSP)
        return JSP_INDENT_OPTIONS;
    if (fileType == StdFileTypes.XML)
        return XML_INDENT_OPTIONS;
    return OTHER_INDENT_OPTIONS;
