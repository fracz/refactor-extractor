private void handleMouseEvent(MouseEvent e) {
    Object oldKey = myKey;
    myKey = getCellKeyForPoint(e.getPoint());
    if (myKey == null || !myComponent.isShowing()) {
        hideHint();
        return;
    }
    Point location = createToolTipImage(myKey);
    if (location == null) {
        hideHint();
    } else if (myHint == null) {
        show(location);
    } else if (!Comparing.equal(oldKey, myKey)) {
        hideHint();
        show(location);
    }
