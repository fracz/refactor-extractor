private void append(List<MavenProject> updated, String text) {
    log += text + " ";
    if (updated.isEmpty()) {
        log += "<none> ";
        return;
    }
    log += StringUtil.join(updated, new Function<MavenProject, String>() {

        public String fun(MavenProject each) {
            return each.getMavenId().artifactId;
        }
    }, " ") + " ";
