private Set<String> getProjectArtifactIds(String groupId) {
    Set<String> result = new THashSet<String>();
    for (MavenId each : getProjectsIds()) {
        if (groupId.equals(each.groupId)) {
            result.add(each.artifactId);
        }
    }
    return result;
