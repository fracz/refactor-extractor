@Override
public void branchHasChanged(@NotNull String branchName) {
    VcsConfiguration vcsConfiguration = VcsConfiguration.getInstance(myProject);
    if (!vcsConfiguration.RELOAD_CONTEXT)
        return;
    String contextName = getContextName(branchName);
    if (!myContextManager.hasContext(contextName))
        return;
    myContextManager.clearContext();
    myContextManager.loadContext(contextName);
    NOTIFICATION.createNotification(null, null, "Branch context has been loaded<br>" + "<a href='off'>Turn off the feature</a>", NotificationType.INFORMATION, (notification, event) -> {
        if (event.getEventType() != HyperlinkEvent.EventType.ACTIVATED)
            return;
        if ("off".equals(event.getDescription())) {
            vcsConfiguration.RELOAD_CONTEXT = false;
            notification.expire();
        }
    }).notify(myProject);
