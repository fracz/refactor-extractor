                getScope(vcs).addDirtyFile(new FilePathImpl(file));

            if (vcs != null) {
                somethingDirty = true;
                getScope(vcs).addDirtyDirRecursively(new FilePathImpl(dir));
            }
        }
    }
    if (somethingDirty) {
        myChangeListManager.scheduleUpdate();
    }
