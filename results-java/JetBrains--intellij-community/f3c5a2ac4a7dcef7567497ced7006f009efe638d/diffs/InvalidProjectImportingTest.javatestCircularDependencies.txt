public void testCircularDependencies() throws Exception {
    if (ignore())
        return;
    createProjectPom("<groupId>test</groupId>" + "<artifactId>project</artifactId>" + "<packaging>pom</packaging>" + "<version>1</version>" + "<modules>" + "  <module>m1</module>" + "  <module>m2</module>" + "  <module>m3</module>" + "</modules>");
    createModulePom("m1", "<groupId>test</groupId>" + "<artifactId>m1</artifactId>" + "<version>1</version>" + "<dependencies>" + "  <dependency>" + "    <groupId>test</groupId>" + "    <artifactId>m2</artifactId>" + "    <version>1</version>" + "  </dependency>" + "</dependencies>");
    createModulePom("m2", "<groupId>test</groupId>" + "<artifactId>m2</artifactId>" + "<version>1</version>" + "<dependencies>" + "  <dependency>" + "    <groupId>test</groupId>" + "    <artifactId>m3</artifactId>" + "    <version>1</version>" + "  </dependency>" + "</dependencies>");
    createModulePom("m3", "<groupId>test</groupId>" + "<artifactId>m3</artifactId>" + "<version>1</version>" + "<dependencies>" + "  <dependency>" + "    <groupId>test</groupId>" + "    <artifactId>m1</artifactId>" + "    <version>1</version>" + "  </dependency>" + "</dependencies>");
    importProject();
    MavenProjectModel.Node root = getRootProjects().get(0);
    assertTrue(root.isValid());
    assertProblems(root);
    assertProblems(root.getSubProjects().get(0));
    assertProblems(root.getSubProjects().get(1));
    assertProblems(root.getSubProjects().get(2));
