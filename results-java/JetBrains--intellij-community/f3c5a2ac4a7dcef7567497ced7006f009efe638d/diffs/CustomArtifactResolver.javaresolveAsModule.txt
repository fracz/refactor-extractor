private boolean resolveAsModule(Artifact a) {
    if (myMapping == null)
        return false;
    VirtualFile f = myMapping.get(new ProjectId(a));
    if (f == null)
        return false;
    a.setResolved(true);
    a.setFile(new File(f.getPath()));
    return true;
