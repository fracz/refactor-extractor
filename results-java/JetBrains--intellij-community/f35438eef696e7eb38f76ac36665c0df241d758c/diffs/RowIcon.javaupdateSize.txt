private void updateSize() {
    int width = 0;
    int height = 0;
    for (Icon icon : getIcons()) {
        if (icon == null)
            continue;
        width += icon.getIconWidth();
        // height += icon.getIconHeight();
        height = Math.max(height, icon.getIconHeight());
    }
    myWidth = width;
    myHeight = height;
