        public boolean value(String s) {
            return s.contains("http-proxy-port");
        }
    });
    if (!StringUtil.isEmpty(proxyHostParameter) && myUrl != null && myProxyAuthentication != null) {
        String group = getHostGroup(proxyHostParameter);

    }
