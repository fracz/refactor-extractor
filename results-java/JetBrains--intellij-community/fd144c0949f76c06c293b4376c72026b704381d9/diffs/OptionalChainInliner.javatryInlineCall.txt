    BiConsumer<CFGBuilder, PsiMethodCallExpression> terminalInliner = TERMINAL_MAPPER.mapFirst(call);
    if (terminalInliner != null) {
        PsiExpression qualifierExpression = call.getMethodExpression().getQualifierExpression();
        if (!pushOptionalValue(builder, PsiUtil.skipParenthesizedExprDown(qualifierExpression), call.getMethodExpression()))
            return false;
        terminalInliner.accept(builder, call);
        return true;
    }
    if (pushIntermediateOperationValue(builder, call)) {
        builder.ifNotNull().push(builder.getFactory().getOptionalFactory().getOptional(true)).elseBranch().push(builder.getFactory().getOptionalFactory().getOptional(false)).endIf();
        return true;
    }
    return false;
