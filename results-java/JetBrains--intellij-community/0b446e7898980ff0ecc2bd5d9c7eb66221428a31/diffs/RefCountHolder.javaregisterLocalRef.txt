private void registerLocalRef(PsiElement ref, PsiElement refElement) {
    // filter self-recursive calls
    if (refElement instanceof PsiMethod && PsiTreeUtil.isAncestor(refElement, ref, true))
        return;
    // filter inner use of itself
    if (refElement instanceof PsiClass && PsiTreeUtil.isAncestor(refElement, ref, true))
        return;
    myLocalRefsMap.put(ref, refElement);
    if (refElement instanceof PsiNamedElement) {
        PsiNamedElement namedElement = (PsiNamedElement) refElement;
        if (!myUsedElements.contains(namedElement)) {
            myUsedElements.add(namedElement);
            addStatistics(namedElement);
        }
    }
