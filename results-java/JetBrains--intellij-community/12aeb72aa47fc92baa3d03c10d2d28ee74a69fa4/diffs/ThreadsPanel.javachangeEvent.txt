        startLabelsUpdate();
    } else if (DebuggerSession.EVENT_PAUSE == event || DebuggerSession.EVENT_DETACHED == event || DebuggerSession.EVENT_DISPOSE == event) {
        myUpdateLabelsAlarm.cancelAllRequests();
    }

        stateManager.removeListener(this);
    }
