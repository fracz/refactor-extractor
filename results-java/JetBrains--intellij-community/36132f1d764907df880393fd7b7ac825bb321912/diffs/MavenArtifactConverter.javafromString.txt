public String fromString(@Nullable @NonNls String s, ConvertContext context) {
    if (s == null)
        return null;
    Dependency dep = MavenArtifactConverterHelper.getMavenDependency(context);
    if (dep != null) {
        if ("system".equals(dep.getScope().getStringValue())) {
            return s;
        }
    }
    try {
        Project p = context.getFile().getProject();
        return isValid(p, MavenProjectIndicesManager.getInstance(p), MavenArtifactConverterHelper.getId(context), context) ? s : null;
    } catch (MavenIndexException e) {
        MavenLog.info(e);
        return s;
    }
