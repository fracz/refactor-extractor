@Override
public void mouseClicked(MouseEvent e) {
    if (myLinkListener.onClick(e, e.getClickCount())) {
        return;
    }
    if (e.getClickCount() > 1) {
        expandOrCollapseRoots(e);
    } else if (e.getClickCount() == 1) {
        performAction(e, MyGraphMouseAction.Type.CLICK);
    }
