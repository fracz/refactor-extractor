@Nullable
public static PsiElement getChildWhichContainsElement(@NotNull PsiCodeBlock ancestor, @NotNull PsiElement descendant) {
    PsiElement element = descendant;
    while (!element.equals(ancestor)) {
        descendant = element;
        element = descendant.getParent();
        if (element == null) {
            return null;
        }
    }
    return descendant;
