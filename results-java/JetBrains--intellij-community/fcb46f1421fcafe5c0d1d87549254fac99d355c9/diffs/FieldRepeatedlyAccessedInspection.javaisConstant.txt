private boolean isConstant(PsiField field) {
    if (!field.hasModifierProperty(PsiModifier.STATIC) || !field.hasModifierProperty(PsiModifier.FINAL)) {
        return false;
    }
    final PsiType type = field.getType();
    if (type == null) {
        return false;
    }
    return ClassUtils.isImmutable(type);
