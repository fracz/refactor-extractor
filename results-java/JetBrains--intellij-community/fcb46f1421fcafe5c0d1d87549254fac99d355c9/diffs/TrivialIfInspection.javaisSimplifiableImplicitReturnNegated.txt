public static boolean isSimplifiableImplicitReturnNegated(PsiIfStatement ifStatement) {
    if (ifStatement.getElseBranch() != null) {
        return false;
    }
    PsiStatement thenBranch = ifStatement.getThenBranch();
    thenBranch = ConditionalUtils.stripBraces(thenBranch);
    final PsiElement nextStatement = PsiTreeUtil.skipSiblingsForward(ifStatement, new Class[] { PsiWhiteSpace.class });
    if (!(nextStatement instanceof PsiStatement)) {
        return false;
    }
    final PsiStatement elseBranch = (PsiStatement) nextStatement;
    return ConditionalUtils.isReturn(thenBranch, PsiKeyword.FALSE) && ConditionalUtils.isReturn(elseBranch, PsiKeyword.TRUE);
