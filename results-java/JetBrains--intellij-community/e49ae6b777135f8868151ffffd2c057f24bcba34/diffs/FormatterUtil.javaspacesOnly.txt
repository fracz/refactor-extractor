private static boolean spacesOnly(TreeElement node) {
    if (node.getElementType() == TokenType.WHITE_SPACE || node.getTextLength() == 0)
        return true;
    for (FormatterUtilHelper helper : ourHelpers) {
        if (helper.containsWhitespacesOnly(node))
            return true;
    }
    return false;
