private void delete(File file) {
    if (file.isDirectory()) {
        File[] files = file.listFiles();
        for (File fileToDelete : files) {
            delete(fileToDelete);
        }
    }
    boolean b = file.delete();

        assertTrue("Can't delete " + file.getAbsolutePath() + " in " + getFullName(), false);
    }
