public void applyFix(@NotNull Project project, @NotNull ProblemDescriptor problemDescriptor) {
    groups.add(myGroupName);
    final InspectionProfile inspectionProfile = InspectionProjectProfileManager.getInstance(project).getInspectionProfile(problemDescriptor.getPsiElement());
    // correct save settings
    ((InspectionProfileImpl) inspectionProfile).isProperSetting(HighlightDisplayKey.find(SHORT_NAME));
    try {
        inspectionProfile.save();
    } catch (IOException e) {
        Messages.showErrorDialog(project, e.getMessage(), CommonBundle.getErrorTitle());
    }
