private void fireBeforeSelectionChanged(TabInfo oldInfo, TabInfo newInfo) {
    if (oldInfo != newInfo) {
        myOldSelection = oldInfo;
        try {
            for (TabsListener eachListener : myTabListeners) {
                eachListener.beforeSelectionChanged(oldInfo, newInfo);
            }
        } finally {
            myOldSelection = null;
        }
    }
