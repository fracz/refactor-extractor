private static PsiReferenceExpression getVariableReferenceFromCondition(PsiExpression condition) {
    final PsiExpression strippedCondition = ParenthesesUtils.stripParentheses(condition);
    final PsiBinaryExpression binaryExp = (PsiBinaryExpression) strippedCondition;
    if (binaryExp == null) {
        return null;
    }
    final PsiExpression rhs = binaryExp.getROperand();
    if (rhs == null) {
        return null;
    }
    PsiExpression strippedRhs = ParenthesesUtils.stripParentheses(rhs);
    if (strippedRhs instanceof PsiMethodCallExpression) {
        final PsiMethodCallExpression methodCallExpression = (PsiMethodCallExpression) strippedRhs;
        strippedRhs = methodCallExpression.getMethodExpression();
    }
    if (strippedRhs == null) {
        return null;
    }
    final PsiReferenceExpression referenceExpression = (PsiReferenceExpression) strippedRhs;
    return (PsiReferenceExpression) referenceExpression.getQualifierExpression();
