public static Object getData(final TestProxy testProxy, final String dataId, final JUnitRunningModel model) {

        return null;
    if (TestProxy.DATA_CONSTANT.equals(dataId))
        return testProxy;
    if (DataConstants.NAVIGATABLE.equals(dataId))
        return getOpenFileDescriptor(testProxy, model);
    final PsiLocator testInfo = testProxy.getInfo();

        return location != null ? location.getPsiElement() : null;
    }
    if (Location.LOCATION.equals(dataId))

    return null;
