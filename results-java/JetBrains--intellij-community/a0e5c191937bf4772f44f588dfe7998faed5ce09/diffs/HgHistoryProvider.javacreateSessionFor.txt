    if (vcsRoot == null) {
        return null;
    }
    HgFile hgFile = new HgFile(vcsRoot, filePath);

    final List<VcsFileRevision> result = new LinkedList<VcsFileRevision>(revisions);

            return new HgWorkingCopyRevisionsCommand(project).firstParent(vcsRoot);
        }

            return null;
