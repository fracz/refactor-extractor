private static boolean hasConstructor(PsiClass aClass) {
    final PsiMethod[] methods = aClass.getMethods();
    for (int i = 0; i < methods.length; i++) {
        final PsiMethod method = methods[i];
        if (method.isConstructor()) {
            return true;
        }
    }
    return false;
