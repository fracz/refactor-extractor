    if (mySteps.size() > 1)
        toolbar.addSeparator();
    toolbar.addAction(myPrevChangeAction);
    toolbar.addAction(myNextChangeAction);
    if (myActionsFactory != null) {
        toolbar.addSeparator();
        for (AnAction action : actions) {
            toolbar.addAction(action);
        }
    }
