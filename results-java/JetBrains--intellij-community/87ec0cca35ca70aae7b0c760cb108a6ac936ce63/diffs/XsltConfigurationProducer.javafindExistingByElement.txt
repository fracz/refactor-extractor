@Override
protected RunnerAndConfigurationSettings findExistingByElement(Location location, @NotNull RunnerAndConfigurationSettings[] existingConfigurations, ConfigurationContext context) {
    final XmlFile file = PsiTreeUtil.getParentOfType(location.getPsiElement(), XmlFile.class, false);
    if (file != null && file.isPhysical() && XsltSupport.isXsltFile(file)) {
        for (RunnerAndConfigurationSettings existingConfiguration : existingConfigurations) {
            final RunConfiguration configuration = existingConfiguration.getConfiguration();
            if (configuration instanceof XsltRunConfiguration) {
                if (file.getVirtualFile().getPath().replace('/', File.separatorChar).equals(((XsltRunConfiguration) configuration).getXsltFile())) {
                    return existingConfiguration;
                }
            }
        }
    }
    return null;
