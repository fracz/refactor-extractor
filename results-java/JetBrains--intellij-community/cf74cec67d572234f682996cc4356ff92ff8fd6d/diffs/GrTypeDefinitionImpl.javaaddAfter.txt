    if (anchor == null) {
        add(element);
        return element;
    }
    final GrTypeDefinitionBody body = getBody();
    assert anchor.getParent() == body;
    final PsiElement nextChild = anchor.getNextSibling();
    if (nextChild == null) {
        add(element);
        return element;

    ASTNode node = element.getNode();
    assert node != null;
    body.getNode().addChild(node, nextChild.getNode());
    return element;
