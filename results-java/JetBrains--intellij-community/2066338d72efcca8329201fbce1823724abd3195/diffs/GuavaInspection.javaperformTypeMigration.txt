private void performTypeMigration(List<PsiElement> elements, List<PsiType> types) {
    if (!FileModificationService.getInstance().preparePsiElementsForWrite(elements))
        return;
    final Project project = elements.get(0).getProject();
    final TypeMigrationRules rules = new TypeMigrationRules();
    rules.setBoundScope(GlobalSearchScopesCore.projectProductionScope(project).union(GlobalSearchScopesCore.projectTestScope(project)));
    rules.addConversionRuleSettings(new GuavaConversionSettings(ignoreJavaxNullable));
    TypeMigrationProcessor.runHighlightingTypeMigration(project, null, rules, elements.toArray(new PsiElement[elements.size()]), createMigrationTypeFunction(elements, types), true);
