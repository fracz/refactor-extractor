/**
 * Generates blocks for binary expressions
 *
 * @return
 * @param node
 * @param groovySettings
 */
private static List<Block> generateForBinaryExpr(final ASTNode node, Wrap myWrap, CommonCodeStyleSettings mySettings, GroovyCodeStyleSettings groovySettings, Map<PsiElement, Alignment> alignments) {
    final ArrayList<Block> subBlocks = new ArrayList<Block>();
    Alignment alignment = mySettings.ALIGN_MULTILINE_BINARY_OPERATION ? Alignment.createAlignment() : null;
    GrBinaryExpression binary = (GrBinaryExpression) node.getPsi();
    LOG.assertTrue(binary != null);
    addBinaryChildrenRecursively(binary, subBlocks, Indent.getContinuationWithoutFirstIndent(), alignment, myWrap, mySettings, groovySettings, alignments);
    return subBlocks;
