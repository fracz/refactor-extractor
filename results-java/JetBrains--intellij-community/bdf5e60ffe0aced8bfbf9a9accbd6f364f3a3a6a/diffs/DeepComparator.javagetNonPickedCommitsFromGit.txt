@NotNull
private Set<CommitId> getNonPickedCommitsFromGit(@NotNull Project project, @NotNull VirtualFile root, @NotNull String currentBranch, @NotNull String comparedBranch) {
    GitLineHandler handler = new GitLineHandler(project, root, GitCommand.CHERRY);
    // upstream - current branch; head - compared branch
    handler.addParameters(currentBranch, comparedBranch);
    final Set<CommitId> pickedCommits = ContainerUtil.newHashSet();
    handler.addLineListener(new GitLineHandlerAdapter() {

        @Override
        public void onLineAvailable(String line, Key outputType) {
            // - c3b9b90f6c26affd7e597ebf65db96de8f7e5860
            if (line.startsWith("+")) {
                try {
                    line = line.substring(2).trim();
                    int firstSpace = line.indexOf(' ');
                    if (firstSpace > 0) {
                        // safety-check: take just the first word for sure
                        line = line.substring(0, firstSpace);
                    }
                    Hash hash = HashImpl.build(line);
                    pickedCommits.add(new CommitId(hash, root));
                } catch (Exception e) {
                    LOG.error("Couldn't parse line [" + line + "]");
                }
            }
        }
    });
    handler.runInCurrentThread(null);
    return pickedCommits;
