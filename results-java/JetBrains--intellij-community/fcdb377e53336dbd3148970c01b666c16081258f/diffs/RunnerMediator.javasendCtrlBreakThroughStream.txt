private void sendCtrlBreakThroughStream() {
    OutputStream os = getProcessInput();
    PrintWriter pw = new PrintWriter(os);
    try {
        pw.print(IAC);
        pw.print(BRK);
        pw.flush();
    } finally {
        pw.close();
    }
