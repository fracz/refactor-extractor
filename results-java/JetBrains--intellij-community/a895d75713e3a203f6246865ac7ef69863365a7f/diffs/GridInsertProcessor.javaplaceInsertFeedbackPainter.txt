private void placeInsertFeedbackPainter(final GridInsertLocation insertLocation) {
    final Rectangle cellRect = SwingUtilities.convertRectangle(insertLocation.getContainer().getDelegee(), insertLocation.getCellRect(), myEditor.getActiveDecorationLayer());
    cellRect.grow(3, 3);
    myInsertFeedbackPainter.setVertical(insertLocation.getMode() == GridInsertMode.ColumnBefore || insertLocation.getMode() == GridInsertMode.ColumnAfter);
    int w = 4;
    switch(insertLocation.getMode()) {
        case ColumnBefore:
            myInsertFeedbackPainter.setBounds((int) cellRect.getMinX() - w, (int) cellRect.getMinY(), 2 * w, (int) cellRect.getHeight());
            break;
        case ColumnAfter:
            myInsertFeedbackPainter.setBounds((int) cellRect.getMaxX() - w, (int) cellRect.getMinY(), 2 * w, (int) cellRect.getHeight());
            break;
        case RowBefore:
            myInsertFeedbackPainter.setBounds((int) cellRect.getMinX(), (int) cellRect.getMinY() - w, (int) cellRect.getWidth(), 2 * w);
            break;
        case RowAfter:
            myInsertFeedbackPainter.setBounds((int) cellRect.getMinX(), (int) cellRect.getMaxY() - w, (int) cellRect.getWidth(), 2 * w);
            break;
    }
