public void load(@NotNull Consumer<String> consumer) {
    File file = new File(url);
    FileInputStream stream = null;
    try {
        stream = new FileInputStream(file);
        StreamLoader loader = new StreamLoader(stream);
        loader.load(consumer);
    } catch (Exception e) {
        LOG.error(e);
    } finally {
        try {
            stream.close();

        }
    }
