public Icon getIcon(boolean expanded) {
    final PsiElement element = getElement().getElement();
    if (element != null) {
        final int flags = Iconable.ICON_FLAG_VISIBILITY | Iconable.ICON_FLAG_READ_STATUS;
        if (getElement().isSyntheticJSP()) {

        return null;
