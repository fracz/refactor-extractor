    assert lineFeeds < MAX_LF_COUNT;
    final int flags = myFlags;
    myFlags &= ~0xFFFFFF80;
    myFlags |= (lineFeeds << LF_COUNT_SHIFT);
    assert getLineFeeds() == lineFeeds;
