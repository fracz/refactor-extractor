private void initAuthor() {
    AUTHOR = new ColumnInfo<Object, String>("Author") {

        private final TableCellRenderer myRenderer = new HighLightingRenderer(HIGHLIGHT_TEXT_ATTRIBUTES, SimpleTextAttributes.REGULAR_ATTRIBUTES, myUsersSearchContext);

        @Override
        public String valueOf(Object o) {
            if (o instanceof GitCommit) {
                return ((GitCommit) o).getAuthor();
            }
            return "";
        }

        @Override
        public TableCellRenderer getRenderer(Object o) {
            return myRenderer;
        }
    };
