    if (getBuilder().isSmartExpand() && node.getChildCount() == 1) {
        // "smart" expand
        TreeNode childNode = node.getChildAt(0);
        if (isLoadingNode(childNode))
            return;
        final TreePath childPath = new TreePath(node.getPath()).pathByAddingChild(childNode);
        expand(childPath);
    }
