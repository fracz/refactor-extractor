private void onTabSelection(JTabbedPane tabs) {
    int i = tabs.getSelectedIndex();
    tabs.setComponentAt(myLangSelectionIndex, createDummy());
    tabs.setComponentAt(i, getEditor().getComponent());
    myLangSelectionIndex = i;

    if (lang != null) {
        setLanguage(lang);
