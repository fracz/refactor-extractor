@Override
public boolean execute(PsiElement element, ResolveState state) {
    if (element instanceof PsiMethod || element instanceof PsiField) {
        String propertyName;
        PsiType type;
        if (element instanceof PsiMethod) {
            PsiMethod method = (PsiMethod) element;
            if (!GroovyPropertyUtils.isSimplePropertySetter(method))
                return true;
            propertyName = GroovyPropertyUtils.getPropertyNameBySetter(method);
            if (propertyName == null)
                return true;
            type = method.getParameterList().getParameters()[0].getType();
        } else {
            type = ((PsiField) element).getType();
            propertyName = ((PsiField) element).getName();
        }
        if (((PsiModifierListOwner) element).hasModifierProperty(PsiModifier.STATIC))
            return true;
        if (myResult.containsKey(propertyName) || propertyName.equals(METACLASS))
            return true;
        PsiSubstitutor substitutor = state.get(PsiSubstitutor.KEY);
        if (substitutor != null) {
            type = substitutor.substitute(type);
        }
        myResult.put(propertyName, new TypeCondition(type, element, myContext));
    }
    return true;
