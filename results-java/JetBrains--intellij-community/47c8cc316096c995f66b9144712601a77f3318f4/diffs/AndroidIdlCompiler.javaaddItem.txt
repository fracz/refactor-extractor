private void addItem(VirtualFile file, AndroidFacet facet, IAndroidTarget target, String packageName, List<GenerationItem> items) {
    Module module = facet.getModule();
    String sourceRootPath = facet.getAidlGenSourceRootPath();
    if (sourceRootPath == null) {
        myContext.addMessage(CompilerMessageCategory.ERROR, AndroidBundle.message("android.compilation.error.apt.gen.not.specified", module.getName()), null, -1, -1);
        return;
    }
    ProjectFileIndex fileIndex = ProjectRootManager.getInstance(myProject).getFileIndex();
    items.add(new IdlGenerationItem(module, file, fileIndex.isInTestSourceContent(file), target, packageName));
