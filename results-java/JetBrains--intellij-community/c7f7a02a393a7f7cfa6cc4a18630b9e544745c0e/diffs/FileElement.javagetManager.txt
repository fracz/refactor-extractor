public PsiManager getManager() {
    final PsiManager manager = getUserData(MANAGER_KEY);
    if (manager == null) {
        if (parent != null)
            return parent.getManager();
        else
            // TODO: cache?
            return SourceTreeToPsiMap.treeElementToPsi(this).getManager();
    } else {
        return manager;
    }
