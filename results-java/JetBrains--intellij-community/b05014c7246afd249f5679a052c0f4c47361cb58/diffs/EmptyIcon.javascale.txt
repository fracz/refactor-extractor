@Override
public Icon scale(float scale) {
    if (getScale() == scale) {
        return this;
    }
    if (myScaledCache != null && myScaledCache.getScale() == scale) {
        return myScaledCache;
    }
    myScaledCache = createScaledInstance(scale);
    myScaledCache.setScale(scale);
    return myScaledCache;
