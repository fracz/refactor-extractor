@Override
public Icon scale(float scale) {
    if (scale == 1f)
        return this;
    // force state update & cache reset
    getRealIcon();
    Icon icon = myScaledIconsCache.getOrScaleIcon(getJBUIScale(ScaleType.PIX), scale);
    if (icon != null) {
        return icon;
    }
    return this;
