public void fireUpdatedIfNecessary(Object message) {
    if (updatedProjectsWithChanges.isEmpty() && deletedProjects.isEmpty())
        return;
    List<MavenProject> mavenProjects = deletedProjects.isEmpty() ? Collections.<MavenProject>emptyList() : new ArrayList<MavenProject>(deletedProjects);
    List<Pair<MavenProject, MavenProjectChanges>> updated = updatedProjectsWithChanges.isEmpty() ? Collections.<Pair<MavenProject, MavenProjectChanges>>emptyList() : MavenUtil.mapToList(updatedProjectsWithChanges);
    fireProjectsUpdated(updated, mavenProjects, message);
