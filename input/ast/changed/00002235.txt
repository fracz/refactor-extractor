    protected function nodesList(DOMCrawler $nodes, $contains = null)
    {
        $output = "";
        foreach ($nodes as $node) {
            if ($contains) {
                if (strpos($node->nodeValue, $contains) === false) {
                    continue;
                }
            }
            $output .= "\n+ <info>" . $node->C14N() . "</info>";
        }
        return $output;
    }
