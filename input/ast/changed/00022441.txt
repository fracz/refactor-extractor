                }
                $this->savepointpresent = false;
            }

                $res = @pg_query($this->pgsql, "ROLLBACK TO SAVEPOINT moodle_pg_savepoint");
                if ($res) {
                    pg_free_result($res);
                }
                $this->savepointpresent = false;
            }
            throw $e;
        }
