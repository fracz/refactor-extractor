    public function updateDocument(Document $document, array $options = array())
    {
        if (!$document->hasId()) {
            throw new InvalidException('Document id is not set');

        return $this->getIndex()->getClient()->updateDocument(
            $document->getId(),
            $document,
            $this->getIndex()->getName(),
            $this->getName(),
            $options
        );
    }
