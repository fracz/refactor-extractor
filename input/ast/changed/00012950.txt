	public function crud()
	{
		R::freeze( TRUE );

		for( $i = 0; $i < 100; $i++ ) {
			$book = R::dispense( 'book' );
			$book->title = 'Book '.$i;
			$page = R::dispense('page');
			$page->content = 'Content '.$i;
			$tag = R::dispense('tag');
			$tag->label = 'Tag '.$i;
			$book->noLoad()->ownPage[] = $page;
			$book->noLoad()->sharedTag[] = $tag;
			R::store( $book );
			$book = $book->fresh();
			$book->ownPage;
			$book->sharedTag;
			R::trash( $book );
		}
	}
