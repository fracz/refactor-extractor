	public function getColumnTranslation( $columnName )
	{
		if( isset($this->columnsTranslations[$columnName]) )
		{
			return html_entity_decode($this->columnsTranslations[$columnName], ENT_COMPAT, 'UTF-8');
		}
		else

	}
