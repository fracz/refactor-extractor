	function selectEmail( $v )
	{
		$s = $this->select($v, 'email');
		if(empty($s)){
			$s = $this->select($v, 'yahooid');
			if(!empty($s) && isset($s->value) && strpos($s->value,"@")===FALSE)
				$s->value .= "@yahoo.com";
		}
		return ($s)?$s->value:"";
	}
