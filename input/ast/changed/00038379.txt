    protected function renderList($block)
    {
        foreach ($block['items'] as $item => $itemLines) {
            if (preg_match('~\[([^\]]+)\]\(([^\)]+)\)(.*)~', implode("\n", $itemLines), $matches)) {
                $this->_chapters[$this->_chapter]['content'][] = [
                    'headline' => $matches[1],
                    'file' => $matches[2],
                    'teaser' => $matches[3],
                ];
            }
        }
        return parent::renderList($block);
    }
