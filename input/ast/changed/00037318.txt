    public function insertOne($row)
    {
        $row = $this->formatRow($row);
        if ($this->useValidation) {
            $row = $this->validateRow($row);
        }
        $csv = $this->getCsv();
        $csv->fputcsv($row, $this->delimiter, $this->enclosure);
        if ("\n" !== $this->newline) {
            $csv->fseek(-1, SEEK_CUR);
            $csv->fwrite($this->newline);
        }

        return $this;
    }
