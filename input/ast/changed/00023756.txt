	{
		if (empty($args[1]) || !isset($this->attrs[$m])) { // set
			$this->attrs[$m] = $args[0];

		} elseif ($args[0] == NULL) {
			// append empty value -> ignore

		} elseif (is_array($this->attrs[$m])) { // append to array
			$this->attrs[$m][] = $args[0];

		} else { // append to string
			$this->attrs[$m] .= ' ' . $args[0];
		}
		return $this;
	}
