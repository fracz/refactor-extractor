    public static function getTplFile($view)
    {
        $views = array(
            'install', 'import', 'export', 'config', 'tags',
            'edit-tags', 'view', 'login', 'error', 'tag'
            );

        if (in_array($view, $views)) {
            return $view . '.twig';
        }

        return 'home.twig';
    }
