    public function isExclusivelyArrayLike() : bool
    {
        if ($this->isEmpty()) {
            return false;
        }

        return \array_reduce($this->getTypeSet()->toArray(),
            function (bool $is_exclusively_array, Type $type) : bool {
                return (
                    $is_exclusively_array
                    && $type->isArrayLike()
                    && !$type->getIsNullable()
                );
            }, true);
    }
