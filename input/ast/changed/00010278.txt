			'value' => 1,
			'checked' => false,
			'disabled' => false,
		];
		$attrs = $this->_templates->formatAttributes(
			$data,
			['name', 'value']
		);

		return $this->_templates->format('checkbox', [
			'name' => $data['name'],
			'value' => $data['value'],
			'attrs' => $attrs
		]);
	}
