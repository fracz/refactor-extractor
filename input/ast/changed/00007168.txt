	public function testExplicitSet() {
		$doc = new Document();
		$doc->set('id', 4);
		$doc->set('name', 'Four');
		$doc->set('content',  'Lorem ipsum four');

		$expected = array('id' => 4, 'name' => 'Four', 'content' => 'Lorem ipsum four');
		$result = $doc->data();
		$this->assertEqual($expected, $result);
	}
