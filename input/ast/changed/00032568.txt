	public function inc( $file ){
		// $_ erstellen
		$_ = $this->vars;

		// Einbinden
		ob_start();
		include( $this->path.$file.'.php' );
		$data = ob_get_contents();
		ob_end_clean();

		// Daten zurückgeben
		return $data;
	}
