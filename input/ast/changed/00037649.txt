    function delete($restriction) {
        $fileName = $this->getEntityFilename($restriction['vp_id']);
        if (is_file($fileName)) {
            $entity = $this->loadEntity($restriction['vp_id']);
            unlink($fileName);
            $this->notifyChangeListeners($entity, $entity, 'delete');
        }
    }
