	public function config($key, $value=null)
	{
		$key = strtolower( $key );

		if( $value ){
			$_SESSION["HA::CONFIG"][$key] = serialize( $value );
		}
		elseif( isset( $_SESSION["HA::CONFIG"][$key] ) ){
			return unserialize( $_SESSION["HA::CONFIG"][$key] );
		}

		return NULL;
	}
