	public function __isset( $name ) {
		if (
			0 === (int) $this->comment_post_ID
			|| property_exists( $this, $name )
		) {
			return;
		}

		$post = get_post( $this->comment_post_ID );
		if ( $post ) {
			return property_exists( $post, $name );
		}
	}
