	public function render($template, $data = array(), $options = array()) {
		$this->_context = $options['context'] + $this->_context;

		$__t__ = $this->_config['protocol'] . '://' . $template;
		unset($options);
		extract($data, EXTR_OVERWRITE);
		ob_start();
		include $__t__;
		return ob_get_clean();
	}
