  function email($email, $name='', $class='', $more='') {
    if (!$email) return $name;
    $email = obfuscate($email);
    if (!$name) $name = $email;
    $class = "class='".($class ? $class : 'mail')."'";
    return "<a href='mailto:$email' $class title='$email' $more>$name</a>";
  }
