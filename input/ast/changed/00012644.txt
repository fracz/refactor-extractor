    protected static function arguments(array $tokens)
    {
        return array_filter(array_map(function ($token) {
            if (starts_with($token, '{') && !starts_with($token, '{--')) {
                return static::parseArgument(trim($token, '{}'));
            }
        }, $tokens));
    }
