	public function setItems(array $items, $useKeys = TRUE)
	{
		if (!$useKeys) {
			foreach ($items as $key => $value) {
				unset($items[$key]);
				if (is_array($value)) {
					foreach ($value as $val) {
						$items[$key][(string) $val] = $val;
					}
				} else {
					$items[(string) $value] = $value;
				}
			}
		}
		$this->items = $items;
		$this->flattenItems = Nette\Utils\Arrays::flatten($items, TRUE);
		return $this;
	}
