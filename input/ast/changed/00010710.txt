	function testWritingWithConfig() {

		Cache::config('sessions');

		Cache::write('test_somthing', 'this is the test data', 'tests');

		$expected = array(
			'path' => TMP . 'sessions',
			'prefix' => 'cake_',
			'lock' => false,
			'serialize' => true,
			'duration' => 3600,
			'probability' => 100,
			'engine' => 'File',
			'isWindows' => (substr(PHP_OS, 0, 3) == "WIN")
		);
		$this->assertEqual($expected, Cache::settings('File'));
	}
