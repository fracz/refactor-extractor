    public function containsErrors (FormView $form) {
        if (count($form->vars['errors'])) {
            return true;
        }
        foreach ($form->children as $child) {
            if (count($child->vars['errors'])) {
               return true;
            }

            if (count($child->children)) {

                if ($hasErrors) {
                    return true;
                }
            }
