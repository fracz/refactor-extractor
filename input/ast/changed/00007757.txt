	public function rewind() {
		$data = parent::rewind();
		$key = key($this->_data);

		if (is_a($data, $this->_classes['entity'])) {
			return $data;
		}

		if ($this->_isComplexType($data)) {
			$this->_data[$key] = $this->_relation('entity', $key, $this->_data[$key]);
		}
		return isset($this->_data[$key]) ? $this->_data[$key] : null;
	}
