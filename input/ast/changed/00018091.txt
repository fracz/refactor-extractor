	protected function validate_mimes($attribute, $parameters)
	{
		$file = IoC::container()->resolve('laravel.file');

		foreach ($parameters as $extension)
		{
			if ($file->is($extension, $this->attributes[$attribute]['tmp_name'])) return true;
		}

		return false;
	}
