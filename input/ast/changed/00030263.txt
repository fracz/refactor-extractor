	protected function generate()
	{
		if($this->subperiodsProcessed)
		{
			return;
		}
		parent::generate();

		$year = $this->date->get("Y");
		for($i=1; $i<=12; $i++)
		{
			$this->addSubperiod( new Piwik_Period_Month(
									new Piwik_Date("$year-$i-01")
								)
							);
		}
	}
