	private static function fillTemplate($template, $countryCode) {
		$bic = self::generateBic($countryCode);
		$bicPos = 0;
		$len = strlen($template);
		$unsigned = '';
		for ($i=0; $i<$len; $i++) {
			$c = $template[$i];
			if (strtoupper($c) === $c) {
				$unsigned .= $c;
				continue;
			}
			if ($c === 'i') {
				$unsigned .= $bic{$bicPos++};
				continue;
			}
			if ($c === 'k') {
				$unsigned .= '_';
				continue;
			}
			$unsigned .= Utils::generateRandomAlphanumericStr('x');
		}
		return self::recalculateChecksum($unsigned);
	}
