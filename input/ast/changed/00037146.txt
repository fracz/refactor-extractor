    protected function convertToUtf8(Traversable $iterator)
    {
        if ('UTF-8' == $this->encodingFrom) {
            return $iterator;
        }

        return new MapIterator($iterator, function ($row) {
            foreach ($row as &$value) {
                $value = mb_convert_encoding($value, 'UTF-8', $this->encodingFrom);
            }
            unset($value);

            return $row;
        });
    }
