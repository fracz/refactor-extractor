	public function read($length = null, $offset = null) {
		if (!is_resource($this->_resource)) {
			return false;
		}

		$buffer = null;
		if (is_null($length)) {
			$buffer = stream_get_contents($this->_resource);
		} else {
			$buffer = stream_get_contents($this->_resource, $length, $offset);
		}

		return $buffer;
	}
