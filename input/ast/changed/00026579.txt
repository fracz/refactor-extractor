  public function shouldUseHTTP() {
    $uri = new PhutilURI($this->getRemoteURI());
    $protocol = $uri->getProtocol();
    if ($this->isHTTPProtocol($protocol)) {
      return (bool)$this->getDetail('http-login');
    } else {
      return false;
    }
  }
