	public function stream_close() {

		$this->flush();

		if (
		$this->meta['mode']!='r'
		and $this->meta['mode']!='rb'
		) {

			\OC\Files\Filesystem::putFileInfo( $this->relPath, array( 'encrypted' => true, 'size' => $this->size ), '' );

		}

		return fclose( $this->handle );

	}
