	public function returnNameTable($alias = false, $tables)
	{
		if ($alias)
		{
			foreach ($tables as $table)
			{
				if ($table['alias'] == $alias)
					return array($table['table']);
			}
		}
		else if (!$alias && (count($tables) > 1))
		{
			$this->error_sql['returnNameTable'] = false;
			return false;
		}
		else
		{
			foreach ($tables as $table)
				$tab[] = $table['table'];
			return $tab;
		}
	}
