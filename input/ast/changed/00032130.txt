    function delete($ids)
    {
        if (!is_array($ids))
            $ids = explode(',', $ids);

        $ids = join(',', array_map(array($this->db, 'quote'), $ids));

        // flag record as deleted
        $this->db->query(
            "UPDATE ".$this->db_name.
            " SET del=1, changed=".$this->db->now().
            " WHERE user_id=?".
                " AND contact_id IN ($ids)",
            $this->user_id
        );

        $this->cache = null;

        return $this->db->affected_rows();
    }
