	public function add($autodate = true, $nullValues = false)
	{
		if (!empty($this->zones))
			$this->addZoneProducts($this->zones);
		if (!empty($this->categories))
			$this->addCategories($this->categories);

		return parent::add($autodate, $nullValues);
	}
