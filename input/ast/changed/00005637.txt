	public static function isFeatureActive()
	{
		if (self::$feature_active === null)
			self::$feature_active = (bool)Db::getInstance(_PS_USE_SQL_SLAVE_)->getValue('
				SELECT `id_group`
				FROM `'._DB_PREFIX_.'group`
			');
		return self::$feature_active;
	}
}
