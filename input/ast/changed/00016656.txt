	public function insertMulti($col, $docs = [], $cb = NULL, $params = []) {
		if (strpos($col, '.') === false) {
			$col = $this->dbname . '.' . $col;
		}

		$ids  = [];
		$bson = '';

		foreach ($docs as &$doc) {
			if (!isset($doc['_id'])) {
				$doc['_id'] = new MongoId();
			}

			$bson .= bson_encode($doc);

			$ids[] = $doc['_id'];
		}

		$this->request(self::OP_INSERT,
					   "\x00\x00\x00\x00"
					   . $col . "\x00"
					   . $bson
		, false, null, function ($conn, $reqId = null) use ($cb, $col, $params) {
			if ($cb !== NULL) {
				$this->lastError($col, $cb, $params, $conn);
			}
		});
		return $ids;
	}
