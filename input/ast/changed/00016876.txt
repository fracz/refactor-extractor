	public function s2c($reqId, $data, $ts) {
		if (!isset($this->requests[$reqId])) {
			return;
		}
		$req = $this->requests[$reqId];
		if ($req->jsid) {
			$body = 'Response' . $req->jsid . ' = ' . $data . ";\n";
		} else {
			$body = '<script type="text/javascript">WebSocket.onmessage(' . $data . ");</script>\n";
		}
		$req->out($body);
		$req->finish();
	}
