    private static function getWhere($whereConditions) {
        $where = array_filter($whereConditions, function ($w) {
            if (is_object($w)) {
                return !$w->isOperator;
            } else {
                return true;
            }
        });

        $where = array_map(function ($w) {
            if (is_object($w)) {
                return $w->expr;
            } else {
                return $w;
            }
        }, $where);

        return array_values($where);
    }
