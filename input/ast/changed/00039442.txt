    public function route(Request $request)
    {
        $requestUri = $request->server->get('REQUEST_URI');
        $requestUri = ltrim($requestUri, '/');

        if (!$requestUri) {
            return $this->setDefaultRoute($request);
        }

        $request->setRoute($requestUri);
    }
