	function getStaticContentCount( ) {
		static $count;

		if (!isset ($count)) {
			$count = false;
		}

		if ($count !== false) {
			global $database;

			$query = "SELECT COUNT( id )"
			."\n FROM #__menu "
			."\n WHERE type = 'content_typed'"
			."\n AND published = 1"
			;
			$database->setQuery( $query );
			$count = $database->loadResult();
		}
		return $count;
	}
