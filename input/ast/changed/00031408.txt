    public function format(array $record, Log $logger)
    {
        $record = $this->next($record, $logger);

        if (! is_string($record['message'])) {
            return $record;
        }

        if (!Common::isPhpCliMode()) {
            $record['message'] = '<pre>' . Common::sanitizeInputValue($record['message']) . '</pre>';
        }

        return $record;
    }
