	function getLanguagePath($basePath = JPATH_BASE, $language = null, $addTrailingSlash = true)
	{
		$dir = $basePath.DS.'language'.DS;
		if (isset ($language)) {
			$dir .= $language.DS;
		}
		return JPath::clean($dir, $addTrailingSlash);
	}
