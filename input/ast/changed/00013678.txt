    public function getKeys($regexp = null)
    {
        $keys = array_keys($this->data);

        // If a regular expression was set, filter the keys
        if ($regexp) {
            $keys = array_filter($keys, function($key) use ($regexp) {
                return preg_match($regexp, $key);
            });
        }

        return $keys;
    }
