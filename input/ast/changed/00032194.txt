    function encrypt($text, $keys)
    {
        try {
            foreach ($keys as $key) {
                $this->gpg->addEncryptKey($key);
            }

            $dec = $this->gpg->encrypt($text, true);
            return $dec;
        }
        catch (Exception $e) {
            return $this->get_error_from_exception($e);
        }
    }
