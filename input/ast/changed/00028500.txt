    protected function matches($other)
    {

        $match = @preg_match($this->expectedMessage, $other->getMessage());
        if (false !== $match) {
            $this->regexBased = true;

            return (bool) $match;
        }

        return strpos($other->getMessage(), $this->expectedMessage) !== false;
    }
