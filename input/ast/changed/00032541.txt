    function checkLock($path)
    {
        $result = false;
        $query = "SELECT *
                  FROM locks
                 WHERE path = '$path'
               ";
            $res = OC_DB::select($query);
        if ($res) {
				$row=$res[0];
                OC_DB::free_result($res);

            if ($row) {
                $result = array( "type"    => "write",
                                 "scope"   => $row["exclusivelock"] ? "exclusive" : "shared",
                                 "depth"   => 0,
                                 "owner"   => $row['owner'],
                                 "token"   => $row['token'],
                                 "created" => $row['created'],
                                 "modified" => $row['modified'],
                                 "expires" => $row['expires']
                                 );
            }
        }

        return $result;
    }
