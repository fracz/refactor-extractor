    public function load(array $configs, ContainerBuilder $container)
    {
        $dbal = $orm = array();
        foreach ($configs as $config) {
            if (isset($config['dbal'])) {
                $dbal[] = $config['dbal'];
            }

            if (isset($config['orm'])) {
                $orm[] = $config['orm'];
            }
        }

        if (!empty($dbal)) {
            $this->dbalLoad($dbal, $container);
        }

        if (!empty($orm)) {
            $this->ormLoad($orm, $container);
        }
    }
