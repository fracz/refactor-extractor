	protected function findConstraints($table)
	{
		$sql = "PRAGMA foreign_key_list({$table->quotedName})";
		$keys = $this->connection->createCommand($sql)->queryAll();
		foreach ($keys as $key) {
			$table->foreignKeys[] = array($key['table'], $key['from'] => $key['to']);
		}
	}
