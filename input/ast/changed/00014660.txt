	function login()
	{
		global $mainframe;

		$username	= JRequest::getVar( 'username' );
		$password	= JRequest::getVar( 'password' );

		$result = $mainframe->login($username, $password);

		if (!JError::isError($result)) {
			$mainframe->redirect('index.php');
		}

		LoginController::display();
	}
