    public function dump($language = 'en')
    {
        $definitions = '';

        foreach ($this->dispatcher->getDefinitions() as $regex => $definition) {
            $regex = $this->dispatcher->translateDefinitionRegex($regex, $language);

            $definitions .= $definition->getType() . " $regex\n\n";
        }

        return $definitions;
    }
