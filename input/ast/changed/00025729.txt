  public function getEpoch() {
    if ($this->isDisabled()) {
      return null;
    }

    $year = $this->valueYear;
    $month = $this->valueMonth;
    $day = $this->valueDay;
    $time = $this->valueTime;
    $zone = $this->getTimezone();

    if (!strlen($time)) {
      return null;
    }

    try {
      $date = new DateTime("{$year}-{$month}-{$day} {$time}", $zone);
      $value = $date->format('U');
    } catch (Exception $ex) {
      $value = null;
    }
    return $value;
  }
