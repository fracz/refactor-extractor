    public function assignRole(...$roles)
    {
        collect($roles)
            ->flatten()
            ->map(function ($role) {
                return $this->getStoredRole($role);
            })
            ->each(function (Role $role) {
                return $this->roles()->save($role);
            });

        $this->forgetCachedPermissions();

        return $this;
    }
