    public function get_children() {
        $all = self::get_all_ids();
        $rv = array();
        if (!empty($all[$this->id])) {
            foreach ($all[$this->id] as $id) {
                if ($coursecat = self::get($id, IGNORE_MISSING)) {
                    // do not return invisible
                    $rv[$coursecat->id] = $coursecat;
                }
            }
        }
        return $rv;
    }
