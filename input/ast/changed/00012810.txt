	protected function _rounded($radius, $sides, $antialias)
	{
		if ($radius < 0)
			$radius = 0;
		$tl = $tr = $bl = $br = $sides == null;
		if ($sides != null)
		{
			$sides = explode(" ", $sides);
			foreach ($sides AS $side)
			{
				if ($side == 'tl' || $side == 'tr' || $side == 'bl' || $side == 'br')
					$$side = true;
			}
		}
		if ($antialias == null)
			$antialias = 1;
		return Array(
			'radius' => $radius,
			'tl' => $tl,
			'tr' => $tr,
			'bl' => $bl,
			'br' => $br,
			'antialias' => $antialias
		);
	}
