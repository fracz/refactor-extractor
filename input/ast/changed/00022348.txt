    public static function ksort(array &$arr, $sortflag = collatorlib::SORT_STRING) {
        $keys = array_keys($arr);
        if (!self::asort($keys, $sortflag)) {
            return false;
        }
        // This is a bit slow, but we need to keep the references
        $original = $arr;
        $count = count($arr);
        for($i=0; $i<$count; $i++) {
            array_pop($arr);
        }
        foreach ($keys as $key) {
            $arr[$key] = $original[$key];
        }

        return true;
    }
}
