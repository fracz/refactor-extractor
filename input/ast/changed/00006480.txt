	public function processSave($token)
	{
		if ((int)Tools::getValue('id_feature_value') <= 0 && $this->display == 'add'
			|| (int)Tools::getValue('id_feature') <= 0 && $this->display != 'add')
			return $this->processAdd($token);
		else
			return $this->processUpdate($token);
	}
