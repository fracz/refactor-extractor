    public static function setcookie($name, $value, $exp = 0)
    {
        if (headers_sent()) {

        }

        $cookie = session_get_cookie_params();
        $secure = self::https_check();

        setcookie($name, $value, $exp, $cookie['path'], $cookie['domain'], $secure, true);
    }
