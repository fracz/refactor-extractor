	function test_getArrayFromQueryString()
	{
		$expected = array(
			'a' => false,
			'b' => '',
			'c' => '1',
			'd' => array( false ),
			'e' => array(''),
			'f' => array('a'),
			'g' => array('b', 'c'),
		);
		$this->assertEqual(serialize(Piwik_Common::getArrayFromQueryString('a&b=&c=1&d[]&e[]=&f[]=a&g[]=b&g[]=c')), serialize($expected));
	}
