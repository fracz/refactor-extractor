	public function noise2D($x, $z, $normalized = false){
		$result = 0;
		$amp = 1;
		$laq = 1;
		$max = 0;

		$x *= $this->frequency;
		$z *= $this->frequency;

		for($i = 0; $i < $this->octaves; ++$i){
			$result += $this->getNoise2D($x * $laq, $z * $laq) * $amp;
			$max += $amp;
			$laq *= $this->lacunarity;
			$amp *= $this->amplitude;
		}
		if($normalized === true){
			$result /= $max;
		}

		return $result;
	}
