	private function load($module, $file, $language)
	{
		if (isset(static::$lines[$module][$language.$file])) return;

		if (file_exists($path = Module::path($module).'lang/'.$language.'/'.$file.EXT))
		{
			static::$lines[$module][$language.$file] = require $path;
		}
	}
