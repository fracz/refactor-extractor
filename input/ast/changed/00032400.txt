	public function getConfig($key = '')
	{
		$config = $this->getParam('config');
		if (empty($key)) {
			return $config;
		}

		if (!array_key_exists($key, $config[$key])) {
			throw new Elastica_Exception_Invalid('Config key is not set: ' . $key);
		}

		return $this->$config[$key];
	}
