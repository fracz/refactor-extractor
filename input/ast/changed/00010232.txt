	function __construct($path = false, $create = false, $mode = false) {
		parent::__construct();
		if (empty($path)) {
			$path = TMP;
		}
		if ($mode) {
			$this->mode = intval($mode, 8);
		}
		if (!file_exists($path) && $create == true) {
			$this->create($path, $this->mode);
		}
		if($path{0} != '/') {
			$path = realpath($path);
		}
		$this->cd($path);
	}
