        if (!$this->menu) {

             *     public function addMenuItems()
             *     {
             *         MenuAdmin::getInstance()->add(
             *             'MenuName',
             *             'SubmenuName',
             *             array('module' => 'MyPlugin', 'action' => 'index'),
             *             $showOnlyIf = Piwik::hasUserSuperUserAccess(),
             *             $order = 6
             *         );

            Piwik::postEvent('Menu.Admin.addItems');
