	function _createReport($data)
	{
		$idReport = Piwik_PDFReports_API::getInstance()->addReport(
			$data['idsite'],
			$data['description'],
			$data['period'],
			$data['format'],
			$data['display_format'],
			$data['reports'],
			$data['email_me'],
			$data['additional_emails']);
		return $idReport;
	}
