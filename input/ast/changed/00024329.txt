	public function presentDelete($id = 0)
	{
		if (!$this->request->isPost()) return; // continue to renderDelete()...

		$del = $this->request->post['del'];
		if ($del == 'Yes' && $id > 0) {
			$album = new Albums();
			$album->delete($id);
		}

		$this->redirect('default');
	}
