    private static function mapErrorCode($subData)
    {
        switch ($subData) {
            case 0x00:
                return '#NULL!';
                break;
            case 0x07:
                return '#DIV/0!';
                break;
            case 0x0F:
                return '#VALUE!';
                break;
            case 0x17:
                return '#REF!';
                break;
            case 0x1D:
                return '#NAME?';
                break;
            case 0x24:
                return '#NUM!';
                break;
            case 0x2A:
                return '#N/A';
                break;
            default:
                return false;
        }
    }
