	public function renderPreviousAndNext()
	{
		if (!$this->displayPreviousAndNext) {
			return;
		}

				'class' => 'left carousel-control',
				'data-slide' => 'prev'
			)) .
			"\n" .
			Html::a($this->nextLabel, '#' . $this->options['id'], array(
				'class' => 'right carousel-control',
				'data-slide' => 'next'
			)) .
			"\n";
	}
