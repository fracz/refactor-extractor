	public static function unlink($path, $cb = null, $pri = EIO_PRI_DEFAULT) {
		if (!self::$supported) {
			$r = unlink($path);
			if ($cb) {
				call_user_func($cb, $path, $r);
			}
			return;
		}
		return eio_unlink($path, $pri, $cb, $path);
	}
