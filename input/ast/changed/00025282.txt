    public function getImages()
    {
        $images = array();

        if ($this->getType() === 'photo') {
            $images[] = $this->bag->get('url');
        }

        if ($this->bag->has('thumbnail_url')) {
            $images[] = $this->bag->get('thumbnail_url');
        }

        return $images;
    }
