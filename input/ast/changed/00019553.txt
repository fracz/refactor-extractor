    public static function handleException(Exception $e) {
        try {
            static::error($e);
        }
        catch (Exception $ex) {
            error_log($ex->getMessage());
            exit(
                '<h1>500 Internal Server Error</h1>'.
                '<h3>'.$ex->getMessage().'</h3>'.
                '<pre>'.$ex->getTraceAsString().'</pre>'
            );
        }
    }
