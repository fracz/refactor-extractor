	private function has($table, $idCol, $id) {
		$query = $this->dbc->getQueryBuilder();
		$query->select($idCol)
			->from($table)
			->where($query->expr()->eq($idCol, $query->createNamedParameter($id)))
			->setMaxResults(1);

		$result = $query->execute();
		$row = $result->fetch();
		$result->closeCursor();

		return (bool) $row;
	}
