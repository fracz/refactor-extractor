	function _checkMenuItems(& $items, $parts)
	{
		foreach($parts as $id => $part)
		{
			if(isset($items[$part]))
			{
				foreach($items[$part] as $item)
				{
					if (($item->published) && (@$item->link_parts['id'] == $id)) {
						return $item;
					}
				}
			}
		}

	}
