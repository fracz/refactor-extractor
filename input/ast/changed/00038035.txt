    public function make($mDataSupport = false, $orderFirst = false)
    {
        $this->m_data_support = $mDataSupport;
        $this->totalRecords   = $this->count();

        if ($orderFirst) {
            $this->ordering();
        }

        if ($this->autoFilter && $this->request->isSearchable()) {
            $this->filtering();
        }
        $this->columnSearch();
        $this->filteredRecords = $this->count();

        if ( ! $orderFirst) {
            $this->ordering();
        }
        $this->paginate();
        $this->setResults();

        return $this->render();
    }
