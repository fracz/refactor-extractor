	protected function write_cookie(\Laravel\Cookie $cookie)
	{
		if ( ! headers_sent())
		{
			$config = Config::get('session');

			extract($config);

			$minutes = ($expire_on_close) ? 0 : $lifetime;

			$cookie->put('laravel_session', $this->session['id'], $minutes, $path, $domain, $https, $http_only);
		}
	}
