    public function getRule($name)
    {
        if ($name === null) {
            return null;
        }
        $query = new Query;
        $query->select(['data'])->from($this->ruleTable)->where(['name' => $name]);
        $row = $query->createCommand($this->db)->queryOne();
        return $row === false ? null : unserialize($row['data']);
    }
