	public function validateParser($tab, $in = false, $sql)
	{
		if (!$tab)
			return false;
		else if (isset($tab['UNION']))
		{
			$union = $tab['UNION'];
			foreach ($union as $tab)
			{
				if (!$this->validateSql($tab, $in, $sql))
					return false;
			}
			return true;
		}
		else
			return $this->validateSql($tab, $in, $sql);
	}
