	protected function getSlugSource()
	{
		$config = $this->getSluggableConfig();
		$from = $config['build_from'];

		if ( is_null($from) )
		{
			return $this->__toString();
		}

		$source = array_map(
			function($attribute)
			{
				return $this->{$attribute};
			},
			(array) $from
		);

		return join($source, ' ');
	}
