    public function setPath($path)
    {
        if (is_array($path)) {
            $this->path = '/' . implode('/', $path);
        } else {
            $this->path = $path;
            if ($this->path != '*' && substr($this->path, 0, 1) != '/') {
                $this->path = '/' . $path;
            }
        }

        return $this;
    }
