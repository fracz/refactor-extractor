	public function export() {
		$className = __CLASS__;
		foreach ($this->_updated as $key => $val) {
			if ($val instanceof $className) {
				$path = $this->_pathKey ? "{$this->_pathKey}." : '';
				$this->_updated[$key]->_pathKey = "{$path}{$key}";
			}
		}
		return parent::export() + array('key' => $this->_pathKey);
	}
