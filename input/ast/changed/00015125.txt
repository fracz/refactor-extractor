	function getContentItemLinkCount( ) {
		global $database;

		$query = "SELECT COUNT( id )"
		."\n FROM #__menu "
		."\n WHERE type = 'content_item_link'"
		."\n AND published = 1"
		;
		$database->setQuery( $query );
		$count = $database->loadResult();
		return $count;
	}
