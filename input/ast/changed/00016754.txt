	public function onReadEv($bev) {
		if (!$this->ready) {
			$this->wRead = true;
			return;
		}
		if ($this->finished) {
			return;
		}
		try {
			$this->reading = !$this->onRead();
		} catch (Exception $e) {
			Daemon::uncaughtExceptionHandler($e);
		}
	}
