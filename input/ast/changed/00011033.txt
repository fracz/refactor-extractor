	protected function set_date($user_date=NULL)
	{
		$curr_date = !empty($user_date) ? $user_date : time();

		switch ($this->date_format)
		{
			case 'int':
				return $curr_date;
				break;
			case 'datetime':
				return date('Y-m-d H:i:s', $curr_date);
				break;
			case 'date':
				return date( 'Y-m-d', $curr_date);
				break;
		}

	}//end set_date()
