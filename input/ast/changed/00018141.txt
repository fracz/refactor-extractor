	private static function paths($module, $file)
	{
		$module = str_replace('.', '/', $module);

		$paths = array(CONFIG_PATH, Module::path($module).'config/');

		if (isset($_SERVER['LARAVEL_ENV']))
		{
			$paths[] = Module::path($module).'/config/'.$_SERVER['LARAVEL_ENV'].'/';
		}

		return $paths;
	}
