    public function log($message)
    {
        if ($message instanceof Elastica_Request) {
            $message = $this->_convertRequest($message);
        }

        $this->_lastMessage = $message;

        if (is_string($this->_log)) {
            error_log($message . PHP_EOL, 3, $this->_log);
        } else {
            error_log($message);
        }

    }
