	public function getForm()
	{
		// Get the form.
		try {
			$form = parent::getForm('com_config.application', 'application', array('control' => 'jform'));
		} catch (Exception $e) {
			$this->setError($e->getMessage());
			return false;
		}

		return $form;
	}
