	public function attributExistInTable($attr, $table)
	{
		if (is_array($table) && (count($table) == 1))
			$table = $table[0];
		$attributs = $this->getAttributesByTable($table);
		foreach ($attributs as $attribut)
			if ($attribut['Field'] == trim($attr))
				return true;
		return false;
	}
