    protected function add($id, $fullPath, $bottom, $priority)
    {
        if (filesize($fullPath) === 0) {
            return;
        }

        $position = $bottom ? 'bottom' : 'head';

        $this->list[$position][$id] = array(
            'id' => $id,
            'uri' => $this->getFQDN().$this->getUriFromPath($fullPath),
            'priority' => $priority,
        );
    }
