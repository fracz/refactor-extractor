    function url_get_path() {
        global $CFG;

        $adminroot = admin_get_root();

        $root = $adminroot->locate($this->section);
        if (is_a($root, 'admin_externalpage')) {
            return $root->url;
        } else {
            return ($CFG->wwwroot . '/' . $CFG->admin . '/settings.php');
        }
    }
