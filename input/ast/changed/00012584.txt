    protected function clearUserDataFromStorage()
    {
        $this->session->remove($this->getName());

        if (! is_null($this->getRecaller())) {
            $recaller = $this->getRecallerName();

            $this->getCookieJar()->queue($this->getCookieJar()->forget($recaller));
        }
    }
