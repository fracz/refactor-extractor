	public function testToContext() {
		$expected = array(
			'http' => array(
				'method' => 'GET',
				'content' => '',
				'header' => array(
					'Host: localhost:80',
					'Connection: Close',
					'User-Agent: Mozilla/5.0 (Lithium)'
				),
				'ignore_errors' => true,
				'timeout' => 1
			)
		);
		$result = $this->request->to('context');
		$this->assertEqual($expected, $result);
	}
}
