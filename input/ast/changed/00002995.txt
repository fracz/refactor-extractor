    public function exists($key)
    {
        if (!$this->isConnected()) {
            $this->connect();
        }

        $files = ftp_nlist($this->connection, dirname($this->computePath($key)));
        foreach ($files as $file) {
            if ($key === $file) {
                return true;
            }
        }

        return false;
    }
