	public static function build_cache($ids) {

		$idlist = '(' . implode(',',$ids) . ')';

		$sql = "SELECT * FROM `tag` WHERE `id` IN $idlist";
		$db_results = Dba::query($sql);

		while ($row = Dba::fetch_assoc($db_results)) {
			parent::add_to_cache('tag',$row['id'],$row);
		}

	} // build_cache
