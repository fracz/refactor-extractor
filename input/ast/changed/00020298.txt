    public function get_field_sql($sql, array $params=null) {
        if ($mrs = $this->get_recordset_sql($sql, $params, 0, 1)) {
            if ($mrs->valid()) {
                $record = $mrs->current();
                return reset($record); // first column
            }
            $mrs->close();
        }
        return false;
    }
