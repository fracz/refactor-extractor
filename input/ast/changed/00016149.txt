	function & getInstance($lang)
	{
		static $instances;

		if (!isset ($instances)) {
			$instances = array ();
		}

		if (empty ($instances[$lang])) {
			$instances[$lang] = new JLanguage($lang);
		}

		return $instances[$lang];
	}
