    public function addUniqueIndex( $table,$col1,$col2 ) {
        $r = $this->adapter->get("SHOW INDEX FROM $table");
        $name = "UQ_".$col1."_".$col2;
        if ($r) {
            foreach($r as $i) {
                if ($i["Key_name"]==$name) {
                    return;
                }
            }
        }
        $sql = "ALTER IGNORE TABLE `$table`
                ADD UNIQUE INDEX `$name` (`$col1`, `$col2`)";
        $this->adapter->exec($sql);
    }
