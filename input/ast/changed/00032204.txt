      list($subject, $str, $msgs, $charset) = $subject;
    if ($msgs != null && !is_array($msgs))

    $this->search_subject = $subject;
    $this->search_string = $str;
    $this->search_set = (array)$msgs;
    $this->search_charset = $charset;
    }
