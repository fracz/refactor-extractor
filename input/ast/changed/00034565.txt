    protected function setValueInOptGroups($value) {
        $value_exists = false;
        /** @var OptGroup $optGroup */
        foreach (array_merge(array($this->options), $this->optGroups) as $optGroup) {
            $value_exists = $optGroup->storeValue($value) || $value_exists;
            if ($value_exists) {
                $value = null;
            }
        }
        return $value_exists;
    }
