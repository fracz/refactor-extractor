    public static function getSqlAccessSite($select)
    {
        return "SELECT " . $select . "
				FROM " . Common::prefixTable('access') . " as t1
				JOIN " . Common::prefixTable('site') . " as t2 USING (idsite) " .
              " WHERE login = ?";
