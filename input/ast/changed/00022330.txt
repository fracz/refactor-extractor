    function display_field (&$form) {
        if (empty($this->field)) {
            return false;
        }

        /// Check that this field is visible to current user
        if (!$this->_is_visible()) {
            return false;
        }

        $this->display_field_add($form);
        $this->display_field_lock($form);
        $this->display_field_default($form);
        $this->display_field_required($form);

        return true;
    }
