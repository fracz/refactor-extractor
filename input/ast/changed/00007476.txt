	protected static function _locatePath($type, $params) {
		if (!isset(static::$_paths[$type])) {
			return;
		}
		foreach (static::$_paths[$type] as $path) {
			if (is_dir($path = str_replace('\\', '/', String::insert($path, $params)))) {
				return $path;
			}
		}
	}
