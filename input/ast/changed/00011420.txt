    protected function formatParameters($action)
    {
        $parameterList = array_map(
            function ($item) {
                if (is_callable($item)) {
                    return 'inline_function(...)';
                }
                if (is_array($item)) {
                    return var_export($item, true);
                }
                if (is_object($item)) {
                    return '[object]';
            //                    return var_export($item, true);
                }
                if (is_string($item)) {
                    return "'$item'";
                }
                return $item;
            },
            $action
        );
        return implode(', ', $parameterList);
    }
