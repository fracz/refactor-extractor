	public static function to($url, $status = 302, $method = 'location', $https = false)
	{
		$url = URL::to($url, $https);

		if ($method == 'location')
		{
			return static::make('', $status)->header('Refresh', '0;url='.$url);
		}
		else
		{
			return static::make('', $status)->header('Location', $url);
		}
	}
