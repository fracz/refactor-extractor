    protected function printTable(TableNode $table, $color = null)
    {
        $string = strtr('      ' . (string) $table, array("\n" => "\n      "));

        if (null !== $color) {
            $this->writeln("{+$color}$string{-$color}");
        } else {
            $this->writeln($string);
        }
    }
