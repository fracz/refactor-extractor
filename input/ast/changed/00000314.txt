    public function runScenario(Background $scenario, array $values = array())
    {
        $statuses = $this->initStatusesArray();
        $skip = false;

        foreach ($scenario->getSteps() as $step) {
            $status = $this->runStep($step, $values, $skip);
            if ('failed' === $status) {
                $skip = true;
            }
            $statuses[$status]++;
        }

        return $statuses;
    }
