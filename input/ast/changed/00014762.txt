	function _load()
	{
		global $mainframe;

		static $plugins;

		if (isset($plugins)) {
			return $plugins;
		}

		$db		= & $mainframe->getDBO();
		$user	= & $mainframe->getUser();

		if (is_object( $user )) {
			$gid = $user->get('gid');
		} else {
			$gid = 0;
		}

		$query = "SELECT id, name, folder, element, published, params"
			. "\n FROM #__plugins"
			. "\n WHERE published >= 1"
			. "\n AND access <= $gid"
			. "\n ORDER BY ordering"
			;

		$db->setQuery( $query );

		if (!($plugins = $db->loadObjectList())) {
			JError::raiseWarning( 'SOME_ERROR_CODE', "Error loading Plugins: " . $db->getErrorMsg());
			return false;
		}

		return $plugins;
	}

}
