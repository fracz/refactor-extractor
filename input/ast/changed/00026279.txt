  private function getIconOptions() {
    $options = array();

    $set = new PhabricatorProjectIconSet();
    foreach ($set->getIcons() as $icon) {
      $options[$icon->getKey()] = array(
        id(new PHUIIconView())
          ->setIconFont($icon->getIcon()),
        ' ',
        $icon->getLabel(),
      );
    }

    return $options;
  }
