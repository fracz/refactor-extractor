    public function get_option($config = ''){
        if($config==='api_key'){
            return trim(get_config('flickr', 'api_key'));
        } else {
            $options['api_key'] = trim(get_config('flickr', 'api_key'));
        }
        $options = parent::get_option($config);
        return $options;
    }
