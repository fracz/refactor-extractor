    public function make($mDataSupport = false)
    {
        $this->m_data_support = $mDataSupport;
        $this->totalRecords   = $this->count();

        if ($this->autoFilter && $this->request->isSearchable()) {
            $this->filtering();
        }

        $this->columnSearch();
        $this->filteredRecords = $this->count();

        $this->ordering();
        $this->paging();

        $this->setResults();
        $this->initColumns();
        $this->regulateArray();

        return $this->output();
