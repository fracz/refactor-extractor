	public function checkRole() {
		$modifyRole = false;
		$user = ClassRegistry::init('User')->findById($this->Auth->user('id'));
		if (isset($user['User'])) {
			$role = ClassRegistry::init('Role')->findById($user['User']['role_id']);
			if ($role['Role']['perm_modify_org']) {
				$modifyRole = true;
			}
		}
		return $modifyRole;
	}
