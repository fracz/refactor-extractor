	private function includeApiFile($fileName)
	{
		$potentialPaths = array( "plugins/". $fileName ."/API.php", );

		$found = false;
		foreach($potentialPaths as $path)
		{
			if(Zend_Loader::isReadable($path))
			{
				require_once $path;
				$found = true;
				break;
			}
		}

		if(!$found)
		{
			throw new Exception("API module $fileName not found.");
		}
	}
