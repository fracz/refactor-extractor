    protected static function splitMessage($message)
    {
        $data = trim($message);

        if (empty($data)) {
            return [];
        }

        $data = array_filter(array_map(
            function ($data) {
                if ($data[0] !== '{') {
                    $data = '{' . $data;
                }
                if ($data[strlen($data) - 1] !== '}') {
                    $data = $data . '}';
                }

                return $data;
            },
            explode('}{', $data)
        ));

        return $data;
    }
