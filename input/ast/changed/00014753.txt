	function &getUser()
	{
		/*
		 * If there is a userid in the session, load the application user
		 * object with the logged in user.
		 */
		if (JSession::get('userid')){
			$this->_user = & JUser::getInstance(JSession::get('userid'));
		} else {
			$this->_user = & JUser::getInstance();
		}
		return $this->_user;
	}
