    protected function convertToArray(array $row, $index)
    {
        $data = [];
        foreach ($row as $key => $value) {
            if (is_object($this->engine->result_object[$index])) {
                $data[$key] = $this->engine->result_object[$index]->$key;
            } else {
                $data[$key] = $value;
            }
        }

        return $data;
    }
