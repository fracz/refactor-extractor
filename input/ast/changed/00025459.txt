	protected function _getInput()
	{
		if (empty($this->getAttribute('href'))) {
			return parent::_getInput();
		}

		$element = preg_replace('~^<input(.*)/>$~', 'a\1', parent::_getInput());

		return <<<EOT
	<{$element}>{$this->_title}</a>
EOT;
	}
