	public function drainIfMatch($str) {
		if (!isset($this->bev)) {
			return false;
		}
		$l = strlen($str);
		$ll = $this->bev->input->length;
		if ($ll < $l) {
			$read = $this->read($ll);
			$this->bev->input->prepend($read);
			return strncmp($read, $str, $ll) === 0;
		}
		$read = $this->read($l);
		if ($read === $str) {
			return true;
		}
		$this->bev->input->prepend($read);
		return false;
	}
