	public function getCreditCardData($ccBrand) {
		$data = array();
		reset($this->creditCardData);
		while (list($currBrand, $ccData) = each($this->creditCardData)) {
			if ($ccBrand != $currBrand) {
				continue;
			}
			$data = $ccData;
		}
		return $data;
	}
