    public function buildForm(FormBuilderInterface $builder, array $options)
    {

        $builder->add(
            'company',
            'company_list',
            [
                'label'       => 'mautic.company.selectcompany',
                'label_attr'  => ['class' => 'control-label'],
                'attr'        => [
                    'class'    => 'form-control',
                    'tooltip'  => 'mautic.company.choose.company_descr'
                ],
                'multiple'    => false,
                'required'    => true,
                'constraints' => [
                    new NotBlank(
                        ['message' => 'mautic.company.choosecompany.notblank']
                    )
                ]
            ]
        );

        $windowUrl = $this->factory->getRouter()->generate(
            'mautic_company_action',
            [
                'objectAction' => 'new',
                'contentOnly'  => 1,
                'updateSelect' => 'campaignevent_properties_company'
            ]
        );

        $builder->add(
            'newCompanyButton',
            'button',
            [
                'attr'  => [
                    'class'   => 'btn btn-primary btn-nospin',
                    'onclick' => 'Mautic.loadNewCompanyWindow({
                    "windowUrl": "'.$windowUrl.'"
                })',
                    'icon'    => 'fa fa-plus'
                ],
                'label' => 'mautic.company.new.company'
            ]
        );
    }
