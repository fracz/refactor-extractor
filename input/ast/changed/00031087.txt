	public function userExists( $userLogin )
	{
		Piwik::checkUserHasSomeAdminAccess();
		$count = Piwik_FetchOne("SELECT count(*)
													FROM ".Piwik::prefixTable("user"). "
													WHERE login = ?", $userLogin);
		return $count != 0;
	}
