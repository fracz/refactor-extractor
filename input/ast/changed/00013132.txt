    protected static function jsonDecode($json)
    {
        $obj = json_decode($json);

        if (json_last_error() === JSON_ERROR_NONE) {
            return $obj;
        } else {
            // @todo throw an exception
            Output::err('JSON ERROR: ' . $json);
        }
    }
