	function getSubtype($subtype_id) {
		global $SUBTYPE_CACHE;

		if (!$subtype_id) {
			return '';
		}

		if ($SUBTYPE_CACHE === null) {
			$this->populateCache();
		}

		if (isset($SUBTYPE_CACHE[$subtype_id])) {
			return $SUBTYPE_CACHE[$subtype_id]->subtype;
		}

		return false;
	}
