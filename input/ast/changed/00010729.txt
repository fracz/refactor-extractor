	function beforeRedirect(&$controller, $url) {
		if (!$this->isAjax()) {
			return;
		}
		foreach ($_POST as $key => $val) {
			unset($_POST[$key]);
		}
		Router::reload();

		if (is_array($url)) {
			$url = Router::url(array_merge(array('base' => false), $url));
		}
		echo $this->requestAction($url, array('return'));
		$this->stop();
	}
