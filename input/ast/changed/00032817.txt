	public function append( $a, $b ){
		if( array_key_exists( $a, $this->vars )){
			if( is_array( $this->vars[$a] )){
				$this->vars[$a][] = $b;
			}
			else
			{
				$array = array( $this->vars[$a], $b );
				$this->vars[$a] = $array;
			}
		}
		else{
			$this->vars[$a] = $b;
		}
	}
