                        continue;

                        $message = $this->normalizeToken($tokens[$key + $id]);

                        continue;

                        break;
                    }
                }

                $message = trim($message, '\'"');

                if ($message) {
                    $catalog->set($message, $this->prefix.$message);
                    break;
                }
            }
        }
