	public function stdin($c) {
		if ($c !== '') {
			$this->attrs->stdinbuf .= $c;
			$this->attrs->stdinlen += strlen($c);
		}
		if (
			!isset($this->attrs->server['HTTP_CONTENT_LENGTH'])
			|| ($this->attrs->server['HTTP_CONTENT_LENGTH'] <= $this->attrs->stdinlen)
		) {
			$this->attrs->stdin_done = true;
			$this->postPrepare();
			if ($this->sleepTime === 0) {
				$this->wakeup();
			}
		}
		$this->parseStdin();
	}
