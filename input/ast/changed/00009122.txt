    private function parseClass()
    {
        $klass = get_class($this);

        if ($klass === __CLASS__) {
            return '';
        }

        $service = substr($klass, strrpos($klass, '\\') + 1, -6);
        $this->commandException = "Aws\\{$service}\\Exception\\{$service}Exception";

        return strtolower($service);
    }
