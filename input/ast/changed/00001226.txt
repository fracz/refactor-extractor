	protected function _backtrace()
	{
		if (function_exists('debug_backtrace'))
		{
			$back = debug_backtrace();
			return array(
					'file' => (isset($back[1]['file']) ? $back[1]['file'] : ''),
					'line' => (isset($back[1]['line']) ? $back[1]['line'] : '')
				);
		}
		return array('file' => 'Unknown', 'line' => 'Unknown');
	}
