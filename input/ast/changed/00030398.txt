	protected function request($url)
	{
		$url = $this->piwikUrl.$url . '&trigger=archivephp';
//		$this->log($url);
		$response = trim(@file_get_contents($url));

		if(empty($response)
			|| stripos($response, 'error')) {
			$this->logError("Got invalid response from $url. Response was '$response'");
		}
		return $response;
	}
