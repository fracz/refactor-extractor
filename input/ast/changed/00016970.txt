	public function write($s) {
		if (!$this->alive) {
			Daemon::log('Attempt to write to dead IOStream ('.get_class($this).')');
			return false;
		}
		if (!$this->buffer) {
			return false;
		}
		if (!strlen($s)) {
			return true;
		}
 		$this->sending = true;
		return event_buffer_write($this->buffer, $s);
	}
