	public function insert($col, $doc = [], $cb = NULL, $params = []) {
		if (strpos($col, '.') === false) {
			$col = $this->dbname . '.' . $col;
		}

		if (!isset($doc['_id'])) {
			$doc['_id'] = new \MongoId;
		}
		if ($this->safeMode) {
			static::safeModeEnc($doc);
		}
		$this->request(self::OP_INSERT,
								"\x00\x00\x00\x00"
								. $col . "\x00"
								. bson_encode($doc)
		, false, null, function ($conn, $reqId = null) use ($cb, $col, $params) {
			if ($cb !== NULL) {
				$this->lastError($col, $cb, $params, $conn);
			}
		});

		return $doc['_id'];
	}
