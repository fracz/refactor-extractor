    public function generatePreviousLink()
    {
        $html = '<span class="pagination-previous">';

        if ($this->offset > 0) {
            $html .= $this->container['helper']->a(
                '&larr; '.t('Previous'),
                $this->controller,
                $this->action,
                $this->getUrlParams($this->page - 1, $this->order, $this->direction)
            );
        }
        else {
            $html .= '&larr; '.t('Previous');
        }

        $html .= '</span>';

        return $html;
    }
