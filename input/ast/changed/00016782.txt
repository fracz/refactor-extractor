	public function sighup() {
		if (Daemon::$settings['logsignals']) {
			Daemon::log('Worker ' . $this->pid . ' caught SIGHUP (reload config).');
		}

		if (isset(Daemon::$parsedSettings['configfile'])) {
			Daemon::loadConfig(Daemon::$parsedSettings['configfile']);
		}

		$this->update = TRUE;
	}
