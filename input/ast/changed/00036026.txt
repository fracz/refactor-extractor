                    ->method('walkConstraint')
                    ->with($this->equalTo($constraint), $this->equalTo($value), $this->equalTo('MyGroup'), $this->equalTo('bar['.$key.']'));

        }

        $this->assertTrue($this->validator->isValid($data, new Collection(array(

        ))));
    }
