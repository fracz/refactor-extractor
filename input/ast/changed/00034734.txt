    public function updateEmailNotify($id)
    {
        $user = User::findOrFail($id);
        $this->authorize('update', $user);
        $user->email_notify_enabled = $user->email_notify_enabled == 'yes' ? 'no' : 'yes';
        $user->save();

        return redirect(route('users.show', $id));
    }
