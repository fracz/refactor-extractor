	function &getLanguage( )
	{
		if(is_null($this->_lang)) {
			$this->setLanguage();
		}

		$lang =& JLanguage::getInstance( $this->_lang );
		$lang->setDebug( $this->getCfg('debug') );

		return $lang;
	}
