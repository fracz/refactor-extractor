	function populateCache() {
		global $SUBTYPE_CACHE;

		$results = _elgg_services()->db->getData("SELECT * FROM {$this->CONFIG->dbprefix}entity_subtypes");

		$SUBTYPE_CACHE = array();
		foreach ($results as $row) {
			$SUBTYPE_CACHE[$row->id] = $row;
		}
	}
