    public function writetext($raw, $indent = 0, $short = true) {
        $indent = str_repeat('  ', $indent);

        // if required add CDATA tags
        if (!empty($raw) && htmlspecialchars($raw) != $raw) {
            $raw = "<![CDATA[$raw]]>";
        }

        if ($short) {
            $xml = "$indent<text>$raw</text>\n";
        } else {
            $xml = "$indent<text>\n$raw\n$indent</text>\n";
        }

        return $xml;
    }
