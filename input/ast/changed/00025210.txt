	public function __construct (Url $Url) {
		$Xml = $Url->getXMLContent();

		if ($Xml) {
			$data = self::parseRss($Xml) ?: self::parseAtom($Xml);

			if (is_array($data)) {
				$this->sourceUrl = $Url->getUrl();
				$this->providerUrl = $data['url'];
				$this->items = $data['items'];
				$this->valid = true;
			}
		}
	}
