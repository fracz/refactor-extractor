	public static function replaceIllegalChars($value) {
		$replace_pairs = array(
				'|' => '|7c|', // Needs to stay on top !
				'"' => '|22|',
				';' => '|3b|',
				':' => '|3a|',
				'\\' => '|5c|',
				'0x' => '|30 78|'
		);
		return strtr($value, $replace_pairs);
	}
