  public function saveTransaction() {
    $key = $this->getTransactionKey();
    $stack = &$this->getTransactionStack($key);

    if (!count($stack)) {
      throw new Exception(
        "No open transaction! Unable to save transaction, since there ".
        "isn't one.");
    }

    array_pop($stack);

    if (!count($stack)) {
      $this->query('COMMIT');
    }
  }
