	protected static function formatTemplateLayoutFiles($presenter)
	{
		$presenter = str_replace(':', 'Module/', $presenter);
		$module = substr($presenter, 0, (int) strrpos($presenter, '/'));
		return array(
			"$presenter/@layout.phtml",
			"$presenter.@layout.phtml",
			"$module/@layout.phtml",
			"@layout.phtml",
			"layout.phtml", // back compatibility
		);
	}
