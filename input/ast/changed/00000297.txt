    protected function getNextPyString()
    {
        $value  = '';

        if (
            $this->moveToNextLine() &&
            preg_match($this->regex->getPyStringStarterRegex(), $this->currentLine)
        ) {
            while (
                $this->moveToNextLine() &&
                !preg_match($this->regex->getPyStringStarterRegex(), $this->currentLine)
            ) {
                $value .= $this->currentLine . "\n";
            }
        }

        return trim($value);
    }
