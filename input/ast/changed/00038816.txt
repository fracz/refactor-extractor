	public function one()
	{
		// TODO add support for orderBy
		$data = $this->executeScript('One');
		if ($data === array()) {
			return null;
		}
		$row = array();
		$c = count($data);
		for($i = 0; $i < $c; ) {
			$row[$data[$i++]] = $data[$i++];
		}
		if ($this->asArray) {
			$model = $row;
		} else {
			/** @var $class ActiveRecord */
			$class = $this->modelClass;
			$model = $class::create($row);
		}
		if (!empty($this->with)) {
			$models = array($model);
			$this->populateRelations($models, $this->with);
			$model = $models[0];
		}
		return $model;
	}
