  public function getAddedLines() {
    $lines = array();
    $n = $this->newOffset;
    foreach (explode("\n", $this->changes) as $diff_line) {
      if ($diff_line == '' || $diff_line[0] == '\\') {
        continue;
      }
      if ($diff_line[0] == '+') {
        $lines[$n] = (string)substr($diff_line, 1); // substr('+', 1) === false
      }
      if ($diff_line[0] != '-') {
        $n++;
      }
    }
    return $lines;
  }
