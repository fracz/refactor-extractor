    protected function getStreamFilterPath()
    {
        $this->assertStreamable();
        if (!$this->stream_filters) {
            return $this->stream_uri;
        }

        return 'php://filter/'
            .$this->getStreamFilterPrefix()
            .implode('|', $this->stream_filters)
            .'/resource='.$this->stream_uri;
    }
