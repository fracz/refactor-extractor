    {
        $hooks = isset($this->hooks[$name]) ? $this->hooks[$name] : array();

        foreach ($hooks as $hook) {
            if ($hook instanceof FilterableHook) {
                if ($hook->filterMatches($event)) {
                    $hook->run($event);
                }
            } else {

            }
        }
