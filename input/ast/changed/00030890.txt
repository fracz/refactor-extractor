	static public function splitUrl($url)
	{
		$matches = array();
		$split_arr = array();
		$n = preg_match("#://[^/]+(/)#",$url, $matches, PREG_OFFSET_CAPTURE);
		if( $n )
		{
			$host = substr($url, 0, $matches[1][1]);

			$split_arr = array($host, $url);
		}
		else
		{
			$split_arr = array($url);
		}
		return $split_arr;
	}
