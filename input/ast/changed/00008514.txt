    {

        $id = $user->getKey();
        $idColumn = $user->getKeyName();

        $query = $this->app['db']->connection()->table($usersTable)
            ->where('email',$user->email);

        if($user->username)

            $query = $query->orWhere('username',$user->username);
        }
