    public function send()
    {
        if ($this->config['run'] == false) {
            return;
        }

        $resp = (yield $this->httpClient->post($this->config['uri'], $this->data));
        $result = $resp->getResponseJson();

        $this->data = [];

        if (!isset($result['code']) || $result['code'] != self::SUCCESS) {
            //TODO: 上报失败
            var_dump("hawk上报失败");
        }
    }
