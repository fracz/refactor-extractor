  function loadConfig(){

    $path = DOKU_PLUGIN.$this->getPluginName().'/conf/';
    $conf = array();

    if (!@file_exists($path.'default.php')) return false;

    // load default config file
    include($path.'default.php');

    return $conf;
  }
