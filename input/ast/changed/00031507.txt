    public function getTableColumns($tableName)
    {
        $db = Db::get();

        $allColumns = $db->fetchAll("SHOW COLUMNS FROM . $tableName");

        $fields = array();
        foreach ($allColumns as $column) {
            $fields[trim($column['Field'])] = $column;
        }

        return $fields;
    }
