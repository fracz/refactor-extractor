    function parse_co($qrec, $question)
    {
        $question->usecase = 0;
        $answer = trim($qrec['answer'][0]['#']);
        $answers = explode("\n",$answer);

        foreach($answers as $key => $value) {
            $value = trim($value);
            if (strlen($value) > 0) {
                $question->answer[$key] = addslashes($value);
                $question->fraction[$key] = 1;
                $question->feedback[$key] = "Correct";
            }
        }
        return $question;
    }
