    public function getDefaultOptions(array $options)
    {
        $defaultOptions = array(
            'template' => 'repeated',
            'first_key' => 'first',
            'second_key' => 'second',
        );

        // Lazy creation of the prototype
        if (!isset($options['prototype'])) {
            $defaultOptions['prototype'] = $this->getInstance('text');
        }

        return $defaultOptions;
    }
