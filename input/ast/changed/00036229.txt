    public function accept()
    {
        $fileinfo = $this->getInnerIterator()->current();

        if (!$fileinfo->isFile()) {
            return true;
        }

        $filedate = $fileinfo->getMTime();

        if (
            (false !== $this->minDate && $filedate < $this->minDate)
            ||
            (false !== $this->maxDate && $filedate > $this->maxDate)
        ) {
            return false;
        }

        return true;
    }
