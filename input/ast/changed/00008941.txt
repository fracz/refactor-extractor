    protected function attachApiFilters(Route $route)
    {
        $filters = $route->beforeFilters();

        foreach (['api.auth', 'api.throttle'] as $filter) {
            if (! isset($filters[$filter])) {
                $route->before($filter);
            }
        }

        return $route;
    }
