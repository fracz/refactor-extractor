    public function testMultipleHooks()

            ->setMethods(array('render'))
            ->getMock();

            )
            ->will($this->returnValue('tpl1_content'));

            ->expects($this->at(1))
            ->method('render')
            ->with(
                $this->equalTo('tpl2'),
