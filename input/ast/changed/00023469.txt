	final public function render()
	{
		$s = $this->startTag();

		// empty elements are finished now
		if ($this->isEmpty) {
			return $s;
		}

		// add content
		foreach ($this->children as $child) {
			if (is_object($child)) {
				$s .= $child->render();
			} else {
				$s .= $child;
			}
		}

		// add end tag
		return $s . $this->endTag();
	}
