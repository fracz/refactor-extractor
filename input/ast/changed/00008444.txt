	public function getErrorList() {
		$asp_enabled   = (boolean) ini_get( 'asp_tags' );
		$short_enabled = (boolean) ini_get( 'short_open_tag' );

		$errors = array(
			6 => 1,
		);

		if ( true === $asp_enabled ) {
			$errors[4] = 1;
		}
		if ( true === $asp_enabled && ( true === $short_enabled || defined( 'HHVM_VERSION' ) === true ) ) {
			$errors[5] = 1;
		}

		return $errors;
	} // end getErrorList()
