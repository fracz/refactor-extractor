	public static function get($id, $meta = 0, $count = 1){
		if(isset(self::$list[$id])){
			$class = self::$list[$id];
			if($id < 256){
				$item = new ItemBlock(new $class($meta), $meta, $count);
			}else{
				$item = new $class($meta, $count);
			}
		}else{
			$item = new Item($id, $meta, $count);
		}

	}
