    public function render($__template_name, array $__template_args = array())
    {
        $this->beforeRender($__template_name);

        extract($__template_args);
        ob_start();
        include $this->getTemplateFile($__template_name);
        $html = ob_get_clean();

        $this->afterRender($__template_name);

        return $html;
    }
