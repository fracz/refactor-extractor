	public static function firstWildcardInPattern($pattern)
	{
		$wildcards = array('*','?','[','\\');
		$wildcardSearch = function($r, $c) use ($pattern) {
			$p = strpos($pattern, $c);
			return $r===false ? $p : ($p===false ? $r : min($r, $p));
		};
		return array_reduce($wildcards, $wildcardSearch, false);
	}
