	function _updateReport($idReport, $data)
	{
		//$idReport, $idSite, $description, $period, $reports, $emailMe = true, $additionalEmails = false)
		$idReport = Piwik_PDFReports_API::getInstance()->updateReport(
			$idReport,
			$data['idsite'],
			$data['description'],
			$data['period'],
			$data['format'],
			$data['display_format'],
			$data['reports'],
			$data['email_me'],
			$data['additional_emails']);
		return $idReport;
	}
