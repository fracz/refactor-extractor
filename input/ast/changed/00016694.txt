	public static function callAutoGC() {
		if (
			(Daemon::$config->autogc->value > 0)
			&& (Daemon::$process->counterGC > 0)
			&& (Daemon::$process->counterGC % Daemon::$config->autogc->value === 0)
		) {
			gc_collect_cycles();
			++Daemon::$process->counterGC;
		}
	}
