	private function _hash($input, $alphabet) {

		$ret = '';
		$alphabet_length = strlen($alphabet);

		do {
			$rem = $input % $alphabet_length;
			$input = (int)($input / $alphabet_length);
			$ret = $alphabet[$rem] . $ret;
		} while ($input);

		return $ret;

	}
