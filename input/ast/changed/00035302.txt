    public static function getMaxFilesize()
    {
        $max = trim(ini_get('upload_max_filesize'));

        if ('' === $max) {
            return PHP_INT_MAX;
        }

        if (preg_match('#^(\d+)([bkmgt])#i', $max, $match)) {
            $shift = array('b' => 0, 'k' => 10, 'm' => 20, 'g' => 30, 't' => 40);
            $max = ($match[1] * (1 << $shift[strtolower($match[2])]));
        }

        return (integer) $max;
    }
