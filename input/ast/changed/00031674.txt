    protected function setUp()
    {
        $this->io = new IO\NullIO;
        $this->config = $this->prophesize('Composer\Config')
                ->get('cache-files-dir')
                ->willReturn('tests/workspace/')

                ->get('prestissimo')
                ->willReturn(null)
            ->getObjectProphecy()
            ->reveal();
        $this->composer = new Composer($this->io);
        $this->composer->setConfig($this->config);
    }
