    public function open($base_path = '/')
    {
        session_set_cookie_params(
            self::SESSION_LIFETIME,
            $base_path,
            null,
            isset($_SERVER['HTTPS']) && $_SERVER['HTTPS'] === 'on',
            true
        );

        session_start();
    }
