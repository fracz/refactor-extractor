	function tearDown()
	{

		\OC_FileProxy::$enabled = true;
		\OC_FileProxy::clearProxies();

		// reset app files_trashbin
		if ($this->stateFilesTrashbin) {

		} else {
			OC_App::disable('files_trashbin');
		}
	}
