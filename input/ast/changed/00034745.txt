  {
    /* -- RFC 2046, 5.1.1.

    $headers = $this->_createHeaderSet(array(), false);

    $child1 = $this->_createChild(Swift_Mime_MimeEntity::LEVEL_ALTERNATIVE,

      "foobar"
      );

    $child2 = $this->_createChild(Swift_Mime_MimeEntity::LEVEL_ALTERNATIVE,
      "Content-Type: text/html\r\n" .
      "\r\n" .
      "<b>foobar</b>"
      );
