  public function renderPropertyViewValue() {
    $user = $this->getObject();

    $roles = array();
    if ($user->getIsAdmin()) {
      $roles[] = pht('Administrator');
    }
    if ($user->getIsDisabled()) {
      $roles[] = pht('Disabled');
    }
    if ($user->getIsSystemAgent()) {
      $roles[] = pht('Bot');
    }

    if ($roles) {
      return implode(', ', $roles);
    }

    return null;
  }
