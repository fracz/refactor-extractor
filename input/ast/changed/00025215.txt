	public function getImages () {
		$images = array();

		foreach ($this->providers as $Provider) {
			$imgs = (array)$Provider->getImage();

			foreach ($imgs as $imgs) {
				if (!empty($imgs)) {
					$images[] = $this->Url->getAbsolute($imgs);
				}
			}
		}

		return array_unique($images);
	}
