    {
        if (null !== $threshold) {
            $threshold = $this->filterMinRange($threshold, 1, 'The flush threshold must be a valid positive integer or null');
        }

        if ($threshold === $this->flush_threshold) {
            return $this;
        }

        $this->flush_threshold = $threshold;
