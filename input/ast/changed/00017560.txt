    public function searchVideosRelatedToSong(Song $song, $pageToken = '')
    {
        $q = $song->title;

        // If the artist is worth noticing, include them into the search.
        if (!$song->artist->isUnknown() && !$song->artist->isVarious()) {
            $q .= ' '.$song->artist->name;
        }

        return $this->search($q, $pageToken);
    }
