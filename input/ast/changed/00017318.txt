    public function remove()
    {
        $this->checkCSRFParam();
        $link = $this->getLink();

        if ($this->link->remove($link['id'])) {
            $this->session->flash(t('Link removed successfully.'));
        }
        else {
            $this->session->flashError(t('Unable to remove this link.'));
        }

        $this->response->redirect($this->helper->url('link', 'index'));
    }
