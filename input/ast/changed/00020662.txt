    public function load_required_main_version() {

        if (empty($this->rootdir)) {
            return;
        }

        $versionfile = $this->rootdir . '/version.php';

        if (is_readable($versionfile)) {
            include($versionfile);
            if (isset($module->requires)) {
                $this->versionrequires = $module->requires;
            }
        }
    }
