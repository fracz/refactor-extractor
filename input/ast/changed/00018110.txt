	private function link($page, $text, $class)
	{
		$append = '';

		foreach ($this->append as $key => $value)
		{
			$append .= '&'.$key.'='.$value;
		}

		return HTML::link(Request::uri().'?page='.$page.$append, $text, array('class' => $class), Request::is_secure());
	}
