	function getid3_tar(&$fd, &$ThisFileInfo) {
		$ThisFileInfo['fileformat'] = 'tar';

		@fseek($fd, 0);
		$filebuffer = @fread($fd, $ThisFileInfo['filesize']);
		return $this->read_tar($filebuffer, $ThisFileInfo);
