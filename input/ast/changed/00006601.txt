	public function setCompareDateTo($value)
	{
		if (!isset($value) || $value == '')
			$value = date('Y-m-d');

		if (!is_string($value))
			throw new PrestaShopException('Date must be a string');

		$this->_compare_date_to = $value;
		return $this;
	}
