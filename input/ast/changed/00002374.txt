	public function __destruct() {
		global $CONFIG;

		foreach ($this->entities as $e) {
			$e->delete();
		}

		// manually remove subtype entries since there is no way
		// to using the API.
		$subtype_arr = array();
		foreach ($this->subtypes as $type => $subtypes) {
			foreach ($subtypes as $subtype) {
				remove_subtype($type, $subtype);
			}
		}

		parent::__destruct();
	}
