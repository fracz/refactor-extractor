	function _reindex_segments()
	{
		$i = 1;

		foreach ($this->segments as $val)
		{
			$this->segments[$i++] = $val;
		}

		unset($this->segments[0]);

		$i = 1;

		foreach ($this->rsegments as $val)
		{
			$this->rsegments[$i++] = $val;
		}

		unset($this->rsegments[0]);
	}
