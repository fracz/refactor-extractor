    public static function persistCache()
    {
        if (self::$isDirty) {
            if (SettingsServer::isTrackerApiRequest()) {
                $mode = 'tracker';
            } else {
                $mode = 'ui';
            }

            self::getStorage()->set('StaticCache-' . $mode, self::$content);
        }
    }
