    public function fetch($annotation)
    {
        $docBlock = $this->currentReflectedItem->getDocComment();
        $matched = array();
        $res = preg_match(sprintf(self::$regex, $annotation), $docBlock, $matched);
        if (!$res) return null;
        return $matched[1];
    }
