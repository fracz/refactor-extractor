	public function cutJoin($attrs, $from)
	{
		$attrs = explode('=', str_replace(' ', '', $attrs));
		foreach ($attrs as $attr)
		{
			if ($attribut = $this->cutAttribute($attr, $from))
				$tab[] = $attribut;
			else
				return false;
		}
		return $tab;
	}
