	public static function get($id, $meta = 0, Position $pos = null){
		if(isset(self::$list[$id])){
			$block = self::$list[$id];
			$block = new $block($meta);
		}else{
			$block = new Block($id, $meta);
		}

		if($pos !== null){
			$block->x = $pos->x;
			$block->y = $pos->y;
			$block->z = $pos->z;
			$block->level = $pos->level;
		}

		return $block;
