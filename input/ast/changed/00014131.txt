        public function delete() {
            $query = array();
            $query[] = "DELETE FROM";
            $query[] = $this->table_name;
            $query[] = "WHERE";
            $query[] = $this->get_id_column_name();
            $query[] = "= ?";
            $query = join(" ", $query);
            self::setup_db();
            $statement = self::$db->prepare($query);
            return $statement->execute(array($this->id()));
        }
