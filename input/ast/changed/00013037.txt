    public function load($type, $id) {
		$bean = $this->dispense( $type );
        $row =  $this->writer->selectRecord($type,$id);
		if (!$row) return $this->dispense($type);
        foreach($row as $p=>$v) {
            //populate the bean with the database row
                $bean->$p = $v;
        }
        $this->signal( "open", $bean );

		return $bean;
    }
