    public function params($mask = null) {
        global $__params;
        $params = $__params;
        if (null !== $mask) {
            if (!is_array($mask)) {
                $mask = func_get_args();
            }
            $params = array_intersect_key($params, array_flip($mask));
            //Make sure each key in $mask has at least a null value
            foreach ($mask as $key) {
                if (!isset($params[$key])) $params[$key] = null;
            }
        }
        return $params;
    }
