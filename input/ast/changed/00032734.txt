	public static function getUserGroups($uid) {
		$user = self::$userManager->get($uid);
		if ($user) {
			$groups = self::getManager()->getUserGroups($user);
			$groupIds = array();
			foreach ($groups as $group) {
				$groupIds[] = $group->getGID();
			}
			return $groupIds;
		} else {
			return array();
		}
	}
