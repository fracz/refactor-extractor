    public function update(array $values)
    {
        $this->prepare($values);
        $result = $this->db->table(self::TABLE)->eq('id', $values['id'])->update($values);

        // If the user is connected refresh his session
        if (Session::isOpen() && $_SESSION['user']['id'] == $values['id']) {
            $this->updateSession();
        }

        return $result;
    }
