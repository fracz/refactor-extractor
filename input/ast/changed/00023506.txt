	public function canonicalize()
	{
		$this->path = $this->path == '' ? '/' : self::unescape($this->path);

		$this->host = strtolower($this->host);

		if ($this->query !== '') {
			$tmp = explode('&', self::unescape($this->query));
			sort($tmp);
			$this->query = implode('&', $tmp);
		}
	}
