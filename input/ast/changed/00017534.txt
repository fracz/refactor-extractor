    public function getPreferencesAttribute($value)
    {
        $preferences = unserialize($value) ?: [];

        // Hide the user's secrets away!
        foreach ($this->hiddenPreferences as $key) {
            if (isset($preferences[$key])) {
                $preferences[$key] = 'hidden';
            }
        }

        return $preferences;
    }
