	protected function _headerDetector($detect) {
		$headers = getallheaders();
		if (isset($headers['Accept'])) {
			$headers = explode(',', $headers['Accept']);
			foreach ($detect['header'] as $header) {
				if (in_array($header, $headers)) {
					return true;
				}
			}
		}
		return false;
	}
