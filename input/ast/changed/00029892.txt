	function renderTable($table)
	{
		$renderer = new Piwik_DataTable_Renderer_PHP($table, $serialize = false);
		$array = $renderer->render();
		$xmlStr = $this->array_to_simplexml($array)->asXML();
		$xmlStr = str_replace("<","\n<",$xmlStr);

		return $xmlStr;
	}
