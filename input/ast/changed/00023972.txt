	public function getHtmlId()
	{
		if ($this->htmlId === FALSE) {
			return NULL;

		} elseif ($this->htmlId === NULL) {
			$this->htmlId = str_replace(
				array('%f', '%n'),
				array($this->getForm()->getName(), $this->getHtmlName()),
				self::$idMask
			);
			$this->htmlId = str_replace(array('[', ']'), array('-', ''), $this->htmlId);
		}
		return $this->htmlId;
	}
