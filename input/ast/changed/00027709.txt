    public static function build(OptionsFragment $fragment)
    {
        $options = array();
        foreach ($fragment->options as $option) {
            if (is_array($option)) {
                $options[] = $option['name'] . '=' . $option['value'];
            } else {
                $options[] = $option;
            }
        }
        return implode(' ', $options);
    }
