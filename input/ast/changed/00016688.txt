	public function applyConfig() {
		parent::applyConfig();
		if (isset($this->config->servers)) {
			$servers = array_filter(array_map('trim',explode(',', $this->config->servers->value)), 'strlen');
			foreach ($servers as $s) {
				if (strpos($s, '://') !== false) {
					$this->addServer($s);
				} else {
					$e = explode(':', trim($s));
					$this->addServer($e[0], isset($e[1]) ? $e[1] : NULL);
				}
			}
		}
	}
