	public function get($key)
	{
		$key = strtolower( $key );

		if( isset( $_SESSION["HA::STORE"], $_SESSION["HA::STORE"][$key] ) ){
			return unserialize( $_SESSION["HA::STORE"][$key] );
		}

		return NULL;
	}
