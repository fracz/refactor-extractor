    public static function requestGithubApi($url, $maxAge = self::DEFAULT_CACHE_MAX_AGE)
    {
        $paramStart = false === strpos($url, '?') ? '?' : '&';
        $url = 'https://api.github.com' . $url . $paramStart . 'per_page=100';
        $content = self::requestCache($url, $maxAge) ?: array();
        return $content;
    }
