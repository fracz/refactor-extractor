	function &getTable($type = null)
	{
		static $tabletype;

		//Set the default tabletype;
		if(!isset($tabletype)) {
			$tabletype = 'user';
		}

		//Set a custom table type is defined
		if(isset($type)) {
			$tabletype = $type;
		}

		// Create the user table object
		$table 	=& JTable::getInstance( $tabletype);
		return $table;
	}
