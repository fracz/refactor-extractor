        $table->add_field('id', XMLDB_TYPE_INTEGER, '10', XMLDB_UNSIGNED, XMLDB_NOTNULL, XMLDB_SEQUENCE, null, null, null);

        $dbman->create_table($table);
        $this->tables[$table->getName()] = $table;

        $active = $DB->begin_sql();
