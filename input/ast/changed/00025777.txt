  final public function linkHistory($path) {
    $href = $this->getDiffusionRequest()->generateURI(
      array(
        'action' => 'history',
        'path'   => $path,
      ));

    return phutil_tag(
      'a',
      array(
        'href' => $href,
      ),
      pht('History'));
  }
