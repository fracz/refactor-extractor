	public static function set($key, $value)
	{
		list($file, $key) = static::parse($key);

		static::load($file);

		if (is_null($key))
		{
			Arr::set(static::$items, $file, $value);
		}
		else
		{
			Arr::set(static::$items[$file], $key, $value);
		}
	}
