    public function link($reset = true)
    {
        if ($this->mode !== 'source') {
            $this->display('source');
        }

        $this->linkAttributes['href'] = $this->url();

        $this->thumbnail('auto');
        $thumb = $this->display('thumbnail');
        $thumb->linked = true;

        return $thumb;
    }
