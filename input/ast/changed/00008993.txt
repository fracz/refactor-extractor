    public function isProtected()
    {
        $middleware = $this->getMiddleware();

        if (isset($middleware['api.auth'])) {
            if (! $this->usesController()) {
                return true;
            }

            return $this->optionsApplyToControllerMethod($middleware['api.auth']);
        }

        return false;
    }
