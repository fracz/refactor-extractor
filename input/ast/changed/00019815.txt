    public function readLong()
    {
        $str = $this->_fread(8);

        /**
         * PHP uses long as largest integer. fseek() uses long for offset.
         * long has 4 bytes in a lot of systems. 4 bytes are discarded to prevent
         * conversion to float.
         * So, largest index segment file is 2Gb
         */
        return  /* ord($str{0}) << 56  | */
                /* ord($str{1}) << 48  | */
                /* ord($str{2}) << 40  | */
                /* ord($str{3}) << 32  | */
                ord($str{4}) << 24  |
                ord($str{5}) << 16  |
                ord($str{6}) << 8   |
                ord($str{7});
    }
