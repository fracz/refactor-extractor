	protected function makePost($title, $subtitle=null, $slug=null)
	{
		$post = new Post;
		$post->title = $title;
		if ($subtitle)
		{
			$post->subtitle = $subtitle;
		}
		if ($slug)
		{
			$post->slug = $slug;
		}
		return $post;
	}
