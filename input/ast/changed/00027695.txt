    public static function build(ArrayFragment $fragment)
    {
        $values = array();
        if (!empty($fragment->raw)) {
            $values = $fragment->raw;
        } else {
            foreach ($fragment->values as $value) {
                $values[] = $value;
            }
        }
        return '(' . implode(', ', $values) . ')';
    }
