    public function nonNullableClone() : UnionType
    {
        $result = new UnionType();
        foreach ($this->getTypeSet() as $type) {
            if (!$type->getIsNullable()) {
                $result->addType($type);
                continue;
            }
            if ($type === NullType::instance(false)) {
                continue;
            }

            $result->addType($type->withIsNullable(false));
        }
        return $result;
    }
