	private function prepare($query, $parameters) {
		if(!$parameters)
		{
			$parameters = array();
		}
		foreach($parameters as $i=>$p)
		{
			$parameters[$i] = addslashes($p);
		}
		$query = str_replace('?', "'%s'", $query);
		array_unshift($parameters, $query);
		$query = call_user_func_array('sprintf', $parameters);
		return $query;
	}
