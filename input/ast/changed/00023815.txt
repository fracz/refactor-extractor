	public function storeRequest()
	{
		$session = Environment::getSession('Nette.Application.Request');
		do {
			$key = /*Nette::*/Tools::uniqueId();
		} while (isset($session->rq[$key]));

		$session->rq[$key] = end($this->requests);
		$session->setExpiration(10 * 60, 'rq');
		return $key;
	}
