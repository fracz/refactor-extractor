	public function addTemplate($name, $file)
	{
		if ($file instanceof self) {
			$this->add($name, $file);
			return $file;

		} else {
			$tpl = clone $this;
			$this->absolutize($file);
			$tpl->setFile($file);
			$tpl->params = & $this->params;
			$this->add($name, $tpl);
			return $tpl;
		}
	}
