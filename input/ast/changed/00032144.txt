    function countMessages($mailbox, $refresh = false)
    {
	    if ($refresh) {
		    $this->selected = '';
	    }

	    $this->select($mailbox);
	    if ($this->selected == $mailbox) {
		    return $this->exists;
	    }

        return false;
    }
