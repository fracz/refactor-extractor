    function validation($data){
        $errors = array();
        $answers = $data['answer'];
        $answercount = 0;
        foreach ($answers as $answer){
            $trimmedanswer = trim($answer);
            if ($trimmedanswer!=''){
                $answercount++;
            }
        }
        if ($answercount==0){
            $errors['answer[0]'] = get_string('notenoughanswers', 'qtype_numerical');
        }
        return $errors;
    }
