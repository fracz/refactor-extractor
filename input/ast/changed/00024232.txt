	public function setExpiration($seconds, $variables = NULL)
	{
		if (is_string($seconds) && !is_numeric($seconds)) {
			$seconds = strtotime($seconds);
		}

		if ($seconds <= 0) {
			$seconds = 0;

		} elseif ($seconds <= /*Nette\*/Tools::YEAR) {
			$seconds += time();
		}

		if ($variables === NULL) {
			// to entire namespace
			$this->meta['EXP'][''] = $seconds;

		} elseif (is_array($variables)) {
			// to variables
			foreach ($variables as $variable) {
				$this->meta['EXP'][$variable] = $seconds;
			}

		} else {
			$this->meta['EXP'][$variables] = $seconds;
		}
	}
