    public function getSessionKey($token)
    {
        $query = $this->buildAuthCallParams([
            'method' => 'auth.getSession',
            'token' => $token,
        ], true);

        try {
            $response = $this->get("/?$query", [], false);

            return (string) $response->session->key;
        } catch (Exception $e) {
            Log::error($e);

            return false;
        }
    }
