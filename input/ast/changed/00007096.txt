	public function out($str = null, $newlines = 1, $style = "") {
		if (is_array($str)) {
			foreach ($str as $string) {
				$this->out($string, $newlines);
			}
			return;
		}
		if($style !== "") {
			$str = "{:$style}" . $str . "{:end}";
		}
		if ($newlines) {
			$str = $str . str_pad("\n", $newlines, "\n");
		}
		return $this->response->output($str);
	}
