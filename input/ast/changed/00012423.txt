  protected function matchLabel($name = 'foo', $required = false)
  {
    $text = str_replace('[]', null, ucfirst($name));
    if ($required) $text .= '*';

    return array(
      'tag' => 'label',
      'content' => $text,
      'attributes' => array(
        'for'   => $name,
        'class' => 'control-label',
      ),
    );
  }
