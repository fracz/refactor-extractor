  public function getRef() {
    $group = null;
    if ($this->docblockMeta) {
      $group = $this->getDocblockMetaValue('group');
    }

    return id(new DivinerAtomRef())
      ->setBook($this->getBook())
      ->setContext($this->getContext())
      ->setType($this->getType())
      ->setName($this->getName())
      ->setGroup($group);
  }
