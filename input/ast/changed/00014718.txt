	function show(&$model, &$access, &$menu)
	{
		$type = JRequest::getVar( 'type', 'html' );

		require_once (dirname(__FILE__).DS.'view'.DS.'blog'.DS.'blog.'.$type.'.php');

		/*
		 * Need to cache this for speed
		 */
		JViewBlog::show($model, $access, $menu);
	}
