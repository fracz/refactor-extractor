    public static function createFromID($id, $changefreq = null, $priority = null) {
        $id = trim($id);
        $date = @filemtime(wikiFN($id));
        if(!$date) return NULL;
        return new SitemapItem(wl($id, '', true), $date, $changefreq, $priority);
    }
