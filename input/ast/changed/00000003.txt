	public function resolve($path)
	{
		$absolutePath = $this->rootDir . '/' . ltrim($path, DIRECTORY_SEPARATOR);
		if (file_exists($absolutePath)) {
			return $absolutePath;
		}

		throw new ConfigurationException('Config "' . $path . "' was not found");
	}
