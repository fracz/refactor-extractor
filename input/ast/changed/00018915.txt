    public function requireCategoryPageBySlug($slug)
    {
        $model = $this->model->where('slug', '=', $slug)->first();

        if ( ! $model) {
            throw new EntityNotFoundException("Could not find forum category: {$slug}");
        }

        return $model;
    }
