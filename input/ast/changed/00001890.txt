    public function load($sql)
    {
        $query = "";
        foreach ($sql as $sql_line) {

            $parsed = $this->sqlLine($sql_line);
            if ($parsed) continue;

            $query .= $sql_line;

            if (substr(rtrim($query), -1, 1) == ';') {
                $this->sqlQuery($query);
                $query = "";
            }
        }
    }
