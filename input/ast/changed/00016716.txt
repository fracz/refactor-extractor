	public function close() {
		if (isset($this->event)) {
			event_del($this->event);
			event_free($this->event);
			$this->event = null;
		}
		if (isset($this->buffer)) {
			event_buffer_free($this->buffer);
			$this->buffer = null;
		}
		if (isset($this->fd)) {
			$this->closeFd();
		}
	}
