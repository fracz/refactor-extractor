	public function postFinishHandler() {
		if (!$this->headers_sent) {
			$this->out('');
		}
		$this->sendfp = null;
		if (isset($this->attrs->files)) {
			foreach ($this->attrs->files as &$f) {
				if (
					($f['error'] === UPLOAD_ERR_OK)
					&& file_exists($f['tmp_name'])
				) {
					FS::unlink($f['tmp_name']);
				}
			}
		}
		if (isset($this->attrs->session)) {
			$this->sessionCommit();
		}
	}
