	public function addJob($name, $cb) {
		if (isset($this->jobs[$name])) {
			return false;
		}
		$this->jobs[$name] = $cb;
		++$this->jobsNum;
		if (($this->state === self::STATE_RUNNING) || ($this->state === self::STATE_DONE)) {
			$cb($name, $this);
		}
		return true;
	}
