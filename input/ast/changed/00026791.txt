    protected function do_close()
    {
        if (isset($this->input) && $this->input) {
            $this->input->close();
            $this->input = null;
        }

        if (is_resource($this->sock)) {
            if ($this->debug) {
              MiscHelper::debug_msg("closing socket");
            }

            @fclose($this->sock);
            $this->sock = null;
        }
    }
