    public function getUsers(array $userLogins)
    {
        $where = '';
        $bind  = array();

        if (!empty($userLogins)) {
            $where = 'WHERE login IN (' . Common::getSqlStringFieldsArray($userLogins) . ')';
            $bind  = $userLogins;
        }

        $users = Db::get()->fetchAll("SELECT * FROM " . $this->table . "
                                      $where
                                      ORDER BY login ASC", $bind);

        return $users;
    }
