    public function authCheck(Event $event)
    {
        if (isset($this->earlyAuthTest)) {
            if ($this->_config['checkAuthIn'] !== $event->name()) {
                return;
            }
            $this->authCheckCalledFrom = $event->name();

            return;
        }

        return parent::authCheck($event);
    }
