    private function formatResponse(Tracker $tracker)
    {
        $result = array(
            'status' => 'success',
            'tracked' => $tracker->getCountOfLoggedRequests(),
            'invalid' => count($this->invalidRequests),
        );

        if ($this->isAuthenticated) {
            $result['invalid_indices'] = $this->invalidRequests;
        }

        return $result;
    }
