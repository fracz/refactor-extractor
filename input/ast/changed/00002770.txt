    public function getSourcePath($line = null)
    {
        if (null === $this->project) {
            return;
        }

        if (!$root = $this->project->getSourceRoot()) {
            return;
        }

        if (null === $this->plainFile) {
            return;
        }

        $url = $root . $this->plainFile;

        if (null !== $line) {
            $url .= '#L' . (int) $line;
        }

        return $url;
    }
