    private function setPath($path)
    {
        if ($this->getScheme() === 'data') {
            $this->info['content'] = $path;
            $this->info['path'] = $this->info['file'] = $this->info['extension'] = null;

            return;
        }

        $file = substr(strrchr($path, '/'), 1);

        if ($file !== '') {
            $path = substr($path, 0, -strlen($file));
        }

        if ($file) {
            if (preg_match('/(.*)\.([\w]+)$/', $file, $match)) {
                $this->info['file'] = $match[1];
                $this->info['extension'] = $match[2];
            } else {
                $this->info['file'] = $file;
                $this->info['extension'] = null;
            }
        }

        $this->info['path'] = [];

        foreach (explode('/', $path) as $dir) {
            $dir = trim(urldecode($dir));

            if ($dir !== '') {
                $this->info['path'][] = $dir;
            }
        }
    }
