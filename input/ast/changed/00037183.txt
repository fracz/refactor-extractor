    {
        if (! $this->iterator_sort_by) {
            return $iterator;

        $res = iterator_to_array($iterator, false);

        uasort($res, function ($rowA, $rowB) {
            $res = 0;
            foreach ($this->iterator_sort_by as $callable) {
                $res = $callable($rowA, $rowB);
                if (0 !== $res) {
                    break;
                }
            }

            return $res;
        });

        $this->clearSortBy();

        return new ArrayIterator($res);
    }
