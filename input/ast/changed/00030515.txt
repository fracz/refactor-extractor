    public static function sortByRequiredFields($a, $b)
    {
        $fields = $a->getRequiredVisitFields();

        if (empty($fields)) {
            return -1;
        }

        if (in_array($b->columnName, $fields)) {
            return 1;
        }

        return 0;
    }
