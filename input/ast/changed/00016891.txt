	public static function realpath($path, $cb, $pri = EIO_PRI_DEFAULT) {
		if (!self::$supported) {
			call_user_func($cb, $path, realpath($path));
			return;
		}
		return eio_realpath($path, $pri, $cb, $path);
	}
