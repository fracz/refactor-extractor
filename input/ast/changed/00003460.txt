	public function log()
	{
		$stack = [];

		for ($index = 0, $length = count($this->logs); $index < $length; $index++)
		{
			$log = $this->logs[ count($this->logs) - 1 ];

			$stack[] = $this->generate($log[ 0 ], $log[ 1 ]);
		}

		return $stack;
	}
