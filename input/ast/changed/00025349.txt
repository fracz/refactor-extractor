    public function getNote($note_id)
    {

        $sth = $this->db->prepare("SELECT * FROM note WHERE user_id = :user_id AND note_id = :note_id;");
        $sth->execute(array(
            ':user_id' => $_SESSION['user_id'],
            ':note_id' => $note_id));

        return $sth->fetch();
    }
