	public function parseUrl($url) {
		if (strpos($url, '://') !== false) { // URL
			$u = parse_url($url);
			if (isset($u['host']) && (substr($u['host'], 0, 1) === '[')) {
				$u['host'] = substr($u['host'], 1, -1);
			}
			if (!isset($u['port'])) {
				$u['port'] = $this->pool->config->port->value;
			}
		} else {
			$e = explode(':', $url, 2);
			$u = array(
				'scheme' => 'tcp',
				'host' => $e[0],
				'port' => isset($e[1]) ? $e[1] : $this->pool->config->port->value,
			);
		}
		return $u;
	}
