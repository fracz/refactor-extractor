	public function __toString()
	{
		$args = func_get_args();
		ob_start();
		try {
			call_user_func_array(array($this, 'render'), $args);
			return ob_get_clean();

		} catch (\Exception $e) {
			ob_end_clean();
			if ($args && $args[0]) {
				throw $e;
			} else {
				Nette\Diagnostics\Debugger::toStringException($e);
			}
		}
	}
