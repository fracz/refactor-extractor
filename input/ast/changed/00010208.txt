	public function orWhere($conditions, $types = []) {
		$where = $this->_parts['where'];

		if ($where->type() === 'OR') {
			$where->add($conditions, $types);
		} else {
			$where = new QueryExpression([$conditions, $where], $types, 'OR');
		}

		$this->_parts['where'] = $where;
		$this->_dirty = true;
		return $this;
	}
