	public function getLabel()
	{
		$label = clone $this->label;
		$label->for = $this->getHtmlId();
		$text = $this->getOption('label');
		if (is_string($text)) {
			if ($this->getTranslator() !== NULL) {
				$text = $this->getTranslator()->translate($text);
			}
			$label->setText($text);
		} else {
			$label->add($text);
		}
		return $label;
	}
