    public function transaction(\Closure $callback, $isolationLevel = null)
    {
        $transaction = $this->beginTransaction($isolationLevel);

        try {
            $result = $callback($transaction);
            $transaction->commit();
        } catch (\Exception $e) {
            $transaction->rollBack();
            throw $e;
        }

        return $result;
    }
