	public function testGridFsDeleteWithCustomPrefix() {

		$data = array('_id' => new MongoId);

		$db = new MongoDb($this->_testConfig + array('gridPrefix' => 'custom'));
		$db->server = new MockMongoConnection();

		$db->server->connected = true;
		$model::$connection = $db;
