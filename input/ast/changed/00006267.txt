	public function add($autodate = true, $null_values = false)
	{
		if (!$this->id_lang)
			$this->id_lang = Configuration::get('PS_LANG_DEFAULT');
		if (Context::getContext()->shop->getGroup()->share_order == true)
			$this->id_shop = null;
		$return = parent::add($autodate);
		Hook::exec('cart');

		return $return;
	}
