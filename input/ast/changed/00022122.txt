    public function get_setting() {
        $value = $this->config_read($this->name);
        $fix = $this->config_read('fix_' . $this->name);
        if (is_null($value) or is_null($fix)) {
            return NULL;
        }
        return array('value' => $value, 'fix' => $fix);
    }
