	function change_dir($path = '', $supress_debug = false)
	{
		if ($path == '' or ! $this->_is_conn())
		{
			return false;
		}

		$result = @ftp_chdir($this->_conn_id, $path);

		if ($result === false)
		{
			if ($this->_debug == true and $supress_debug == false)
			{
				throw new \Exception('ftp_unable_to_change_dir');
			}
			return false;
		}

		return true;
	}
