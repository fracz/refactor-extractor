    public function params($mask = null) {
        global $__params;
        if (null !== $mask) {
            if (!is_array($mask)) {
                $mask = func_get_args();
            }
            $mask = array_flip($mask);
            $params = array_intersect_key($__params, $mask);
            $params = array_merge($mask, $params); //Ensure each key is present
        }
        return $params;
    }
