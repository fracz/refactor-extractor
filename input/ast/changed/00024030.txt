	public function getLabel()
	{
		$label = clone $this->label;
		$label->for = $this->getHtmlId();
		$text = $label->getText();
		$translator = $this->getTranslator();
		if ($translator !== NULL) {
			$text = $translator->translate($text);
		}
		$label->setText(str_replace('%l', $text, self::$labelMask));
		return $label;
	}
