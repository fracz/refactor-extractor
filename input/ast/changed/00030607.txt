    public function getUsersHavingSuperUserAccess()
    {
        $users = Db::get()->fetchAll("SELECT login, email
                                      FROM " . Common::prefixTable("user") . "
                                      WHERE superuser_access = 1
                                      ORDER BY date_registered ASC");

        return $users;
    }
