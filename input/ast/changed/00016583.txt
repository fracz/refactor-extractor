	public function sendPacket($p) {
		if (!$this->client) {
			return;
		}
		if (is_string($p['method']) && ctype_digit($p['method'])) {
			$p['method'] = (int) $p['method'];
		}
		$this->client->sendFrame($this->toJson($p) . "\n");
	}
