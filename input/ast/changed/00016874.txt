	public function onFailureEvent($stream, $arg = null) {
		$this->close();
		if ($this->finished) {
			return;
		}
		$this->finished = true;
		$this->onFinish();
		if ($this->pool) {
			$this->pool->detach($this);
			$this->pool = null;
		}

		event_base_loopexit(Daemon::$process->eventBase);
	}
