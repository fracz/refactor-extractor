    private static function tokensTrim(array $tokens)
    {
        foreach ($tokens as $index => $token) {
            if (self::tokenIsIgnored($token)) {
                unset($tokens[$index]);
            } else {
                break;
            }
        }

        $tokens = array_reverse($tokens);

        foreach ($tokens as $index => $token) {
            if (self::tokenIsIgnored($token)) {
                unset($tokens[$index]);
            } else {
                break;
            }
        }

        return array_reverse($tokens);
    }
