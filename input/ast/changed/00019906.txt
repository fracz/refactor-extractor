    function fetch($params) {
        if ($outcome = grade_object::fetch_helper('grade_outcomes', 'grade_outcome', $params)) {
            if (!empty($outcome->scaleid)) {
                $outcome->scale = new grade_scale(array('id'=>$outcome->scaleid));
                $outcome->scale->load_items();
            }
            return $outcome;

        } else {
            return false;
        }
    }
