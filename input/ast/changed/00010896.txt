		$out = $this->fetchRow("SELECT COUNT(" . $Model->primaryKey . ") " . $this->alias . "count FROM " . $this->fullTableName($Model) . ' ' . ($sql ? ' ' . $sql : 'WHERE 1 = 1'));

			return $out[0]['count'];
		}
		return false;
	}
