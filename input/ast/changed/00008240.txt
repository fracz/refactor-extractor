	public function __construct( $meta_query = false ) {
		$this->db = $GLOBALS['wpdb'];

		if ( ! $meta_query ) {
			return;
		}

		if ( isset( $meta_query['relation'] ) && strtoupper( $meta_query['relation'] ) == 'OR' ) {
			$this->relation = 'OR';
		} else {
			$this->relation = 'AND';
		}

		$this->queries = $this->sanitize_query( $meta_query );
	}
