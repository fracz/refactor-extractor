	public function statvfs($cb, $pri = EIO_PRI_DEFAULT) {
		if (!$this->fd) {
			if ($cb) {
				call_user_func($cb, $this, false);
			}
			return false;
		}
		if (!FS::$supported) {
			if ($cb) {
				call_user_func($cb, $this, false);
			}
			return;
		}
		if ($this->statvfs) {
			call_user_func($cb, $this, $this->statvfs);
		} else {
			eio_fstatvfs($this->fd, $pri, function ($file, $stat) use ($cb) {
				$file->statvfs = $stat;
				call_user_func($cb, $file, $stat);
			}, $this);
		}
	}
