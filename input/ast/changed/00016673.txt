	public function checkIfReady() {
		if (
			!$this->attrs->params_done
			|| !$this->attrs->stdin_done
		) {
			return false;
		} else {
			if (isset($this->appInstance->passphrase)) {
				if (
					!isset($this->attrs->server['PASSPHRASE'])
					|| ($this->appInstance->passphrase !== $this->attrs->server['PASSPHRASE'])
				) {
					$this->finish();
				}
			}
		}
		return true;
	}
