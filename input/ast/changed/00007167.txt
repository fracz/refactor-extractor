	public function testBooleanValues() {
		$doc = new Document();

		$doc->tall = false;
		$doc->fat = true;
		$doc->set('hair', true);
		$doc->set('fast', false);

		$expected = array('tall','fat','hair','fast');
		$result = array_keys($doc->data());
		$this->assertEqual($expected, $result);
	}
}
