	public function run()
	{
		if (PHP_SAPI !== 'cli' || realpath($_SERVER['argv'][0]) !== $this->getName())
		{
			require_once(__DIR__ . '/../../scripts/runners/autorunner.php');
		}
		else
		{
			parent::run();

			if ($this->help === true)
			{
				$this->help();
			}

			if ($this->infos === true)
			{
				$this->infos();
			}

			if ($this->signature === true)
			{
				$this->signature();
			}

			if ($this->extract !== false)
			{
				$this->extract();
			}
		}

		return $this;
	}
