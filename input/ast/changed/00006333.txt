	public static function getRequestSql()
	{
		if (!$result = Db::getInstance(_PS_USE_SQL_SLAVE_)->executeS('SELECT * FROM `'._DB_PREFIX_.'request_sql` ORDER BY `id_request_sql`'))
			return false;
		$request_sql = array();
		foreach ($result as $row)
			$request_sql[] = $row['sql'];
		return $request_sql;
	}
