    public function seek($position)
    {
        $pos = $this->filterMinRange((int) $position, 0, 'Can\'t seek stream to negative line %d');
        foreach ($this as $key => $value) {
            if ($key === $pos || feof($this->stream)) {
                $this->offset--;
                break;
            }
        }

        $this->current();
    }
