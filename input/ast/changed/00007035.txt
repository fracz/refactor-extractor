		public function register( $plugin ) {
			if ( ! isset( $plugin['slug'] ) || ! isset( $plugin['name'] ) ) {
				return;
			}

			foreach ( $this->plugins as $registered_plugin ) {
				if ( $plugin['slug'] === $registered_plugin['slug'] ) {
					return;
				}
			}

			$this->plugins[] = $plugin;

		}
