    public static function getDependencies($docComment, array $dependencies = array())
    {
        if (preg_match_all(self::REGEX_DEPENDS, $docComment, $matches)) {
            $dependencies = array_unique(array_merge($dependencies, $matches[1]));
        }

        return $dependencies;
    }
