	public function filterQuery(&$query, $model, &$selects)
	{
		//if this field has a min/max range, set it
		if ($this->minMax)
		{
			if ($this->minValue)
			{
				$query->where($model->getTable().'.'.$this->field, '>=', $this->minValue);
			}

			if ($this->maxValue)
			{
				$query->where($model->getTable().'.'.$this->field, '<=', $this->maxValue);
			}
		}
	}
