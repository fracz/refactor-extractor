	protected function _watermark($filename, $x, $y)
	{
		extract(parent::_watermark($filename, $x, $y));
		$image = '"'.$this->image_temp.'"';
		$filename = '"'.$filename.'"';
		if ($x >= 0)
			$x = '+'.$x;
		if ($y >= 0)
			$y = '+'.$y;
		$this->exec(
				'composite',
				'-compose atop -geometry '.$x.$y.' '.
				'-dissolve '.$this->config['watermark_alpha'].'% '.
				$filename.' '.$image.' '.$image
		);
	}
