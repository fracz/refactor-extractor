  public function getPassthroughRequestData() {
    $data = $this->getRequestData();

    // Remove magic parameters like __dialog__ and __ajax__.
    foreach ($data as $key => $value) {
      if (strncmp($key, '__', 2)) {
        unset($data[$key]);
      }
    }

    return $data;
  }
