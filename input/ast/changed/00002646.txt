	function getClass($type, $subtype) {
		global $SUBTYPE_CACHE;

		if ($SUBTYPE_CACHE === null) {
			$this->populateCache();
		}

		// use the cache before going to the database
		$obj = $this->retrieveFromCache($type, $subtype);
		if ($obj) {
			return $obj->class;
		}

		return null;
	}
