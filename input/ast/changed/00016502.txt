	public function shutdown($signo = false) {
		$this->shutdown = true;
		$this->waitAll(true);

		if (Daemon::$shm_wstate) {
			shmop_delete(Daemon::$shm_wstate);
		}

		file_put_contents(Daemon::$config->pidfile->value,'');

		exit(0);
	}
