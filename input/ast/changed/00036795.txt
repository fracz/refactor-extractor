    private function initialize()
    {
        if (!$this->initialized) {
            $this->initialized = true;

            // Make sure that plugins and set variables are applied in the
            // order they were added
            foreach ($this->changes as $key => $value) {
                if ($value instanceof PluginInterface) {
                    $value->setUp($this);
                } else {
                    $this->vars[$key] = $value;
                }
            }
        }
    }
