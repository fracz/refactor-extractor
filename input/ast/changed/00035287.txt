    {
        $node

            ->scalarNode('token_provider')->end()
        ;

        foreach ($this->options as $name => $value) {
            if (is_bool($value)) {
                $node->booleanNode($name)->defaultValue($value);
            } else {
                $node->scalarNode($name)->defaultValue($value);
