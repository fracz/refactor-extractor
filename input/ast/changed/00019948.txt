    public function get_entries($start=0, $limit=10) {
        global $DB;

        if (empty($this->entries)) {
            if ($sql_array = $this->get_entry_fetch_sql()) {
                $this->entries = $DB->get_records_sql($sql_array['sql'], $sql_array['params'], $start, $limit);
            } else {
                return false;
            }
        }

        return $this->entries;
    }
