   function _doThread ($data, $root)
   {
      $out = array();

      for ($ii=0; $ii<sizeof($data); $ii++)
      {
         if ($data[$ii]['parent_id'] == $root)
         {
            $tmp = $data[$ii];
            $tmp['children'] = isset($data[$ii]['id'])? $this->_doThread($data, $data[$ii]['id']): null;
            $out[] = $tmp;
         }
      }

      return $out;
   }
