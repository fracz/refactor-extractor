    public function needsReload($key)
    {
        if (!$this->cache->exists($key)) {
            return true;
        }

        try {
            $dateCache = $this->cache->mtime($key);

            if (time() - $this->ttl > $dateCache) {
                $dateSource = $this->source->mtime($key);

                return $dateCache < $dateSource;
            }
            else {
                return false;
            }
        } catch (\RuntimeException $e) {
            return true;
        }
    }
