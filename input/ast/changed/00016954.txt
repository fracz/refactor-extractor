	public static function parseLabels(&$data) {
		$domain = '';
		while (strlen($data) > 0) {
			if (binarySubstr($data, 0, 2) === "\xc0\x0c") {
				$data = binarySubstr($data, 2);
				continue;
			}
			$l = ord($data[0]);
     		$p = substr($data, 1, $l);
     		$domain .= $p.(($l !== 0)?'.':'');
     		$data = substr($data, $l + 1);
       		if ($l === 0) {
     			break;
     		}
    	}
    	return $domain;
	}
