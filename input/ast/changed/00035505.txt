    public function toArray()
    {
        $exceptions = array();
        foreach (array_merge(array($this), $this->getPreviouses()) as $exception) {
            $exceptions[] = array(
                'code'     => $exception->getStatusCode(),
                'message'  => $exception->getMessage(),
                'class'    => $exception->getClass(),
                'trace'    => $exception->getTrace(),
            );
        }

        return $exceptions;
    }
