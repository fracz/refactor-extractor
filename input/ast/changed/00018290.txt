	protected function of($name, $data = array())
	{
		if ( ! is_null($view = $this->composer->name($name)))
		{
			return new View($view, $data, $this->path($view), $this->composer, $this);
		}

		throw new \Exception("Named view [$name] is not defined.");
	}
