    protected function render_route($route, $params = []) {

        $output = false;

        if (isset($this->routes[$route])) {

            if (is_callable($this->routes[$route])){
                $ret = call_user_func($this->routes[$route], $params);
            }

            if ( !is_null($ret) ){
                return $ret;
            }
        }

        return $output;
    }
