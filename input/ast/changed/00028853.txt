    public function numberSort($rowA, $rowB)
    {
        if (isset($rowA[0]) && isset($rowB[0])) {
            if ($rowA[0] != $rowB[0] || !isset($rowA[1])) {
                return $this->sign * ($rowA[0] < $rowB[0] ? -1 : 1);
            } else {
                return -1 * $this->sign * strnatcasecmp($rowA[1], $rowB[1]);
            }
        } elseif (!isset($rowB[0])) {
            return -1;
        } elseif (!isset($rowA[0])) {
            return 1;
        }

        return 0;
    }
