	function connect() {
		if ($this->connected) {
			return true;
		}
		$config = $this->config;

		if ($config) {
			$this->_conn = ociplogon($config['login'], $config['password'], $config['database']);
		}
		if($this->_conn){
			$this->connected = true;
			$this->execute('alter session set nls_sort=binary_ci');
		}
		if (!$this->connected) {
			return false;
		}
	}
