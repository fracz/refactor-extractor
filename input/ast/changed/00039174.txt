	protected function renderItem($item)
	{
		if (is_string($item)) {
			return $item;
		}
		$config = ArrayHelper::getValue($item, 'options', array());
		$config['class'] = ArrayHelper::getValue($item, 'class', 'yii\bootstrap\Nav');
		$widget = \Yii::createObject($config);
		ob_start();
		$widget->run();
		return ob_get_clean();
	}
