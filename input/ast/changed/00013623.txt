    public static function fromString($query)
    {
        $q = new static();

        if (!empty($query)) {
            if ($query[0] == '?') {
                $query = substr($query, 1);
            }
            foreach (explode('&', $query) as $kvp) {
                $parts = explode('=', $kvp);
                $key   = rawurldecode($parts[0]);
                $value = '';

                if (substr($key, -2) == '[]') {
                    $key = substr($key, 0, -2);
                }

                if (isset($parts[1])) {
                    $value = rawurldecode(str_replace('+', '%20', $parts[1]));
                }

                $q->add($key, $value);
            }
        }

        return $q;
    }
