	public function offsetSet($offset, $data) {
		if ($schema = $this->schema()) {
			$options = array('pathKey' => $this->_pathKey);
			$data = $schema->cast($this, array($offset => $data), $options);
			$data = reset($data);
		}
		return $offset ? ($this->_data[$offset] = $data) : ($this->_data[] = $data);
	}
