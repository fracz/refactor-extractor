    public function initializeTokens($contents, $filename = null)
    {
        $tokens = token_get_all($contents);
        $this->debug(
            count($tokens) . ' tokens found in class ' . $this->getName()
        );

        $tokens = new DocBlox_Token_Iterator($tokens);
        if ($filename != null) {
            $tokens->setFilename($filename);
        }

        return $tokens;
    }
