    public function refresh() {
        $redirect = isset($_SERVER['REQUEST_URI']) ? $_SERVER['REQUEST_URI'] : '/';
        if (isset($_SERVER['QUERY_STRING']) && $_SERVER['QUERY_STRING']) {
            $redirect .= '?' . $_SERVER['QUERY_STRING'];
        }
        $this->redirect($redirect);
    }
