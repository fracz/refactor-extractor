    protected function getResultColorCode($result)
    {
        switch ($result) {
            case StepTester::PASSED:
                return 'passed';
            case StepTester::SKIPPED:
                return 'skipped';
            case StepTester::PENDING:
                return 'pending';
            case StepTester::UNDEFINED:
                return 'undefined';
            case StepTester::FAILED:
                return 'failed';
        }
    }
