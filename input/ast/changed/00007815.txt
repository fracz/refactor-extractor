	public function __construct(array $config = array()) {

			'adapter'	 => null,
			'persistent' => false,
			'scheme'     => 'http',
			'host'       => 'localhost',
			'version'    => '1.1',
			'auth'       => null,
			'login'      => '',
			'password'   => '',
			'port'       => 80,
			'timeout'    => 30,
			'encoding'   => 'UTF-8'
		);

		$config['username'] = $config['login'];
		parent::__construct($config);
	}
