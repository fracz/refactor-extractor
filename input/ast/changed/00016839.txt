	public function readahead($length, $offset = null, $cb = null, $pri = EIO_PRI_DEFAULT) {
		if (!$this->fd) {
			if ($cb) {
				call_user_func($cb, $this, false);
			}
			return false;
		}
		if (!FS::$supported) {
			if ($cb) {
				call_user_func($cb, $this, false);
			}
			return;
		}
		$this->offset += $length;
		eio_readahead(
			$this->fd,
			$length,
			$offset !== null ? $offset : $this->pos,
			$pri,
			$cb,
			$this
		);
		return true;
	}
