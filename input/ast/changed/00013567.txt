    public function stopTimer($name)
    {
        if (in_array($name, $this->timers) && ($name[0] == '_' || $this->grav['config']->get('system.debugger.enabled'))) {
            $this->debugbar['time']->stopMeasure($name);
        }

        return $this;
    }
