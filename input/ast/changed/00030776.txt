    public static function getOsFamily($osLabel)
    {
        $osShortName = substr($osLabel, 0, 3);

        foreach (self::$osFamilies as $osFamily => $osShortNames) {
            if (in_array($osShortName, $osShortNames)) {
                return $osFamily;
            }
        }

        return 'Other';
    }
