	public function readSession($id)
	{
		$query = new Query;
		$row = $query->select(['data'])
			->from($this->sessionCollection)
			->where([
				'expire' => ['$gt' => time()],
				'id' => $id
			])
			->one($this->db);
		return $row === false ? '' : $row['data'];
	}
