					$object = $object->attributes;
				} else {
					return null;
				}
			}
		} else {
			$out = array();
			if (is_object($object)) {
				$keys = get_object_vars($object);
				foreach ($keys as $key => $value) {
					if (is_array($value)) {
						$out[$key] = (array)Set::reverse($value);
					} else {
						$out[$key] = Set::reverse($value);

				}
