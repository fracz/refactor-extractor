	public function andWhere($conditions, $types = []) {
		$where = $this->_parts['where'];

		if ($where->type() === 'AND') {
			$where->add($conditions, $types);
		} else {
			$where = new QueryExpression([$conditions, $where], $types, 'AND');
		}

		$this->_parts['where'] = $where;
		$this->_dirty = true;
		return $this;
	}
