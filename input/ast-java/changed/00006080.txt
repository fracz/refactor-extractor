private static String showDialogUrl(@NotNull final String initialUrl) {
    final String url = Messages.showInputDialog("IPython Notebook URL:", "Start IPython Notebook", null, initialUrl, new InputValidator() {

        @Override
        public boolean checkInput(String inputString) {
            try {
                new URL(inputString);
            } catch (MalformedURLException e) {
                return false;
            }
            return !inputString.isEmpty();
        }

        @Override
        public boolean canClose(String inputString) {
            return true;
        }
    });
    return url == null ? null : StringUtil.trimEnd(url, "/");
