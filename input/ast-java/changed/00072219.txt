@Override
public boolean process(Set<? extends TypeElement> annotations, RoundEnvironment roundEnv) {
    for (Element element : roundEnv.getElementsAnnotatedWith(BugPattern.class)) {
        try {
            FileObject manifest = processingEnv.getFiler().createResource(StandardLocation.SOURCE_OUTPUT, "", "bugPatterns.txt");
            PrintWriter pw = new PrintWriter(new OutputStreamWriter(manifest.openOutputStream()));
            // 0
            pw.print(element.toString() + "\t");
            BugPattern annotation = element.getAnnotation(BugPattern.class);
            // 1
            pw.print(annotation.name() + "\t");
            // 2
            pw.print(annotation.category() + "\t");
            // 3
            pw.print(annotation.severity() + "\t");
            // 4
            pw.print(annotation.maturity() + "\t");
            // 5
            pw.print(annotation.summary() + "\t");
            // 6
            pw.println(annotation.explanation().replace("\n", "\\n"));
            pw.close();
        } catch (IOException e) {
            // TODO(eaftan): Auto-generated catch block
            e.printStackTrace();
        }
    }
    return true;
