private static boolean resetTaskFile(@NotNull final Document document, @NotNull final Project project, TaskFile taskFile, String name) {
    if (!resetDocument(document, taskFile, name)) {
        return false;
    }
    taskFile.getTask().setStatus(StudyStatus.Unchecked);
    resetAnswerPlaceholders(taskFile, project);
    ProjectView.getInstance(project).refresh();
    StudyUtils.updateToolWindows(project, taskFile.getTask());
    return true;
