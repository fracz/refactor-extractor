public static String getSimpleQualifiedName(Object... names) {
    StringBuilder name = new StringBuilder();
    for (Object namePart : names) {
        if (namePart == null) {
            continue;
        }
        if (name.length() > 0) {
            name.append('.');
        }
        name.append(namePart);
    }
    return name.toString();
