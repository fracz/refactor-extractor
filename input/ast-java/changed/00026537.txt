        // TODO: correct container for scripts (KScript?)
        StackValue value = generateClassLiteralReference(typeMapper, ((ScriptDescriptor) container).getClassDescriptor().getDefaultType());
        value.put(K_CLASS_TYPE, iv);
    } else {
        iv.aconst(null);
    }
