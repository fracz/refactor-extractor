@Override
public void close() {
    synchronized (conn) {
        checkClosed();
        if (connectRunnable != null) {
            // Server side
            if (rejected) {
                throw new IllegalStateException("Cannot close websocket, it has been rejected");
            }
            if (!connected && !closed) {
                connect();
            }
        }
        super.close();
    }
