/**
 * When define is called with an object literal, assign it to module.exports and
 * we're done.
 */
private void handleDefineObjectLiteral(Node parent, Node onlyExport, Node script) {
    onlyExport.getParent().removeChild(onlyExport);
    script.replaceChild(parent, IR.exprResult(IR.assign(compiler.newQualifiedNameNode("module.exports"), onlyExport)).copyInformationFromForTree(onlyExport));
    compiler.reportCodeChange();
