            lexer.nextToken();
        } else if (lexer.equalToken(Token.IDENTIFIER)) {
            result = lexer.getLiterals();
            lexer.nextToken();
        } else {

                case MATCHED:
                case ERRORS:
                case REJECT:
                case UNLIMITED:
                case BEGIN:
                case EXCLUSIVE:
                case MODE:
                case ADVISE:
                case TYPE:
                    result = lexer.getLiterals();
                    lexer.nextToken();
                    return result;
                case QUESTION:
                    result = Token.QUESTION.getName();
                    lexer.nextToken();
                default:
                    break;
            }
        }
        if (null != result) {
            while (lexer.equalToken(Token.DOT)) {
                lexer.nextToken();

            return result;
        }
        if (lexer.equalToken(Token.LEFT_PAREN)) {
            return null;
        }
        throw new ParserException(String.format("Syntax error, token is '%s'.", lexer.getToken()));
    }
    if (lexer.equalToken(Token.IDENTIFIER) || lexer.equalToken(Token.USER) || lexer.equalToken(Token.END)) {
        result = lexer.getLiterals();
        lexer.nextToken();
