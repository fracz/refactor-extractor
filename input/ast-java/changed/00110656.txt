private void addArgs(CodeWriter code, InsnNode insn, int k) throws CodegenException {
    code.add('(');
    for (int i = k; i < insn.getArgsCount(); i++) {
        code.add(arg(insn.getArg(i)));
        if (i < insn.getArgsCount() - 1)
            code.add(", ");
    }
    code.add(')');
