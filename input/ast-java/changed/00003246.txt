public Artifact downloadArtifact(MavenProjectModel mavenProject, MavenId id, MavenGeneralSettings coreSettings, MavenConsole console) throws MavenProcessCanceledException {
    MavenEmbedderWrapper e = MavenEmbedderFactory.createEmbedderForExecute(coreSettings, console);
    try {
        Artifact artifact = e.createArtifact(id.groupId, id.artifactId, id.version, MavenConstants.JAR_TYPE, null);
        artifact.setScope(Artifact.SCOPE_COMPILE);
        e.resolve(artifact, mavenProject.getRepositories());
        return artifact;
    } finally {
        e.release();
    }
