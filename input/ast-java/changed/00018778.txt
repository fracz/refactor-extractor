private void onTextChange() {
    if (!myIsInUpdate) {
        myCurrentScope = null;
        try {
            myCurrentScope = PackageSetFactory.getInstance().compile(myPatternField.getText());
            myErrorMessage = null;
            myCanceled = true;
            rebuild(false);
        } catch (Exception e) {
            myErrorMessage = e.getMessage();
            showErrorMessage();
        }
    } else {
        myErrorMessage = null;
    }
