private void failWebSocket(Exception e, Response response) {
    Streams streamsToClose;
    synchronized (this) {
        // Already failed.
        if (failed)
            return;
        failed = true;
        streamsToClose = this.streams;
        this.streams = null;
    }
    try {
        listener.onFailure(this, e, response);
    } finally {
        closeQuietly(streamsToClose);
    }
