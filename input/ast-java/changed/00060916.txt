@Test
public void testDefault() throws IOException {
    Settings settings = Settings.settingsBuilder().put("index.analysis.filter.limit_default.type", "limit").put("path.home", createTempDir().toString()).build();
    AnalysisService analysisService = AnalysisTestsHelper.createAnalysisServiceFromSettings(settings);
    {
        TokenFilterFactory tokenFilter = analysisService.tokenFilter("limit_default");
        String source = "the quick brown fox";
        String[] expected = new String[] { "the" };
        Tokenizer tokenizer = new WhitespaceTokenizer();
        tokenizer.setReader(new StringReader(source));
        assertTokenStreamContents(tokenFilter.create(tokenizer), expected);
    }
