            if (upper.x == lower.x && upper.width == lower.width) {
                unmerged.remove(upper);
                lower.y = upper.y;
                lower.height = upper.height + 1;
                break;
            } else if (upper.x > lower.x || upper.x + upper.width > lower.x) {

            }
        }
    }
