@Override
public final void load() throws IOException, InterruptedException {
    DataSpec loadDataSpec = Util.getRemainderDataSpec(dataSpec, bytesLoaded);
    try {
        // Create and open the input.
        ExtractorInput input = new DefaultExtractorInput(dataSource, loadDataSpec.absoluteStreamPosition, dataSource.open(loadDataSpec));
        if (bytesLoaded == 0) {
            // Set the target to ourselves.
            DefaultTrackOutput trackOutput = getTrackOutput();
            trackOutput.formatWithOffset(sampleFormat, sampleOffsetUs);
            extractorWrapper.init(this, trackOutput);
        }
        // Load and parse the sample data.
        try {
            int result = Extractor.RESULT_CONTINUE;
            while (result == Extractor.RESULT_CONTINUE && !loadCanceled) {
                result = extractorWrapper.read(input);
            }
        } finally {
            bytesLoaded = (int) (input.getPosition() - dataSpec.absoluteStreamPosition);
        }
    } finally {
        dataSource.close();
    }
