private static void writeData(String header, Iterator<String> iterator, File file, ProgressListener progress, int count) throws IOException {
    System.out.println("Writing " + file.getAbsolutePath());
    try (Writer out = new BufferedWriter(new FileWriter(file), 102 * 1024 * 10)) {
        out.write(header);
        out.append('\n');
        for (int i = 0; i < count; i++) {
            out.write(iterator.next());
            out.append('\n');
            progress.add(1);
        }
    }
