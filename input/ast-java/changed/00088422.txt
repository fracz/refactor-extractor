@Override
public Response createNode(ForceMode force, String body) {
    Transaction transaction = graph.beginTx();
    try {
        Response response = restfulGraphDatabase.createNode(force, body);
        if (response.getStatus() < 300) {
            transaction.success();
        }
        return response;
    } finally {
        transaction.finish();
    }
