        return Comparing.strEqual(MAIN_CLASS_NAME, ExecutionUtil.getRuntimeQualifiedName(aClass)) && Comparing.strEqual(METHOD_NAME, ((PsiMethod) element).getName());
    } else if (element instanceof PsiClass) {
        return Comparing.strEqual(MAIN_CLASS_NAME, ExecutionUtil.getRuntimeQualifiedName((PsiClass) element));
    }
    return false;
