    final long wrapPoint = sequence - bufferSize;
    if (wrapPoint > minGatingSequence.get()) {
        long minSequence;
        while (wrapPoint > (minSequence = getMinimumSequence(dependentSequences))) {
            LockSupport.parkNanos(1L);
        }
        minGatingSequence.set(minSequence);
    }
