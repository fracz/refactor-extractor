private void checkExpression(PsiExpression expression) {
    if (!isArithmetic(expression)) {
        return;
    }
    if (isParentArithmetic(expression)) {
        return;
    }
    final int numTerms = countTerms(expression);
    if (numTerms <= getLimit()) {
        return;
    }
    registerError(expression);
