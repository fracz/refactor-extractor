public void showTooltipByMouseMove(@NotNull final Editor editor, @NotNull MouseEvent e, final TooltipRenderer tooltipObject, final boolean alignToRight, @NotNull final TooltipGroup group) {
    myTooltipAlarm.cancelAllRequests();
    if (myCurrentTooltip == null || !myCurrentTooltip.isVisible()) {
        myCurrentTooltipObject = null;
    }
    if (Comparing.equal(tooltipObject, myCurrentTooltipObject)) {
        return;
    }
    hideCurrentTooltip();
    if (tooltipObject != null) {
        final Point p = SwingUtilities.convertPoint((Component) e.getSource(), e.getPoint(), editor.getComponent().getRootPane().getLayeredPane());
        p.x += alignToRight ? -10 : 10;
        myTooltipAlarm.addRequest(new Runnable() {

            public void run() {
                Project project = editor.getProject();
                if (project != null && !project.isOpen())
                    return;
                if (editor.getContentComponent().isShowing()) {
                    showTooltip(editor, p, tooltipObject, alignToRight, group);
                }
            }
        }, 50);
    }
