@Override
public DBDContentStorage getContents(DBRProgressMonitor monitor) throws DBCException {
    if (storage == null && xml != null) {
        try {
            storage = StringContentStorage.createFromReader(xml.getCharacterStream());
        } catch (Exception e) {
            throw new DBCException(e);
        }
        // Free blob - we don't need it anymore
        try {
            xml.free();
        } catch (Exception e) {
            log.warn(e);
        } finally {
            xml = null;
        }
    }
    return storage;
