public int stop() {
    String location = "unknown location";
    try {
        if (server != null) {
            server.stop();
        }
        log.info("Successfully shutdown Neo Server on port [%s], database [%s]", serverPort, location);
        removeShutdownHook();
        life.shutdown();
        return 0;
    } catch (Exception e) {
        log.error("Failed to cleanly shutdown Neo Server on port [%s], database [%s]. Reason [%s] ", serverPort, location, e.getMessage(), e);
        return 1;
    }
