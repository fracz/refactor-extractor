public int acceptInteger() {
    if (lexer.token() == Token.LITERAL_INT) {
        int intVal = ((Integer) lexer.integerValue()).intValue();
        lexer.nextToken();
        return intVal;
    } else {
        throw new ParserException("syntax error, expect int, actual " + lexer.token() + " " + lexer.info());
    }
