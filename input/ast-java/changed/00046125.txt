static Mutation makeStartSessionMutation(ByteBuffer sessionId, InetAddress client, Map<String, String> parameters, String request, long startedAt, String command, int ttl) {
    Mutation mutation = new Mutation(NAME, sessionId);
    ColumnFamily cells = mutation.addOrGet(TraceKeyspace.Sessions);
    CFRowAdder adder = new CFRowAdder(cells, cells.metadata().comparator.builder().build(), FBUtilities.timestampMicros(), ttl);
    adder.add("client", client).add("coordinator", FBUtilities.getBroadcastAddress()).add("request", request).add("started_at", new Date(startedAt)).add("command", command);
    for (Map.Entry<String, String> entry : parameters.entrySet()) adder.addMapEntry("parameters", entry.getKey(), entry.getValue());
    return mutation;
