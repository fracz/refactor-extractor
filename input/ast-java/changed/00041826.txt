@Override
public void setHotspot(float x, float y) {
    if (mState.mPinned && !circleContains(mHotspotBounds, x, y)) {
        x = mHotspotBounds.exactCenterX();
        y = mHotspotBounds.exactCenterY();
    }
    if (mHotspot == null) {
        mHotspot = new Ripple(this, mHotspotBounds, x, y);
        if (mActive) {
            activateHotspot();
        }
    } else {
        mHotspot.move(x, y);
    }
