    if (paint.getStyle() == Paint.Style.STROKE && paint.getStrokeWidth() == 0f) {
        return;
    }
    super.draw(canvas);
    float halfWidth = rectSize.getValue().x / 2f;
    float halfHeight = rectSize.getValue().y / 2f;
    fillRect.set(rectPosition.getValue().x - halfWidth, rectPosition.getValue().y - halfHeight, rectPosition.getValue().x + halfWidth, rectPosition.getValue().y + halfHeight);
    if (rectCornerRadius.getValue() == 0) {

    }
