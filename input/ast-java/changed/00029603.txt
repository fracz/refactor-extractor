private GenericRequestBuilder buildThumbnailGlideRequest(Slide slide, MasterSecret masterSecret) {
    final GenericRequestBuilder builder;
    if (slide.isDraft() && slide.isEncrypted())
        builder = buildEncryptedDraftGlideRequest(slide, masterSecret);
    else if (slide.isDraft())
        builder = buildDraftGlideRequest(slide);
    else
        builder = buildEncryptedPartGlideRequest(slide, masterSecret);
    return builder;
