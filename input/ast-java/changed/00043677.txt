/**
 * @see AudioManager#stopBluetoothSco()
 */
public void stopBluetoothSco(IBinder cb) {
    if (!checkAudioSettingsPermission("stopBluetoothSco()")) {
        return;
    }
    ScoClient client = getScoClient(cb, false);
    if (client != null) {
        client.decCount();
    }
