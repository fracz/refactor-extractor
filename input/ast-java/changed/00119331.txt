@Override
public synchronized void rollback() {
    Preconditions.checkArgument(isOpen(), "The transaction has already been closed");
    try {
        txHandle.rollback();
    } catch (Exception e) {
        throw new TitanException("Could not rollback transaction due to exception", e);
    } finally {
        close();
    }
