@Override
public ByteBuffer toByteBuffer(RowBasedKey key) {
    keyBuffer.rewind();
    final int dimStart;
    if (includeTimestamp) {
        keyBuffer.putLong((long) key.getKey()[0]);
        dimStart = 1;
    } else {
        dimStart = 0;
    }
    for (int i = dimStart; i < key.getKey().length; i++) {
        if (!serdeHelpers.get(i - dimStart).putToKeyBuffer(key, i)) {
            return null;
        }
    }
    keyBuffer.flip();
    return keyBuffer;
