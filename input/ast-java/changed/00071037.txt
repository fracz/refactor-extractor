/**
 * Compare this node to node2 recursively and return the first pair
 * of nodes that differs doing a preorder depth-first traversal.
 * Package private for testing. Returns null if the nodes are equivalent.
 */
Pair<Node, Node> checkTreeEqualsImpl(Node node2) {
    boolean eq = false;
    if (type == node2.getType() && getChildCount() == node2.getChildCount() && getClass() == node2.getClass()) {
        eq = this.isEquivalentTo(node2);
    }
    if (!eq) {
        return Pair.of(this, node2);
    }
    Pair<Node, Node> res = null;
    Node n, n2;
    for (n = first, n2 = node2.first; res == null && n != null; n = n.next, n2 = n2.next) {
        res = n.checkTreeEqualsImpl(n2);
        if (res != null) {
            return res;
        }
    }
    return res;
