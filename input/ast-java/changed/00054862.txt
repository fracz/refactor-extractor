/* Remove taskLock with given ID */
public boolean removeLock(final long lockId) {
    return retryingHandle(new HandleCallback<Boolean>() {

        @Override
        public Boolean withHandle(Handle handle) throws Exception {
            return handle.createStatement(String.format("DELETE FROM %s WHERE id = :id", config.getTaskLockTable())).bind("id", lockId).execute() == 1;
        }
    });
