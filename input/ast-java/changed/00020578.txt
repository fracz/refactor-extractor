private void copyRow() {
    int selected = getSelectedIndex();
    if (selected < 0)
        return;
    TemplateImpl orTemplate = getTemplate(selected);
    LOG.assertTrue(orTemplate != null);
    TemplateImpl template = orTemplate.copy();
    myTemplateOptions.put(System.identityHashCode(template), template.createOptions());
    myTemplateContext.put(System.identityHashCode(template), template.createContext());
    EditTemplateDialog dialog = new EditTemplateDialog(this, CodeInsightBundle.message("dialog.copy.live.template.title"), template, getTemplateGroups(), (String) myExpandByCombo.getSelectedItem(), getOptions(template), getContext(template));
    dialog.show();
    if (!dialog.isOK())
        return;
    dialog.apply();
    addTemplate(template);
    isModified = true;
