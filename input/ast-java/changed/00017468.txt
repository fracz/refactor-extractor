    GrListOrMap map = getElement();

    if (conversionType == null)
        return null;
    PsiType type = map.getType();

    if (ownType != null && TypesUtil.isAssignableWithoutConversions(conversionType.rawType(), ownType, map))
        return null;

            return null;
        if (InheritanceUtil.isInheritor(resolved, CommonClassNames.JAVA_UTIL_LIST))
            return null;
    }
    return conversionType;
