@Override
public void autofill(AutofillValue value) {
    if (!isEnabled())
        return;
    if (!value.isDate()) {
        Log.w(LOG_TAG, value + " could not be autofilled into " + this);
    }
    mDelegate.setDate(value.getDateValue());
