        getLexer().nextToken();
        getLexer().accept(Token.BY);
        while (true) {
            if (getLexer().identifierEquals("GROUPING")) {
                throw new UnsupportedOperationException("Cannot support GROUPING SETS");
            }
            addGroupByItem(getExprParser().expr());
            if (!getLexer().equalToken(Token.COMMA)) {
                break;

            getLexer().nextToken();
        }

            getLexer().nextToken();
            getExprParser().expr();
        }
