 * Refreshes mappings if they are not the same between original and parsed version
 */
public void refreshMapping(final String index, final String indexUUID) {
    final RefreshTask refreshTask = new RefreshTask(index, indexUUID);
    clusterService.submitStateUpdateTask("refresh-mapping [" + index + "]", refreshTask, ClusterStateTaskConfig.build(Priority.HIGH), refreshExecutor, (source, e) -> logger.warn("failure during [{}]", e, source));
