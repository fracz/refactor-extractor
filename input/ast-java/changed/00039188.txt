void killAppAtUsersRequest(ProcessRecord app, Dialog fromDialog, boolean crashed) {
    synchronized (this) {
        app.crashing = false;
        app.crashingReport = null;
        app.notResponding = false;
        app.notRespondingReport = null;
        if (app.anrDialog == fromDialog) {
            app.anrDialog = null;
        }
        if (app.waitDialog == fromDialog) {
            app.waitDialog = null;
        }
        if (app.pid > 0 && app.pid != MY_PID) {
            if (crashed) {
                handleAppCrashLocked(app);
            }
            Log.i(ActivityManagerService.TAG, "Killing process " + app.processName + " (pid=" + app.pid + ") at user's request");
            Process.killProcess(app.pid);
        }
    }
