protected void showBouncerOrKeyguard() {
    if (mBouncer.needsFullscreenBouncer()) {
        // The keyguard might be showing (already). So we need to hide it.
        mPhoneStatusBar.hideKeyguard();
        mBouncer.show(true);
    } else {
        mPhoneStatusBar.showKeyguard();
        mBouncer.hide(false);

    }
