@Override
public void close() {
    mRequestThreadManager.quit();
    mCallbackHandlerThread.quitSafely();
    mResultThread.quitSafely();
    try {
        mCallbackHandlerThread.join();
    } catch (InterruptedException e) {
        Log.e(TAG, String.format("Thread %s (%d) interrupted while quitting.", mCallbackHandlerThread.getName(), mCallbackHandlerThread.getId()));
    }
    try {
        mResultThread.join();
    } catch (InterruptedException e) {
        Log.e(TAG, String.format("Thread %s (%d) interrupted while quitting.", mResultThread.getName(), mResultThread.getId()));
    }
