private void updateBarChart() {
    mBarChart.reset();
    BarSet barSet = new BarSet();
    Bar bar;
    for (int i = 0; i < 7; i++) {
        bar = new Bar(barLabels[i], barValues[0][i]);
        if (i == 4)
            bar.setColor(this.getResources().getColor(R.color.bar_highest));
        else
            bar.setColor(this.getResources().getColor(R.color.bar_fill1));
        barSet.addBar(bar);
    }
    mBarChart.addData(barSet);
    barSet = new BarSet();
    barSet.addBars(barLabels, barValues[1]);
    barSet.setColor(this.getResources().getColor(R.color.bar_fill2));
    mBarChart.addData(barSet);
    mBarChart.setSetSpacing(Tools.fromDpToPx(1));
    mBarChart.setBorderSpacing(0).setAxisBorderValues(BAR_MIN, BAR_MAX, 2).setGrid(mBarGridPaint).setYAxis(false).setXLabels(XController.LabelPosition.OUTSIDE).setYLabels(YController.LabelPosition.NONE).show(getAnimation(true).setEndAction(mEnterEndAction));
