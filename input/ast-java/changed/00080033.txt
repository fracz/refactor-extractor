@Override
public boolean touchDown(float x, float y, int pointer) {
    if (pointer != 0)
        return false;
    if (handleBounds.contains(x, y)) {
        touchDrag = true;
        lastPoint.set(x, y);
        handlePos.set(handleBounds.x, handleBounds.y);
        focus(this, 0);
        return true;
    }
    return super.touchDown(x, y, pointer);
