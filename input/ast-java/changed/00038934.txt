public void dumpWithTimeout(int pid) {
    Process.sendSignal(pid, Process.SIGNAL_QUIT);
    synchronized (this) {
        try {
            // Wait for traces file to be closed.
            wait(TRACE_DUMP_TIMEOUT_MS);
        } catch (InterruptedException e) {
            Slog.wtf(TAG, e);
        }
    }
    if (!mClosed) {
        Slog.w(TAG, "Didn't see close of " + mTracesPath + " for pid " + pid + ". Attempting native stack collection.");
        Debug.dumpNativeBacktraceToFileTimeout(pid, mTracesPath, TRACE_DUMP_TIMEOUT_SECONDS);
    }
    mClosed = false;
