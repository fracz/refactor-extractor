private static float discreteScale(float scale) {
    int intPart = (int) Math.floor(scale);
    float fractPart = scale - intPart;
    if (fractPart == 0)
        return scale;
    for (float f = 0; f < 1; f += DISCRETE_SCALE_RESOLUTION) {
        if (fractPart < (f + DISCRETE_SCALE_RESOLUTION)) {
            return intPart + f;
        }
    }
    // unreachable
    return intPart;
