@Override
public String getName() {
    if (name == null) {
        TitanProperty p;
        if (isLoaded()) {
            StandardTitanTx tx = tx();
            p = (TitanProperty) Iterables.getOnlyElement(RelationConstructor.readRelation(this, tx.getGraph().getSchemaCache().getSchemaRelations(getLongId(), BaseKey.SchemaName, Direction.OUT, tx()), tx), null);
        } else {
            p = Iterables.getOnlyElement(query().type(BaseKey.SchemaName).properties(), null);
        }
        Preconditions.checkState(p != null, "Could not find type for id: %s", getLongId());
        name = p.getValue();
    }
    assert name != null;
    return TitanSchemaCategory.getName(name);
