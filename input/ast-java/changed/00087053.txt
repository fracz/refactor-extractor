void lock(PageLock lockType) {
    if (lockType == PageLock.SHARED) {
        lock.readLock().lock();
    } else if (lockType == PageLock.EXCLUSIVE) {
        lock.writeLock().lock();
        dirty = true;
    } else {
        throw new IllegalArgumentException("Unknown lock type: " + lockType);
    }
