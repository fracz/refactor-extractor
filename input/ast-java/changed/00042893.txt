@Override
public Intent createShortcutIntent(int launcherUserId, @NonNull String callingPackage, @NonNull String packageName, @NonNull String shortcutId, int userId) {
    // Calling permission must be checked by LauncherAppsImpl.
    Preconditions.checkStringNotEmpty(packageName, "packageName can't be empty");
    Preconditions.checkStringNotEmpty(shortcutId, "shortcutId can't be empty");
    synchronized (mLock) {
        getLauncherShortcutsLocked(callingPackage, userId, launcherUserId).attemptToRestoreIfNeededAndSave(ShortcutService.this);
        // Make sure the shortcut is actually visible to the launcher.
        final ShortcutInfo si = getShortcutInfoLocked(launcherUserId, callingPackage, packageName, shortcutId, userId);
        // "si == null" should suffice here, but check the flags too just to make sure.
        if (si == null || !(si.isDynamic() || si.isPinned())) {
            return null;
        }
        return si.getIntent();
    }
