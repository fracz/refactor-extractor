/**
 * Records the given type as one to skip for this property.
 */
void addTypeToSkip(JSType type) {
    for (JSType skipType : typeSystem.getTypesToSkipForType(type)) {
        typesToSkip.add(skipType);
        getTypes().union(skipType, type);
    }
