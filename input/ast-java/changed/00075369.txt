        ModuleRevisionId moduleRevisionId = dd.getDependencyRevisionId();

        Artifact pomArtifact = new DefaultArtifact(moduleRevisionId, null, moduleRevisionId.getName(), "pom", "pom", moduleRevisionId.getExtraAttributes());

    return null;
