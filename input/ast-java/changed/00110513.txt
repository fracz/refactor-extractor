private static boolean declareAtAssign(Usage u) {
    RegisterArg arg = u.getArg();
    InsnNode parentInsn = arg.getParentInsn();
    if (!arg.equals(parentInsn.getResult())) {
        return false;
    }
    parentInsn.add(AFlag.DECLARE_VAR);
    processVar(arg);
    return true;
