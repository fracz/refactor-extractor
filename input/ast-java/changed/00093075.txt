/**
 * Avoid to close it but rather release itself to the owner pool.
 */
@Override
public void close() {
    if (isClosed())
        return;
    vertexBaseClass = null;
    edgeBaseClass = null;
    checkOpeness();
    rollback();
    getMetadata().close();
    ((ODatabaseRaw) underlying.getUnderlying()).callOnCloseListeners();
    getLevel1Cache().clear();
    final OGraphDatabasePool pool = ownerPool;
    ownerPool = null;
    pool.release(this);
