        for (StorageDir storageDir : storageTier.getStorageDirs()) {
            try {
                dirInfos.add(new WorkerDirInfo(storageDir.getStorageDirId(), storageDir.getDirPath().toString(), CommonUtils.objectToByteBuffer(storageDir.getUfsConf())));

        }
    }
    return dirInfos;
