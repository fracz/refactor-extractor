private void parseSample() {
    if (getSqlParser().skipIfEqual(OracleKeyword.SAMPLE)) {
        getSqlParser().skipIfEqual(OracleKeyword.BLOCK);
        getSqlParser().skipParentheses();
        if (getSqlParser().skipIfEqual(OracleKeyword.SEED)) {
            getSqlParser().skipParentheses();
        }
    }
