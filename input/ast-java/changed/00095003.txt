/**
 * Creates the index.
 *
 * @param iDatabase
 * Current Database instance
 * @param iProperty
 * Owner property
 * @param iClusterIndexName
 * Cluster name where to place the TreeMap
 * @param iProgressListener
 * Listener to get called on progress
 */
public OIndex create(final ODatabaseRecord iDatabase, final String iClusterIndexName, final int[] iClusterIdsToIndex, final OProgressListener iProgressListener) {
    configuration = new ODocument(iDatabase);
    if (iClusterIdsToIndex != null)
        for (int id : iClusterIdsToIndex) clustersToIndex.add(iDatabase.getClusterNameById(id));
    map = new OMVRBTreeDatabaseLazySave<String, List<ORecordId>>(iDatabase, iClusterIndexName, OStreamSerializerString.INSTANCE, OStreamSerializerListRID.INSTANCE);
    rebuild(iProgressListener);
    return this;
