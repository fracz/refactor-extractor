static XmlResourceValue parsePlurals(XMLEventReader eventReader) throws XMLStreamException {
    Map<String, String> values = new HashMap<>();
    for (XMLEvent element = eventReader.nextTag(); !isEndTag(element, TAG_PLURALS); element = eventReader.nextTag()) {
        if (isItem(element)) {
            values.put(getElementAttributeByName(element.asStartElement(), ATTR_QUANTITY), eventReader.getElementText());
        }
    }
    return PluralXmlResourceValue.of(values);
