    BytesStreamOutput out = new BytesStreamOutput();
    int expectedSize = (BigArrays.BYTE_PAGE_SIZE * 2) + 1;
    byte[] expectedData = randomizedByteArrayWithSize(expectedSize);
    // write byte-by-byte
    for (int i = 0; i < expectedSize; i++) {
