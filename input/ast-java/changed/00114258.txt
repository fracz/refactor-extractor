private int getFirstVarGroup(String v) {
    // Trim the variable...
    v = v.trim();
    for (int i = 0; i < varGroupBindings.varnames.length; i++) {
        String s = varGroupBindings.varnames[i];
        if (v.equals(s)) {
            if (matchedGroups[i] != null) {
                return i;
            }
        }
    }
    return -1;
