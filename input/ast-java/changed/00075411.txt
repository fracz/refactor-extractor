public PluginDescriptor findPluginDescriptor(String pluginId) {
    URL resource = classLoader.getResource(String.format("META-INF/gradle-plugins/%s.properties", pluginId));
    if (resource == null) {
        return null;
    } else {
        return new PluginDescriptor(resource);
    }
