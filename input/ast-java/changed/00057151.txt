static <T> T copyIfRequired(T obj) {
    if (obj instanceof byte[]) {
        // Copy it
        byte[] bytes = (byte[]) obj;
        byte[] copy = new byte[bytes.length];
        System.arraycopy(bytes, 0, copy, 0, bytes.length);
        return (T) copy;
    } else if (obj instanceof Buffer) {
        // Copy it
        return (T) ((Buffer) obj).copy();
    } else {
        return obj;
    }
