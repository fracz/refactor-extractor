    boolean backet = false;
    if (getLexer().equalToken(Token.LEFT_BRACKET)) {
        getLexer().nextToken();
        backet = true;
    }
    expr = super.dotRest(expr);
    if (backet) {
        accept(Token.RIGHT_BRACKET);
    }
    return expr;
