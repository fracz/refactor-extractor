private void fixFocusableAndClickableSettings() {
    if (mMovement != null || (mEditor != null && getEditor().mKeyListener != null)) {
        setFocusable(true);
        setClickable(true);
        setLongClickable(true);
    } else {
        setFocusable(false);
        setClickable(false);
        setLongClickable(false);
    }
