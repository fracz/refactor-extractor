private void closeIndex() {
    try {
        if (myIndices != null) {
            myIndices.save();
            myIndices.close();
        }
        if (myEmbedder != null)
            myEmbedder.stop();
        myIndices = null;
        myEmbedder = null;
    } catch (MavenEmbedderException e) {
        MavenLog.info(e);
    } catch (MavenIndexException e) {
        logError(e);
    }
