    myEmbedder.getDefaultRequest().setTransferListener(new TransferListenerAdapter());

        WagonManager wagon = (WagonManager) myEmbedder.getPlexusContainer().lookup(WagonManager.ROLE);

        MavenLog.info(e);
    }
