private JSError[] toArray(CheckLevel level) {
    List<JSError> errors = new ArrayList<JSError>(messages.size());
    for (Pair<JSError, CheckLevel> p : messages) {
        if (p.second == level) {
            errors.add(p.first);
        }
    }
