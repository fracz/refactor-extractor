@Override
public boolean onOptionsItemSelected(MenuItem item) {
    switch(item.getItemId()) {
        case R.id.refresh:
            onRefresh(true);
            return true;
        case R.id.accounts:
            onAccounts();
            return true;
        case R.id.compose:
            onCompose();
            return true;
        case R.id.account_settings:
            onEditAccount();
            return true;
        case R.id.empty_trash:
            onEmptyTrash(mAccount);
            return true;
        default:
            return super.onOptionsItemSelected(item);
    }
