@Property(viewable = false, expensive = true, order = 21)
public synchronized Long getRealRowCount(DBRProgressMonitor monitor) {
    if (realRowCount != null) {
        return realRowCount;
    }
    if (!isPersisted()) {
        // Do not count rows for views
        return null;
    }
    if (realRowCount == null) {
        // Query row count
        try (DBCSession session = getDataSource().getDefaultContext(false).openSession(monitor, DBCExecutionPurpose.META, "Read row count")) {
            realRowCount = countData(session, null);
        } catch (DBException e) {
            log.debug("Can't fetch row count", e);
        }
    }
    if (realRowCount == null) {
        realRowCount = -1L;
    }
    return realRowCount;
