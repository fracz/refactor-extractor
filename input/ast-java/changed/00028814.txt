/**
 * 移除插件及其已释放的Dex、Native库等文件
 *
 * @param info 插件信息
 */
public static void forceDelete(PluginInfo info) {
    if (info == null) {
        return;
    }
    try {
        FileUtils.forceDelete(info.getApkFile());
        FileUtils.forceDelete(info.getDexFile());
        FileUtils.forceDelete(info.getNativeLibsDir());
    } catch (IOException e) {
        if (LogRelease.LOGR) {
            e.printStackTrace();
        }
    }
