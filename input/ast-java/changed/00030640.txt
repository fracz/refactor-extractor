static Mask newMask(JSONObject json, LottieComposition composition) {
    MaskMode maskMode;
    switch(json.optString("mode")) {
        case "a":
            maskMode = MaskMode.MaskModeAdd;
            break;
        case "s":
            maskMode = MaskMode.MaskModeSubtract;
            break;
        case "i":
            maskMode = MaskMode.MaskModeIntersect;
            break;
        default:
            maskMode = MaskMode.MaskModeUnknown;
    }
    AnimatableShapeValue maskPath = new AnimatableShapeValue(json.optJSONObject("pt"), composition);
    return new Mask(maskMode, maskPath);
