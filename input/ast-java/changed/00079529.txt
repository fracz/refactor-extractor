private boolean tryToPerformRefresh() {
    if (mStatus != PTR_STATUS_PREPARE) {
        return false;
    }
    if ((mCurrentPos >= mHeaderHeight && mAutoScrollRefreshTag > 0) || mCurrentPos >= mOffsetToRefresh) {
        mStatus = PTR_STATUS_LOADING;
        performRefresh();
    }
    return false;
