            @Override
            public void initChannel(SctpChannel ch) throws Exception {
                ch.pipeline().addLast(new LoggingHandler(LogLevel.INFO), new SctpEchoClientHandler(firstMessageSize));
            }
        });
        // Start the client.
        ChannelFuture f = b.connect().sync();
