@Override
public MultiMap headers() {
    // Lazily decode headers
    if (headers == null) {
        // The message has been read from the wire
        if (headersPos != 0) {
            readHeaders();
        }
        if (headers == null) {
            headers = new CaseInsensitiveHeaders();
        }
    }
    return headers;
