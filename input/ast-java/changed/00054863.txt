/**
 * Update task status.
 *
 * @return true if status of the task with the given id has been updated successfully
 */
public boolean setStatus(final String taskId, final boolean active, final TaskStatusType status) {
    return retryingHandle(new HandleCallback<Boolean>() {

        @Override
        public Boolean withHandle(Handle handle) throws Exception {
            return handle.createStatement(String.format("UPDATE %s SET active = :active, status_payload = :status_payload WHERE id = :id AND active = TRUE", config.getTasksTable())).bind("id", taskId).bind("active", active).bind("status_payload", jsonMapper.writeValueAsBytes(status)).execute() == 1;
        }
    });
