/**
 * Returns the Householder vectors.
 * @return  Lower trapezoidal matrix whose columns define the reflections
 */
public double[][] getH() {
    int m = QR.nrows();
    int n = QR.ncols();
    double[][] H = new double[m][n];
    for (int i = 0; i < m; i++) {
        for (int j = 0; j < n; j++) {
            if (i >= j) {
                H[i][j] = QR.get(i, j);
            } else {
                H[i][j] = 0.0;
            }
        }
    }
    return H;
