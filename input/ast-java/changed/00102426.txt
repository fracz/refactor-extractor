            free(lastValues);

            lastValues = null;
            nextIterPos = 0;
            if (!tryAgain()) {
                return false;

            finished = false;

        do {
            prevIterPos = nextIterPos;

            if (lastValues != null) {
                free(lastValues);
            }
            lastValues = convert(res.getMap());
            client = res.getRedisClient();
            if (nextIterPos == 0 && firstValues == null) {
                firstValues = lastValues;
                lastValues = null;
                if (firstValues.isEmpty() && tryAgain()) {
                    client = null;
                    firstValues = null;
                    prevIterPos = -1;
                }
            } else {
                if (firstValues.isEmpty()) {
                    firstValues = lastValues;
                    lastValues = null;
                    if (firstValues.isEmpty() && tryAgain()) {

                    removeExecuted = false;
                    client = null;

                    lastValues = null;
                    nextIterPos = 0;
                    prevIterPos = -1;
                    if (tryAgain()) {
                        continue;
                    }
                    finished = true;
                    return false;
                }

            lastIter = res.getMap().entrySet().iterator();
