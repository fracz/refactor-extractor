@Override
public PinnablePage load(PageSwapper io, long pageId, PageLock lock) throws IOException {
    StandardPinnablePage page = nextFreePage();
    if (page.pin(null, UNBOUND_PAGE_ID, lock)) {
        page.reset(io, pageId);
        page.load();
        monitor.pageFault(pageId, io);
    } else {
        throw new IOException("Tried to load a page (page fault) but the page in the free-list " + "was already bound to a file and a pageId: " + page);
    }
    return page;
