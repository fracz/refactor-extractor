private void createTransaction(GraphDatabaseAPI db) {
    Transaction tx = db.beginTx();
    try {
        db.createNode();
        tx.success();
    } catch (RuntimeException e) {
        e.printStackTrace();
        throw e;
    } finally {
        tx.finish();
    }
