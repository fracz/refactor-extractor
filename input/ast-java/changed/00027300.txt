/*
     * statement
     *  : expression
     *  : declaration
     *  ;
     */
private void parseStatement() {
    if (!parseLocalDeclaration()) {
        if (!atSet(EXPRESSION_FIRST)) {
            errorAndAdvance("Expecting a statement");
        } else {
            parseExpression();
        }
    }
