private static boolean markChannelNotReadable(ChannelHandlerContext ctx) {
    // Catch any unexpected exception when marking the channel.
    ctx.setAttachment(new DiscardEvent());
    try {
        ctx.getChannel().setReadable(false);
    } catch (Exception ex) {
        log.debug(ex);
        return false;
    }
    return true;
