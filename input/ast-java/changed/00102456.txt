public void testRetainAllNoModify() {
    RSetCacheReactive<Integer> set = redisson.getSetCache("set");
    sync(set.add(1));
    sync(set.add(2));
    // nothing changed
    Assert.assertFalse(sync(set.retainAll(Arrays.asList(1, 2))));
