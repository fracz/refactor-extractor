    }

        for (int r = 0; r < REL_COUNT; r++) {
            for (int j = 1; j <= EDGE_COUNT; j++) {
                nodes[i].addEdge("rel" + r, nodes[wrapAround(i + j, NODE_COUNT)]);
            }
        }
    }
