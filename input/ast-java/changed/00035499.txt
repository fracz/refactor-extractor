static boolean isSpecial(char ch, int features) {
    if (ch == ' ') {
        return false;
    }
    if (ch == '/' && SerializerFeature.isEnabled(features, SerializerFeature.WriteSlashAsSpecial)) {
        return true;
    }
    if (ch > '#' && ch != '\\') {
        return false;
    }
    if (ch <= 0x1F || ch == '\\' || ch == '"') {
        return true;
    }
    return false;
