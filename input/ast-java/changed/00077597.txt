/**
 * Processes the payload of a status frame.
 */
private void processStatus() {
    try {
        int statusCode = nextFrame.readUnsignedShort();
        Transport.Code code = Transport.Code.valueOf(statusCode);
        notifyStatus(code != null ? new Status(code) : new Status(Transport.Code.UNKNOWN, "Unknown status code " + statusCode));
    } finally {
        nextFrame.close();
        nextFrame = null;
    }
