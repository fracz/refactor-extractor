private TitanTransaction getAutoStartTx() {
    if (txs == null)
        ExceptionFactory.graphShutdown();
    TitanTransaction tx = txs.get();
    if (tx == null) {
        tx = newThreadBoundTransaction();
        txs.set(tx);
        openTx.put(tx, Boolean.TRUE);
    }
    return tx;
