@Override
public int compareTo(IntTuple o) {
    int commonLen = Math.min(o.length(), length());
    for (int i = 0; i < commonLen; i++) {
        int a = get(i);
        int b = o.get(i);
        if (a < b)
            return -1;
        if (b < a)
            return 1;
    }
    if (o.length() == length()) {
        return 0;
    } else {
        return (length() < o.length()) ? -1 : 1;
    }
