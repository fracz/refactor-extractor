private void checkWorkspaceIsRunning(WorkspaceImpl workspace, String operation) throws ConflictException {
    if (workspace.getStatus() != RUNNING) {
        throw new ConflictException(format("Could not %s the workspace '%s:%s' because its status is '%s'.", operation, workspace.getNamespace(), workspace.getConfig().getName(), workspace.getStatus()));
