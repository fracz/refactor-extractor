private PsiType handleConversion(PsiType paramType, PsiType argType) {
    final GroovyPsiElement context = myPlace;
    if (!TypesUtil.isAssignable(TypeConversionUtil.erasure(paramType), argType, context, true) && TypesUtil.isAssignableByMethodCallConversion(paramType, argType, context, true)) {
        return paramType;
    }
    return argType;
