/**
 * 关闭所有activity
 */
public void killAll() {
    LinkedList<BaseActivity> copy;
    synchronized (AppManager.class) {
        copy = new LinkedList<>(getActivityList());
    }
    for (BaseActivity baseActivity : copy) {
        baseActivity.finish();
    }
