@Override
public void partyAction(Action action) {
    String s = this + " ";
    switch(action) {
        case ENEMY:
            s = s + "runs for cover";
            break;
        case GOLD:
            s = s + "takes his share of the gold";
            break;
        case HUNT:
            s = s + "arrives for dinner";
            break;
        case TALE:
            s = s + "comes to listen";
            break;
        default:
            break;
    }
    System.out.println(s);
