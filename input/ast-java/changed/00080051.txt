@Override
public boolean touchDown(float x, float y, int pointer) {
    if (pointer != 0)
        return false;
    if (hit(x, y) != null) {
        parent.keyboardFocus(this);
        keyboard.show(true);
        x = x - renderOffset;
        for (int i = 0; i < glyphPositions.size; i++) {
            float pos = glyphPositions.items[i];
            if (pos > x) {
                cursor = Math.max(0, i - 1);
                break;
            }
        }
        return true;
    } else
        return false;
