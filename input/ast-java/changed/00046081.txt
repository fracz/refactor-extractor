private static List<Row> readRegular(List<ReadCommand> commands, ConsistencyLevel consistencyLevel) throws UnavailableException, ReadFailureException, ReadTimeoutException {
    long start = System.nanoTime();
    List<Row> rows = null;
    try {
        rows = fetchRows(commands, consistencyLevel);
    } catch (UnavailableException e) {
        readMetrics.unavailables.mark();
        throw e;
    } catch (ReadTimeoutException e) {
        readMetrics.timeouts.mark();
        throw e;
    } catch (ReadFailureException e) {
        readMetrics.failures.mark();
        throw e;
    } finally {
        long latency = System.nanoTime() - start;
        readMetrics.addNano(latency);
        // TODO avoid giving every command the same latency number.  Can fix this in CASSADRA-5329
        for (ReadCommand command : commands) Keyspace.open(command.ksName).getColumnFamilyStore(command.cfName).metric.coordinatorReadLatency.update(latency, TimeUnit.NANOSECONDS);
    }
    return rows;
