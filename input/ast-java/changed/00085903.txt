@Override
public ContainerRequest filter(ContainerRequest request) {
    try {
        List<String> headers = request.getRequestHeader("User-Agent");
        if (!headers.isEmpty()) {
            userAgents.add(headers.get(0).split(" ")[0]);
        }
    } catch (RuntimeException e) {
    // We're fine with that
    }
    return request;
