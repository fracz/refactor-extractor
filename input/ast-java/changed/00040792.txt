public int getIsTargetServiceActive(ComponentName cname, int userId) {
    synchronized (mAuthorities) {
        if (cname != null) {
            AuthorityInfo authority = getAuthorityLocked(new EndPoint(cname, userId), "get service enabled");
            if (authority == null) {
                return -1;
            }
            return authority.syncable;
        }
        return -1;
    }
