private String makeArgName(RegisterArg arg) {
    if (fallback) {
        return getFallbackName(arg);
    }
    String name = arg.getName();
    String varName;
    if (name != null) {
        if ("this".equals(name)) {
            return name;
        }
        varName = name;
    } else {
        varName = makeNameForType(arg.getType());
    }
    if (NameMapper.isReserved(varName)) {
        return varName + "R";
    }
    return varName;
