/**
 * Stops the communication with the RWE Smarthome Central.
 */
public void stop() {
    runPoller = false;
    if (rweSmarthomeSession != null) {
        logger.info("Shutting down RWE Smarthome communicator");
        try {
            rweSmarthomeSession.destroy();
        } finally {
            rweSmarthomeSession = null;
        }
    }
