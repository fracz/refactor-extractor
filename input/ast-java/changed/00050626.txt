@Override
public void downloadTree(Digest rootDigest, Path rootLocation) throws IOException, CacheNotFoundException {
    Directory directory = Directory.parseFrom(downloadBlob(rootDigest));
    for (FileNode file : directory.getFilesList()) {
        downloadFileContents(file.getDigest(), rootLocation.getRelative(file.getName()), file.getIsExecutable());
    }
    for (DirectoryNode child : directory.getDirectoriesList()) {
        downloadTree(child.getDigest(), rootLocation.getRelative(child.getName()));
    }
