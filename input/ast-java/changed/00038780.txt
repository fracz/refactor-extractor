@Override
public void finishOperation(int code, int uid, String packageName) {
    uid = handleIncomingUid(uid);
    synchronized (this) {
        Op op = getOpLocked(code, uid, packageName);
        if (op == null) {
            return;
        }
        if (op.duration != -1) {
            Slog.w(TAG, "Ignoring finishing op not started: uid " + uid + " pkg " + packageName + " code " + code + " time=" + op.time + " duration=" + op.duration);
            return;
        }
        op.duration = (int) (System.currentTimeMillis() - op.time);
    }
