@Test
public void testInheritHierarchy() {
    // Test hierarchies
    // A defaults to A
    Setting<String> a = setting("A", STRING, "A").build();
    // B defaults to B unless A is defined
    Setting<String> b = setting("B", STRING, "B").inherits(a).build();
    // C defaults to C unless B is defined
    Setting<String> c = setting("C", STRING, "C").inherits(b).build();
    // D defaults to B
    Setting<String> d = setting("D", STRING).inherits(b).build();
    // E defaults to D (hence B)
    Setting<String> e = setting("E", STRING).inherits(d).build();
    assertThat(c.apply(map(stringMap("C", "X"))), equalTo("X"));
    assertThat(c.apply(map(stringMap("B", "X"))), equalTo("X"));
    assertThat(c.apply(map(stringMap("A", "X"))), equalTo("X"));
    assertThat(c.apply(map(stringMap("A", "Y", "B", "X"))), equalTo("X"));
    assertThat(d.apply(map(stringMap())), equalTo("B"));
    assertThat(e.apply(map(stringMap())), equalTo("B"));
