@Override
public void dispose() {
    if (myDisposed)
        return;
    UIUtil.invokeLaterIfNeeded(new Runnable() {

        @Override
        public void run() {
            if (myDisposed)
                return;
            myDisposed = true;
            onDispose();
            Disposer.dispose(myViewer);
            myToolbarStatusPanel.setContent(null);
            myToolbarPanel.setContent(null);
            myContentPanel.setContent(null);
            myBottomPanel.setContent(null);
        }
    });
