private static void closeStorage(File root, @Nullable AbstractStateStorage<?, ?> storage) {
    if (storage != null) {
        synchronized (storage) {
            try {
                storage.close();
            } catch (IOException e) {
                LOG.error(e);
                FileUtil.delete(root);
            }
        }
    }
