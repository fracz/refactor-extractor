private int[] getCheckedRows() {
    int count = 0;
    for (MemberInfo info : myMemberInfos) {
        if (info.isChecked()) {
            count++;
        }
    }
    int[] rows = new int[count];
    int currentRow = 0;
    for (int idx = 0; idx < myMemberInfos.length; idx++) {
        if (myMemberInfos[idx].isChecked()) {
            rows[currentRow++] = idx;
        }
    }
    return rows;
