@Override
SimpleConfigList resolveSubstitutions(final ResolveContext context, ResolveSource source) throws NotPossibleToResolve {
    if (resolved)
        return this;
    if (context.isRestrictedToChild()) {
        // so nothing to do.
        return this;
    } else {
        final ResolveSource sourceWithParent = source.pushParent(this);
        try {
            return modifyMayThrow(new Modifier() {

                @Override
                public AbstractConfigValue modifyChildMayThrow(String key, AbstractConfigValue v) throws NotPossibleToResolve {
                    return context.resolve(v, sourceWithParent);
                }
            }, ResolveStatus.RESOLVED);
        } catch (NotPossibleToResolve e) {
            throw e;
        } catch (RuntimeException e) {
            throw e;
        } catch (Exception e) {
            throw new ConfigException.BugOrBroken("unexpected checked exception", e);
        }
    }
