    Node n = newNode(transformBinaryTokenType(exprNode.operator.type), transform(exprNode.left), transform(exprNode.right));
    if (isAssignmentOp(n)) {
        Node target = n.getFirstChild();
        if (!target.isValidAssignmentTarget()) {

        }
    }
    return n;
