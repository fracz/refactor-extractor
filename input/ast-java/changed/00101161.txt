@Override
public void process() {
    adminReceiveBuffer.read((eventTypeId, buffer, index, length) -> {
        switch(eventTypeId) {
            case ControlProtocolEvents.ADD_CHANNEL:
                channelMessage.reset(buffer, index);
                onAddChannel(channelMessage.destination(), channelMessage.sessionId(), channelMessage.channelId());
                return;
            case ControlProtocolEvents.REMOVE_CHANNEL:
                channelMessage.reset(buffer, index);
                onRemoveChannel(channelMessage.destination(), channelMessage.sessionId(), channelMessage.channelId());
                return;
        }
    });
