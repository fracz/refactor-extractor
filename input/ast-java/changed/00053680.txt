public void putStaticInfo(String sessionID, String workerID, byte[] staticInfo) {
    logIDs(sessionID, workerID);
    SessionTypeWorkerId id = new SessionTypeWorkerId(sessionID, workerID);
    this.staticInfo.put(id, staticInfo);
