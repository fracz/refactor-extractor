static void writeThrowsList(StringBuilder text, PsiReferenceList throwsList, PsiClassType[] exceptions, final ClassNameProvider classNameProvider) {
    if (exceptions.length <= 0)
        return;
    text.append("throws ");
    for (int i = 0; i < exceptions.length; i++) {
        PsiClassType exception = exceptions[i];
        if (i != 0) {
            text.append(",");
        }
        writeType(text, exception, throwsList, classNameProvider);
        text.append(" ");
    }
