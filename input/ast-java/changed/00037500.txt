void tearDownPipes() {
    // null out the pipes twice.
    synchronized (this) {
        if (mPipes != null) {
            try {
                mPipes[0].close();
                mPipes[0] = null;
                mPipes[1].close();
                mPipes[1] = null;
            } catch (IOException e) {
                Slog.w(RefactoredBackupManagerService.TAG, "Couldn't close agent pipes", e);
            }
            mPipes = null;
        }
    }
