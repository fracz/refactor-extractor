private static int gravityIntToGravity(@GravityInt int gravity) {
    switch(gravity) {
        case START:
            return Gravity.START;
        case CENTER:
            return Gravity.CENTER_HORIZONTAL;
        case END:
            return Gravity.END;
        default:
            return Gravity.START;
    }
