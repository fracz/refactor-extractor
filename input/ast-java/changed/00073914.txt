private void runStop(Stop stop, Connection<Object> connection) {
    try {
        connection.dispatch(stop);
        try {
            connection.receive();
        } catch (Exception e) {
            LOGGER.info("After sending stop command to the daemon we couldn't receive a message. The daemon is already stopped.");
        }
    } finally {
        connection.stop();
    }
