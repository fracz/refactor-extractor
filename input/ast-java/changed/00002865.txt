/**
 * 退出应用程序
 */
public void AppExit() {
    try {
        killAll();
        ActivityManager activityMgr = (ActivityManager) mApplication.getSystemService(Context.ACTIVITY_SERVICE);
        activityMgr.killBackgroundProcesses(mApplication.getPackageName());
        System.exit(0);
    } catch (Exception e) {
    }
