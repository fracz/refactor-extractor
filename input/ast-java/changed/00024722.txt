                return null;
            ClassDescriptor classDescriptor = semanticServices.getDescriptorResolver().resolveClass(new FqName(innerClass.getQualifiedName()), DescriptorSearchRule.IGNORE_IF_FOUND_IN_KOTLIN);
            if (classDescriptor != null) {
                return classDescriptor;

        }
    }
