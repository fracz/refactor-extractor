/**
 * Adds a view to the graph.
 *
 * @param view The view to be added as a node to the graph.
 */
void add(View view) {
    final int id = view.getId();
    if (id != View.NO_ID) {
        mNodes.put(id, Node.acquire(view));
    } else {
        mNakedRoots.add(view);
    }
