protected void setFps(float fps) {
    if (!setFps || targetFps != fps) {
        if (60 < fps) {
            // Disables v-sync
            gl.setSwapInterval(0);
        } else if (30 < fps) {
            gl.setSwapInterval(1);
        } else {
            gl.setSwapInterval(2);
        }
        targetFps = currentFps = fps;
        setFps = true;
    }
