@LargeTest
public void testPerformAccessibilityActionFocus() throws Exception {
    final long startTimeMillis = SystemClock.uptimeMillis();
    try {
        // find a view and make sure it is not focused
        AccessibilityNodeInfo button = AccessibilityInteractionClient.getInstance().findAccessibilityNodeInfoByViewIdInActiveWindow(sConnectionId, R.id.button5);
        assertFalse(button.isFocused());
        // focus the view
        assertTrue(button.performAction(ACTION_FOCUS));
        // find the view again and make sure it is focused
        button = AccessibilityInteractionClient.getInstance().findAccessibilityNodeInfoByViewIdInActiveWindow(sConnectionId, R.id.button5);
        assertTrue(button.isFocused());
    } finally {
        if (DEBUG) {
            final long elapsedTimeMillis = SystemClock.uptimeMillis() - startTimeMillis;
            Log.i(LOG_TAG, "testPerformAccessibilityActionFocus: " + elapsedTimeMillis + "ms");
        }
    }
