public static JLabel createErrorLabel(String text) {
    JLabel label = new HintLabel();
    label.setText(text);
    label.setIcon(ERROR_ICON);
    // label.setBorder(BorderFactory.createLineBorder(Color.black));
    label.setBorder(BorderFactory.createCompoundBorder(new SideBorder2(Color.white, Color.white, Color.gray, Color.gray, 1), BorderFactory.createEmptyBorder(2, 2, 2, 2)));
    label.setForeground(Color.black);
    label.setFont(getBoldFont());
    label.setBackground(ERROR_COLOR);
    label.setOpaque(true);
    return label;
