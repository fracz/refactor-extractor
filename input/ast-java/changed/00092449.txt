public static String java2unicode(final String iInput) {
    final StringBuffer result = new StringBuffer();
    char ch;
    String hex;
    for (int i = 0; i < iInput.length(); i++) {
        ch = iInput.charAt(i);
        if (// Does the char need to be converted to unicode?
        ch >= 0x0020 && ch <= 0x007e)
            // No.
            result.append(ch);
        else // Yes.
        {
            // standard unicode format.
            result.append("\\u");
            // Get hex value of the char.
            hex = Integer.toHexString(iInput.charAt(i) & 0xFFFF);
            for (int j = 0; j < 4 - hex.length(); j++) // Prepend zeros because unicode requires 4 digits
            result.append('0');
            // standard unicode format.
            result.append(hex.toLowerCase());
        // ostr.append(hex.toLowerCase(Locale.ENGLISH));
        }
    }
    return result.toString();
