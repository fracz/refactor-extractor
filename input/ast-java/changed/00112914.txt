@Test
public void jsonPojoFlux() throws Exception {
    HttpUrl baseUrl = server.url("/pojos");
    this.server.enqueue(new MockResponse().setHeader("Content-Type", "application/json").setBody("[{\"bar\":\"bar1\",\"foo\":\"foo1\"},{\"bar\":\"bar2\",\"foo\":\"foo2\"}]"));
    ClientRequest<Void> request = ClientRequest.GET(baseUrl.toString()).accept(MediaType.APPLICATION_JSON).build();
    Flux<Pojo> result = this.webClient.exchange(request).flatMap(response -> response.body(toFlux(Pojo.class)));
    TestSubscriber.subscribe(result).awaitAndAssertNextValuesWith(p -> assertThat(p.getBar(), Matchers.is("bar1")), p -> assertThat(p.getBar(), Matchers.is("bar2"))).assertValueCount(2).assertComplete();
    RecordedRequest recordedRequest = server.takeRequest();
    assertEquals(1, server.getRequestCount());
    assertEquals("/pojos", recordedRequest.getPath());
    assertEquals("application/json", recordedRequest.getHeader(HttpHeaders.ACCEPT));
