private void printInstanceHelp(PrintStream out, Command instance) {
    boolean firstLine = true;
    for (String line : instance.getDescription().split("\n")) {
        String prefix;
        if (firstLine) {
            prefix = instance.getUsage() + ":\t";
            firstLine = false;
        } else {
            prefix = "\t\t";
        }
        System.out.println(prefix + line);
    }
