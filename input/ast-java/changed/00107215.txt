@NotNull
public static Image getTypeImage(DBSTypedObject column) {
    Image image = null;
    if (column instanceof DBPImageProvider) {
        image = ((DBPImageProvider) column).getObjectImage();
    }
    if (image == null) {
        image = DBIcon.TREE_COLUMN.getImage();
    }
    return image;
