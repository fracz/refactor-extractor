@Override
public Token LT(int k) {
    lazyInit();
    if (k == 0)
        return null;
    if (k < 0)
        return LB(-k);
    int i = p + k - 1;
    sync(i);
    if (i >= tokens.size()) {
        // EOF must be last token
        return tokens.get(tokens.size() - 1);
    }
    // if ( i>range ) range = i;
    return tokens.get(i);
