HashSet verifyUUIDSet(int count, UUIDGenerator uuidSource) {
    HashSet<String> uuidSet = new HashSet<>();
    for (int i = 0; i < count; ++i) {
        uuidSet.add(uuidSource.getBase64UUID());
    }
    assertEquals(count, uuidSet.size());
    return uuidSet;
