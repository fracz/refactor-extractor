public String getPassword(Account account) throws RemoteException {
    SQLiteDatabase db = mOpenHelper.getReadableDatabase();
    Cursor cursor = db.query(TABLE_ACCOUNTS, new String[] { ACCOUNTS_PASSWORD }, ACCOUNTS_NAME + "=? AND " + ACCOUNTS_TYPE + "=?", new String[] { account.mName, account.mType }, null, null, null);
    try {
        if (cursor.moveToNext()) {
            return cursor.getString(0);
        }
        return null;
    } finally {
        cursor.close();
    }
