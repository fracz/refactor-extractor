@Override
public Flux<String> decode(Publisher<DataBuffer> inputStream, ResolvableType elementType, MimeType mimeType, Object... hints) {
    Flux<DataBuffer> inputFlux = Flux.from(inputStream);
    if (this.splitOnNewline) {
        inputFlux = inputFlux.flatMap(dataBuffer -> {
            List<DataBuffer> tokens = DataBufferUtils.tokenize(dataBuffer, b -> b == '\n' || b == '\r');
            return Flux.fromIterable(tokens);
        });
    }
    Charset charset = getCharset(mimeType);
    return inputFlux.map(dataBuffer -> {
        CharBuffer charBuffer = charset.decode(dataBuffer.asByteBuffer());
        DataBufferUtils.release(dataBuffer);
        return charBuffer.toString();
    });
