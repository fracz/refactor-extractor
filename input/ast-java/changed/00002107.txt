private void copyUrl() {
    String url = mWebView.getUrl();
    if (!TextUtils.isEmpty(url)) {
        ClipboardUtils.copyText(mWebView.getTitle(), url, this);
    } else {
        ToastUtils.show(R.string.webview_copy_url_empty, this);
    }
