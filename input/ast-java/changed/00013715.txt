/**
 * Adds all children of specified element to given list
 *
 * @param elem
 * @param list
 * @param indent
 * @param alignment
 * @param groovySettings
 */
private static void addBinaryChildrenRecursively(PsiElement elem, List<Block> list, Indent indent, Alignment alignment, Wrap myWrap, CommonCodeStyleSettings mySettings, GroovyCodeStyleSettings groovySettings, Map<PsiElement, Alignment> alignments) {
    if (elem == null)
        return;
    // For binary expressions
    if ((elem instanceof GrBinaryExpression)) {
        GrBinaryExpression myExpr = ((GrBinaryExpression) elem);
        if (myExpr.getLeftOperand() instanceof GrBinaryExpression) {
            addBinaryChildrenRecursively(myExpr.getLeftOperand(), list, Indent.getContinuationWithoutFirstIndent(), alignment, myWrap, mySettings, groovySettings, alignments);
        }
        PsiElement op = ((GrBinaryExpression) elem).getOperationToken();
        for (ASTNode childNode : visibleChildren(elem.getNode())) {
            PsiElement psi = childNode.getPsi();
            if (!(psi instanceof GrBinaryExpression)) {
                list.add(new GroovyBlock(childNode, op == psi ? alignments.get(op) : alignment, indent, myWrap, mySettings, groovySettings, alignments));
            }
        }
        if (myExpr.getRightOperand() instanceof GrBinaryExpression) {
            addBinaryChildrenRecursively(myExpr.getRightOperand(), list, Indent.getContinuationWithoutFirstIndent(), alignment, myWrap, mySettings, groovySettings, alignments);
        }
    }
