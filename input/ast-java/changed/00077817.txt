 * Returns the matrix transpose.
 */
@Override
public Matrix transpose() {
    int m = A.length;
    int n = A[0].length;
    double[][] B = new double[n][m];
    for (int i = 0; i < m; i++) {
        for (int j = 0; j < n; j++) {
            B[j][i] = A[i][j];
        }
    }
    return new Matrix(B);
