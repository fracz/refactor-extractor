@Override
public void clearReleasables() {
    if (clearables != null) {
        Throwable th = null;
        for (Releasable releasable : clearables) {
            try {
                releasable.release();
            } catch (Throwable t) {
                if (th == null) {
                    th = t;
                }
            }
        }
        clearables.clear();
        if (th != null) {
            throw new RuntimeException(th);
        }
    }
