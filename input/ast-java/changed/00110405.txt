private void declareVars(CodeWriter code, IContainer cont) {
    DeclareVariablesAttr declVars = (DeclareVariablesAttr) cont.getAttributes().get(AttributeType.DECLARE_VARIABLES);
    if (declVars != null) {
        for (RegisterArg v : declVars.getVars()) {
            code.startLine();
            declareVar(code, v);
            code.add(';');
        }
    }
