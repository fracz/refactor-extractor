public void addArtifact(File repository, MavenId artifact) {
    if (!isInitialized)
        return;
    for (MavenIndex each : getIndices()) {
        if (repository.equals(each.getRepositoryFile())) {
            try {
                each.addArtifact(artifact);
            } catch (MavenIndexException e) {
                MavenLog.error(e);
            }
            return;
        }
    }
