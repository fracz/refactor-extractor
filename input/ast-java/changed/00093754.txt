@Override
public void close() {
    if (getLockRead().tryAcquireLock())
        try {
            readCondition.signalAll();
        } finally {
            releaseReadLock();
        }
    super.close();
    if (serviceThread != null) {
        final OAsynchChannelServiceThread s = serviceThread;
        serviceThread = null;
        if (s != null)
            // CHECK S BECAUSE IT COULD BE CONCURRENTLY RESET
            s.sendShutdown();
    }
