@Override
public void onReceive(Context context, Intent intent) {
    final String action = intent.getAction();
    if (action.equals(AudioManager.RINGER_MODE_CHANGED_ACTION)) {
        mSoundButton.setAlpha(getSilentMode() ? 0x7F : 0xFF);
    } else if (action.equals(Intent.ACTION_BATTERY_CHANGED)) {
        updateBattery(intent);
    } else if (action.equals(WifiManager.RSSI_CHANGED_ACTION) || action.equals(WifiManager.WIFI_STATE_CHANGED_ACTION) || action.equals(WifiManager.SUPPLICANT_STATE_CHANGED_ACTION) || action.equals(WifiManager.NETWORK_STATE_CHANGED_ACTION)) {
        updateWifi(intent);
    }
