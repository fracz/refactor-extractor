@Override
public final void activityDestroyed(IBinder token) {
    if (DEBUG_SWITCH)
        Slog.v(TAG, "ACTIVITY DESTROYED: " + token);
    synchronized (this) {
        ActivityStack stack = ActivityRecord.getStackLocked(token);
        if (stack != null) {
            stack.activityDestroyedLocked(token);
        }
    }
