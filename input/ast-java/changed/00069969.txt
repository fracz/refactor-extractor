        retrieveCookieIfNecessary();
        retrieveWork();

            LOG.debug("[Agent Loop] Successfully retrieved work.");
        }
    } catch (Exception e) {
        if (isCausedBySecurity(e)) {
            handleIfSecurityException(e);

            if (LOG.isDebugEnabled()) {

            }
        } else {
            LOG.error("[Agent Loop] Error occurred during loop: ", e);
        }
    }
