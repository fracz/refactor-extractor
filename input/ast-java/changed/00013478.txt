@Nullable
public PsiClass resolve() {
    ResolveResult[] results = multiResolve();
    if (results.length == 0)
        return null;
    if (results.length == 1) {
        PsiElement only = results[0].getElement();
        return only instanceof PsiClass ? (PsiClass) only : // constructor
        only instanceof PsiMethod ? ((PsiMethod) only).getContainingClass() : null;
    }
    PsiElement first = results[0].getElement();
    return // constructor
    first instanceof PsiMethod ? ((PsiMethod) first).getContainingClass() : null;
