        maxRetries = cfgMaxRetries;

    if (dumpEveryMs != null && dumpEveryMs > 0) {
        dumpTask = new TimerTask() {

            @Override

                dumpProgress();

        };

        t.end();
    }

        source.begin();
    extractor.begin();
