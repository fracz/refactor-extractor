/**
 * Asks the user for a decision when moving existent resource.
 */
private void resolveMoveConflict(String cause) {
    ChoiceDialog dialog = dialogFactory.createChoiceDialog("Move conflict", cause, "Rename", "Skip", "Overwrite", new ConfirmCallback() {

        @Override
        public void accepted() {
            /**
             * Rename
             */
            moveWithNewName();
        }
    }, new ConfirmCallback() {

        @Override
        public void accepted() {
            /**
             * Skip
             */
            move();
        }
    }, new ConfirmCallback() {

        @Override
        public void accepted() {
            /**
             * Overwrite
             */
            moveWithOverwriting();
        }
    });
    dialog.show();
