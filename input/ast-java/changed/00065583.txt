public void stop() {
    running.set(false);
    Thread joinThread = currentJoinThread.getAndSet(null);
    if (joinThread != null) {
        try {
            joinThread.interrupt();

        try {
            joinThread.join(10000);
        } catch (InterruptedException e) {
            Thread.currentThread().interrupt();
        }
    }
