private TypeParameterDescriptor resolveTypeParameter(DeclarationDescriptor containingDescriptor, WritableScope extensibleScope, JetTypeParameter typeParameter) {
    // TODO: other bounds from where-clause
    JetTypeReference extendsBound = typeParameter.getExtendsBound();
    JetType bound = extendsBound == null ? JetStandardClasses.getDefaultBound() : typeResolver.resolveType(extensibleScope, extendsBound);
    TypeParameterDescriptor typeParameterDescriptor = new TypeParameterDescriptor(containingDescriptor, AttributeResolver.INSTANCE.resolveAttributes(typeParameter.getModifierList()), typeParameter.getVariance(), typeParameter.getName(), Collections.singleton(bound), bound);
    extensibleScope.addTypeParameterDescriptor(typeParameterDescriptor);
    trace.recordDeclarationResolution(typeParameter, typeParameterDescriptor);
    return typeParameterDescriptor;
