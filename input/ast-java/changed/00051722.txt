/**
 * Returns true if the type is acceptable to be returned to the Skylark language.
 */
public static boolean isSkylarkAcceptable(Class<?> c) {
    return // implements SkylarkValue
    SkylarkValue.class.isAssignableFrom(c) || // basic values
    c.equals(String.class) || c.equals(Integer.class) || c.equals(Boolean.class) || // there is a registered Skylark ancestor class (useful e.g. when using AutoValue)
    SkylarkInterfaceUtils.getSkylarkModule(c) != null || // will be converted to SkylarkDict
    ImmutableMap.class.isAssignableFrom(c) || // will be converted to SkylarkNestedSet
    NestedSet.class.isAssignableFrom(c) || // other known class
    c.equals(PathFragment.class);
