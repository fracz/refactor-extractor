    int len = strArray.length;

    int end = start;
    while (true) {
        if (end + 1 >= len) {
            break;
        }
        if (strArray[end] == 0 && strArray[end + 1] == 0) {
            break;
        }
        end += 2;
    }
    return new String(strArray, start, end - start, ParserStream.STRING_CHARSET_UTF16);
