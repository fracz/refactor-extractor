/**
 * Initializes state of task file
 *
 * @param task task which task file belongs to
 */
public void init(Task task) {
    myTask = task;
    for (Window window : windows) {
        window.init(this);
    }
    Collections.sort(windows);
    for (int i = 0; i < windows.size(); i++) {
        windows.get(i).setIndex(i);
    }
