/**
 * @return The number of times the the predicate is true for the node
 * or any of its children.
 */
static int getCount(Node n, Predicate<Node> pred) {
    int total = 0;
    if (pred.apply(n)) {
        total++;
    }
    for (Node c = n.getFirstChild(); c != null; c = c.getNext()) {
        total += getCount(c, pred);
    }
    return total;
