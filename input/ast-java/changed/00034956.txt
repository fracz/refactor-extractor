            lexer.nextToken();

            item.setKeyOrIndex(Token.KEY.name);
            lexer.nextToken();
        }
    } else {
        if (lexer.token() == Token.INDEX) {

            accept(Token.INDEX);

            item.setKeyOrIndex(Token.KEY.name);
            accept(Token.KEY);
        }
    }
    if (lexer.token() == Token.LPAREN) {
        lexer.nextToken();

        item.setName(this.exprParser.name());
        accept(Token.LPAREN);

        break;
