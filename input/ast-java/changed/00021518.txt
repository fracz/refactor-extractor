private static int indentWithInVisualColumns(String indent, int tabSize) {
    if (tabSize <= 1) {
        return indent.length();
    }
    int result = 0;
    for (int i = 0; i < indent.length(); i++) {
        char c = indent.charAt(i);
        if (c == '\t') {
            result += tabSize - result % tabSize;
        } else {
            result++;
        }
    }
    return result;
