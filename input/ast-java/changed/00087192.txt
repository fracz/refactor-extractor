public Response<Integer> createRelationshipType(SlaveContext context, String name) {
    // Does this type exist already?
    Integer id = getConfig().getRelationshipTypeHolder().getIdFor(name);
    if (id != null) {
        // OK, return
        return packResponse(context, id, ALL);
    }
    // No? Create it then
    Config config = getConfig();
    id = config.getRelationshipTypeCreator().getOrCreate(txManager, config.getIdGeneratorModule().getIdGenerator(), config.getPersistenceModule().getPersistenceManager(), config.getRelationshipTypeHolder(), name);
    return packResponse(context, id, ALL);
