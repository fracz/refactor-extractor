public void keyPressed(KeyEvent e) {
    if (!silverGhost.isDataLoaded) {
        openArchive();

    }
    if (KeyUtils.isLeftArrowPressed(e)) {
        openArchive();

    final String textFromTypingArea = processKeyPressWithTypedText(e, toolbar.getText());
    final boolean isViewTopClassKeyPressed = KeyUtils.isRightArrowPressed(e) || KeyUtils.isCommandKeyPressed(e);
