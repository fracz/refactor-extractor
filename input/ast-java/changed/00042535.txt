        Slog.d(TAG, "onServiceConnected():" + name);

        mService = IAutoFillService.Stub.asInterface(service);

            Slog.w(TAG, "Exception on service.onConnected(): " + e);
            return;

        if (!mQueuedRequests.isEmpty()) {
