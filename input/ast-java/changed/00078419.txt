@Override
public void loadMessageForViewHeadersAvailable(Account account, String folder, String uid, final Message message) {
    if (!mMessageReference.uid.equals(uid) || !mMessageReference.folderName.equals(folder) || !mMessageReference.accountUuid.equals(account.getUuid())) {
        return;
    }
    MessageView.this.mMessage = message;
    if (!message.isSet(Flag.X_DOWNLOADED_FULL) && !message.isSet(Flag.X_DOWNLOADED_PARTIAL)) {
        mHandler.post(new Runnable() {

            public void run() {
                mMessageContentView.loadUrl("file:///android_asset/downloading.html");
                updateDecryptLayout();
            }
        });
    }
    try {
        mHandler.setHeaders(message);
        mHandler.showHeaderContainer();
    } catch (MessagingException me) {
        Log.e(K9.LOG_TAG, "loadMessageForViewHeadersAvailable", me);
    }
