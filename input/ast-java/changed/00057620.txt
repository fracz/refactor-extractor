    if (client != null) {
        try {
            client.close();
        } catch (IllegalStateException ignore) {
        // Client was already closed by the test
        }
    }
    if (server != null) {
        CountDownLatch latch = new CountDownLatch(1);
        server.close((asyncResult) -> {
            assertTrue(asyncResult.succeeded());
            latch.countDown();
        });
        awaitLatch(latch);
    }
    super.tearDown();
