@Test
public // https://github.com/elasticsearch/elasticsearch/issues/3200
void testResetTwice() throws Exception {
    assertAcked(prepareCreate("test").setSettings(settingsBuilder().put(indexSettings()).put("analysis.analyzer.my_analyzer.type", "pattern").put("analysis.analyzer.my_analyzer.pattern", "\\s+").build()).addMapping("type", "text", "type=string,analyzer=my_analyzer"));
    ensureGreen();
    client().prepareIndex("test", "type", "1").setSource("text", "elasticsearch test").get();
    refresh();
    SearchResponse response = client().prepareSearch("test").setQuery(QueryBuilders.matchQuery("text", "test").type(MatchQuery.Type.BOOLEAN)).highlighter(new HighlightBuilder().field("text")).execute().actionGet();
    // PatternAnalyzer will throw an exception if it is resetted twice
    assertHitCount(response, 1l);
