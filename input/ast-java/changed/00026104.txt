        JetExpression expression = argument.getArgumentExpression();
        if (i < parameters.size()) {

            JetType parameterType = parameters.get(i).getType();

                typesToShorten.add(argumentType);

            changeSignatureData.addParameter(parameterInfo);
        }
    }
