@Override
public void run() {
    myCell.removeCellOutputs();
    myViewPanel.removeAll();
    final GridBagConstraints c = new GridBagConstraints();
    c.fill = GridBagConstraints.HORIZONTAL;
    c.gridx = 0;
    c.gridy = 0;
    c.gridwidth = 1;
    addPromptPanel(myViewPanel, myCell.getPromptNumber(), IpnbEditorUtil.PromptType.In, myCodeSourcePanel, c);
    for (Map.Entry<String, String> entry : outputMap.entrySet()) {
        final String type = entry.getKey();
        final String output = entry.getValue();
        myCell.addCellOutput(new CellOutput(new String[] { output }));
        c.gridx = 0;
        c.gridy += 1;
        addPromptPanel(myViewPanel, myCell.getPromptNumber(), "text/plain".equals(type) ? IpnbEditorUtil.PromptType.Out : IpnbEditorUtil.PromptType.None, new CodeOutputPanel(output), c);
    }
    revalidate();
    repaint();
