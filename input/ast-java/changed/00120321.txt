    Preconditions.checkArgument(StringUtils.isBlank(name), "Need to specify name");

    for (int i = 0; i < 2; i++) Preconditions.checkArgument(!hasUniqueLock[i] || isUnique[i], "Must be unique in order to have a lock");
    checkPrimaryKey(primaryKey);
    checkSignature(signature);
