@Override
public void run() {
    // as long as no interrupt is requested, continue running
    while (!interrupted) {
        if (datapointsToInitialize.size() > 0) {
            // we first clone the map, so that it stays unmodified
            HashMap<Datapoint, Integer> clonedMap = new HashMap<Datapoint, Integer>(datapointsToInitialize);
            initializeDatapoints(clonedMap);
        }
        // just wait before looping again
        try {
            sleep(1000L);
        } catch (InterruptedException e) {
            interrupted = true;
        }
    }
