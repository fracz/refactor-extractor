/**
 * Returns the orthogonal factor.
 */
public double[][] getQ() {
    int m = QR.nrows();
    int n = QR.ncols();
    double[][] Q = new double[m][n];
    for (int k = n - 1; k >= 0; k--) {
        for (int i = 0; i < m; i++) {
            Q[i][k] = 0.0;
        }
        Q[k][k] = 1.0;
        for (int j = k; j < n; j++) {
            if (QR.get(k, k) != 0) {
                double s = 0.0;
                for (int i = k; i < m; i++) {
                    s += QR.get(i, k) * Q[i][j];
                }
                s = -s / QR.get(k, k);
                for (int i = k; i < m; i++) {
                    Q[i][j] += s * QR.get(i, k);
                }
            }
        }
    }
    return Q;
