public LookupEx showLookup(final Editor editor, @NotNull LookupElement[] items, final String prefix, @NotNull final LookupArranger arranger) {
    for (LookupElement item : items) {
        assert item != null;
    }
    final LookupImpl lookup = createLookup(editor, items, prefix, arranger);
    lookup.show();

        lookup.setStartCompletionWhenNothingMatches(true);

    return lookup;
