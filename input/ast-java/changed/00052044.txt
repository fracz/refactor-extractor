        try {
            waitersLock.lock();

            // count is 0.
            waiters.remove(key);
            if (waiters.count(key) == 0) {
                // No other thread is waiting to access this key, so we garbage collect the lock.

            }
        } finally {
            waitersLock.unlock();
        }
    } finally {
        unlock();
    }
