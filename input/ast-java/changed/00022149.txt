    if (result != null)

    }

        protected PsiElement doProcessParent(VirtualFile parentFile) {
            MavenDomProjectModel parentProjectDom = MavenDomUtil.getMavenDomProjectModel(myProject, parentFile);

        }
    }.process(projectDom);
    if (result != null)
