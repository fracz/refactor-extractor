@Override
public Description matchBinary(BinaryTree tree, VisitorState state) {
    switch(tree.getKind()) {
        case AND:
        case OR:
            break;
        default:
            return NO_MATCH;
    }
    if (!isSameType(getType(tree), state.getSymtab().booleanType, state)) {
        return NO_MATCH;
    }
    return describeMatch(tree, SuggestedFix.replace(state.getEndPosition(tree.getLeftOperand()), ((JCTree) tree.getRightOperand()).getStartPosition(), tree.getKind() == Tree.Kind.AND ? " && " : " || "));
