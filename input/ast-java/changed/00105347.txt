DBCEntityIdentifier getVirtualEntityIdentifier() {
    if (!singleSourceCells || CommonUtils.isEmpty(metaColumns)) {
        return null;
    }
    DBCEntityIdentifier identifier = metaColumns[0].getRowIdentifier().getEntityIdentifier();

        return identifier;
    } else {
        return null;
    }
