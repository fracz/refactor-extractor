private EnvTypePair analyzeTemplateLitBwd(Node expr, TypeEnv outEnv) {
    TypeEnv env = outEnv;
    for (Node elm = expr.getLastChild(); elm != null; elm = elm.getPrevious()) {
        EnvTypePair pair = analyzeExprBwd(elm, env);
        env = pair.env;
    }
    return new EnvTypePair(env, STRING);
