@Override
public String toString() {
    if (isFinished()) {
        if (isSuccess()) {
            return "success";
        }
        String cause = cause().toString();
        StringBuilder buf = new StringBuilder(cause.length() + 17);
        buf.append("failure(");
        buf.append(cause);
        buf.append(')');
        return buf.toString();
    } else {
        return "unfinished";
    }
