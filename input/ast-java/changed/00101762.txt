private Param[] parameters(String[] params) {
    Param[] parameters = new Param[params.length / 2];
    for (int i = 0; i < parameters.length; i++) {
        parameters[i] = new Param();
        parameters[i].type = params[i * 2];
        parameters[i].name = params[i * 2 + 1];
    }
    return parameters;
