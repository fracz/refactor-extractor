public void release(final OChannelBinaryAsynchClient conn) {
    final OResourcePool<String, OChannelBinaryAsynchClient> pool = connections.get(conn.getServerURL());
    if (pool != null) {
        if (!conn.isConnected()) {

            remove(conn);

        }
    }
