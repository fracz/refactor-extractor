public void probeAll() {
    Set<InetAddress> ipProbeList = new HashSet<InetAddress>();
    synchronized (mLock) {
        ipProbeList.addAll(mIpWatchList);
    }
    for (InetAddress target : ipProbeList) {
        if (!stillRunning()) {
            break;
        }
        probeNeighbor(mInterfaceIndex, target);
    }
