@Override
public void config(final OServer server, OServerParameterConfiguration[] iParams) {
    for (OServerParameterConfiguration param : iParams) {
        if (param.name.equalsIgnoreCase("enabled")) {
            if (!Boolean.parseBoolean(param.value))
                // DISABLE IT
                return;
        } else if (param.name.equalsIgnoreCase("graph.pool.max"))
            graphPoolMax = Integer.parseInt(param.value);
    }
    if (OGremlinHelper.isGremlinAvailable()) {
        enabled = true;
        OLogManager.instance().info(this, "Installed GREMLIN language v.%s - graph.pool.max=%d", OGremlinHelper.getEngineVersion(), graphPoolMax);
        Orient.instance().getScriptManager().registerInjection(this);
    } else
        enabled = false;
    this.server = server;
