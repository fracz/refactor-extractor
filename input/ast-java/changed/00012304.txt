public boolean isCutElement(Object element) {
    if (myRecentData == null)
        return false;
    if (myRecentData.isCopied())
        return false;
    PsiElement[] elements = myRecentData.getElements();
    if (elements == null)
        return false;
    for (int i = 0; i < elements.length; i++) {
        if (elements[i] == element)
            return true;
    }
    return false;
