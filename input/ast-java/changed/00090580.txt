@SuppressWarnings("ResultOfMethodCallIgnored")
private static File toDirectory(String path) {
    if (path == null) {
        return null;
    }
    File f = new File(path);
    if (!f.exists()) {
        f.mkdirs();
    }
    if (!f.isDirectory()) {
        return null;
    }
    try {
        return f.getAbsoluteFile();
    } catch (Exception ignored) {
        return f;
    }
