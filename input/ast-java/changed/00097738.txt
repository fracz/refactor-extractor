public OUser createUser(final String iUserName, final String iUserPassword, final String[] iRoles) {
    acquireExclusiveLock();
    try {
        final OUser user = new OUser(getDatabase(), iUserName, iUserPassword);
        if (iRoles != null)
            for (String r : iRoles) {
                user.addRole(r);
            }
        return user.save();
    } finally {
        releaseExclusiveLock();
    }
