/**
 * Returns true if the block read request is valid.
 *
 * @param request the block read request
 * @return true if the block read request is valid
 */
private String validateReadRequest(Protocol.ReadRequest request) {
    if (mRequest == null) {
        return "";
    }
    if (request.getCancel()) {
        if (request.getId() != mRequest.mId) {
            return String.format("The IDs do not match: [actual: %d, expected: %d].", mRequest.mId, request.getId());
        }
        return "";
    }
    return String.format("Received a read request %s on a busy channel (used by block %d).", request, mRequest.mId);
