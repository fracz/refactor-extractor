private static void safeClose(Object object) {
    if (object == null) {
        return;
    }
    try {
        if (object instanceof IndexWriter) {
            ((IndexWriter) object).close();
        } else if (object instanceof IndexSearcher) {
            ((IndexSearcher) object).close();
        }
    } catch (IOException e) {
    // Ok
    }
