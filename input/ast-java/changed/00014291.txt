private static void addEmptyDocstring(Project project, PyDocStringOwner docStringOwner) {
    if (docStringOwner instanceof PyFunction) {
        PyGenerateDocstringIntention.generateDocstringForFunction(project, getEditor(project, docStringOwner.getContainingFile()), (PyFunction) docStringOwner);
    }
    if (docStringOwner instanceof PyClass) {
        PyFunction init = ((PyClass) docStringOwner).findInitOrNew(false);
        if (init == null)
            return;
        PythonDocumentationProvider.insertDocStub(init, ((PyClass) docStringOwner).getStatementList(), project, getEditor(project, docStringOwner.getContainingFile()));
    }
