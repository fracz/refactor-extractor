public void setPassword(Account account, String password) throws RemoteException {
    ContentValues values = new ContentValues();
    values.put(ACCOUNTS_PASSWORD, password);
    mOpenHelper.getWritableDatabase().update(TABLE_ACCOUNTS, values, ACCOUNTS_NAME + "=? AND " + ACCOUNTS_TYPE + "=?", new String[] { account.mName, account.mType });
    mContext.sendBroadcast(ACCOUNTS_CHANGED_INTENT);
