public static void reformatCode(final PsiElement element) {
    // todo rewrite activity
    final TextRange textRange = element.getTextRange();
    ApplicationManager.getApplication().runWriteAction(new Runnable() {

        public void run() {
            try {
                CodeStyleManager.getInstance(element.getProject()).reformatText(element.getContainingFile(), textRange.getStartOffset(), textRange.getEndOffset());
            } catch (IncorrectOperationException e) {
                e.printStackTrace();
            }
        }
    });
