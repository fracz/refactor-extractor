@Override
public void binderDied() {
    if (LOCAL_LOGV) {
        Slog.v(TAG, "Location listener died");
    }
    synchronized (mLock) {
        removeUpdatesLocked(this);
    }
    synchronized (this) {
        if (mPendingBroadcasts > 0) {
            LocationManagerService.this.decrementPendingBroadcasts();
            mPendingBroadcasts = 0;
        }
    }
