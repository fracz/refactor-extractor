@Override
public void onCancellable(ICancellationSignal cancellation) {
    synchronized (mLock) {
        final boolean cancelled;
        synchronized (mLock) {
            mCancellation = cancellation;
            cancelled = mCancelled;
        }
        if (cancelled) {
            try {
                cancellation.cancel();
            } catch (RemoteException e) {
                Slog.e(LOG_TAG, "Error requesting a cancellation", e);
            }
        }
    }
