public void add(String key) {
    for (int i = 0; i < hashes_; ++i) {
        ISimpleHash hash = hashLibrary_.get(i);
        int hashValue = hash.hash(key);
        int index = Math.abs(hashValue % size_);
        filter_.set(index);
    }
