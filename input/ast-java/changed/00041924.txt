private static byte[] generateDigest(Uri uri) {
    try {
        final MessageDigest digest = MessageDigest.getInstance(SHA_ALGORITHM);
        final byte[] hostBytes = uri.getHost().getBytes();
        return digest.digest(hostBytes);
