@Override
public View getView(int position, View convertView, ViewGroup parent) {
    if (convertView == null) {
        convertView = LayoutInflater.from(parent.getContext()).inflate(R.layout.item_title, parent, false);
    }
    final TextView title = (TextView) convertView.findViewById(android.R.id.title);
    final TextView summary = (TextView) convertView.findViewById(android.R.id.summary);
    final Document cwd = getCurrentDirectory();
    if (cwd != null) {
        title.setText(cwd.displayName);
    } else {
        // No directory means recents
        title.setText(R.string.root_recent);
    }
    summary.setText((String) getItem(position));
    return convertView;
