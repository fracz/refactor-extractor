/**
 * Validates the read request, throwing an exception if it's invalid.
 *
 * @param request the block read request
 */
private void validateReadRequest(Protocol.ReadRequest request) {
    if (request.getId() < 0) {
        throw new IllegalArgumentException(String.format("Invalid blockId (%d) in read request.", request.getId()));
    }
    if (!request.getCancel() && (request.getOffset() < 0 || request.getLength() <= 0)) {
        throw new IllegalArgumentException(String.format("Invalid read bounds in read request %s.", request.toString()));
    }
