@Override
public void postWindowCreate() {
    log.debug("Initialize workbench window");
    final IWorkbenchWindow window = getWindowConfigurer().getWindow();
    UIUtils.updateMainWindowTitle(window);
    try {
        DBeaverApplication.executeCommandLineCommands(DBeaverApplication.getCommandLine(), DBeaverApplication.getInstanceServer());
    } catch (Exception e) {
        log.error("Error processing command line", e);
    }
