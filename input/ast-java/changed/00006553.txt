private void mergeWithDefaultConfiguration() {
    try {
        final Configuration cfg = load(getClass().getClassLoader().getResourceAsStream("/" + COMPONENT_NAME + ".xml"));
        // very strange
        if (cfg == null)
            return;
        importFrom(cfg);
    } catch (Exception e) {
        LOG.warn(e);
    }
