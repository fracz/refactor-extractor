private static String createInaccessibleFieldsConflictDescription(ArrayList<PsiField> inaccessible, PsiElement container) {
    if (inaccessible.size() == 1) {
        final PsiField field = inaccessible.get(0);
        return RefactoringBundle.message("field.0.is.not.accessible", CommonRefactoringUtil.htmlEmphasize(field.getName()), RefactoringUIUtil.getDescription(container, true));
    } else {
        StringBuffer fieldsBuffer = new StringBuffer();
        for (int j = 0; j < inaccessible.size(); j++) {
            PsiField field = inaccessible.get(j);
            if (j > 0) {
                fieldsBuffer.append(", ");
            }
            fieldsBuffer.append(CommonRefactoringUtil.htmlEmphasize(field.getName()));
        }
        return RefactoringBundle.message("fields.0.are.not.accessible", fieldsBuffer.toString(), RefactoringUIUtil.getDescription(container, true));
    }
