private PendingEvent dequeuePendingEventLocked(int seq) {
    PendingEvent p = mFirstPendingEvent;
    if (p == null) {
        return null;
    }
    if (p.mSeq == seq) {
        mFirstPendingEvent = p.mNext;
    } else {
        PendingEvent prev;
        do {
            prev = p;
            p = p.mNext;
            if (p == null) {
                return null;
            }
        } while (p.mSeq != seq);
        prev.mNext = p.mNext;
    }
    p.mNext = null;
    return p;
