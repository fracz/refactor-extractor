protected List<T> runLoadOneByOne(List<K> keys, String traceName) {
    List<T> list = new ArrayList<T>(keys.size());
    startClock(traceName);
    for (K key : keys) {
        list.add(dao.load(key));
    }
    stopClock(keys.size() + " entities");
    return list;
