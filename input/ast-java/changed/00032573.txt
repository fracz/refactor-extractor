    final char first = ch;

    if (charAt(pos + 1) == '@') {

        hash = 31 * hash + ch;
        bufPos++;
    } else if (charAt(pos + 1) == '{') {
        hash = 31 * hash + '"';
        pos++;
        bufPos++;
        mybatisFlag = true;

        bufPos++;
