@Test
public void testMap() {
    Map<String, String> m1 = getMap("One");
    Map<String, String> m2 = getMap("Two");
    @SuppressWarnings("unchecked") Observable<Map<String, String>> observable = Observable.toObservable(m1, m2);
    Observable<String> m = map(observable, new Func1<Map<String, String>, String>() {

        @Override
        public String call(Map<String, String> map) {
            return map.get("firstName");
        }
    });
    m.subscribe(stringObserver);
    verify(stringObserver, never()).onError(any(Exception.class));
    verify(stringObserver, times(1)).onNext("OneFirst");
    verify(stringObserver, times(1)).onNext("TwoFirst");
    verify(stringObserver, times(1)).onCompleted();
