        public boolean canCloseProject(Project project) {

        }

        @Override

        @Override
        public void projectClosing(Project project) {
            if (project != null) {
                for (UsagesCollector usagesCollector : Extensions.getExtensions(UsagesCollector.EP_NAME)) {

                    }
                }
            }
        }
