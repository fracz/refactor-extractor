RevCommit checkinToBranch(String branchName, GoConfigRevision rev) throws Exception {
    try {
        checkout(branchName);
        checkin(rev);
        return getCurrentRevCommit();
    } catch (Exception e) {
        LOGGER.error(String.format("[CONFIG_MERGE] Check-in to branch %s failed", branchName));
        throw e;
    }
