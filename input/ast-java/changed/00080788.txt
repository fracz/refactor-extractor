 * identified by its id
 *
 * @param identifier
 * @param badge
 */
public void updateBadge(int identifier, StringHolder badge) {
    Badgeable badgeable = ((Badgeable) getDrawerItem(identifier));

        updateItem((IDrawerItem) badgeable);
    }
