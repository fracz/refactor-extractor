private static boolean isWinShell(@NotNull String command, @NotNull List<String> parameters) {
    if (command.endsWith(".cmd")) {
        return true;
    }
    return ("cmd".equalsIgnoreCase(command) || "cmd.exe".equalsIgnoreCase(command)) && parameters.size() > 1 && "/c".equalsIgnoreCase(parameters.get(0));
