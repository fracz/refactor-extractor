@Override
public void alterSession(DBCSession session, MySQLSession sessionType, Map<String, Object> options) throws DBException {
    try {
        JDBCPreparedStatement dbStat = ((JDBCSession) session).prepareStatement(Boolean.TRUE.equals(options.get(PROP_KILL_QUERY)) ? "KILL QUERY " + sessionType.getPid() : "KILL CONNECTION " + sessionType.getPid());
        try {
            dbStat.execute();
        } finally {
            dbStat.close();
        }
    } catch (SQLException e) {
        throw new DBException(e);
    }
