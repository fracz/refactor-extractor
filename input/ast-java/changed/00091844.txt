            for (RFXComBindingProvider provider : providers) {

                    boolean inBinding = provider.isInBinding(itemName);
                    if (id1.equals(id2) && inBinding) {
                        RFXComValueSelector parseItem = provider.getValueSelector(itemName);
                        try {
