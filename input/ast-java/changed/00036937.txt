public static Allocation createFromString(RenderScript rs, String str) throws IllegalArgumentException {
    byte[] allocArray = null;
    try {
        allocArray = str.getBytes("UTF-8");
        Allocation alloc = Allocation.createSized(rs, Element.U8(rs), allocArray.length);
        alloc.data(allocArray);
        return alloc;
    } catch (Exception e) {
        Log.e("rs", "could not convert string to utf-8");
    }
    return null;
