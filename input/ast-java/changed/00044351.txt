/**
 * Resets the focused task.
 */
void resetFocusedTask(Task task) {
    if (task != null) {
        TaskView tv = getChildViewForTask(task);
        if (tv != null) {
            tv.setFocusedState(false, false, /* animated */
            false);
        }
    }
    mFocusedTask = null;
