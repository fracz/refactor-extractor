@Override
public ArrayList<ArrayList<Region>> defineRegions(ArrayList<ChartSet> data) {
    final ArrayList<ArrayList<Region>> result = new ArrayList<ArrayList<Region>>();
    for (int i = 0; i < data.size(); i++) result.add(new ArrayList<Region>());
    float drawingOffset;
    BarSet barSet;
    Bar bar;
    for (int i = 0; i < data.get(0).size(); i++) {
        // Set first offset to draw a group of bars
        drawingOffset = data.get(0).getEntry(i).getX() - mDrawingOffset;
        for (int j = 0; j < data.size(); j++) {
            barSet = (BarSet) data.get(j);
            bar = (Bar) barSet.getEntry(i);
            if (bar.getValue() > 0)
                result.get(j).add(new Region((int) drawingOffset, (int) bar.getY(), (int) (drawingOffset += barWidth), (int) this.getZeroPosition()));
            else
                result.get(j).add(new Region((int) drawingOffset, (int) this.getZeroPosition(), (int) (drawingOffset += barWidth), (int) bar.getY()));
            // If last bar of group no set spacing is necessary
            if (j != data.size() - 1)
                drawingOffset += style.mSetSpacing;
        }
    }
    return result;
