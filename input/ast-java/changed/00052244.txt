 */
private void evict() {
    while (data.size() > maximumSize) {
        Node node = sentinel.next;
        if (node == sentinel) {
            return;
        } else if (policy.onEvict(node)) {
            policyStats.recordEviction();
            data.remove(node.key);
            node.remove();
        }
    }
