public void saveState() {
    String author = myAuthorField.getText();
    if (StringUtil.isEmptyOrSpaces(author)) {
        myNextCommitAuthor = null;
    } else {
        myNextCommitAuthor = GitCommitAuthorCorrector.correct(author);
