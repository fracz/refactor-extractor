private void appendWebsites(final StringBuilder builder, final Map<String, List<ContentValues>> contentValuesListMap) {
    final List<ContentValues> contentValuesList = contentValuesListMap.get(Website.CONTENT_ITEM_TYPE);
    if (contentValuesList != null) {
        for (ContentValues contentValues : contentValuesList) {
            String website = contentValues.getAsString(Website.URL);
            if (website != null) {
                website = website.trim();
            }
            if (!TextUtils.isEmpty(website)) {
                appendVCardLine(builder, VCARD_PROPERTY_URL, website);
            }
        }
    }
