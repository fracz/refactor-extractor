static boolean isSpecial(char ch, int features) {
    if (ch == ' ') {
        // 32
        return false;
    }
    if (ch == '/') {
        // 47
        return SerializerFeature.isEnabled(features, SerializerFeature.WriteSlashAsSpecial);
    }
    if (// 35
    ch > '#' && // 92
    ch != '\\') {
        return false;
    }
    if (// 31
    ch <= 0x1F || // 92
    ch == '\\' || // 34
    ch == '"') {
        return true;
    }
    return false;
