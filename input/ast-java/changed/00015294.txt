private void finish(boolean inSeparateEvent) {
    Editor editor = myCallback.getEditor();
    if (myEndOffset >= 0) {
        editor.getCaretModel().moveToOffset(myEndOffset);
    }

        myListener.finished(inSeparateEvent, true);
    }
