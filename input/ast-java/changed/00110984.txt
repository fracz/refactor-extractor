@Bean
@ConditionalOnMissingBean(name = "remoteDebugHandlerMapper")
public UrlHandlerMapper remoteDebugHandlerMapper(@Qualifier("remoteDebugHttpTunnelServer") HttpTunnelServer server) {
    String url = (this.serverProperties.getContextPath() == null ? "" : this.serverProperties.getContextPath()) + this.properties.getRemote().getContextPath() + "/debug";
    logger.warn("Listening for remote debug traffic on " + url);
    Handler handler = new HttpTunnelServerHandler(server);
    return new UrlHandlerMapper(url, handler);
