/**
 * helper method to close the drawer delayed
 */
protected void closeDrawerDelayed() {
    if (mCloseOnClick && mDrawerLayout != null) {
        if (mDelayOnDrawerClose > -1) {
            new Handler().postDelayed(new Runnable() {

                @Override
                public void run() {
                    mDrawerLayout.closeDrawers();
                }
            }, mDelayOnDrawerClose);
        } else {
            mDrawerLayout.closeDrawers();
        }
    }
