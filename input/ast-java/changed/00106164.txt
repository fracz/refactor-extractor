/**
 * Get parent by level.
 * @param grand 0 - self, 1 - direct parent, 2 - grand parent, etc
 * @return parent or null
 */
@Nullable
public DBDAttributeBinding getParent(int grand) {
    if (grand == 0) {
        return this;
    }
    DBDAttributeBinding p = this;
    for (int i = 0; i < grand; i++) {
        assert p != null;
        p = p.parent;
    }
    return p;
