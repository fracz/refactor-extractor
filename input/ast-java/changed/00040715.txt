        for (int j = mPowerSaveTempWhitelistAppIds.size() - 1; j >= 0; j--) {
            int appId = mPowerSaveTempWhitelistAppIds.keyAt(j);
            int uid = UserHandle.getUid(user.id, appId);

        }
    }
