/**
 * Handler for Sketch &rarr; Export Application
 */
public synchronized void handleExportApplication() {
    if (!handleExportCheckModified())
        return;
    toolbar.activate(EditorToolbar.EXPORT);
    // SwingUtilities.invokeLater(new Runnable() {
    SwingUtilities.invokeLater(new Runnable() {

        public void run() {
            statusNotice("Exporting application...");
            try {
                if (sketch.exportApplication(PConstants.WINDOWS) && sketch.exportApplication(PConstants.MACOSX) && sketch.exportApplication(PConstants.LINUX)) {
                    Base.openFolder(sketch.folder);
                    statusNotice("Done exporting.");
                } else {
                // error message will already be visible
                }
            } catch (Exception e) {
                statusNotice("Error during export.");
                e.printStackTrace();
            }
            // toolbar.clear();
            toolbar.deactivate(EditorToolbar.EXPORT);
        }
    });
