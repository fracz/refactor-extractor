private void stopNavigating() {
    if (DEBUG)
        Log.d(TAG, "stopNavigating");
    if (mStarted) {
        mStarted = false;
        mSingleShot = false;
        native_stop();
        mTTFF = 0;
        mLastFixTime = 0;
        mLocationFlags = LOCATION_INVALID;
        // reset SV count to zero
        updateStatus(LocationProvider.TEMPORARILY_UNAVAILABLE, 0);
    }
