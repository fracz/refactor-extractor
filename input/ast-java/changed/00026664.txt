private void boxIfNeeded(PsiType type) {
    if (type == PsiType.LONG) {
        v.invokestatic("java/lang/Long", "valueOf", "(J)Ljava/lang/Long;");
    } else {
        throw new UnsupportedOperationException("Don't know how to box type " + type);
