private static List<BlockNode> cleanBlockList(List<BlockNode> list) {
    List<BlockNode> ret = new ArrayList<BlockNode>(list.size());
    for (BlockNode block : list) {
        if (!block.getAttributes().contains(AttributeType.EXC_HANDLER)) {
            ret.add(block);
        }
    }
    return ret;
