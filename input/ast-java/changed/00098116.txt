/**
 * Ends the request and unlock the write lock
 */
public void endRequest(final OChannelBinaryAsynchClient iNetwork) throws IOException {
    if (iNetwork == null)
        return;
    try {
        iNetwork.flush();
        iNetwork.releaseWriteLock();
    } catch (IOException e) {
        engine.getConnectionManager().remove(iNetwork);
        throw e;
    }
