public boolean parseStatementListDialect(List<SQLStatement> statementList) {
    switch(lexer.token()) {
        case BEGIN:
            statementList.add(handleBegin());
            return true;
        case START:
            return handleStart(statementList);
        case WITH:
            statementList.add(parseWith());
            return true;
        default:
            return false;
    }
