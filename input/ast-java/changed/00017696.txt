void add(final Content content) {
    if (myContents.containsKey(content))
        return;
    myContents.put(content, null);
    revalidateCell(new Runnable() {

        public void run() {
            myTabs.addTab(createTabInfoFor(content));
        }
    });
    updateSelection(myTabs.getRootPane() != null);
