@Override
public void removeUpdates(ILocationListener listener) {
    try {
        synchronized (mLock) {
            removeUpdatesLocked(getReceiver(listener));
        }
    } catch (SecurityException se) {
        throw se;
    } catch (IllegalArgumentException iae) {
        throw iae;
    } catch (Exception e) {
        Slog.e(TAG, "removeUpdates got exception:", e);
    }
