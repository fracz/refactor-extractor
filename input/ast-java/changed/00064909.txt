@Override
public Facet reduce(List<Facet> facets) {
    if (facets.size() == 1) {
        return facets.get(0);
    }
    int count = 0;
    for (Facet facet : facets) {
        if (facet.name().equals(name)) {
            count += ((QueryFacet) facet).count();
        }
    }
    return new InternalQueryFacet(name, count);
