public static void main(String[] args) throws Exception {
    LogLevelAdjuster.register();
    if (args.length != 2) {
        log.info("Usage: ExecutorMain <task.json> <status.json>");
        System.exit(2);
    }
    final ExecutorNode node = ExecutorNode.builder().build(new ExecutorLifecycleFactory(new File(args[0]), new File(args[1]), System.in));
    final Lifecycle lifecycle = new Lifecycle();
    lifecycle.addManagedInstance(node);
    try {
        lifecycle.start();
        node.join();
        lifecycle.stop();
    } catch (Throwable t) {
        log.info(t, "Throwable caught at startup, committing seppuku");
        System.exit(2);
    }
