public RemoteRepository delRepo(String id) {
    synchronized (repos) {
        Iterator<RemoteRepository> it = repos.iterator();
        if (it.hasNext()) {
            RemoteRepository repo = it.next();
            if (repo.getId().equals(id)) {
                it.remove();
                return repo;
            }
        }
    }
    return null;
