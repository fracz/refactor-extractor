    long size = 0;
    if (useSparse) {
        if (data == null)

    if (hasComplexDeletion)
        size += UnfilteredRowIteratorSerializer.delTimeSerializedSize(data == null ? DeletionTime.LIVE : data.complexDeletion(), header);
    if (data != null) {

    }
