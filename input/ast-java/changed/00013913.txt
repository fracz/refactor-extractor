@Override
public boolean isModified() {
    if (myUpdatesSettingsPanel == null)
        return false;
    UpdateSettings settings = UpdateSettings.getInstance();
    if (settings.CHECK_NEEDED != myUpdatesSettingsPanel.myCheckForUpdates.isSelected())
        return true;
    if (settings.SECURE_CONNECTION != myUpdatesSettingsPanel.myUseSecureConnection.isSelected())
        return true;
    Object channel = myUpdatesSettingsPanel.myUpdateChannels.getSelectedItem();
    return channel != null && !channel.equals(ChannelStatus.fromCode(settings.UPDATE_CHANNEL_TYPE));
