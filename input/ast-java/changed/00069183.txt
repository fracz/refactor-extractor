        @Override
        public DiscoveryNodes nodes() {
            return DiscoveryNodes.builder().put(nodeA).localNodeId("UZP_A").build();
        }

        @Override
        public NodeService nodeService() {
            return null;
        }

        @Override
        public boolean nodeHasJoinedClusterOnce() {
            return false;
        }
    });
