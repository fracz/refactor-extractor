 * Returns true if field corresponding to fieldID is set (has been assigned a value) and false otherwise
 */
public boolean isSet(_Fields field) {
    if (field == null) {
        throw new IllegalArgumentException();
    }
    switch(field) {
        case NAME:
            return isSetName();
        case VALUE:
            return isSetValue();
    }
