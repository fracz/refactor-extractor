public SQLServerTop parseTop() {
    if (getLexer().equalToken(Token.TOP)) {
        SQLServerTop top = new SQLServerTop();
        getLexer().nextToken();
        boolean paren = false;
        if (getLexer().equalToken(Token.LEFT_PAREN)) {
            paren = true;
            getLexer().nextToken();
        }
        top.setExpr(primary());
        if (paren) {
            accept(Token.RIGHT_PAREN);
        }
        if (getLexer().equalToken(Token.PERCENT)) {
            getLexer().nextToken();
            top.setPercent(true);
        }
        return top;
    }
    return null;
