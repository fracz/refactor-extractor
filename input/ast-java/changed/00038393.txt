    if (homeInFront) {
        startHomeActivityLocked(newUserId);
    } else {
        mStackSupervisor.resumeTopActivitiesLocked();
    }

    getUserManagerLocked().userForeground(newUserId);
    sendUserSwitchBroadcastsLocked(oldUserId, newUserId);
