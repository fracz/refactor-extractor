public void clearLog() {
    if (getShell().isDisposed()) {
        return;
    }
    UIUtils.runInUI(getShell(), new Runnable() {

        @Override
        public void run() {
            synchronized (DatabaseWizardPageLog.this) {
                if (!dumpLogText.isDisposed()) {
                    // $NON-NLS-1$
                    dumpLogText.setText("");
                }
            }
        }
    });
