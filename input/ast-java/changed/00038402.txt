            if (r.frontOfTask && task == topTask() && task.isOverHomeStack()) {
                mStackSupervisor.moveHomeStackTaskToTop(task.getTaskToReturnTo());

        }
        ActivityRecord top = mStackSupervisor.topRunningActivityLocked();
        if (top != null) {

        }
    }
