private static void doAddUsed(Phi r, Set<Phi> regs) {
    if (r.used) {
        if (!regs.contains(r)) {
            regs.add(r);
            for (Phi p : r) {
                p.used = true;
                doAddUsed(p, regs);
            }
        }
    }
