@Test
public void testGranularitySpecLegacy() {
    // Deprecated and replaced by granularitySpec, but still supported
    final HadoopDruidIndexerConfig cfg;
    try {
        cfg = jsonReadWriteRead("{" + "\"segmentGranularity\":\"day\"," + "\"intervals\":[\"2012-02-01/P1D\"]" + "}", HadoopDruidIndexerConfig.class);
    } catch (Exception e) {
        throw Throwables.propagate(e);
    }
    final UniformGranularitySpec granularitySpec = (UniformGranularitySpec) cfg.getGranularitySpec();
    Assert.assertEquals("getIntervals", Lists.newArrayList(new Interval("2012-02-01/P1D")), granularitySpec.getIntervals().get());
    Assert.assertEquals("getGranularity", "DAY", granularitySpec.getGranularity().toString());
