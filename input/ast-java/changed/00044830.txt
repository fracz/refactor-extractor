/**
 * Reset the state of the view.
 */
public void reset() {
    if (mShowing) {
        if (mOccluded) {
            mPhoneStatusBar.hideKeyguard();
            mPhoneStatusBar.stopWaitingForKeyguardExit();
            mBouncer.hide(false);
        } else {
            showBouncerOrKeyguard();
        }
        KeyguardUpdateMonitor.getInstance(mContext).sendKeyguardReset();
        updateStates();
    }
