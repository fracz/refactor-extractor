public void layout() {
    if (!needsLayout)
        return;
    needsLayout = false;
    for (int i = 0, n = children.size(); i < n; i++) {
        Actor actor = children.get(i);
        actor.x = 0;
        actor.y = 0;
        actor.width = width;
        actor.height = height;
        if (actor instanceof Layout) {
            Layout layout = (Layout) actor;
            layout.invalidate();
            layout.layout();
        }
    }
