    Preconditions.checkState(keepEdges(), "%s %s", reverseDep, this);
    if (!isDone()) {
        getReverseDepsUtil().removeReverseDep(this, reverseDep);

        getReverseDepsUtil().checkReverseDep(this, reverseDep);
