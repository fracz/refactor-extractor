        return;

    if (editor == null) {
        return;
    }
    PsiDocumentManager documentManager = PsiDocumentManager.getInstance(project);
    Document document = editor.getDocument();

        return;
    }
    new RearrangeCodeProcessor(project, file, editor.getSelectionModel()).run();
