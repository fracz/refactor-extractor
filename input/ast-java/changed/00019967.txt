        return;

    final PsiExpression arrayExpression = arg.getArrayExpression();
    if (!(arrayExpression instanceof PsiReferenceExpression)) {
        return;

    final PsiElement referent = ((PsiReference) arrayExpression).resolve();
    if (referent == null) {
        return;
    }
    if (referent.equals(variable)) {
        accessed = true;
