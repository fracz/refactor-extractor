/**
 * Get quantity of leased lines in right Editor
 *
 * @return quantity of leased lines
 */
public int getQuantityLeasedLineInRightEditor() {
    new WebDriverWait(seleniumWebDriver, LOAD_PAGE_TIMEOUT_SEC).until(ExpectedConditions.frameToBeAvailableAndSwitchToIt(By.xpath(Locators.PREVIEW_EDITOR_FRAME)));
    int lineQuantity = new WebDriverWait(seleniumWebDriver, REDRAW_UI_ELEMENTS_TIMEOUT_SEC).until(ExpectedConditions.presenceOfAllElementsLocatedBy(By.xpath(Locators.LEASED_LINE_RIGHT_EDITOR))).size();
    seleniumWebDriver.switchTo().parentFrame();
    return lineQuantity;
