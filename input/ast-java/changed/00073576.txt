public static String intToBinaryString(int i) {
    // Max number of binary digits in an int
    int bufLen = 32;
    char[] buf = new char[bufLen];
    int cursor = bufLen;
    do {
        buf[--cursor] = DIGITS[i & 1];
    } while ((i >>>= 1) != 0);
    return new String(cursor, bufLen - cursor, buf);
