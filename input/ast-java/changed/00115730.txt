@Override
public String toString() {
    StringBuilder sb = new StringBuilder("CRFDatum[\n");
    sb.append("    label=").append(label).append('\n');
    for (int i = 0, sz = features.size(); i < sz; i++) {
        sb.append("    features(").append(i).append("):").append(features.get(i));

        sb.append('\n');
