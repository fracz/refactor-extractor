public static void visit(InsnNode insn) {
    InsnArg res = insn.getResult();
    if (res != null && !res.getType().isTypeKnown()) {
        selectType(res);
    }
    for (InsnArg arg : insn.getArguments()) {
        if (!arg.getType().isTypeKnown()) {
            selectType(arg);
        }
    }
