@Override
public String getIndexJobStatus(TitanIndex index) {
    IndexIdentifier indexId = new IndexIdentifier(index);
    StandardScanner.ScanResult result = graph.getBackend().getScanJobStatus(indexId);
    if (result == null)
        return "There is currently no active indexing job for index: " + index;
    if (result.isDone())
        return "Indexing Job completed for index: " + index;
    if (result.isCancelled())
        return "Indexing Job was canceled for index: " + index;
    ScanMetrics metrics = result.getIntermediateResult();
    return String.format("Indexing Job [%s] processed %s records successfully and failed on % records", index, metrics.get(ScanMetrics.Metric.SUCCESS), metrics.get(ScanMetrics.Metric.FAILURE));
