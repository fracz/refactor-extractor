private static List<MavenResource> convenrtResources(List<Resource> resources) {
    List<MavenResource> result = new ArrayList<>();
    if (resources != null) {
        for (Resource res : resources) {
            result.add(new MavenResource(res.getDirectory(), res.isFiltering(), res.getTargetPath(), patternsOrEmptyList(res.getIncludes()), patternsOrEmptyList(res.getExcludes())));
        }
    }
    return result;
