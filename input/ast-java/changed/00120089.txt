public static String byteBufferToString(ByteBuffer b) {
    try {
        return new String(b.array(), b.position() + b.arrayOffset(), b.remaining(), "UTF-8");
    } catch (UnsupportedEncodingException e) {
        throw new RuntimeException(e);
    }
