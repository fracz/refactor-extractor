        @Override
        protected Void doInBackground() throws Exception {
            long start = System.currentTimeMillis();
            allClassesInArchive = reducer.reduce("");
            System.out.println("Archive Reading " + (System.currentTimeMillis() - start) + " ms ");
            return null;
        }

        @Override
        protected void done() {
            if (allClassesInArchive.isEmpty() || (allClassesInArchive.size() == 1 && allClassesInArchive.contains("AndroidManifest.xml"))) {
                filesTree.fillArchive(new File("XXXXX"), new ArrayList<String>());
                displayArea.displayError();
                return;
            }
            filesTree.fillArchive(loadedFile, allClassesInArchive);
            displayArea.displaySharkey();
        }
    };
    worker.execute();
