@Override
public Object getObject(Result result, String columnName) {
    byte[] family = this.getFamily(columnName);
    byte[] bytes;
    if (isRow(columnName)) {
        bytes = result.getRow();
    } else {
        byte[] qualifier = Bytes.toBytes(columnName);
        bytes = result.getValue(family, qualifier);
    }
    return bytes;
