public static void waitForReindex(TitanIndex index, TitanManagement mgmt) {
    Set<String> finishMsg = ImmutableSet.of("There is currently no active indexing job for index", "Indexing Job completed for index");
    while (true) {
        String msg = mgmt.getIndexJobStatus(index);
        System.out.println("Index [" + index + "] job status: " + msg);
        if (finishMsg.contains(msg))
            break;
        try {
            Thread.sleep(500);
        } catch (InterruptedException e) {
            throw new RuntimeException(e);
        }
    }
