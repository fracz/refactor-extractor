private void forceHigherLayerIfNeeded(WindowState w, WindowStateAnimator winAnimator, AppWindowToken wtoken) {
    boolean force = false;
    if (wtoken.mWillReplaceWindow && wtoken.mReplacingWindow != w && wtoken.mAnimateReplacingWindow) {
        // We know that we will be animating a relaunching window in the near future,
        // which will receive a z-order increase. We want the replaced window to
        // immediately receive the same treatment, e.g. to be above the dock divider.
        force = true;
    }
    if (!force) {
        final TaskStack stack = w.getStack();
        if (stack != null && (StackId.isAlwaysOnTop(stack.mStackId) || stack.mStackId == DOCKED_STACK_ID)) {
            // If the window's stack is always on top, we want to make it above other windows
            // also when these windows are animating.
            force = true;
        }
    }
    if (force) {
        w.mLayer += TYPE_LAYER_OFFSET;
        winAnimator.mAnimLayer += TYPE_LAYER_OFFSET;
    }
