private void updateStateForChildTransitioningInBottom(StackScrollAlgorithmState algorithmState, float transitioningPositionStart, float bottomPeakStart, float currentYPosition, StackScrollState.ViewState childViewState, int childHeight) {
    // This is the transitioning element on top of bottom stack, calculate how far we are in.
    algorithmState.partialInBottom = 1.0f - ((transitioningPositionStart - currentYPosition) / (childHeight + mPaddingBetweenElements));
    // the offset starting at the transitionPosition of the bottom stack
    float offset = mBottomStackIndentationFunctor.getValue(algorithmState.partialInBottom);
    algorithmState.itemsInBottomStack += algorithmState.partialInBottom;
    childViewState.yTranslation = transitioningPositionStart + offset - childHeight - mPaddingBetweenElements;
    // We want at least to be at the end of the top stack when collapsing
    clampPositionToTopStackEnd(childViewState, childHeight);
    childViewState.location = StackScrollState.ViewState.LOCATION_MAIN_AREA;
