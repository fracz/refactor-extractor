final Intent verifyBroadcastLocked(Intent intent) {
    // Refuse possible leaked file descriptors
    if (intent != null && intent.hasFileDescriptors() == true) {
        throw new IllegalArgumentException("File descriptors passed in Intent");
    }
    int flags = intent.getFlags();
    if (!mProcessesReady) {
        // ahead and allow the broadcast without launching any receivers
        if ((flags & Intent.FLAG_RECEIVER_REGISTERED_ONLY_BEFORE_BOOT) != 0) {
        // This will be turned into a FLAG_RECEIVER_REGISTERED_ONLY later on if needed.
        } else if ((flags & Intent.FLAG_RECEIVER_REGISTERED_ONLY) == 0) {
            Slog.e(TAG, "Attempt to launch receivers of broadcast intent " + intent + " before boot completion");
            throw new IllegalStateException("Cannot broadcast before boot completed");
        }
    }
    if ((flags & Intent.FLAG_RECEIVER_BOOT_UPGRADE) != 0) {
        throw new IllegalArgumentException("Can't use FLAG_RECEIVER_BOOT_UPGRADE here");
    }
    if ((flags & Intent.FLAG_RECEIVER_FROM_SHELL) != 0) {
        switch(Binder.getCallingUid()) {
            case Process.ROOT_UID:
            case Process.SHELL_UID:
                break;
            default:
                Slog.w(TAG, "Removing FLAG_RECEIVER_FROM_SHELL because caller is UID " + Binder.getCallingUid());
                intent.removeFlags(Intent.FLAG_RECEIVER_FROM_SHELL);
                break;
        }
    }
    return intent;
