@Override
public boolean onTrackballEvent(MotionEvent ev) {
    if (DEBUG)
        L("trackball: %s", ev);
    switch(ev.getAction()) {
        case MotionEvent.ACTION_DOWN:
            poke();
            return true;
        case MotionEvent.ACTION_UP:
            unpoke();
            return true;
    }
    return false;
