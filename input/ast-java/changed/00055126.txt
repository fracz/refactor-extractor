@Override
public void insert(Task task, TaskStatus status) {
    giant.lock();
    try {
        Preconditions.checkNotNull(task, "task");
        Preconditions.checkNotNull(status, "status");
        Preconditions.checkArgument(task.getId().equals(status.getId()), "Task/Status ID mismatch[%s/%s]", task.getId(), status.getId());
        if (tasks.containsKey(task.getId())) {
            throw new TaskExistsException(task.getId());
        }
        log.info("Inserting task %s with status: %s", task.getId(), status);
        tasks.put(task.getId(), new TaskStuff(task, status));
    } finally {
        giant.unlock();
    }
