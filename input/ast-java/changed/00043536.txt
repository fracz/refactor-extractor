public void onServiceAdded(int status, int srvcType, int srvcInstId, ParcelUuid srvcId) {
    UUID srvcUuid = srvcId.getUuid();
    if (DBG)
        Log.d(TAG, "onServiceAdded() - service=" + srvcUuid + "status=" + status);
    BluetoothGattService service = getService(srvcUuid, srvcInstId, srvcType);
    if (service == null)
        return;
    try {
        mCallback.onServiceAdded((int) status, service);
    } catch (Exception ex) {
        Log.w(TAG, "Unhandled exception in callback", ex);
    }
