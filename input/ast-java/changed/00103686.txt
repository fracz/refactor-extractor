public static boolean isContextActive(String contextId) {
    Collection<?> contextIds = PlatformUI.getWorkbench().getActiveWorkbenchWindow().getService(IContextService.class).getActiveContextIds();
    for (Object id : contextIds) {
        if (contextId.equals(id)) {
            return true;
        }
    }
    return false;
