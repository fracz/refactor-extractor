public void create() {

        tree = new OMVRBTreeDatabaseLazySave<String, T>(underlyingDatabase, clusterName, OStreamSerializerLiteral.INSTANCE, OStreamSerializerAnyRecord.INSTANCE);

        OLogManager.instance().error(this, "Can't create the local database's dictionary", e, ODatabaseException.class);
