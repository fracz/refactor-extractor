    // first one - is a head revision: head hash + head number;
    for (int i = 1; i < branchesWithHeads.length; ++i) {
        Matcher matcher = BRANCH_PATTERN.matcher(branchesWithHeads[i]);

            branches.add(matcher.group(2));
