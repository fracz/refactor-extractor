/**
 * Retrieves a list of all possible Names that this site is referring to.
 */
public List<Name> getReferencesAt(Node site) {
    Preconditions.checkArgument(NodeUtil.isGetProp(site) || NodeUtil.isName(site));
    List<Name> result = new ArrayList<Name>();
    for (DiGraphEdge<Name, Reference> edge : referenceMap.get(site)) {
        Name target = edge.getDestination().getValue();
        result.add(target);
    }
    return result;
