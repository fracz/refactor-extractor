    oldPos = pos;
    pos = OSQLHelper.nextWord(text, textUpperCase, oldPos, word, true);
    if (pos == -1 || !word.toString().equals(KEYWORD_PROPERTY))

    oldPos = pos;

        throw new OCommandSQLParsingException("Expected <class>.<property>", text, oldPos);
