protected void onMapFull() {
    Map.Entry<MapKey, CachedValue> entry = accessMap.pollFirstEntry();
    map.remove(entry.getValue().getKey(), entry.getValue());
    if (entry.getValue().getAccessCount() == 0) {
        return;

    }
