private static boolean isSelectionContiguous(RadContainer selectionParent, ArrayList<RadComponent> selection) {
    assert selectionParent.isGrid();
    Rectangle rc = getSelectionBounds(selection);
    for (RadComponent c : selectionParent.getComponents()) {
        if (!selection.contains(c) && constraintsIntersect(true, c.getConstraints(), rc) && constraintsIntersect(false, c.getConstraints(), rc)) {
            return false;
        }
    }
    return true;
