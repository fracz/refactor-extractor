        nodesCreatedAndDeletedInTx.add(nodeId);
    }
    if (hasNodeStatesMap()) {
        NodeState nodeState = nodeStatesMap.remove(nodeId);
        if (nodeState != null) {
            DiffSets<Integer> diff = nodeState.labelDiffSets();

            nodeState.clear();

    }
    hasChanges = true;
