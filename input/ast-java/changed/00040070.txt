/**
 * Fling the scroll view
 *
 * @param velocityX The initial velocity in the X direction. Positive
 * numbers mean that the finger/curor is moving down the screen,
 * which means we want to scroll towards the left.
 */
public void fling(int velocityX) {
    if (getChildCount() > 0) {
        int width = getWidth() - mPaddingRight - mPaddingLeft;
        int right = getChildAt(0).getWidth();
        mScroller.fling(mScrollX, mScrollY, velocityX, 0, 0, Math.max(0, right - width), 0, 0, width / 2, 0);
        final boolean movingRight = velocityX > 0;
        View newFocused = findFocusableViewInMyBounds(movingRight, mScroller.getFinalX(), findFocus());
        if (newFocused == null) {
            newFocused = this;
        }
        if (newFocused != findFocus() && newFocused.requestFocus(movingRight ? View.FOCUS_RIGHT : View.FOCUS_LEFT)) {
            mScrollViewMovedFocus = true;
            mScrollViewMovedFocus = false;
        }
        awakenScrollBars(mScroller.getDuration());
        invalidate();
    }
