                scale += Math.abs(U.get(k, i));
            }

                for (k = i; k < m; k++) {
                    U.div(k, i, scale);
                    s += U.get(k, i) * U.get(k, i);
                }
                f = U.get(i, i);
                g = -Math.copySign(Math.sqrt(s), f);
                h = f * g - s;
                U.set(i, i, f - g);
                for (j = l - 1; j < n; j++) {
                    for (s = 0.0, k = i; k < m; k++) {
                        s += U.get(k, i) * U.get(k, j);
                    }
                    f = s / h;
                    for (k = i; k < m; k++) {
                        U.add(k, j, f * U.get(k, i));
                    }
                }
                for (k = i; k < m; k++) {
                    U.mul(k, i, scale);
                }
            }

            for (k = l - 1; k < n; k++) {

            }

                for (k = l - 1; k < n; k++) {
                    U.div(i, k, scale);
                    s += U.get(i, k) * U.get(i, k);
                }
                f = U.get(i, l - 1);
                g = -Math.copySign(Math.sqrt(s), f);

                }
                for (j = l - 1; j < m; j++) {
                    for (s = 0.0, k = l - 1; k < n; k++) {
                        s += U.get(j, k) * U.get(i, k);
                    }
                    for (k = l - 1; k < n; k++) {
                        U.add(j, k, s * rv1[k]);
                    }
                }
                for (k = l - 1; k < n; k++) {

                }
            }
        }

                V.set(j, k, -V.get(j, k));
