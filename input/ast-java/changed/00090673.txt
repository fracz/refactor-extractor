/**
 * Write a {@link DefaultFileRegion}
 *
 * @param region        the {@link DefaultFileRegion} from which the bytes should be written
 * @return amount       the amount of written bytes
 */
private boolean writeFileRegion(ChannelOutboundBuffer in, DefaultFileRegion region) throws Exception {
    final long regionCount = region.count();
    if (region.transfered() >= regionCount) {
        in.remove();
        return true;
    }
    final long baseOffset = region.position();
    boolean done = false;
    long flushedAmount = 0;
    for (; ; ) {
        final long offset = region.transfered();
        final long localFlushedAmount = Native.sendfile(fd().intValue(), region, baseOffset, offset, regionCount - offset);
        if (localFlushedAmount == 0) {
            // Returned EAGAIN need to set EPOLLOUT
            setEpollOut();
            break;
        }
        flushedAmount += localFlushedAmount;
        if (region.transfered() >= regionCount) {
            done = true;
            break;
        }
    }
    if (flushedAmount > 0) {
        in.progress(flushedAmount);
    }
    if (done) {
        in.remove();
    }
    return done;
