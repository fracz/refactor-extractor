@Override
void stop() {
    synchronized (mStateLock) {
        if (mDone) {
            return;
        }
        if (mStatusCode == TextToSpeech.STOPPED) {
            return;
        }
        mStatusCode = TextToSpeech.STOPPED;
        cleanUp();
        if (mDispatcher != null) {
            mDispatcher.dispatchOnStop();
        }
    }
