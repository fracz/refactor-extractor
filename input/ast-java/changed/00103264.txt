public String getSelectionText() {
    GridPos selection = resultSet.getSelection().getFirstElement();
    if (selection == null) {
        return "";
    }
    Spreadsheet spreadsheet = resultSet.getSpreadsheet();

    return CommonUtils.toString(value);
