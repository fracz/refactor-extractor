protected void sizeDatabase() throws IOException {
    setDataCommandInfo("Database size");
    if (!isConnectionAlive())
        return;
    beginResponse();
    try {
        sendOk(clientTxId);
        channel.writeLong(connection.database.getStorage().getSize());
    } finally {
        endResponse();
    }
