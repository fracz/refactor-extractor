private static String toString(String[] strs) {
    StringBuilder builder = new StringBuilder();
    for (int i = 0, n = strs.length; i < n; i++) {
        builder.append(strs[i]);
        if (i < n - 1) {
            builder.append(' ');
        }
    }
    return builder.toString();
