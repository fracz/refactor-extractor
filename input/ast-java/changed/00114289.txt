private static boolean lastOfSeveralNNP(Tree parent, Tree t) {
    Tree last = null;
    int numNNP = 0;
    for (Tree kid : parent.children()) {
        if (kid.value().startsWith("NNP")) {
            numNNP++;
            last = kid;
        }
    }
    return numNNP > 1 && t.equals(last);
