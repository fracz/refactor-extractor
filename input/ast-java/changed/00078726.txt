@Override
public void onClick(View view) {
    switch(view.getId()) {
        case R.id.additional_headers_row:
        case R.id.show_additional_headers_area:
            {
                onShowAdditionalHeaders();
                break;
            }
        case R.id.from:
            {
                onAddSenderToContacts();
                break;
            }
        case R.id.to:
        case R.id.cc:
            {
                expand((TextView) view, ((TextView) view).getEllipsize() != null);
                layoutChanged();
            }
    }
