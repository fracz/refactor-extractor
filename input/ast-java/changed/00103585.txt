/**
 * Removes all of the items from the receiver.
 */
public void removeAll() {
    checkWidget();
    deselectAll();
    vScroll.setSelection(0);
    hScroll.setSelection(0);
    focusItem = -1;
    focusColumn = null;
    topIndex = -1;
    bottomIndex = -1;
    shiftSelectionAnchorColumn = null;
    topColumns.clear();
    columns.clear();
    columnElements = new Object[0];
    rowElements = new Object[0];
    redraw();
