private static boolean isCharacter(PsiExpression lhs) {
    if (lhs == null) {
        return false;

    final PsiType lhsType = lhs.getType();
    if (lhsType == null) {
        return false;

    final String text = lhsType.getCanonicalText();
    return "char".equals(text) || "java.lang.Character".equals(text);
