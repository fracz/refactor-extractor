/**
 * ATOM        ::= CLOSURE | GROUP
 * CLOSURE     ::= GROUP QUANTIFIER
 */
private boolean parseAtom(PsiBuilder builder) {
    PsiBuilder.Marker marker = parseGroup(builder);
    if (marker == null) {
        return false;
    }
    marker = marker.precede();
    if (parseQuantifier(builder)) {
        marker.done(RegExpElementTypes.CLOSURE);
    } else {
        marker.drop();
    }
    return true;
