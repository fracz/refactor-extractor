public static String getPathWithoutScheme(Path path) {
    Path ori = path;
    String ret = "";
    while (path != null) {
        if (ret.equals("")) {
            ret = path.getName();
        } else {
            ret = path.getName() + "/" + ret;
        }
        path = path.getParent();
    }
    if (Config.DEBUG) {
        LOG.info("Utils getPathWithoutScheme(" + ori + ") result: " + ret);
    }
    return ret;
