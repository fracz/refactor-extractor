private Job getRunningJob(String paragraphId) {
    Job foundJob = null;
    Collection<Job> jobsRunning = getScheduler().getJobsRunning();
    for (Job job : jobsRunning) {
        if (job.getId().equals(paragraphId)) {
            foundJob = job;
        }
    }
    return foundJob;
