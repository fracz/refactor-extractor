void complete() {
    Collection<Runnable> callbacks = this.callbacks;
    if (callbacks != null) {
        Runnable[] targets;
        synchronized (callbacks) {
            targets = callbacks.toArray(new Runnable[callbacks.size()]);
            this.callbacks = null;
        }
        for (Runnable target : targets) {
            try {
                target.run();
            } catch (Exception e) {
                e.printStackTrace();
            }
        }
    }
