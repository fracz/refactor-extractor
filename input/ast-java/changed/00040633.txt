@Override
public void onReceive(Context context, Intent intent) {
    if (intent.getAction() == Intent.ACTION_BOOT_COMPLETED) {
        synchronized (mLock) {
            populateAccessibilityServiceListLocked();
            // get the accessibility enabled setting on boot
            mIsEnabled = Settings.Secure.getInt(mContext.getContentResolver(), Settings.Secure.ACCESSIBILITY_ENABLED, 0) == 1;
            // if accessibility is enabled inform our clients we are on
            if (mIsEnabled) {
                updateClientsLocked();
            }
            manageServicesLocked();
            updateInputFilterLocked();
        }
        return;
    }
    super.onReceive(context, intent);
