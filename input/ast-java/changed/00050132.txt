public void commit(Request request) {
    if (stopped || request == null) {
        return;
    }
    if (LOG.isDebugEnabled()) {
        LOG.debug("Committing request:: " + request);
    }
    committedRequests.add(request);
    if (!isProcessingCommit()) {
        wakeup();
    }
