private List<Pattern> collectPatternsFor(String groupName) {
    if (isAllExcludeDefinedGroup(groupName)) {
        return myAllPatterns;
    } else {
        if (!myPatterns.containsKey(groupName)) {
            return EMPTY_LIST;
