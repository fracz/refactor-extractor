private int evaluateJobPriorityLocked(JobStatus job) {
    int priority = job.getPriority();
    if (priority >= JobInfo.PRIORITY_FOREGROUND_APP) {
        return priority;
    }
    if (mForegroundUids.get(job.getSourceUid())) {
        return JobInfo.PRIORITY_FOREGROUND_APP;
    }
    return priority;
