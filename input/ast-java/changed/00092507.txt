public synchronized int rollback(final ODatabaseDocumentTx database) {
    ODistributedServerLog.debug(this, db.getManager().getLocalNodeName(), null, ODistributedServerLog.DIRECTION.NONE, "Distributed transaction: rolling back transaction %s (%d ops)", reqId, undoTasks.size());
    for (ORemoteTask task : undoTasks) {
        try {
            task.execute(reqId, db.getManager().getServerInstance(), db.getManager(), database);
        } catch (Exception e) {
            ODistributedServerLog.error(this, db.getManager().getLocalNodeName(), null, ODistributedServerLog.DIRECTION.NONE, "Error on rolling back transaction %s task %s", e, reqId, task);
        }
    }
    unlock();
    return undoTasks.size();
