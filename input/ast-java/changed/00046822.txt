/**
 * Stop services common to both active and standby states
 * @throws IOException
 */
void stopCommonServices() {
    writeLock();
    try {
        if (blockManager != null)
            blockManager.close();
        cacheManager.deactivate();
    } finally {
        writeUnlock();
    }
    RetryCache.clear(retryCache);
