    c.applyTo(new Snapshot() {

        @Override
        public Entry getEntry(Path path) {
            return new FileEntry(null, null, "content");
        }
    });
    os.writeChange(c);
    Change read = is.readChange();
    assertEquals(ChangeFileContentChange.class, read.getClass());
