    for (int i = 0; i < 100; i++) {
        Child child = database.newInstance(Child.class);
        child.setName(String.valueOf(i));
        test.getSet().add(child);
    }
    Assert.assertNotNull(test.getSet());

    database.save(test);

    close();
    open();

    Assert.assertNotNull(test.getSet());

        Assert.assertNotNull(child.getName());

    database.delete(test);
