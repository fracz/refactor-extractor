public Subscriber<? super T> apply(Subscriber<? super U> t) {
    U coll;
    try {
        coll = collectionSupplier.get();
    } catch (Throwable e) {
        t.onSubscribe(EmptySubscription.INSTANCE);
        t.onError(e);
        return CancelledSubscriber.INSTANCE;
    }
    return new ToListSubscriber<>(t, coll);
