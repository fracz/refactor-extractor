private void executeDeletes(int number, int txNumber, ODatabase database) {
    int txCounter = 1;
    if (txNumber > 0) {
        database.begin();
    }
    for (int j = 0; j < number; j++) {
        if (txNumber > 0 && txCounter % txNumber == 0) {
            database.commit();
            txCounter = 1;
            database.begin();
        }
        ODatabaseUtils.deleteOperation(insertedDocs.get(j));
        consoleWriter.addDelete();
    }
    if (txNumber > 0) {
        database.commit();
    }
