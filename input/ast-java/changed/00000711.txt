    testFile.createNewFile();

    fos.close();
    mFsShell.copyFromLocal(new String[] { "copyFromLocal", testFile.getAbsolutePath(), "/testFile" });

    Assert.assertNotNull(fileInfo);
    Assert.assertEquals(SIZE_BYTES, fileInfo.length);
