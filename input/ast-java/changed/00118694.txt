        // Simple query does not apply
        return null;

    if (types.length > 0) {
        type = getType(types[0]);
        if (type == null)
            return null;
    }
    switch(relationType) {
        case PROPERTY:
            return new SimpleVertexQueryProcessor(vertex, (TitanKey) type);
        case EDGE:
            return new SimpleVertexQueryProcessor(vertex, dir, (TitanLabel) type);
        default:
            throw new IllegalArgumentException("Invalid relation type: " + relationType);
    }
