protected String as() {
    String alias = null;
    final Token token = lexer.token();
    if (token == Token.COMMA) {
        return null;
    }
    if (token == Token.AS) {
        lexer.nextToken();
        alias = alias();
        if (alias != null) {
            while (lexer.token() == Token.DOT) {
                lexer.nextToken();
                alias += ('.' + lexer.token().name());
                lexer.nextToken();
            }
            return alias;
        }
        if (lexer.token() == Token.LPAREN) {
            return null;
        }
        throw new ParserException("Error : " + lexer.token());
    }
    if (lexer.token() == Token.LITERAL_ALIAS) {
        alias = '"' + lexer.stringVal() + '"';
        lexer.nextToken();
    } else if (lexer.token() == Token.IDENTIFIER) {
        alias = lexer.stringVal();
        lexer.nextToken();
    } else if (lexer.token() == Token.LITERAL_CHARS) {
        alias = "'" + lexer.stringVal() + "'";
        lexer.nextToken();
    } else {
        switch(lexer.token()) {
            case CASE:
            case USER:
            case LOB:
            case END:
            case DEFERRED:
            case OUTER:
                alias = lexer.stringVal();
                lexer.nextToken();
                break;
            default:
                break;
        }
    }
    switch(lexer.token()) {
        case KEY:
        case INTERVAL:
        case CONSTRAINT:
            alias = lexer.token().name();
            lexer.nextToken();
            return alias;
        default:
            break;
    }
    return alias;
