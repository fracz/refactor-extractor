List<Provider<?>> findParameterProviders(Method method) {
    List<Provider<?>> parameterProviders = new ArrayList<Provider<?>>();
    Type[] parameterTypes = method.getGenericParameterTypes();
    Annotation[][] parameterAnnotations = method.getParameterAnnotations();
    for (int i = 0; i < parameterTypes.length; i++) {
        Annotation bindingAnnotation = findBindingAnnotation(parameterAnnotations[i]);
        Key<?> key = bindingAnnotation == null ? Key.get(parameterTypes[i]) : Key.get(parameterTypes[i], bindingAnnotation);
        Provider<?> provider = getProvider(key);
        parameterProviders.add(provider);
    }
    return parameterProviders;
