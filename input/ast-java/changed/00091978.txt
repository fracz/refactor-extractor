@Bean
SpanStore spanStore() {
    SpanStore result;
    if (datasource != null && server.getStore().getType() == Type.mysql) {
        result = new JDBCSpanStore(datasource, new Settings().withRenderSchema(false), listener);
    } else {
        result = new InMemorySpanStore();
    }
    return result;
