@Override
public void notifyListeners() {
    // only show wifi in the cluster if connected or if wifi-only
    boolean wifiVisible = mCurrentState.enabled && (mCurrentState.connected || !mHasMobileData);
    String wifiDesc = wifiVisible ? mCurrentState.ssid : null;
    boolean ssidPresent = wifiVisible && mCurrentState.ssid != null;
    String contentDescription = getStringIfExists(getContentDescription());
    int length = mSignalsChangedCallbacks.size();
    for (int i = 0; i < length; i++) {
        mSignalsChangedCallbacks.get(i).onWifiSignalChanged(mCurrentState.enabled, mCurrentState.connected, getQsCurrentIconId(), ssidPresent && mCurrentState.activityIn, ssidPresent && mCurrentState.activityOut, contentDescription, wifiDesc);
    }
    int signalClustersLength = mSignalClusters.size();
    for (int i = 0; i < signalClustersLength; i++) {
        mSignalClusters.get(i).setWifiIndicators(wifiVisible, getCurrentIconId(), contentDescription);
    }
