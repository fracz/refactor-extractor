@Test
public void testComposeHandlerFailAfterCompletion() {
    Future<String> f1 = Future.future();
    Future<Integer> f2 = Future.future();
    RuntimeException cause = new RuntimeException();
    f1.compose(string -> {
        f2.complete(46);
        throw cause;
    }, f2);
    try {
        f1.complete("foo");
    } catch (Exception e) {
        assertEquals(cause, e);
    }
    assertSucceeded(f2, 46);
