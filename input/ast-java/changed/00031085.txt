public static String readFromResource(String resource) throws IOException {
    InputStream in = null;
    try {
        in = Thread.currentThread().getContextClassLoader().getResourceAsStream(resource);
        if (in == null) {
            return null;
        }
        String text = IOUtils.read(in);
        return text;
    } finally {
        JdbcUtils.close(in);
    }
