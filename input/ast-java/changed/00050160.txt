public List<String> getChildren(String nodePath) {
    if (connected) {
        try {
            return zooKeeper.getChildren(nodePath, false);
        } catch (Exception e) {
            LoggerFactory.getLogger().error("Error occurred retrieving children of node: " + nodePath, e);
        }
    }
    return null;
