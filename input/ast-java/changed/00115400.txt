@Override
public void saveToFilename(String s) {
    PrintWriter bw = null;
    try {
        bw = IOUtils.getPrintWriter(s);
        for (int i = 0, size = size(); i < size; i++) {
            E o = get(i);
            if (o != null) {
                bw.printf("%d=%s%n", i, o.toString());
            }
        }
        bw.close();
    } catch (IOException e) {
        e.printStackTrace();
    } finally {
        if (bw != null) {
            bw.close();
        }
    }
