static void doAddUsed(Phi r, Set<Phi> regs) {
    if (r.used) {
        if (!regs.contains(r)) {
            regs.add(r);
            for (int i = 0; i < r.size(); i++) {
                Phi p = r.get(i);
                p.used = true;
                doAddUsed(p, regs);
            }
        }
    }
