private static List<DfaMemoryStateImpl> mergeGroup(List<DfaMemoryStateImpl> group) {
    if (group.size() < 2) {
        return group;
    }
    StateMerger merger = new StateMerger();
    while (true) {
        List<DfaMemoryStateImpl> nextStates = merger.mergeByEquality(group);
        if (nextStates == null)
            nextStates = merger.mergeByType(group);
        if (nextStates == null)
            nextStates = merger.mergeByNullability(group);
        if (nextStates == null)
            nextStates = merger.mergeByUnknowns(group);
        if (nextStates == null)
            break;
        group = nextStates;
    }
    return group;
