private final void fullAddCount(long x, CounterHashCode hc, boolean wasUncontended) {
    int h;
    if (hc == null) {
        hc = new CounterHashCode();
        int s = counterHashCodeGenerator.addAndGet(SEED_INCREMENT);
        // Avoid zero
        h = hc.code = (s == 0) ? 1 : s;
        threadCounterHashCode.set(hc);
    } else
        h = hc.code;
    // True if last slot nonempty
    boolean collide = false;
    for (; ; ) {
        CounterCell[] as;
        CounterCell a;
        int n;
        long v;
        if ((as = counterCells) != null && (n = as.length) > 0) {
            if ((a = as[(n - 1) & h]) == null) {
                if (cellsBusy == 0) {
                    // Try to attach new Cell
                    // Optimistic create
                    CounterCell r = new CounterCell(x);
                    if (cellsBusy == 0 && U.compareAndSwapInt(this, CELLSBUSY, 0, 1)) {
                        boolean created = false;
                        try {
                            // Recheck under lock
                            CounterCell[] rs;
                            int m, j;
                            if ((rs = counterCells) != null && (m = rs.length) > 0 && rs[j = (m - 1) & h] == null) {
                                rs[j] = r;
                                created = true;
                            }
                        } finally {
                            cellsBusy = 0;
                        }
                        if (created)
                            break;
                        // Slot is now non-empty
                        continue;
                    }
                }
                collide = false;
            } else if (// CAS already known to fail
            !wasUncontended)
                // Continue after rehash
                wasUncontended = true;
            else if (U.compareAndSwapLong(a, CELLVALUE, v = a.value, v + x))
                break;
            else if (counterCells != as || n >= NCPU)
                // At max size or stale
                collide = false;
            else if (!collide)
                collide = true;
            else if (cellsBusy == 0 && U.compareAndSwapInt(this, CELLSBUSY, 0, 1)) {
                try {
                    if (counterCells == as) {
                        // Expand table unless stale
                        CounterCell[] rs = new CounterCell[n << 1];
                        for (int i = 0; i < n; ++i) rs[i] = as[i];
                        counterCells = rs;
                    }
                } finally {
                    cellsBusy = 0;
                }
                collide = false;
                // Retry with expanded table
                continue;
            }
            // Rehash
            h ^= h << 13;
            h ^= h >>> 17;
            h ^= h << 5;
        } else if (cellsBusy == 0 && counterCells == as && U.compareAndSwapInt(this, CELLSBUSY, 0, 1)) {
            boolean init = false;
            try {
                // Initialize table
                if (counterCells == as) {
                    CounterCell[] rs = new CounterCell[2];
                    rs[h & 1] = new CounterCell(x);
                    counterCells = rs;
                    init = true;
                }
            } finally {
                cellsBusy = 0;
            }
            if (init)
                break;
        } else if (U.compareAndSwapLong(this, BASECOUNT, v = baseCount, v + x))
            // Fall back on using base
            break;
    }
    // Record index for next time
    hc.code = h;
