private MapMaker setKeyStrength(Strength strength) {
    if (keyStrength != Strength.STRONG) {
        throw new IllegalStateException("Key strength was already set to " + keyStrength + ".");
    }
    keyStrength = strength;
    useCustomMap = true;
    return this;
