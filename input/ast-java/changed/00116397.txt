private double getSourceTotal(Object node) {
    double result = 0.0;
    Set<Arc> arcs = getArcsBySource(node);
    if (arcs.isEmpty()) {
        System.err.println("No outbound arcs from node.");
        return result;
    }
    for (Arc arc : arcs) {
        result += ((Double) arc.getOutput()).doubleValue();
    }
    return result;
