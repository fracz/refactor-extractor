private View inflateButton(String button, ViewGroup parent, boolean landscape) {
    View v = null;
    if (HOME.equals(button)) {
        v = (landscape ? mLandscapeInflater : mLayoutInflater).inflate(R.layout.home, parent, false);
        if (landscape && isSw600Dp()) {
            setupLandButton(v);
        }
    } else if (BACK.equals(button)) {
        v = (landscape ? mLandscapeInflater : mLayoutInflater).inflate(R.layout.back, parent, false);
        if (landscape && isSw600Dp()) {
            setupLandButton(v);
        }
    } else if (RECENT.equals(button)) {
        v = (landscape ? mLandscapeInflater : mLayoutInflater).inflate(R.layout.recent_apps, parent, false);
        if (landscape && isSw600Dp()) {
            setupLandButton(v);
        }
    } else if (MENU_IME.equals(button)) {
        v = (landscape ? mLandscapeInflater : mLayoutInflater).inflate(R.layout.menu_ime, parent, false);
    } else if (NAVSPACE.equals(button)) {
        v = (landscape ? mLandscapeInflater : mLayoutInflater).inflate(R.layout.nav_key_space, parent, false);
    } else {
        throw new IllegalArgumentException("Unknown button " + button);
    }
    parent.addView(v);
    addToDispatchers(v);
    return v;
