public List<Integer> emit(int streamId, List<Object> tuple, Object messageId) {
    List<Integer> ret = _collector.emit(streamId, tuple, messageId);
    emitted++;
    transferred += ret.size();
    if (messageId != null)
        transferred++;
    // for the extra ack on branching or finishing
    if (ret.size() != 1) {
        transferred += 1;
    }
    return ret;
