@Override
public void reportSizeConfigurations(IBinder token, int[] horizontalSizeConfiguration, int[] verticalSizeConfigurations) {
    if (DEBUG_CONFIGURATION)
        Slog.v(TAG, "Report configuration: " + token + " " + horizontalSizeConfiguration + " " + verticalSizeConfigurations);
    synchronized (this) {
        ActivityRecord record = ActivityRecord.isInStackLocked(token);
        if (record == null) {
            throw new IllegalArgumentException("reportSizeConfigurations: ActivityRecord not " + "found for: " + token);
        }
        record.setSizeConfigurations(horizontalSizeConfiguration, verticalSizeConfigurations);
    }
