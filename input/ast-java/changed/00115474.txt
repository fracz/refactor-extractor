@Override
public void saveToFilename(String file) {
    BufferedWriter bw = null;
    try {
        bw = new BufferedWriter(new FileWriter(file));
        for (int i = 0, sz = size(); i < sz; i++) {
            bw.write(i + "=" + get(i) + '\n');
        }
        bw.close();
    } catch (IOException e) {
        e.printStackTrace();
    } finally {
        if (bw != null) {
            try {
                bw.close();
            } catch (IOException ioe) {
            // give up
            }
        }
    }
