public LineMarkerInfo getLineMarkerInfo(PsiElement element) {
    if (!(element instanceof PsiMethod)) {
        return null;
    }
    final PsiMethod method = (PsiMethod) element;
    String name = method.getName();
    if (!name.startsWith("test")) {
        return null;
    }
    String testDataPath = getTestDataBasePath(method.getContainingClass());
    if (testDataPath != null) {
        List<String> fileNames = collectTestDataReferences(method, testDataPath, name.substring(4));
        if (fileNames.size() > 0) {
            return new LineMarkerInfo<PsiMethod>(method, method.getTextOffset(), Icons.TEST_SOURCE_FOLDER, Pass.UPDATE_ALL, null, new TestDataNavigationHandler(fileNames));
        }
    }
    return null;
