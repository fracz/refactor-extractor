private static String constructHelpfulDiagnosticsMessage(LockManager lockManager, Transaction tx, Object resource, LockType type) {
    StringBuilder builder = new StringBuilder(format("%s tried to apply local %s lock on %s after acquired on master. Currently these locks exist:%n", tx, type, resource));
    for (LockInfo lock : lockManager.getAllLocks()) {
        if (lock.getReadCount() > 0 || lock.getWriteCount() > 0) {
            builder.append(format(lock.toString()));
        }
    }
    return builder.toString();
