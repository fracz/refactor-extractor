public void resized(int w, int h, Rect coveredInsets, Rect visibleInsets, boolean reportDraw) {
    SurfaceView surfaceView = mSurfaceView.get();
    if (mCompatibilityInfo.mScalingRequired) {
        float scale = mCompatibilityInfo.mApplicationInvertedScale;
        w *= scale;
        h *= scale;
        coveredInsets.scale(scale);
        visibleInsets.scale(scale);
    }
    if (surfaceView != null) {
        if (localLOGV)
            Log.v("SurfaceView", surfaceView + " got resized: w=" + w + " h=" + h + ", cur w=" + mCurWidth + " h=" + mCurHeight);
        synchronized (this) {
            if (mCurWidth != w || mCurHeight != h) {
                mCurWidth = w;
                mCurHeight = h;
            }
            if (reportDraw) {
                try {
                    surfaceView.mSession.finishDrawing(surfaceView.mWindow);
                } catch (RemoteException e) {
                }
            }
        }
    }
