public static JavaSdkVersion getJavaSdkVersion(@NotNull PsiElement element) {

    if (module != null) {
        final Sdk sdk = ModuleRootManager.getInstance(module).getSdk();

            return JavaSdk.getInstance().getVersion(sdk);
