private void doClose(CloseCode code, String reason, boolean initiatedByRemote) {
    if (state == State.CLOSED) {
        return;
    }
    if (in != null) {
        try {
            in.close();
        } catch (IOException e) {
            e.printStackTrace();
        }
    }
    if (out != null) {
        try {
            out.close();
        } catch (IOException e) {
            e.printStackTrace();
        }
    }
    state = State.CLOSED;
    onClose(this, code, reason, initiatedByRemote);
