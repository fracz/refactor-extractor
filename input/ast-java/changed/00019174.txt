private static PsiElement[] toPsiElementArray(Collection<PsiCrumb> items) {
    final PsiElement[] elements = new PsiElement[items.size()];
    int index = 0;
    for (PsiCrumb item : items) {
        elements[index++] = item.anchor.retrieve();
    }
    return elements;
