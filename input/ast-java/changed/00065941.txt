        adapter.invokeStatic(method.owner.type, method.method);

        adapter.invokeInterface(method.owner.type, method.method);
    } else {
        adapter.invokeVirtual(method.owner.type, method.method);
    }

        adapter.checkCast(method.rtn.type);
