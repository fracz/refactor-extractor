    getDefaultBounds().printShortString(pw);

    mService.getStackBounds(PINNED_STACK_ID, mTmpRect);

    getMovementBounds(mTmpRect).printShortString(pw);

    pw.println(prefix + "  mInInteractiveMode=" + mInInteractiveMode);
