        if (converted.getSecond()) {
            return returnCall;

        if (returnCall == null) {
            returnCall = converted.getFirst();

        if (converted.getFirst() == null) {
            return returnCall;

            newMethodCall = (PsiMethodCallExpression) expression;
        } else {
            return returnCall;
        }
    }
