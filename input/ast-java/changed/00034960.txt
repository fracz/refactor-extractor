public SQLDropUserStatement parseDropUser() {
    accept(Token.USER);
    SQLDropUserStatement stmt = new SQLDropUserStatement(getDbType());
    for (; ; ) {
        SQLExpr expr = this.exprParser.expr();
        stmt.getUsers().add(expr);

            lexer.nextToken();
            continue;
        }
        break;
    }
    return stmt;
