@Test
public void testStartStop() throws InterruptedException, IOException, TException {
    RemoteInterpreterServer server = new RemoteInterpreterServer("localhost", RemoteInterpreterUtils.findRandomAvailablePortOnAllLocalInterfaces());
    assertEquals(false, server.isRunning());
    server.start();
    long startTime = System.currentTimeMillis();
    boolean running = false;
    while (System.currentTimeMillis() - startTime < 10 * 1000) {
        if (server.isRunning()) {
            running = true;
            break;
        } else {
            Thread.sleep(200);
        }
    }
    assertEquals(true, running);
    assertEquals(true, RemoteInterpreterUtils.checkIfRemoteEndpointAccessible("localhost", server.getPort()));
    server.shutdown();
    while (System.currentTimeMillis() - startTime < 10 * 1000) {
        if (server.isRunning()) {
            Thread.sleep(200);
        } else {
            running = false;
            break;
        }
    }
    assertEquals(false, running);
