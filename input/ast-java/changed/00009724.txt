private void load() {
    if (!myIndicesDir.exists())
        return;
    File[] indices = myIndicesDir.listFiles();
    if (indices == null)
        return;
    for (File each : indices) {
        try {
            myIndices.add(new MavenIndex(each));
        } catch (Exception e) {
            FileUtil.delete(each);
            MavenLog.warn(e);
        }
    }
