public void configure(AtmosphereConfig config) {

    if (s != null) {
        method = s;
    }
    s = config.getInitParameter(ATMOSPHERERESOURCE_INTERCEPTOR_TIMEOUT);

        timeoutInSeconds = Integer.valueOf(s);
