@Override
public void lock(long maxTime) {
    long start = Timestamps.SYSTEM().getTime();
    first.lock(maxTime);
    long remaining = Math.max(0, maxTime - (start - Timestamps.SYSTEM().getTime()));
    try {
        second.lock(remaining);
    } catch (RuntimeException e) {
        first.unlock();
        throw e;
    }
