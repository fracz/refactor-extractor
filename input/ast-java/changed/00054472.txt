/**
 * Shuts down all thread pools created by this class in an orderly fashion.
 */
public void shutdown() {
    synchronized (threadPools) {
        for (ExecutorService executor : threadPools.values()) {
            executor.shutdown();
        }
        threadPools.clear();
    }
