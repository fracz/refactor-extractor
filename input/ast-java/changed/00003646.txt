private void analyzeNullLiteralMethodArguments(PsiMethod method, ProblemsHolder holder, boolean isOnTheFly) {
    if (REPORT_NULLS_PASSED_TO_NOT_NULL_PARAMETER) {
        for (PsiParameter parameter : NullParameterConstraintChecker.checkMethodParameters(method, isOnTheFly)) {
            final String name = parameter.getName();
            holder.registerProblem(parameter.getNameIdentifier(), InspectionsBundle.message("dataflow.method.fails.with.null.argument", name), ProblemHighlightType.GENERIC_ERROR_OR_WARNING, NullableStuffInspectionBase.getWrappedUiDependentQuickFix(this::createNavigateToNullParameterUsagesFix, parameter, isOnTheFly));
        }
    }
