 * {@inheritDoc}
 */
@Override
public byte[] getBodyPartBytes() {
    byte[] bytes = contentBytes.get();
    if (bytes != null) {
        return bytes;
    }
    final Buffer b = content.getContent();
    final int origPos = b.position();
    bytes = new byte[b.remaining()];
    b.get(bytes);
    b.flip();
    b.position(origPos);
    contentBytes.compareAndSet(null, bytes);
    return bytes;
