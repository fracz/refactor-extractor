/**
 * Returns the total number of items in the data set hold by the adapter.
 *
 * @return The total number of items in this adapter.
 */
@Override
public int getItemCount() {
    int offset = 0;
    if (hasHeaderView())
        offset++;
    if (enableLoadMore())
        offset++;
    return getAdapterItemCount() + offset;
