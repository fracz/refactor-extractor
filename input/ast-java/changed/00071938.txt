/**
 * Closure's goog.inherits adds a {@code superClass_} property to the
 * subclass, and a {@code constructor} property.
 */
@Override
public void applySubclassRelationship(FunctionType parentCtor, FunctionType childCtor, SubclassType type) {
    super.applySubclassRelationship(parentCtor, childCtor, type);
    if (type == SubclassType.INHERITS) {
        childCtor.defineDeclaredProperty("superClass_", parentCtor.getPrototype(), childCtor.getSource());
        childCtor.getPrototype().defineDeclaredProperty("constructor", // and return types of G.prototype.constructor.
        childCtor.forgetParameterAndReturnTypes(), childCtor.getSource());
    }
