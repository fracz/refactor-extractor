public void endDraw() {

        popFramebuffer();
        gl.glClearColor(0, 0, 0, 0);
        gl.glClear(GL10.GL_COLOR_BUFFER_BIT | GL10.GL_DEPTH_BUFFER_BIT);
        renderDrawTexture();
    } else {
        if (!clear && screenTexID[0] != 0) {
            copyFrameToScreenTexture();
        }
    }
    gl.glFlush();
    report("top endDraw()");
