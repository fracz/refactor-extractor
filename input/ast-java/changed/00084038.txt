    assert xidIdentMap.get(identifier) != null;
    assert txId != -1;
    try {
        writeBuffer.put(TX_1P_COMMIT).putInt(identifier).putLong(txId);

        throw new XAException("Logical log unable to mark 1P-commit [" + identifier + "] " + e);
