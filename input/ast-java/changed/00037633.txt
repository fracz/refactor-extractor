private void startTextSelectionMode() {
    if (!mIsInTextSelectionMode) {
        if (!hasSelectionController()) {
            Log.w(LOG_TAG, "TextView has no selection controller. Action mode cancelled.");
            return;
        }
        if (!requestFocus()) {
            return;
        }
        selectCurrentWord();
        mIsInTextSelectionMode = true;
    }
