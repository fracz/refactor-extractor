private void addIndexEntry(TitanProperty prop, StoreMutator mutator) {
    TitanKey pt = prop.getPropertyKey();
    assert pt.isSimple();
    if (pt.hasIndex()) {
        if (pt.isUnique()) {
            mutator.mutateIndex(getIndexKey(prop.getAttribute()), Lists.newArrayList(new Entry(getKeyedIndexColumn(pt), getIndexValue(prop))), null);
        } else {
            mutator.mutateIndex(getIndexKey(prop.getAttribute()), Lists.newArrayList(new Entry(getIndexColumn(pt, prop.getID()), getIndexValue(prop))), null);
        }
    }
