private String ditrans(Tree t) {
    int n = 0;
    for (Tree kid : t.children()) {
        String childStr = kid.label().value();
        if (childStr.startsWith("NP") && !childStr.contains("-TMP")) {
            n++;
        } else if (englishTrain.markDitransV == 1 && childStr.startsWith("S")) {
            n++;
        }
    }
    if (n >= 2) {
        return "^2Arg";
    } else {
        return "";
    }
