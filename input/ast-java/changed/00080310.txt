public Headers getHeader() {
    do {
        if (namesAndValues != null) {
            break;
        }
        if (nameAndValuesString == null) {
            break;
        }
        synchronized (this) {
            if (namesAndValues != null) {
                break;
            }
            namesAndValues = FileDownloadUtils.convertHeaderString(nameAndValuesString);
        }
    } while (false);
    return Headers.of(namesAndValues);
