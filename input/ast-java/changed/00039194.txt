private void logBootFile(ContentResolver cr, DropBoxManager db, String filename, String tag) throws IOException {
    // Logging disabled
    if (cr == null || db == null || !db.isTagEnabled(tag))
        return;
    File file = new File(filename);
    long fileTime = file.lastModified();
    // File does not exist
    if (fileTime <= 0)
        return;
    String setting = "logfile:" + filename;
    long lastTime = Settings.Secure.getLong(cr, setting, 0);
    // Already logged this particular file
    if (lastTime == fileTime)
        return;
    db.addFile(tag, file, DropBoxManager.IS_TEXT);
