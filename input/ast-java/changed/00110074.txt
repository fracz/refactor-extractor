    int position = offset + length;
    while (!('*' == charAt(position) && '/' == charAt(position + 1))) {
        if (CharTypes.EOI == charAt(position)) {
            throw new UnterminatedSignException("*/");
        }
        position++;
        length++;
    }
    length += 2;
    return offset + length;
