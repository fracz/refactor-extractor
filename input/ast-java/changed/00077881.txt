/**
 * Returns the lower triangular factor.
 */
public double[][] getL() {
    int m = LU.nrows();
    int n = LU.ncols();
    double[][] L = new double[m][n];
    for (int i = 0; i < m; i++) {
        for (int j = 0; j < n; j++) {
            if (i > j) {
                L[i][j] = LU.get(i, j);
            } else if (i == j) {
                L[i][j] = 1.0;
            } else {
                L[i][j] = 0.0;
            }
        }
    }
    return L;
