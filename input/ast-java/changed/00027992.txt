void registerTaskResult(ProcessingResult result, int count) {
    switch(result) {
        case Success:
            numberOfSuccessfulExecutions.addAndGet(count);
            break;
        case TransientError:
            numberOfTransientError.addAndGet(count);
            break;
        case PermanentError:
            numberOfPermanentError.addAndGet(count);
            break;
    }
