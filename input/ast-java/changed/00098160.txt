public void flush() {
    acquireExclusiveLock();
    try {
        flushSplitBucketsBuffer(true);
        saveState();
        metadataStore.synch();
        treeStateStore.synch();
        for (int i = 0; i < filesMetadata.length; i++) if (filesMetadata[i] != null)
            buffer.flushFile(fileLevelIds[i]);
    } catch (IOException e) {
        throw new OIndexException("Error during hash table flush", e);
    } finally {
        releaseExclusiveLock();
    }
