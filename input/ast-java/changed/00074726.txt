public List<DaemonStatus> getAll() {
    List<DaemonStatus> out = new LinkedList<DaemonStatus>();
    List<File> files = getAllFiles();
    for (File file : files) {
        DaemonStatus daemonStatus = openRegistry(file).get();
        if (daemonStatus != null) {
            out.add(daemonStatus);
        }
    }
    return out;
