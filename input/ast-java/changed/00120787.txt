@Override
public Screen pop(String navigatorId) {
    super.pop(navigatorId);
    for (ScreenStack stack : mScreenStacks) {
        if (stack.peek().navigatorId.equals(navigatorId)) {
            Screen popped = stack.pop();
            updateStyles();
            return popped;
        }
    }
    return null;
