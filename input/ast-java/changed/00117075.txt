/**
 * Annotate for mention (statistical or hybrid)
 */
public Annotator mention(Properties properties) {
    // TO DO: split up coref and mention properties
    Properties corefProperties = PropertiesUtils.extractPrefixedProperties(properties, Annotator.STANFORD_COREF + ".", true);
    Properties mentionProperties = PropertiesUtils.extractPrefixedProperties(properties, Annotator.STANFORD_MENTION + ".", true);
    // handle default for coref.useConstituencyParse based on whether parse is used or not
    String[] annotatorsList = properties.getProperty("annotators", "").replaceAll("\\s+", "").split(",");
    if (Arrays.asList(annotatorsList).contains(Annotator.STANFORD_PARSE) && !corefProperties.containsKey(Annotator.STANFORD_COREF + ".useConstituencyParse")) {
        corefProperties.setProperty(Annotator.STANFORD_COREF + ".useConstituencyParse", "true");
    }
    Properties allPropsForCoref = new Properties();
    allPropsForCoref.putAll(corefProperties);
    allPropsForCoref.putAll(mentionProperties);
    return new MentionAnnotator(allPropsForCoref);
