Map<K, V> orderedMap(LinkedDeque<Node<K, V>> deque, Function<V, V> transformer, boolean ascending, int limit) {
    Caffeine.requireArgument(limit >= 0);
    evictionLock.lock();
    try {
        drainBuffers();
        final int initialCapacity = (weigher == Weigher.singleton()) ? Math.min(limit, evicts() ? (int) adjustedWeightedSize() : size()) : 16;
        final Map<K, V> map = new LinkedHashMap<K, V>(initialCapacity);
        final Iterator<Node<K, V>> iterator = ascending ? deque.iterator() : deque.descendingIterator();
        while (iterator.hasNext() && (limit > map.size())) {
            Node<K, V> node = iterator.next();
            K key = node.getKey();
            V value = transformer.apply(node.getValue());
            if ((key != null) && (value != null)) {
                map.put(key, value);
            }
        }
        return unmodifiableMap(map);
    } finally {
        evictionLock.unlock();
    }
