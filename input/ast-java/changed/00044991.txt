public void validate(ByteBuffer bytes) throws MarshalException {
    // 0-127
    for (int i = bytes.position(); i < bytes.limit(); i++) {
        byte b = bytes.get(i);
        if (b < 0 || b > 127)
            throw new MarshalException("Invalid byte for ascii: " + Byte.toString(b));
