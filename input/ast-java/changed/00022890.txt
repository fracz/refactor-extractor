    final NegationCountVisitor visitor = new NegationCountVisitor();
    method.accept(visitor);
    final int negationCount = visitor.getCount();
    if (negationCount <= 3) {
        return;
    }
    if (m_ignoreInEquals && MethodUtils.isEquals(method)) {
        return;
    }
    registerMethodError(method);
