 * Resolves a named type by looking it up in the registry.
 */
private void resolveViaRegistry(ErrorReporter t, StaticScope<JSType> enclosing) {
    JSType type = registry.getType(reference);

        setReferencedType((ObjectType) type, t, enclosing);
