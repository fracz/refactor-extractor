        @Override
        public StoreChannel answer(InvocationOnMock invocation) throws Throwable {

        xaLogicalLog.done(identifier);
    }
    // when
    xaLogicalLog.rotate();
    // then
    assertThat("should not read excessively from the logical log file channel", reads, lessThan(10));
