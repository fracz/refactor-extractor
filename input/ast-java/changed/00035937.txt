        next();
        if (ch == '=') {
            next();
            op = BinaryCompareOperator.LE;
        } else {
            op = BinaryCompareOperator.LT;
        }
    } else if (ch == '>') {
        next();
        if (ch == '=') {
            next();
            op = BinaryCompareOperator.GE;
        } else {
            op = BinaryCompareOperator.GT;
        }
    }

        if ("not".equalsIgnoreCase(name)) {
            skipWhitespace();
            name = readName();
            if ("like".equalsIgnoreCase(name)) {
