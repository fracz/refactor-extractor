@Override
public String toString() {
    if (display != null) {
        return display;
    }
    HttpMessage msg = null;
    try {
        msg = getHttpMessage();
        display = getDisplay(msg);

        display = "";
    } catch (DatabaseException e) {
        display = "";
    }
    return display;
