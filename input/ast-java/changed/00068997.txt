    if (in.readBoolean()) {
        queryHint = in.readUTF();

        scroll = readScroll(in);

    if (size == 0) {
        source = Bytes.EMPTY_ARRAY;

        source = new byte[size];
        in.readFully(source);
    }
    size = in.readInt();

        extraSource = Bytes.EMPTY_ARRAY;

        extraSource = new byte[size];
