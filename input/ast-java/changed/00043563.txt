/**
 * Return the flags of the end of the specified
 * markup object, or 0 if it is not attached to this buffer.
 */
public int getSpanFlags(Object what) {
    int count = mSpanCount;
    Object[] spans = mSpans;
    for (int i = count - 1; i >= 0; i--) {
        if (spans[i] == what) {
            return mSpanFlags[i];
        }
    }
    return 0;
