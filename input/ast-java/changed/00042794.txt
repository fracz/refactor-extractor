private void saveCachedToken(UserAccounts accounts, Account account, String callerPkg, byte[] callerSigDigest, String tokenType, String token, long expiryMillis) {
    if (account == null || tokenType == null || callerPkg == null || callerSigDigest == null) {
        return;
    }
    cancelNotification(getSigninRequiredNotificationId(accounts, account), new UserHandle(accounts.userId));
    synchronized (accounts.cacheLock) {
        TokenCache cache = getTokenCacheForAccountLocked(accounts, account);
        if (cache != null) {
            cache.put(token, tokenType, callerPkg, callerSigDigest, expiryMillis);
        }
        return;
    }
