@Override
public ODistributedTxContext registerTxContext(final ODistributedRequestId reqId) {
    ODistributedTxContextImpl ctx = new ODistributedTxContextImpl(this, reqId);
    final ODistributedTxContextImpl prevCtx = activeTxContexts.putIfAbsent(reqId, ctx);
    if (prevCtx != null) {
        // ALREADY EXISTENT
        ODistributedServerLog.debug(this, localNodeName, null, DIRECTION.NONE, "Distributed transaction: repeating request %s in database '%s'", reqId, databaseName);
        ctx = prevCtx;
    } else
        // REGISTERED
        ODistributedServerLog.debug(this, localNodeName, null, DIRECTION.NONE, "Distributed transaction: registered request %s in database '%s'", reqId, databaseName);
    return ctx;
