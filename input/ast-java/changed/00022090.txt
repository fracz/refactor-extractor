public static IElementType parseReferenceList(PsiBuilder builder, final IElementType startElement, final GrReferenceListElementType<?> clauseType) {
    PsiBuilder.Marker isMarker = builder.mark();
    if (!ParserUtils.getToken(builder, startElement)) {
        isMarker.rollbackTo();
        return NONE;
    }
    ParserUtils.getToken(builder, mNLS);
    if (parseReferenceElement(builder) == fail) {
        isMarker.rollbackTo();
        return WRONGWAY;
    }
    while (ParserUtils.getToken(builder, mCOMMA)) {
        ParserUtils.getToken(builder, mNLS);
        if (parseReferenceElement(builder) == fail) {
            isMarker.rollbackTo();
            return WRONGWAY;
        }
    }
    ParserUtils.getToken(builder, mNLS);
    isMarker.done(clauseType);
    return clauseType;
