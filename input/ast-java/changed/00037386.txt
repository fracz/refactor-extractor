private void notifyUnavailableToClient() {
    if (mCurrentViewState == null) {
        // TODO(b/33197203): temporary sanity check; should never happen
        Slog.w(TAG, "notifyUnavailable(): mCurrentViewState is null");
        return;
    }
    if (!mHasCallback)
        return;
    try {
        mClient.notifyNoFillUi(mWindowToken, mCurrentViewState.mId);
    } catch (RemoteException e) {
        Slog.e(TAG, "Error notifying client no fill UI: windowToken=" + mWindowToken + " id=" + mCurrentViewState.mId, e);
    }
