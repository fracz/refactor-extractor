private void abort(NettyResponseFuture<?> future, Throwable t) {
    if (config.getMaxTotalConnections() != -1) {
        maxConnections.decrementAndGet();
    }
    if (log.isDebugEnabled()) {
        log.debug("abording Future {}", future);
        log.debug(t.getMessage(), t);
    }
    future.abort(t);
