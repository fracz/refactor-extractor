@Override
public boolean mkdirs(TachyonURI path, boolean recursive) throws InvalidPathException, IOException, FileAlreadyExistException {
    FileSystemMasterClient masterClient = mContext.acquireMasterClient();
    try {
        // TODO: Change this RPC's arguments
        boolean result = masterClient.createDirectory(path.getPath(), true);
        if (result) {
            LOG.info("Created directory " + path.getPath());
        }
        return result;
    } finally {
        mContext.releaseMasterClient(masterClient);
    }
