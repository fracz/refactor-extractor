@Override
protected int doReadBytes(ChannelBuffer buf) throws Exception {
    if (socket.isClosed()) {
        return -1;
    }
    int b;
    try {
        b = is.read();
        if (b < 0) {
            return -1;
        }
        is.unread(b);
        int available = is.available();
        buf.ensureWritableBytes(available);
        return buf.writeBytes(is, available);
    } catch (SocketTimeoutException e) {
        return 0;
    }
