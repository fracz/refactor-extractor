/**
 * Unlock the state directory for the given {@link TaskId}
 * @param taskId
 * @throws IOException
 */
void unlock(final TaskId taskId) throws IOException {
    final FileLock lock = locks.remove(taskId);
    if (lock != null) {
        lock.release();
        final FileChannel fileChannel = channels.remove(taskId);
        if (fileChannel != null) {
            fileChannel.close();
        }
    }
