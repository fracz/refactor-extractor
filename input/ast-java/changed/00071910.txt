@Override
protected Node generate(int budget) {
    String name = null;
    // allow variable shadowing
    if (scopeManager.hasNonLocals() && random.nextDouble() < config.optJSONObject("identifier").optDouble("shadow")) {
        name = scopeManager.getRandomSymbol(true).name;
    }
    if (name == null) {
        name = "x_" + snGenerator.getNextNumber();
    }
    scopeManager.addSymbol(new Symbol(name));
    return Node.newString(Token.NAME, name);
