    publication.onStatusMessage(INITIAL_TERM_ID, 0, ALIGNED_FRAME_LENGTH, rcvAddress);

    sender.doWork();
    // should send now
    assertThat(receivedFrames.size(), is(1));
    // skip data frame
    receivedFrames.remove();
