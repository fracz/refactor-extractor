/**
 * Reset all of the state associated with being bound to an input method.
 */
void clearBindingLocked() {
    clearConnectionLocked();
    flushPendingEventsLocked();
    mBindSequence = -1;
    mCurId = null;
    mCurMethod = null;
    if (mCurSender != null) {
        mCurSender.dispose();
        mCurSender = null;
    }
    if (mCurChannel != null) {
        mCurChannel.dispose();
        mCurChannel = null;
    }
