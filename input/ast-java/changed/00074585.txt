List<DaemonStatus> getIdle() {
    List<DaemonStatus> out = new LinkedList<DaemonStatus>();
    List<DaemonStatus> all = getAll();
    for (DaemonStatus d : all) {
        if (d.isIdle()) {
            out.add(d);
        }
    }
    return out;
