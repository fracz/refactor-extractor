/**
 * Returns the record size.
 *
 * @throws IOException
 */
public int getRecordSize(final long iPosition) throws IOException {
    try {
        acquireSharedLock();
        final long[] pos = getRelativePosition(iPosition);
        final OFile file = files[(int) pos[0]];
        return file.readInt(pos[1]);
    } finally {
        releaseSharedLock();
    }
