public void startWebServer() {
    try {
        mServer.getConnectors()[0].close();
        mServer.getConnectors()[0].open();
        mServer.start();
        if (mAddress.getPort() == 0) {
            mAddress = new InetSocketAddress(mAddress.getHostName(), mServer.getConnectors()[0].getLocalPort());
        }
        LOG.info(mServerName + " started @ " + mAddress);
    } catch (Exception e) {
        throw Throwables.propagate(e);
    }
