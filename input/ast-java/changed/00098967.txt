public Set<String> getClusterNames() {
    final boolean locked = acquireSharedLock();
    try {
        Set<String> result = new HashSet<String>();
        for (int i = 0; i < clusters.size(); ++i) result.add(getClusterById(i).getName());
        return result;
    } finally {
        releaseSharedLock(locked);
    }
