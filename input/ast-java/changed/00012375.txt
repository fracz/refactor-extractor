void checkPossibleLooseOfPrecision(@NotNull PsiType targetType, @NotNull GrExpression expression, @NotNull PsiElement toHighlight) {
    PsiType actualType = expression.getType();
    if (actualType == null)
        return;
    if (isPossibleLooseOfPrecision(targetType, actualType, expression))
        registerError(toHighlight, GroovyBundle.message("loss.of.precision", actualType.getPresentableText(), targetType.getPresentableText()), new LocalQuickFix[] { new GrCastFix(targetType, expression, false) }, ProblemHighlightType.GENERIC_ERROR);
