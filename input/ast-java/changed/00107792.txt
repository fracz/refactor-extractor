@Override
public SQLSelect select() {
    SQLServerSelect result = new SQLServerSelect();
    withSubquery(result);
    result.setQuery(query());
    result.setOrderBy(getExprParser().parseOrderBy());
    if (getLexer().equalToken(Token.FOR)) {
        parseFor();
    }
    if (getLexer().identifierEquals("OFFSET")) {
        parseOffset(result);
    }
    return result;
