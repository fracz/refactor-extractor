/**
 * Returns a list of clang flags used for all link and compile actions executed through clang.
 */
private static List<String> commonLinkAndCompileFlagsForClang(ObjcProvider provider, ObjcConfiguration objcConfiguration, AppleConfiguration appleConfiguration) {
    ImmutableList.Builder<String> builder = new ImmutableList.Builder<>();
    Platform platform = appleConfiguration.getPlatform(PlatformType.IOS);
    if (platform == Platform.IOS_SIMULATOR) {
        builder.add("-mios-simulator-version-min=" + objcConfiguration.getMinimumOs());
    } else {
        builder.add("-miphoneos-version-min=" + objcConfiguration.getMinimumOs());
    }
    if (objcConfiguration.generateDebugSymbols() || objcConfiguration.generateDsym()) {
        builder.add("-g");
    }
    return builder.add("-arch", appleConfiguration.getSingleArchitecture(PlatformType.IOS)).add("-isysroot", AppleToolchain.sdkDir()).addAll(commonFrameworkFlags(provider, appleConfiguration)).build();
