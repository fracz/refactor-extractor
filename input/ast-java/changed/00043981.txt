protected boolean shouldShowNotificationIcon(NotificationData.Entry entry, NotificationData notificationData, boolean showAmbient) {
    if (notificationData.isAmbient(entry.key) && !showAmbient && !NotificationData.showNotificationEvenIfUnprovisioned(entry.notification)) {
        return false;
    }
    if (!StatusBar.isTopLevelChild(entry)) {
        return false;
    }
    if (entry.row.getVisibility() == View.GONE) {
        return false;
    }
    return true;
