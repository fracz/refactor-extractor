    NamedNodeMap attrs = this.node.getAttributes();
    Node attr = attrs.getNamedItem(name.convertToString().asJavaString());
    if (attr == null) {
        return context.getRuntime().getNil();

    return constructNode(context.getRuntime(), attr);
