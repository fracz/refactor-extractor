    CustomCommandLine.Builder commandLine = CustomCommandLine.builder().addPath(zipOutput.getExecPath()).add("--platform").add(appleConfiguration.getBundlingPlatform().getLowerCaseNameInPlist()).addExecPath("--output-partial-info-plist", partialInfoPlist).add("--minimum-deployment-target").add(bundling.getMinimumOsVersion().toString());

        commandLine.add("--target-device").add(targetDeviceFamily.name().toLowerCase(Locale.US));
