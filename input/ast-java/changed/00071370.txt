public void testMisplacedTypeAnnotation() throws Exception {
    reportMisplacedTypeAnnotations = false;
    assertNoWarningOrError("var x = /** @type {string} */ y;");
    reportMisplacedTypeAnnotations = true;
    String message = "Type annotations are not allowed here. " + "Are you missing parentheses?";
    assertWarning("var x = /** @type {string} */ y;", RhinoErrorReporter.MISPLACED_TYPE_ANNOTATION, message);
    JSError error = assertWarning("var x = /** @type {string} */ y;", RhinoErrorReporter.MISPLACED_TYPE_ANNOTATION, message);
    assertEquals(1, error.getLineNumber());
    // Rhino uses the "beginning" of the line where the comma appears,
    // for some odd reason.
    assertEquals(0, error.getCharno());
