    if (t instanceof OLockException && t.getCause() instanceof ODatabaseException)

        current = t.getCause();

        current = t;
    while (current != null) {

        channel.writeString(current.getClass().getName());

        current = current.getCause();
    }
    channel.writeByte((byte) 0);
    channel.clearInput();
