@Override
public boolean equals(Object o) {
    if (this == o)
        return true;
    if (o == null || getClass() != o.getClass())
        return false;
    if (!super.equals(o))
        return false;

    if (supportsPasswordBasedAuthentication != that.supportsPasswordBasedAuthentication)
        return false;
    if (supportsWebBasedAuthentication != that.supportsWebBasedAuthentication)
        return false;
    if (displayName != null ? !displayName.equals(that.displayName) : that.displayName != null)
        return false;
    if (displayImageURL != null ? !displayImageURL.equals(that.displayImageURL) : that.displayImageURL != null)
