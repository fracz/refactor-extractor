public synchronized void delete(int fileId, boolean recursive) {
    if (INACTIVE) {
        return;
    }
    try {
        DOS.writeLong(++mTransactionId);
        DOS.writeByte(OP_DELETE);
        DOS.writeInt(fileId);
        DOS.writeBoolean(recursive);
    } catch (IOException e) {
        CommonUtils.runtimeException(e);
    }
