/**
 * Form the eigenvectors of a real nonsymmetric matrix by back transforming
 * those of the corresponding balanced matrix determined by balance.
 */
private static void balbak(double[][] V, double[] scale) {
    int n = V.length;
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            V[i][j] *= scale[i];
        }
    }
