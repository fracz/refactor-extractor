void autoFillApp(Dataset dataset) {
    synchronized (mLock) {
        try {
            if (DEBUG)
                Slog.d(TAG, "autoFillApp(): the buck is on the app: " + dataset);
            mAppCallback.autoFill(dataset);
        } catch (RemoteException e) {
            Slog.w(TAG, "Error auto-filling activity: " + e);
        }
    }
