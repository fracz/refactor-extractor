@Override
protected BackupServer load(KernelData kernel) {
    Integer port = parsePort(kernel);
    if (port != null) {
        TheBackupInterface backup = new BackupImpl(kernel.graphDatabase());
        BackupServer server = new BackupServer(backup, port.intValue(), (String) kernel.getConfig().getParams().get("store_dir"));
        return server;
    }
    return null;
