private void truncateLogicalLog(int size) throws IOException {
    char active = '1';
    FileChannel af = new RandomAccessFile(file("nioneo_logical.log.active"), "r").getChannel();
    ByteBuffer buffer = ByteBuffer.allocate(1024);
    af.read(buffer);
    af.close();
    buffer.flip();
    active = buffer.asCharBuffer().get();
    buffer.clear();
    FileChannel fileChannel = new RandomAccessFile(file("nioneo_logical.log." + active), "rw").getChannel();
    // System.out.println( fileChannel.size() );
    if (fileChannel.size() > size) {
        fileChannel.truncate(size);
    } else {
        fileChannel.position(size);
        ByteBuffer buf = ByteBuffer.allocate(1);
        buf.put((byte) 0).flip();
        fileChannel.write(buf);
    }
    fileChannel.force(false);
    fileChannel.close();
