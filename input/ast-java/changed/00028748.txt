public DataSet copy() {
    ArrayList<Entry> yVals = new ArrayList<Entry>();
    for (int i = 0; i < mYVals.size(); i++) {
        yVals.add(mYVals.get(i).copy());
    }
    DataSet copied = new DataSet(yVals, mLabel);
    return copied;
