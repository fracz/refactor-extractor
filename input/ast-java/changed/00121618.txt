public void commandAction(Command command, Displayable displayable) {
    if (command.getCommandType() == Command.OK) {
        try {
            platformRequest(uri);
        } catch (ConnectionNotFoundException cnfe) {
            showError(cnfe);
        } finally {
            stop();
        }
    } else {
        // cancel
        Display.getDisplay(ZXingMIDlet.this).setCurrent(canvas);
    }
