@Nullable
private FileType getFileTypeByName(String name) {
    Iterator<FileType> itr = myFileTypes.iterator();
    while (itr.hasNext()) {
        FileType fileType = itr.next();
        if (fileType.getName().equals(name))
            return fileType;
    }
    return null;
