private int replay() {
    try {
        final int polled = cursor.controlledPoll(this, REPLAY_SEND_BATCH_SIZE);
        if (cursor.isDone()) {
            lingerSinceMs = epochClock.time();
            state = State.LINGER;
        }
        return polled;
    } catch (final Exception ex) {
        return closeOnError(ex, "Cursor read failed");
    }
