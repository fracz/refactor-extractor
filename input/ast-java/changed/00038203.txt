static String drawStateToString(int state) {
    switch(state) {
        case NO_SURFACE:
            return "NO_SURFACE";
        case DRAW_PENDING:
            return "DRAW_PENDING";
        case COMMIT_DRAW_PENDING:
            return "COMMIT_DRAW_PENDING";
        case READY_TO_SHOW:
            return "READY_TO_SHOW";
        case HAS_DRAWN:
            return "HAS_DRAWN";
        default:
            return Integer.toString(state);
    }
