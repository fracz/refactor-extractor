private void doRemoveRepository() {
    if (!canEdit())
        return;
    try {
        myManager.remove(getSelectedIndexInfo());
        reset();
    } catch (MavenIndexException e) {
        Messages.showErrorDialog(e.getMessage(), getDisplayName());
    }
    setModified(true);
