protected Collection<File> getFilesToRemove(CvsContext context) {
    File[] files = context.getSelectedIOFiles();
    ArrayList<File> result = new ArrayList<File>();
    if (files == null)
        return result;
    for (int i = 0; i < files.length; i++) {
        result.add(files[i]);
    }
    return result;
