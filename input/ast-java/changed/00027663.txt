    String moduleName = getExternalModuleName(descriptor);
    if (moduleName == null)
        return null;
    if (LibrarySourcesConfig.UNKNOWN_EXTERNAL_MODULE_NAME.equals(moduleName))

    }
    return JsAstUtils.pureFqn(moduleId, null);
