void cancelBroadcasts(Provider p) {
    if (p.broadcast != null) {
        mAlarmManager.cancel(p.broadcast);
        long token = Binder.clearCallingIdentity();
        try {
            p.broadcast.cancel();
        } finally {
            Binder.restoreCallingIdentity(token);
        }
        p.broadcast = null;
    }
