 * targets.

 * that was constructed with {@code computeBackwardGraph} {@code false}.
 */

    if (computeBackwardGraph) {

        throw new UnsupportedOperationException("Cannot call " + "getCallsitesPossiblyTargetingFunction() on a Function " + "from a backward-forward CallGraph");
    }
