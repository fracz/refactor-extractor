private static boolean firstOfSeveralNNP(Tree parent, Tree t) {
    boolean firstIsT = false;
    int numNNP = 0;
    for (Tree kid : parent.children()) {
        if (kid.value().startsWith("NNP")) {
            if (t.equals(kid) && numNNP == 0) {
                firstIsT = true;
            }
            numNNP++;
        }
    }
    return numNNP > 1 && firstIsT;
