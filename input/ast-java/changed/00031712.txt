public void writeTabularData(TabularData tabularData) {
    if (tabularData == null) {
        write("null");
        return;
    }
    int entryIndex = 0;
    write('[');
    for (Object item : tabularData.values()) {
        if (entryIndex != 0) {
            write(',');
        }
        CompositeData row = (CompositeData) item;
        writeCompositeData(row);
        entryIndex++;
    }
    write(']');
