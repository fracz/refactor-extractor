@LifecycleStart
public void start() throws Exception {
    log.info("WorkerCuratorCoordinator good to go sir. Server[%s]", worker.getHost());
    synchronized (lock) {
        if (started) {
            return;
        }
        makePathIfNotExisting(getTaskPathForWorker(), CreateMode.PERSISTENT, ImmutableMap.of("created", new DateTime().toString()));
        makePathIfNotExisting(getStatusPathForWorker(), CreateMode.PERSISTENT, ImmutableMap.of("created", new DateTime().toString()));
        announcer.start();
        announcer.announce(getAnnouncementsPathForWorker(), jsonMapper.writeValueAsBytes(worker), false);
        started = true;
    }
