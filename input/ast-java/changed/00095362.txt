protected ODistributedRequest readRequest() throws InterruptedException {
    // GET FROM DISTRIBUTED QUEUE. IF EMPTY WAIT FOR A MESSAGE
    ODistributedRequest req = nextMessage();
    while (distributed.waitForMessageId.get() > -1) {
        if (req != null) {
            if (req.getId() >= distributed.waitForMessageId.get()) {
                // ARRIVED, RESET IT
                ODistributedServerLog.debug(this, manager.getLocalNodeName(), req.getSenderNodeName(), DIRECTION.IN, "reached waited request %d on request=%s sourceNode=%s", distributed.waitForMessageId.get(), req, req.getSenderNodeName());
                distributed.waitForMessageId.set(-1);
                break;
            } else {
                // SKIP IT
                ODistributedServerLog.debug(this, manager.getLocalNodeName(), req.getSenderNodeName(), DIRECTION.IN, "discarded request %d because waiting for %d request=%s sourceNode=%s", req.getId(), distributed.waitForMessageId.get(), req, req.getSenderNodeName());
                // READ THE NEXT ONE
                req = nextMessage();
            }
        }
    }
    if (ODistributedServerLog.isDebugEnabled())
        ODistributedServerLog.debug(this, manager.getLocalNodeName(), req.getSenderNodeName(), DIRECTION.IN, "processing request=%s sourceNode=%s", req, req.getSenderNodeName());
    return req;
