/**
 * Ask a given process to GC right now.
 */
final void performAppGcLocked(ProcessRecord app) {
    try {
        app.lastRequestedGc = SystemClock.uptimeMillis();
        if (app.thread != null) {
            app.thread.processInBackground();
        }
    } catch (Exception e) {
    // whatever.
    }
