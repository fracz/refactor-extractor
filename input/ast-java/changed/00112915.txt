public void jsonPojoMono() throws Exception {
    HttpUrl baseUrl = server.url("/pojo");
    this.server.enqueue(new MockResponse().setHeader("Content-Type", "application/json").setBody("{\"bar\":\"barbar\",\"foo\":\"foofoo\"}"));
    ClientRequest<Void> request = ClientRequest.GET(baseUrl.toString()).accept(MediaType.APPLICATION_JSON).build();
    Mono<Pojo> result = this.webClient.exchange(request).then(response -> response.body(toMono(Pojo.class)));
    TestSubscriber.subscribe(result).awaitAndAssertNextValuesWith(p -> assertEquals("barbar", p.getBar())).assertComplete();
    RecordedRequest recordedRequest = server.takeRequest();
    assertEquals(1, server.getRequestCount());
    assertEquals("/pojo", recordedRequest.getPath());
    assertEquals("application/json", recordedRequest.getHeader(HttpHeaders.ACCEPT));
