public String getDefaultFormat() {
    if (isNumeric()) {
        if (myDtypeKind.equals("f")) {
            return "\'%.5f\'";
        }
        if (myDtypeKind.equals("i") || myDtypeKind.equals("u")) {
            return "\'%d\'";
        }
        if (myDtypeKind.equals("b") || myDtypeKind.equals("c")) {
            myComponent.getFormatTextField().setEditable(false);
            return "\'%s\'";
        }
    }
    return "\'%s\'";
