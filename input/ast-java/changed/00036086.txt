public int scanFieldInt(char[] fieldName) {
    matchStat = UNKOWN;
    if (!charArrayCompare(text, bp, fieldName)) {
        matchStat = NOT_MATCH_NAME;
        return 0;
    }
    int index = bp + fieldName.length;
    char ch = charAt(index++);
    int value;
    if (ch >= '0' && ch <= '9') {
        value = digits[ch];
        for (; ; ) {
            ch = charAt(index++);
            if (ch >= '0' && ch <= '9') {
                value = value * 10 + digits[ch];
            } else if (ch == '.') {
                matchStat = NOT_MATCH;
                return 0;
            } else {
                bp = index - 1;
                break;
            }
        }
        if (value < 0) {
            matchStat = NOT_MATCH;
            return 0;
        }
    } else {
        matchStat = NOT_MATCH;
        return 0;
    }
    if (ch == ',') {
        ch = charAt(++bp);
        matchStat = VALUE;
        token = JSONToken.COMMA;
        return value;
    }
    if (ch == '}') {
        ch = charAt(++bp);
        if (ch == ',') {
            token = JSONToken.COMMA;
            this.ch = charAt(++bp);
        } else if (ch == ']') {
            token = JSONToken.RBRACKET;
            this.ch = charAt(++bp);
        } else if (ch == '}') {
            token = JSONToken.RBRACE;
            this.ch = charAt(++bp);
        } else if (ch == EOI) {
            token = JSONToken.EOF;
        } else {
            matchStat = NOT_MATCH;
            return 0;
        }
        matchStat = END;
    }
    return value;
