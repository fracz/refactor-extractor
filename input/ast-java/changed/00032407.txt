    if (x == null) {
        return;
    }
    if (Boolean.TRUE == getConditionValue(visitor, x, visitor.getConfig().isSelectHavingAlwayTrueCheck())) {
        if (!isSimpleConstExpr(x)) {
            addViolation(visitor, ErrorCode.ALWAYS_TRUE, "having alway true condition not allow", x);
        }
    }
