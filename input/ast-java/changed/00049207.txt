private void broadcastAll(Message m) {
    synchronized (connectedSockets) {
        for (NotebookSocket conn : connectedSockets) {
            try {
                conn.send(serializeMessage(m));
            } catch (IOException e) {
                LOG.error("socket error", e);
            }
        }
    }
