public void encodeMultiLineString() {
    Flux<String> source = Flux.just("foo\nbar", "foo\nbaz");
    MockServerHttpResponse outputMessage = new MockServerHttpResponse();
    messageWriter.write(source, ResolvableType.forClass(String.class), new MediaType("text", "event-stream"), outputMessage, Collections.emptyMap());
    Publisher<Publisher<DataBuffer>> result = outputMessage.getBodyWithFlush();
    TestSubscriber.subscribe(result).assertNoError().assertValuesWith(publisher -> {
        TestSubscriber.subscribe(publisher).assertNoError().assertValuesWith(stringConsumer("data:foo\ndata:bar\n"), stringConsumer("\n"));
    }, publisher -> {

    });
