public void start() {
    if (requests.isEmpty() && transfers.isEmpty()) {
        logger.debug("Session does not have any tasks.");
        closeSession(State.COMPLETE);
        return;
    }
    streamExecutor.execute(new Runnable() {

        public void run() {
            try {
                handler.initiate();
                onInitializationComplete();
            } catch (IOException e) {
                onError(e);
            }
        }
    });
