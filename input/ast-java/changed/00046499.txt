private void initResourcesMap() {
    if (resources == null) {
        resources = new HashMap<>();
        Map<String, ResourceInformation> types = ResourceUtils.getResourceTypes();
        if (types == null) {
            throw new YarnRuntimeException("Got null return value from ResourceUtils.getResourceTypes()");
        }
        for (Map.Entry<String, ResourceInformation> entry : types.entrySet()) {
            resources.put(entry.getKey(), ResourceInformation.newInstance(entry.getValue()));
        }
    }
