    PsiBuilder.Marker attribute = mark();
    PsiBuilder.Marker typeReference = mark();

    typeReference.done(TYPE_REFERENCE);
    parseTypeArgumentList(-1);

        myExpressionParsing.parseValueArgumentList();
    }
    attribute.done(ANNOTATION_ENTRY);
