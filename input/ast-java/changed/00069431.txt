        @Override
        protected ComponentLayout onCreateLayout(ComponentContext c) {
            return Column.create(c).alignItems(YogaAlign.FLEX_START).widthPx(50).heightPx(50).child(TestSizeDependentComponent.create(c).setFixSizes(true).withLayout().paddingPercent(YogaEdge.ALL, 10).backgroundColor(0xFFFF0000)).build();
        }
    };
    InternalNode root = LayoutState.createAndMeasureTreeForComponent(new ComponentContext(RuntimeEnvironment.application), component, SizeSpec.makeSizeSpec(0, SizeSpec.UNSPECIFIED), SizeSpec.makeSizeSpec(0, SizeSpec.UNSPECIFIED));
