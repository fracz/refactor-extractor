@Override
public Integer valueFromObject(Object object, float scale) throws JSONException {
    JSONArray colorArray = (JSONArray) object;
    if (colorArray.length() == 4) {
        boolean shouldUse255 = true;
        for (int i = 0; i < colorArray.length(); i++) {
            double colorChannel = colorArray.getDouble(i);
            if (colorChannel > 1f) {
                shouldUse255 = false;
            }
        }
        float multiplier = shouldUse255 ? 255f : 1f;
        return Color.argb((int) (colorArray.getDouble(3) * multiplier), (int) (colorArray.getDouble(0) * multiplier), (int) (colorArray.getDouble(1) * multiplier), (int) (colorArray.getDouble(2) * multiplier));
    }
    return Color.BLACK;
