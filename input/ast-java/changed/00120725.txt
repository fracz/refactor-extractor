public static boolean setField(Object obj, String name, Object value) {
    try {
        Field field = getField(obj.getClass(), name);
        if (field == null) {
            return false;
        }
        field.setAccessible(true);
        field.set(obj, value);
        return true;
    } catch (Exception e) {
        e.printStackTrace();
    }
    return false;
