@Override
public DocIdSet getDocIdSet(IndexReader reader) throws IOException {
    FixedBitSet result = null;
    TermDocs td = reader.termDocs();
    try {
        td.seek(term);
        if (td.next()) {
            result = new FixedBitSet(reader.maxDoc());
            result.set(td.doc());
            while (td.next()) {
                result.set(td.doc());
            }
        }
    } finally {
        td.close();
    }
    return result;
