/**
 * Perform calculation over this iterable.
 */
public final <T> T reduce(@Nullable T first, @NotNull PairFunction<T, E, T> function) {
    T cur = first;
    for (E e : this) {
        cur = function.fun(cur, e);
    }
    return cur;
