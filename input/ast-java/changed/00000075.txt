@Override
public void start() throws Exception {
    SynchronousQueue<Runnable> synchronousQueue = new SynchronousQueue<>();
    mThreadPool = new ThreadPoolExecutor(mMinNumberOfThreads, mMaxNumberOfThreads, STOP_TIMEOUT_MS, TimeUnit.MILLISECONDS, synchronousQueue);
    try {
        mServerSocket = new ServerSocket(mPort);
    } catch (IOException e) {
        throw new RuntimeException("Failed to bind to port {}.", e);
    }
    mStopped = false;
    while (!mStopped) {
        Socket client;
        try {
            client = mServerSocket.accept();
        } catch (IOException e) {
            throw new RuntimeException("I/O error occured while waiting for connection.", e);
        }
        InetAddress inetAddress = client.getInetAddress();
        AlluxioLog4jSocketNode clientSocketNode = new AlluxioLog4jSocketNode(this, client);
        RetryPolicy retryPolicy = new ExponentialBackoffRetry(BASE_SLEEP_TIME_MS, MAX_SLEEP_TIME_MS, MAX_NUM_RETRY);
        while (true) {
            try {
                mThreadPool.execute(clientSocketNode);
                break;
            } catch (RejectedExecutionException e) {
                if (!retryPolicy.attemptRetry()) {
                    LOG.warn("Connection with {} has been rejected by ExecutorService {} times" + "till timedout, reason: {}", inetAddress.getHostAddress(), retryPolicy.getRetryCount(), e);
                    // an exception. The system admin should increase the thread pool size.
                    throw new RuntimeException("Increase the number of worker threads in the thread pool", e);
                }
            } catch (Error | Exception e) {
                LOG.error("ExecutorService threw error: ", e);
                throw e;
            }
        }
    }
