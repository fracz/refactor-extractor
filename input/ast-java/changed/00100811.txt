    Phi[] distFrame = (Phi[]) dist._ls_forward_frame;

        distFrame = new Phi[localSize];
        dist._ls_forward_frame = distFrame;
    }
    for (int i = 0; i < localSize; i++) {
        Phi srcPhi = frame[i];
        if (srcPhi != null) {
            Phi distPhi = distFrame[i];
            if (distPhi == null) {
                if (!dist._cfg_visited) {
                    distPhi = new Phi();
                    phis.add(distPhi);

    }
