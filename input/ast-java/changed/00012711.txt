private <T> T doIndexTask(IndexTask<T> task, T defaultValue) {
    assert Thread.holdsLock(this);
    if (!isBroken) {
        try {
            return task.doTask();
        } catch (Exception e1) {
            MavenLog.warn(e1);
            cleanupBrokenData();
            try {
                open();
            } catch (MavenIndexException e2) {
                MavenLog.warn(e2);
            }
        }
    }
    isBroken = true;
    myListener.indexIsBroken(this);
    return defaultValue;
