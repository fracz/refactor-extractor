public DruidDataSource getDruidDataSourceById(Integer identity) {
    init();
    if (identity == null) {
        return null;
    }
    for (DruidDataSource datasource : getDruidDataSourceInstances()) {
        if (System.identityHashCode(datasource) == identity) {
            return datasource;
        }
    }
    return null;
