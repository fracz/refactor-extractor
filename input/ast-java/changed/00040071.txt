    // Treat animating scrolls differently; see #computeScroll() for why.
    if (!mScroller.isFinished()) {
        mScrollX = scrollX;
        mScrollY = scrollY;
        if (clampedX) {

        }

        super.scrollTo(scrollX, scrollY);
