public static List<ParseTree> descendants(ParseTree t) {
    List<ParseTree> nodes = new ArrayList<ParseTree>();
    nodes.add(t);
    int n = t.getChildCount();
    for (int i = 0; i < n; i++) {
        nodes.addAll(descendants(t.getChild(i)));
    }
    return nodes;
