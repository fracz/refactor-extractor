@Override
public void onDaemonConnected() {
    // daemon back on main thread.
    if (mConnectedSignal != null) {
        mConnectedSignal.countDown();
        mConnectedSignal = null;
    } else {
        mFgHandler.post(new Runnable() {

            @Override
            public void run() {
                prepareNativeDaemon();
            }
        });
    }
