@Override
public void subscribe(Subscriber<? super T> s) {
    Iterator<? extends T> it;
    try {
        it = source.iterator();
    } catch (Throwable e) {
        s.onSubscribe(EmptySubscription.INSTANCE);
        s.onError(e);
        return;
    }
    s.onSubscribe(new IteratorSourceSubscription<>(it, s));
