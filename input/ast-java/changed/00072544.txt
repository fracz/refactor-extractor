public void testConcatNonemptyEmpty() {
    String[] result = ObjectArrays.concat(new String[] { "a", "b" }, new String[0], String.class);
    assertEquals(String[].class, result.getClass());
    ASSERT.that(result).has().exactly("a", "b").inOrder();
