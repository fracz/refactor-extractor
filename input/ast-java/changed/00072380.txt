@Override
public boolean process(Set<? extends TypeElement> annotations, RoundEnvironment roundEnv) {
    for (Element element : roundEnv.getElementsAnnotatedWith(BugPattern.class)) {
        // 0
        pw.print(element.toString() + "\t");
        BugPattern annotation = element.getAnnotation(BugPattern.class);
        // 1
        pw.print(annotation.name() + "\t");
        // 2
        pw.print(Joiner.on(", ").join(annotation.altNames()) + "\t");
        try {
            // 3
            pw.print(annotation.category() + "\t");
        } catch (EnumConstantNotPresentException e) {
            // TODO(cushon): awful hack to deal with skew between @BugPattern at head and @BugPattern
            // in last release.
            // 3
            pw.print(e.constantName() + "\t");
        }
        // 4
        pw.print(annotation.severity() + "\t");
        // 5
        pw.print(annotation.maturity() + "\t");
        // 6
        pw.print(annotation.suppressibility() + "\t");
        // element. See https://community.oracle.com/thread/1184190.
        try {
            annotation.customSuppressionAnnotation();
            throw new IllegalStateException("Expected a MirroredTypeException");
        } catch (MirroredTypeException e) {
            // 7
            pw.print(e.getTypeMirror().toString() + "\t");
        }
        // 8
        pw.print(annotation.summary() + "\t");
        // 9
        pw.println(annotation.explanation().replace("\n", "\\n"));
    }
    if (roundEnv.processingOver()) {
        // this was the last round, do cleanup
        cleanup();
    }
    return true;
