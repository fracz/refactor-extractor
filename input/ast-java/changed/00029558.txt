@Nullable
public static String getCorrectedMimeType(@Nullable String mimeType) {
    if (mimeType == null)
        return null;
    switch(mimeType) {
        case "image/jpg":
            return MimeTypeMap.getSingleton().hasMimeType(ContentType.IMAGE_JPEG) ? ContentType.IMAGE_JPEG : mimeType;
        default:
            return mimeType;
    }
