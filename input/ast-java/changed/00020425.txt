@Nullable
static Trinity<TextRange, TextRange, TextRange> parseExceptionLine(final String line) {
    int atIndex = line.indexOf(AT_PREFIX);
    if (atIndex < 0)
        return null;
    final int lparenthIndex = line.indexOf('(', atIndex);
    if (lparenthIndex < 0)
        return null;
    final int lastDotIndex = line.lastIndexOf('.', lparenthIndex);
    if (lastDotIndex < 0 || lastDotIndex < atIndex)
        return null;
    final int rparenthIndex = line.indexOf(')', lparenthIndex);
    if (rparenthIndex < 0)
        return null;
    // class, method, link
    return Trinity.create(adjustedRange(line, atIndex + AT_PREFIX.length(), lastDotIndex), adjustedRange(line, lastDotIndex + 1, lparenthIndex), new TextRange(lparenthIndex, rparenthIndex));
