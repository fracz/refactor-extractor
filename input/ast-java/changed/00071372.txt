/**
 * Runs the Compiler and calls System.exit() with the exit status of the
 * compiler.
 */
public final void run() {
    int result;
    try {
        result = doRun();
    } catch (AbstractCompilerRunner.FlagUsageException e) {
        System.err.println(e.getMessage());
        result = -1;
    } catch (Throwable t) {
        t.printStackTrace();
        result = -2;
    }
    System.exit(result);
