private void returnViewConnectionInfo(DruidDataSource datasource, HttpServletRequest request, HttpServletResponse response) throws IOException {
    if (datasource == null)
        return;
    String text = IOUtils.readFromResource(RESOURCE_PATH + "/connectionInfo.html");
    text = text.replaceAll("\\{datasourceId\\}", String.valueOf(System.identityHashCode(datasource)));
    response.getWriter().print(text);
