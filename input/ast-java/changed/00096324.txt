public ResultSet getColumns(final String catalog, final String schemaPattern, final String tableNamePattern, final String columnNamePattern) throws SQLException {
    final OrientJdbcStatement stmt = new OrientJdbcStatement(connection);
    final List<ODocument> records = new ArrayList<ODocument>();
    final OClass clazz = database.getMetadata().getSchema().getClass(tableNamePattern);
    if (clazz != null) {
        final OProperty prop = clazz.getProperty(columnNamePattern);
        if (prop != null) {
            records.add((ODocument) new ODocument().field("TABLE_CAT", database.getName()).field("TABLE_SCHEM", database.getName()).field("TABLE_NAME", clazz.getName()).field("COLUMN_NAME", prop.getName()).field("DATA_TYPE", OrientJdbcResultSetMetaData.getSqlType(prop.getType())).field("COLUMN_SIZE", 1).field("NULLABLE", !prop.isNotNull()).field("IS_NULLABLE", prop.isNotNull() ? "NO" : "YES"));
        }
    }
    return new OrientJdbcResultSet(stmt, records, ResultSet.TYPE_FORWARD_ONLY, ResultSet.CONCUR_READ_ONLY, ResultSet.HOLD_CURSORS_OVER_COMMIT);
