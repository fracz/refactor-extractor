@Test(groups = "standalone")
public void testMaxTotalConnections() {
    AsyncHttpClient client = new AsyncHttpClient(new AsyncHttpClientConfig.Builder().setConnectionTimeoutInMs(100).setRequestTimeoutInMs(100).setAllowPoolingConnection(true).setMaximumConnectionsTotal(1).build());
    String url = getTargetUrl();
    int i;
    Exception exception = null;
    for (i = 0; i < 3; i++) {
        try {
            log.info(String.format("%d requesting url [%s]...", i, url));
            Response response = client.prepareGet(url).execute().get();
            log.info(String.format("%d response [%s].", i, response));
        } catch (Exception ex) {
            ex.printStackTrace();
            exception = ex;
        }
    }
    assertNull(exception);
