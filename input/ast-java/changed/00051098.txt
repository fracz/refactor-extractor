public <T> QueryTaskFuture<Void> eval(final QueryEnvironment<T> env, VariableContext<T> context, QueryExpression expression, final List<Argument> args, Callback<T> callback) {
    String rawPattern = getPattern(args);
    final Pattern compiledPattern;
    try {
        compiledPattern = Pattern.compile(rawPattern);
    } catch (PatternSyntaxException e) {
        return env.immediateFailedFuture(new QueryException(expression, String.format("illegal '%s' pattern regexp '%s': %s", getName(), rawPattern, e.getMessage())));
    }
    // Note that Patttern#matcher is thread-safe and so this Predicate can safely be used
    // concurrently.
    final Predicate<T> matchFilter = new Predicate<T>() {

        @Override
        public boolean apply(T target) {
            for (String str : getFilterStrings(env, args, target)) {
                if ((str != null) && compiledPattern.matcher(str).find()) {
                    return true;
                }
            }
            return false;
        }
    };
    return env.eval(Iterables.getLast(args).getExpression(), context, new FilteredCallback<>(callback, matchFilter));
