        for (DiscoveryDruidNode node : nodes.values()) {
            listenerExecutor.submit(() -> {
                try {

                    log.error(ex, "Exception occured in DiscoveryDruidNode.nodeAdded(node=[%s]) in listener [%s].", node, listener);

            });

        nodeListeners.add(listener);
