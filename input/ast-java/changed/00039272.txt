private Callback addCallbackLocked(Callback head, long dueTime, Runnable action, Object token) {
    Callback callback = obtainCallbackLocked(dueTime, action, token);
    if (head == null) {
        return callback;
    }
    Callback entry = head;
    if (dueTime < entry.dueTime) {
        callback.next = entry;
        return callback;
    }
    while (entry.next != null) {
        if (dueTime < entry.next.dueTime) {
            callback.next = entry.next;
            break;
        }
        entry = entry.next;
    }
    entry.next = callback;
    return head;
