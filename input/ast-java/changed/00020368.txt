    int _y = y;
    for (Icon icon : getIcons()) {
        if (icon == null)
            continue;
        switch(myAlignment) {
            case TOP:
                _y = y;
                break;
            case CENTER:
                _y = y + (myHeight - icon.getIconHeight()) / 2;

            case BOTTOM:
                _y = y + (myHeight - icon.getIconHeight());
