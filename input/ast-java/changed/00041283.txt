@Override
public void onScreenTurnedOff() {
    if (DEBUG)
        Log.d(TAG, "screen off");
    mScreenOn = false;
    mForgotPattern = false;
    mHasOverlay = mUpdateMonitor.getPhoneState() != TelephonyManager.CALL_STATE_IDLE || mHasDialog;
    // Emulate activity life-cycle for both lock and unlock screen.
    if (mLockScreen != null) {
        ((KeyguardScreen) mLockScreen).onPause();
    }
    if (mUnlockScreen != null) {
        ((KeyguardScreen) mUnlockScreen).onPause();
    }
    saveWidgetState();
    // When screen is turned off, need to unbind from FaceLock service if using FaceLock
    stopAndUnbindFromFaceLock();
