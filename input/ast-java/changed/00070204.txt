@Override
public boolean onTouchEvent(MotionEvent ev) {
    if (useController && ev.getActionMasked() == MotionEvent.ACTION_DOWN) {
        if (controller.isVisible()) {
            controller.hide();
        } else {
            controller.show();
        }
    }
    return true;
