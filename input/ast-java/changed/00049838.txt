@Override
public Scheduler getScheduler() {
    int maxConcurrency = 10;
    RemoteInterpreterProcess interpreterProcess = getInterpreterProcess();
    if (interpreterProcess == null) {
        return null;
    } else {
        return SchedulerFactory.singleton().createOrGetRemoteScheduler("remoteinterpreter_" + interpreterProcess.hashCode(), getInterpreterProcess(), maxConcurrency);
    }
