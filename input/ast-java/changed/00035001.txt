            exprList(methodInvokeExpr.getParameters(), methodInvokeExpr);

        accept(Token.RPAREN);
        return primaryRest(methodInvokeExpr);

        SQLExpr value = expr();
        if (lexer.token() != Token.IDENTIFIER) {
            throw new ParserException("Syntax error");

        String unit = lexer.stringVal();
