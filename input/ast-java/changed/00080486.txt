@Override
public long getSofar(final int downloadId) {
    long val = 0;
    if (getService() == null) {
        return val;
    }
    try {
        val = getService().getSofar(downloadId);
    } catch (RemoteException e) {
        e.printStackTrace();
    }
    return val;
