private List<PosterThread> startThreads(CountDownLatch latch, int threadCount, int iterations, Object eventToPost) throws InterruptedException {
    List<PosterThread> threads = new ArrayList<PosterThread>(threadCount);
    for (int i = 0; i < threadCount; i++) {
        PosterThread thread = new PosterThread(latch, iterations, eventToPost);
        thread.start();
        threads.add(thread);
    }
    return threads;
