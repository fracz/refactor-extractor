/**
 * Determines whether machine is running or not.
 *
 * @return
 * true for running machine
 */
private boolean isMachineRunning(String machineName) {
    for (MachineDto machine : machines) {
        if (machine.getConfig().getName().equals(machineName) && "ssh".equals(machine.getConfig().getType()) && machine.getStatus() == MachineStatus.RUNNING) {
            return true;
        }
    }
    return false;
