public Set<String> getRIs() {
    Set<String> ris = new HashSet<String>();
    for (LibraryVersionInfo info : getLibraries().keySet()) {
        String ri = info.getRI();
        if (!StringUtil.isEmptyOrSpaces(ri)) {
            ris.add(ri);
        } else {
            ris.add(UNKNOWN_RI_NAME);
        }
    }
    return ris;
