    client.prepareIndex("test", "type", "4").setSource("field", "value4", "color", "blue").execute().actionGet();
    client.admin().indices().prepareRefresh().execute().actionGet();
    SearchResponse searchResponse = client.prepareSearch("test").setQuery(customFiltersScoreQuery(fuzzyQuery("field", "value")).add(FilterBuilders.idsFilter("type").addIds("1"), 3)).execute().actionGet();
