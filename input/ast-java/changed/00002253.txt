public static String urlEncode(String x) {
    if (x == null || x.isEmpty()) {
        return "";
    }
    try {
        return URLEncoder.encode(x, "UTF-8");
    } catch (UnsupportedEncodingException e) {
        throw new RuntimeException("Unsupported Encoding");
    }
