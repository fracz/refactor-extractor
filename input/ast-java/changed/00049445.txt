public void asyncPut(String jsonNote) throws IOException {
    if (StringUtils.isBlank(jsonNote)) {
        LOG.error("Cannot save empty note/string to ZeppelinHub");
        return;
    }
    client.newRequest(zepelinhubUrl).method(HttpMethod.PUT).header(ZEPPELIN_TOKEN_HEADER, token).content(new StringContentProvider(jsonNote, "UTF-8"), "application/json;charset=UTF-8").send(new BufferingResponseListener() {

        @Override
        public void onComplete(Result res) {
            if (!res.isFailed() && res.getResponse().getStatus() == 200) {
                LOG.info("Successfully saved note to ZeppelinHub with {}", res.getResponse().getStatus());
            } else {
                LOG.warn("Failed to save note to ZeppelinHub with HttpStatus {}", res.getResponse().getStatus());
            }
        }

        @Override
        public void onFailure(Response response, Throwable failure) {
            LOG.error("Failed to save note to ZeppelinHub: {}", response.getReason(), failure);
        }
    });
