@Override
public void query() {
    if (getSqlParser().equalAny(DefaultKeyword.SELECT)) {
        getSqlParser().getLexer().nextToken();
        parseDistinct();
        getSqlParser().skipAll(MySQLKeyword.HIGH_PRIORITY, DefaultKeyword.STRAIGHT_JOIN, MySQLKeyword.SQL_SMALL_RESULT, MySQLKeyword.SQL_BIG_RESULT, MySQLKeyword.SQL_BUFFER_RESULT, MySQLKeyword.SQL_CACHE, MySQLKeyword.SQL_NO_CACHE, MySQLKeyword.SQL_CALC_FOUND_ROWS);
        parseSelectList();
        skipToFrom();
    }
    parseFrom();
    parseWhere();
    parseGroupBy();
    getSqlContext().getOrderByContexts().addAll(parseOrderBy(getSqlContext()));
    if (getSqlParser().equalAny(MySQLKeyword.LIMIT)) {
        getSqlContext().setLimitContext(((MySQLParser) getSqlParser()).parseLimit(getParametersIndex()));
    }
    if (getSqlParser().equalAny(DefaultKeyword.PROCEDURE)) {
        throw new SQLParsingUnsupportedException(getSqlParser().getLexer().getCurrentToken().getType());
    }
    queryRest();
