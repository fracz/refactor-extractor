/**
 * Delegates to {@link Map#containsKey}. Returns {@code false} on {@code
 * ClassCastException}
 */
static boolean safeContainsKey(Map<?, ?> map, Object key) {
    try {
        return map.containsKey(key);
    } catch (ClassCastException e) {
        return false;
    }
