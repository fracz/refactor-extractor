        TransactionRepresentationCommitProcess localCommit = new TransactionRepresentationCommitProcess(appender, applier);

        ReplicatedTransactionStateMachine replicatedTxStateMachine = new ReplicatedTransactionStateMachine(localCommit, localSessionPool.getGlobalSession(), currentReplicatedLockState);
        dependencies.satisfyDependencies(replicatedTxStateMachine);

    };
