@Override
protected final boolean authenticateUsernamePasswordInternal(final UsernamePasswordCredential credentials) throws AuthenticationException {
    final String username = credentials.getUsername();
    final String password = credentials.getPassword();
    try {
        final Connection c = this.getDataSource().getConnection(username, password);
        DataSourceUtils.releaseConnection(c, this.getDataSource());
        return true;
    } catch (final SQLException e) {
        return false;
    }
