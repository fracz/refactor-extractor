protected Texture wrapBackTexture(Texture texture) {
    if (texture == null) {
        texture = new Texture(pg);
        texture.init(pg.width, pg.height, glColorTex.get(backTex), TEXTURE_2D, RGBA, fboWidth, fboHeight, NEAREST, NEAREST, CLAMP_TO_EDGE, CLAMP_TO_EDGE);
        texture.invertedY(true);
        texture.colorBuffer(true);
        pg.setCache(pg, texture);
    } else {
        texture.glName = glColorTex.get(backTex);
    }
    return texture;
