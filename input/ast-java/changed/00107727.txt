    ResultSet rs;
    try {
        rs = ResultSetFactory.getResultSet(new PreparedStatementExecutor(getShardingConnection().getShardingContext().getExecutorEngine(), routeSQL()).executeQuery(), getSqlContext());

        clearRouteContext();
