@Test
public void testJSONGeneration() throws IOException {
    Map<String, Object> vars = new HashMap<>();
    vars.put("template", "filled");
    TemplateQueryBuilder builder = new TemplateQueryBuilder(new Template("I am a $template string", ScriptType.INLINE, null, null, vars));
    XContentBuilder content = XContentFactory.jsonBuilder();
    content.startObject();
    builder.doXContent(content, null);
    content.endObject();
    content.close();
    assertEquals("{\"template\":{\"inline\":\"I am a $template string\",\"lang\":\"mustache\",\"params\":{\"template\":\"filled\"}}}", content.string());
