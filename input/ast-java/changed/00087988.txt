private Transaction beginTx(TxIdElement txId) {
    try {
        txManager.begin();
        Transaction tx = txManager.getTransaction();
        transactions.put(txId, tx);
        return tx;
    } catch (NotSupportedException e) {
        throw new RuntimeException(e);
    } catch (SystemException e) {
        throw new RuntimeException(e);
