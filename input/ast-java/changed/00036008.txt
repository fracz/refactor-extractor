        if (charAt(index++) != 'r') {
            matchStat = NOT_MATCH;
            return false;

        if (charAt(index++) != 'u') {
            matchStat = NOT_MATCH;
            return false;
        }
        if (charAt(index++) != 'e') {
            matchStat = NOT_MATCH;
            return false;
        }
        bp = index;
        ch = charAt(bp);
        value = true;
    } else if (ch == 'f') {
        if (charAt(index++) != 'a') {
            matchStat = NOT_MATCH;
            return false;
        }
        if (charAt(index++) != 'l') {
            matchStat = NOT_MATCH;
            return false;
        }

            matchStat = NOT_MATCH;
            return false;
        }
        if (charAt(index++) != 'e') {
            matchStat = NOT_MATCH;
            return false;
        }
        bp = index;
        ch = charAt(bp);
        value = false;
    } else {
        matchStat = NOT_MATCH;
        return false;

        if (ch == ',') {
            this.ch = charAt(++bp);
            matchStat = VALUE;
            token = JSONToken.COMMA;
            break;

                    this.ch = charAt(++bp);

                    token = JSONToken.RBRACKET;
                    this.ch = charAt(++bp);

                    token = JSONToken.RBRACE;
                    this.ch = charAt(++bp);
                } else if (ch == EOI) {
                    token = JSONToken.EOF;

                    ch = charAt(++bp);

                    matchStat = NOT_MATCH;
                    return false;

                break;

            matchStat = END;
            break;
