public RemoteViews getAppWidgetViews(int appWidgetId) {
    synchronized (mAppWidgetIds) {
        ensureStateLoadedLocked();
        AppWidgetId id = lookupAppWidgetIdLocked(appWidgetId);
        if (id != null) {
            return cloneIfLocalBinder(id.views);
        }
        return null;
    }
