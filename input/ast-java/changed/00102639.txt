private void loadFile(FsFile dir, XmlLoader[] xmlLoaders) throws Exception {
    if (!dir.exists()) {
        throw new RuntimeException("no such directory " + dir);
    }
    for (FsFile file : dir.listFiles(ENDS_WITH_XML)) {
        loadResourceXmlFile(file, xmlLoaders);
    }
