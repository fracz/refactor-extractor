    INode newNode;
    assert hasWriteLock();
    if (underConstruction) {
        newNode = new INodeFileUnderConstruction(permissions, replication, preferredBlockSize, modificationTime, clientName, clientMachine, null);
    } else {
        newNode = new INodeFile(permissions, BlockInfo.EMPTY_ARRAY, replication, modificationTime, atime, preferredBlockSize);
    }
    try {
        newNode = addNode(path, newNode, UNKNOWN_DISK_SPACE);
    } catch (IOException e) {
        return null;
    }
    return newNode;
