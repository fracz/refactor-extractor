private static void objectArrayAppend(StringBuffer sbuf, Object[] a, Map<Object[], Void> seenMap) {
    sbuf.append('[');
    if (!seenMap.containsKey(a)) {
        seenMap.put(a, null);
        final int len = a.length;
        for (int i = 0; i < len; i++) {
            deeplyAppendParameter(sbuf, a[i], seenMap);
            if (i != len - 1) {
                sbuf.append(", ");
            }
        }
        // allow repeats in siblings
        seenMap.remove(a);
    } else {
        sbuf.append("...");
    }
