private static <M extends Member & AnnotatedElement> void addInjectorsForMembers(Class<?> type, Factory<M> factory, boolean statics, Collection<InjectionPoint> injectionPoints, Errors errors) {
    for (M member : factory.getMembers(type)) {
        if (isStatic(member) != statics) {
            continue;
        }
        Inject inject = member.getAnnotation(Inject.class);
        if (inject == null) {
            continue;
        }
        try {
            injectionPoints.add(factory.create(member));
        } catch (ConfigurationException ignorable) {
            if (!inject.optional()) {
                errors.merge(ignorable.getErrorMessages());
            }
        }
    }
