public List<MessageReference> getMessageReferences() {
    List<MessageReference> messageRefs = new ArrayList<MessageReference>();
    for (int i = 0, len = mAdapter.getCount(); i < len; i++) {
        Cursor cursor = (Cursor) mAdapter.getItem(i);
        MessageReference ref = new MessageReference(cursor.getString(ACCOUNT_UUID_COLUMN), cursor.getString(FOLDER_NAME_COLUMN), cursor.getString(UID_COLUMN), null);
        messageRefs.add(ref);
    }
    return messageRefs;
