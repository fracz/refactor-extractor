private String wrappedText(MavenProjectProblem each) {
    String text = each.getDescription();
    StringBuffer result = new StringBuffer();
    int count = 0;
    for (int i = 0; i < text.length(); i++) {
        char ch = text.charAt(i);
        result.append(ch);
        if (count++ > 80) {
            if (ch == ' ') {
                count = 0;
                result.append("<br>");
            }
        }
    }
    return result.toString();
