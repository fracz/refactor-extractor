private void activateAndRun() {
    assert myRunner == null;
    myMessage.clear();
    final IdeFrameImpl frame = getFrame();
    final Component c = ((WindowManagerEx) WindowManager.getInstance()).getFocusedComponent(frame);
    if (c != null) {
        c.requestFocus();
    } else {
        frame.requestFocus();
    }

        public void run() {
            startWhenFrameActive();
        }
    });
