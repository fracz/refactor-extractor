private void runWhenFullyOpen(final Runnable runnable) {
    // may be called from scheduleImport after project started closing and before it is closed.
    if (!isInitialized())
        return;
    if (isNoBackgroundMode()) {
        runnable.run();
        return;
    }
    final Ref<Runnable> wrapper = new Ref<Runnable>();
    wrapper.set(new Runnable() {

        public void run() {
            if (!StartupManagerEx.getInstanceEx(myProject).postStartupActivityPassed()) {
                mySchedulesQueue.queue(new // should not remove previously schedules tasks
                Update(// should not remove previously schedules tasks
                runnable) {

                    public void run() {
                        wrapper.get().run();
                    }
                });
                return;
            }
            runnable.run();
        }
    });
    MavenUtil.runWhenInitialized(myProject, wrapper.get());
