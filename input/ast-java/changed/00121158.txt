private BrowserLauncher getBrowserLauncher() {
    if (!supported) {
        return null;
    }
    if (launcher == null) {
        try {
            launcher = new BrowserLauncher();
        } catch (BrowserLaunchingInitializingException e) {
            supported = false;
        } catch (UnsupportedOperatingSystemException e) {
            supported = false;
        }
    }
    return launcher;
