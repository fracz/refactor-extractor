 * Return the real syntactic parent of vertex.
 */
public IndexedWord getParent(IndexedWord vertex) {
    List<IndexedWord> path = getPathToRoot(vertex);
    if (path != null && path.size() > 0)
        return path.get(0);
    else
        return null;
