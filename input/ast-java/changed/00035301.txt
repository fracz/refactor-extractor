private Object handleBuilder(Object obj) {
    Method buildMethod = beanInfo.getBuildMethod();
    if (buildMethod == null) {
        return obj;
    }
    Object builtObj;
    try {
        builtObj = buildMethod.invoke(obj);
    } catch (Exception e) {
        throw new JSONException("build object error", e);
    }
    return builtObj;
