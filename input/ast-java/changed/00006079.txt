@NotNull
public static Pair<String, String> getHostPortFromUrl(@NotNull String url) {
    String host = null;
    String port = null;
    int index = url.indexOf("://");
    if (index != -1) {
        url = url.substring(index + 3);
    }
    index = url.indexOf(':');
    if (index != -1) {
        host = url.substring(0, index);
        port = url.substring(index + 1);
    }
    return Pair.create(host, port);
