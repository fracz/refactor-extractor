void assertOldIndexWorks(String index) throws Exception {

    true).put(MergePolicyModule.MERGE_POLICY_TYPE_KEY, // disable merging so no segments will be upgraded
    NoMergePolicyProvider.class).build();
    loadIndex(index, settings);
    logMemoryStats();
    assertBasicSearchWorks();

    assertNewReplicasWork();

    unloadIndex();
