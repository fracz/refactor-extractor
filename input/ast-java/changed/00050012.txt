public void getFolderIdTest() {

    // Ordinary case test
    note.setName("this/is/a/folder/noteName");
    assertEquals("this/is/a/folder", note.getFolderId());
    // Normalize test
    note.setName("/this/is/a/folder/noteName");
    assertEquals("this/is/a/folder", note.getFolderId());

    assertEquals(Folder.ROOT_FOLDER_ID, note.getFolderId());
