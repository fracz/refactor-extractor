/**
 * Releases allocated resources. The stream becomes inoperable after calling
 * this method.
 *
 * @throws IllegalStateException if the stream is busy.
 * @see #isBusy()
 */
public void release() {
    if (isBusy()) {
        throw new IllegalStateException("Busy");
    }
    close();
