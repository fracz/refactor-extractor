@Override
public void handleMessage(Message msg) {
    switch(msg.what) {
        case MSG_SYNC_EXTERNAL_STATS:
            updateExternalStats();
            break;
        case MSG_WRITE_TO_DISK:
            updateExternalStats();
            synchronized (mStats) {
                mStats.writeAsyncLocked();
            }
            break;
    }
