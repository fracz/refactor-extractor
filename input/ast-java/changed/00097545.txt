        return OVariableParser.resolveVariables((String) iContent, OSystemVariableResolver.VAR_BEGIN, OSystemVariableResolver.VAR_END, new OVariableParserListener() {

            @Override
            public String resolve(final String iVariable) {
                final Object val = iContext.field(iVariable);
                if (val != null)
                    return val.toString();
                return null;
            }
        });
    return iContent;
