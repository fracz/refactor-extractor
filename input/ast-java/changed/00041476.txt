@Override
public void resolveLayoutDirection() {
    super.resolveLayoutDirection();
    int count = getChildCount();
    for (int i = 0; i < count; i++) {
        final View child = getChildAt(i);
        if (child.isLayoutDirectionInherited()) {
            child.resolveLayoutDirection();
        }
    }
