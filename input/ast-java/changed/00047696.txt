private void handleSaslHandshakeResponse(SaslHandshakeResponse response) {
    Errors error = response.error();
    switch(error) {
        case NONE:
            break;
        case UNSUPPORTED_SASL_MECHANISM:
            throw new UnsupportedSaslMechanismException(String.format("Client SASL mechanism '%s' not enabled in the server, enabled mechanisms are %s", mechanism, response.enabledMechanisms()));
        case ILLEGAL_SASL_STATE:
            throw new IllegalSaslStateException(String.format("Unexpected handshake request with client mechanism %s, enabled mechanisms are %s", mechanism, response.enabledMechanisms()));
        default:
            throw new AuthenticationException(String.format("Unknown error code %s, client mechanism is %s, enabled mechanisms are %s", response.error(), mechanism, response.enabledMechanisms()));
    }
