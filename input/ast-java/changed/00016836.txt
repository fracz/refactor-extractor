 */
@Nullable
String readCurrentRevision() {
    String[] branchesWithHeads = tryLoadFile(myBranchHeadsFile).trim().split("\n");
    String head = branchesWithHeads[0];
    Matcher matcher = BRANCH_PATTERN.matcher(head);
    if (matcher.matches()) {
        return (matcher.group(1));
    }
    return null;
