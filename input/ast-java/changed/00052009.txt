@Subscribe
public void buildStarting(BuildStartingEvent event) {
    RemoteOptions options = event.getRequest().getOptions(RemoteOptions.class);
    if (SimpleBlobStoreFactory.isRemoteCacheOptions(options) || GrpcActionCache.isRemoteCacheOptions(options)) {
        HashFunction hf = FileSystem.getDigestFunction();
        if (hf != HashFunction.SHA1) {
            env.getBlazeModuleEnvironment().exit(new AbruptExitException("Remote cache/execution requires SHA1 digests, got " + hf + ", run with --host_jvm_args=-Dbazel.DigestFunction=SHA1", ExitCode.COMMAND_LINE_ERROR));
        }
    }
