@Nullable
private static PsiElement resolveInDirectory(final String referencedName, final PsiFile containingFile, final PsiDirectory dir, PyResolveUtil.ResolveProcessor processor) {
    final PsiFile file = dir.findFile(referencedName + PY_SUFFIX);
    if (file != null)
        return file;
    final PsiDirectory subdir = dir.findSubdirectory(referencedName);
    if (subdir != null)
        return subdir;
    else {
        // not a subdir, not a file; could be a name in parent/__init__.py
        final PsiFile initPy = dir.findFile(INIT_PY);
        // don't dive into the file we're in
        if (initPy == containingFile)
            return null;
        if (initPy != null) {
            // should not normally happen
            if (processor == null)
                processor = new PyResolveUtil.ResolveProcessor(referencedName);
            // PyResolveUtil.treeWalkUp(processor, initPy, null, importRef);
            return PyResolveUtil.treeCrawlUp(processor, true, initPy);
        }
    }
    return null;
