/**
 * Close the current packet reader.
 */
private void closePacketReader() {
    if (mPacketReader != null) {
        mPacketReader.close();
    }
    if (mCurrentPacket != null) {
        mCurrentPacket.release();
        mCurrentPacket = null;
    }
    mPacketReader = null;
