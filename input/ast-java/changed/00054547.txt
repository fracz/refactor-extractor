@Override
public void run() {
    plumber = initPlumber();
    try {
        Object metadata = plumber.startJob();
        if (fireDepartment.checkFirehoseV2()) {
            firehoseV2 = initFirehoseV2(metadata);
            runFirehoseV2(firehoseV2);
        } else {
            firehose = initFirehose();
            runFirehose(firehose);
        }
    } catch (RuntimeException e) {
        log.makeAlert(e, "[%s] aborted realtime processing[%s]", e.getClass().getSimpleName(), fireDepartment.getDataSchema().getDataSource()).emit();
        normalExit = false;
        throw Throwables.propagate(e);
    } catch (Error e) {
        log.makeAlert(e, "Error aborted realtime processing[%s]", fireDepartment.getDataSchema().getDataSource()).emit();
        normalExit = false;
        throw e;
    } finally {
        CloseQuietly.close(firehose);
        if (normalExit) {
            plumber.finishJob();
            plumber = null;
            firehose = null;
        }
    }
