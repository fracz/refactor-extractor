private static <T extends DeclarationDescriptor> List<T> getDescriptorsFromSupertypes(ClassDescriptor classDescriptor, Class<T> descriptorClass) {
    Set<T> r = Sets.newHashSet();
    for (JetType supertype : classDescriptor.getTypeConstructor().getSupertypes()) {
        r.addAll(getDescriptorsOfType(supertype.getMemberScope(), descriptorClass));
    }
    return new ArrayList<T>(r);
