        boolean done = false;

        @Override
        public OResultSet syncPull(OCommandContext ctx, int nRecords) throws OTimeoutException {
            OInternalResultSet result = new OInternalResultSet();
            if (!done) {
                for (int i = 0; i < 10; i++) {
