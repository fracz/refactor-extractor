public void projectUpdated(final MavenProjectModel.Node n) {
    ApplicationManager.getApplication().invokeLater(new Runnable() {

        public void run() {
            final PomNode pomNode = fileToNode.get(n.getFile());
            if (pomNode != null) {
                pomNode.onFileUpdate();
            } else {
                projectAdded(n);
            }
            tree.repaint();
        }
    });
