    // Note that unlike in the QR Code sampler, we didn't find the center of
    // modules, but the
    // very corners. So there is no 0.5f here; 0.0f is right.
    GridSampler sampler = GridSampler.getInstance();
    // and the result matrix passed down into sampleGrid() and throughout the reader.
    return // p1ToX
    sampler.sampleGrid(// p1ToX
    image.getBlackMatrix(), // p1ToX
    dimension, // p1ToX
    0.0f, // p1ToY
    0.0f, // p2ToX
    dimension, // p2ToY
    0.0f, // p3ToX
    dimension, // p3ToY
    dimension, // p4ToX
    0.0f, // p4ToY
    dimension, // p1FromX
    topLeft.getX(), // p1FromY
    topLeft.getY(), // p2FromX
    topRight.getX(), // p2FromY
    topRight.getY(), // p3FromX
    bottomRight.getX(), // p3FromY
    bottomRight.getY(), // p4FromX
    bottomLeft.getX(), // p4FromY
    bottomLeft.getY());
