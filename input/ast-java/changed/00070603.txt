// Meet two types, location agnostic
public static JSType meet(JSType lhs, JSType rhs) {
    if (lhs.isTop()) {
        return rhs;
    } else if (rhs.isTop()) {
        return lhs;
    } else if (lhs.isUnknown()) {
        return rhs;
    } else if (rhs.isUnknown()) {
        return lhs;
    }
    int newMask = lhs.getMask() & rhs.getMask();
    String newTypevar;
    if (Objects.equals(lhs.getTypeVar(), rhs.getTypeVar())) {
        newTypevar = lhs.getTypeVar();
    } else {
        newTypevar = null;
        newMask = newMask & ~TYPEVAR_MASK;
    }
    return meetEnums(newMask, lhs.getMask() | rhs.getMask(), null, ObjectType.meetSets(lhs.getObjs(), rhs.getObjs()), newTypevar, lhs.getObjs(), rhs.getObjs(), lhs.getEnums(), rhs.getEnums());
