public static String byteToHexString(byte b, boolean upperCase) {
    char[] digits = upperCase ? UPPER_CASE_DIGITS : DIGITS;
    // We always want two digits.
    char[] buf = new char[2];
    buf[0] = digits[(b >> 4) & 0xf];
    buf[1] = digits[b & 0xf];
    return new String(0, 2, buf);
