    if (lastChar == '\\') {
        escaped = true;
    } else {
        escaped = false;
    }
    if (!escaped && current == '\'') {
        singleQuoteStart = singleQuoteStart == -1 ? readerPosition : -1;
    }

        doubleQuoteStart = doubleQuoteStart == -1 ? readerPosition : -1;
    }
    readerPosition++;
    lastChar = current;
    return current;
