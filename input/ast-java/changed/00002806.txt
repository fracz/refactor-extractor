@Override
public void onDestroy() {
    super.onDestroy();
    // 释放资源
    if (mPresenter != null)
        mPresenter.onDestroy();
    EventBus.getDefault().unregister(this);
    RefWatcher watcher = BaseApplication.getRefWatcher(getActivity());
    if (watcher != null) {
        watcher.watch(this);
    }
