static void native_drawPath(int nativeCanvas, int path, int paint) {
    final Path_Delegate pathDelegate = Path_Delegate.getDelegate(path);
    if (pathDelegate == null) {
        assert false;
        return;
    }
    draw(nativeCanvas, paint, false, /*compositeOnly*/
    new GcSnapshot.Drawable() {

        public void draw(Graphics2D graphics, Paint_Delegate paint) {
            Shape shape = pathDelegate.getJavaShape();
            int style = paint.getStyle();
            if (style == Paint.Style.FILL.nativeInt || style == Paint.Style.FILL_AND_STROKE.nativeInt) {
                graphics.fill(shape);
            }
            if (style == Paint.Style.STROKE.nativeInt || style == Paint.Style.FILL_AND_STROKE.nativeInt) {
                graphics.draw(shape);
            }
        }
    });
