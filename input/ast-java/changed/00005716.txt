public void onTestIgnored(@NotNull final String testName, @NotNull final String ignoreComment, @Nullable final String stackTrace) {
    SMRunnerUtil.addToInvokeLater(new Runnable() {

        public void run() {
            final String fullTestName = getFullTestName(testName);
            SMTestProxy testProxy = getProxyByFullTestName(fullTestName);
            if (testProxy == null) {
                final boolean debugMode = SMTestRunnerConnectionUtil.isInDebugMode();
                logProblem("Test wasn't started! " + "TestIgnored event: name = {" + testName + "}, " + "message = {" + ignoreComment + "}. " + cannotFindFullTestNameMsg(fullTestName), debugMode);
                if (debugMode) {
                    return;
                } else {
                    // try to fix
                    // 1. report test opened
                    onTestStarted(testName, null);
                    // 2. report failure
                    testProxy = getProxyByFullTestName(fullTestName);
                }
            }
            if (testProxy == null) {
                return;
            }
            testProxy.setTestIgnored(ignoreComment, stackTrace);
            // fire event
            fireOnTestIgnored(testProxy);
        }
    });
