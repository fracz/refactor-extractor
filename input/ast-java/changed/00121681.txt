@Override
public void handleMessage(Message message) {
    switch(message.what) {
        case R.id.decoding_succeeded_message:
            handleDecode((Result) message.obj);
            break;
        case R.id.decoding_failed_message:
            Context context = getApplication();
            showAlert(context.getString(R.string.title_no_barcode_detected), 0, context.getString(R.string.msg_no_barcode_detected), context.getString(R.string.button_ok), true);
            break;
    }
