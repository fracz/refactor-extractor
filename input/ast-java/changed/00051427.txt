@Override
public Object call(Object[] namedArguments, FuncallExpression ast, Environment env) throws EvalException, InterruptedException {
    SkylarkEnvironment functionEnv = SkylarkEnvironment.createEnvironmentForFunctionCalling(env, definitionEnv, this);
    // Registering the functions's arguments as variables in the local Environment
    int i = 0;
    for (Object arg : namedArguments) {
        functionEnv.update(args.get(i++).getArgName(), arg);
    }
    try {
        for (Statement stmt : statements) {
            stmt.exec(functionEnv);
        }
    } catch (ReturnStatement.ReturnException e) {
        return e.getValue();
    }
    return Environment.NONE;
