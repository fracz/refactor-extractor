boolean updateScrollCoordinates(int x, int y) {
    int oldX = mScrollX;
    int oldY = mScrollY;
    mScrollX = x;
    mScrollY = y;
    if (oldX != mScrollX || oldY != mScrollY) {
        onScrollChanged(mScrollX, mScrollY, oldX, oldY);
        return true;
    } else {
        return false;
    }
