protected String[] parseKey(String key) {
    Pattern p = Pattern.compile("\\A[\\[\\]]*([^\\[\\]]+)\\]*");
    Matcher m = p.matcher(key);
    if (m.find())
        return new String[] { clean(m.group()), key.substring(m.end()) };
    else
        return null;
