/**
 * Forcibly restart Bluetooth if it is on
 */
/* package */
synchronized void restart() {
    if (mBluetoothState != BluetoothAdapter.STATE_ON) {
        return;
    }
    mRestart = true;
    if (!disable(false)) {
        mRestart = false;
    }
