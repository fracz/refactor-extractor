/**
 * Commit all tasks owned by this thread if specified interval time has elapsed
 */
protected void maybeCommit(final long now) {
    if (commitTimeMs >= 0 && lastCommitMs + commitTimeMs < now) {
        log.debug("{} Committing all active tasks {} and standby tasks {} because the commit interval {}ms has elapsed by {}ms", logPrefix, activeTasks.keySet(), standbyTasks.keySet(), commitTimeMs, now - lastCommitMs);
        commitAll();
        lastCommitMs = now;
        processStandbyRecords = true;
    }
