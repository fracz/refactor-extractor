void visit(PsiClass c) {
    if (!result.add(c))

    DirectClassInheritorsSearch.search(c).forEach(candidate -> {
        if (candidate.isInterface() && isFunctionalCompatible(candidate)) {
            visit(candidate);
        }
        return true;
    });
