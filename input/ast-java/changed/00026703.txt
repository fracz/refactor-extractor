    final IntrinsicMethod intrinsic = state.getInjector().getIntrinsics().getIntrinsic(fd);
    if (intrinsic != null) {
        return intrinsic;

    CallableMethod callableMethod;

        callableMethod = typeMapper.asCallableMethod(invoke);

    }
