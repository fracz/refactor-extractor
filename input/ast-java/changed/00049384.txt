    try {
        new WebDriverWait(driver, 5).until(new ExpectedCondition<Boolean>() {

            @Override
            public Boolean apply(WebDriver d) {
                return txt.equals(driver.findElement(by).getText());
            }
        });
        return true;
    } catch (TimeoutException e) {
        return false;
    }
