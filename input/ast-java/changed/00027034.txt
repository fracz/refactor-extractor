    switch(value.receiverSize()) {
        case 0:
            dup(v, type);
            break;
        case 1:
            if (type.getSize() == 2) {
                v.dup2X1();
            } else {
                v.dupX1();
            }
            break;
        case 2:
            if (type.getSize() == 2) {
                v.dup2X2();
            } else {
                v.dupX2();
            }
            break;
        case -1:
            throw new UnsupportedOperationException();
    }
