	private static function _castToBool($c) {
//		echo 'Initial Cast to Boolean<br />';
		$value = isset($c->v) ? (string) $c->v : null;
		if ($value == '0') {
			$value = false;
		} elseif ($value == '1') {
			$value = true;
		} else {
			$value = (bool)$c->v;
		}
		return $value;
	}	//	function _castToBool()
