			case PHPExcel_Style_Font::UNDERLINE_DOUBLE:				return 0x02;
			case PHPExcel_Style_Font::UNDERLINE_SINGLEACCOUNTING:	return 0x21;
			case PHPExcel_Style_Font::UNDERLINE_DOUBLEACCOUNTING:	return 0x22;
			default:												return 0x00;
		}
	}
