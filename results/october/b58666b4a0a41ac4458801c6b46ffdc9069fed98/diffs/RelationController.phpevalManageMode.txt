    protected function evalManageMode()
    {
        switch ($this->relationType) {
            case 'belongsTo':
                $mode = 'list';
                break;

            case 'belongsToMany':
                $mode = (isset($this->config->pivot)) ? 'pivot' : 'list';
                break;

            case 'hasOne':
            case 'hasMany':
                $mode = 'form';
                break;
        }

        return $mode;
    }
