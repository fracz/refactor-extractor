	public function driver($driver)
	{
		if (in_array($driver, array('cookie', 'file', 'database', 'apc', 'memcached')))
		{
			return $this->container->resolve('laravel.session.'.$driver);
		}

		throw new \Exception("Session driver [$driver] is not supported.");
	}
