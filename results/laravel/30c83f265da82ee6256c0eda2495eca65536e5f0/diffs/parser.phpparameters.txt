		// Initialize the array of parameters.

		// Any segment wrapped in parentheses is considered a parameter.
		// --------------------------------------------------------------
		for ($i = 0; $i < count($route_segments); $i++)

			if (strpos($route_segments[$i], '(') === 0)

			}
		}

		return $parameters;
	}
