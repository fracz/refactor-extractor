	private static function base($class, $models)
	{
		$results = array();

		foreach ($models as $model)
		{
			$result = new $class;

			$result->attributes = (array) $model;
			$result->exists = true;

			// -----------------------------------------------------
			// The results are keyed by the ID on the record.
			// -----------------------------------------------------
			$results[$result->id] = $result;
		}

		return $results;
	}
