	public static function reflash()
	{
		$flash = array();

		foreach (static::$session['data'] as $key => $value)
		{
			if (strpos($key, ':old:') === 0)
			{
				$flash[] = str_replace(':old:', '', $key);
			}
		}

		static::keep($flash);
	}
