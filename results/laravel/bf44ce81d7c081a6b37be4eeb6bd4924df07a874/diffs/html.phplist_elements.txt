	private static function list_elements($type, $list, $attributes)
	{
		if ( ! is_array($list))
		{
			return '';
		}

		$html = '';

		foreach ($list as $key => $value)
		{
			$html .= '<li>'.static::entities($value).'</li>';
		}

		return '<'.$type.static::attributes($attributes).'>'.$html.'</'.$type.'>';
	}
