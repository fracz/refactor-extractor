	public static function attributes($attributes)
	{
		$html = array();

		foreach ($attributes as $key => $value)
		{
			if ( ! is_null($value))
			{
				$html[] = $key.'="'.static::entities($value).'"';
			}
		}

		if (count($html) > 0)
		{
			return ' '.implode(' ', $html);
		}
		else
		{
			return '';
		}
	}
