	public function of($attributes, $rules, $messages = array())
	{
		foreach ($rules as $key => &$rule)
		{
			$rule = (is_string($rule)) ? explode('|', $rule) : $rule;
		}

		$this->attributes = $attributes;
		$this->messages = $messages;
		$this->rules = $rules;

		return $this;
	}
