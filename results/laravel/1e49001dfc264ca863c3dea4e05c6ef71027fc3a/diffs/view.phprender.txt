	public function render()
	{
		$this->compose();

		foreach ($this->data as &$data)
		{
			if ($data instanceof Renderable) $data = $data->render();
		}

		ob_start() and extract($this->data, EXTR_SKIP);

		try
		{
			include $this->path;
		}
		catch (\Exception $e)
		{
			Exception\Handler::make(new Exception\Examiner($e, new File))->handle();
		}

		return ob_get_clean();
	}
