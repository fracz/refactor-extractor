	public function age_flash()
	{
		foreach ($this->session['data'] as $key => $value)
		{
			if (strpos($key, ':old:') === 0) $this->forget($key);
		}

		foreach ($this->session['data'] as $key => $value)
		{
			if (strpos($key, ':new:') === 0)
			{
				$this->put(':old:'.substr($key, 5), $value);

				$this->forget($key);
			}
		}
	}
