	public function to($url, $status = 302, $method = 'location', $https = false)
	{
		$url = $this->url->to($url, $https);

		parent::__construct('', $status);

		if ($method == 'location')
		{
			return $this->header('Refresh', '0;url='.$url);
		}
		else
		{
			return $this->header('Location', $url);
		}
	}
