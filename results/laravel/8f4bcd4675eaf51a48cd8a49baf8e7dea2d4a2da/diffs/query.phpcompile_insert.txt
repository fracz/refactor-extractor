	private function compile_insert($values)
	{
		$sql = 'INSERT INTO '.$this->wrap($this->table);

		$columns = array_map(array($this, 'wrap'), array_keys($values));

		return $sql .= ' ('.implode(', ', $columns).') VALUES ('.$this->parameterize($values).')';
	}
