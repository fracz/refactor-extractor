	private function link($page, $text, $class)
	{
		$append = '';

		foreach ($this->append as $key => $value)
		{
			$append .= '&'.$key.'='.$value;
		}

		$request = IoC::container()->core('request');

		return HTML::link($request->uri().'?page='.$page.$append, $text, compact('class'), $request->secure());
	}
