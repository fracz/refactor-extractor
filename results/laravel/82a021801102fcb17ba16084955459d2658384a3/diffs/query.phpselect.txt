	public function select($columns = array('*'))
	{
		$this->select = ($this->distinct) ? 'SELECT DISTINCT ' : 'SELECT ';

		foreach ($columns as $column)
		{
			$wrapped[] = $this->wrap($column);
		}

		$this->select .= implode(', ', $wrapped);

		return $this;
	}
