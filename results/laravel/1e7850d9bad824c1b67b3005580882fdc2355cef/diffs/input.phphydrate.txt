	private function hydrate($method, $spoofed)
	{
		if ($method == 'GET')
		{
			$this->input = $this->get;
		}
		elseif ($method == 'POST')
		{
			$this->input = $this->post;
		}
		elseif ($method == 'PUT' or $method == 'DELETE')
		{
			($spoofed) ? $this->input = $this->post : parse_str(file_get_contents('php://input'), $this->input);
		}
	}
