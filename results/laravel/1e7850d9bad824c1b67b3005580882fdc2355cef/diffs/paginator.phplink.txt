	private function link($page, $text, $class)
	{
		$append = '';

		foreach ($this->append as $key => $value)
		{
			$append .= '&'.$key.'='.$value;
		}

		return HTML::link(Request::active()->uri().'?page='.$page.$append, $text, compact('class'), $this->secure);
	}
