	private static function find_view_for_name($name, $composers)
	{
		foreach ($composers as $key => $value)
		{
			if (is_string($value) and $value == $name)
			{
				return $key;
			}
			elseif (is_array($value) and isset($value['name']) and $value['name'] == $name)
			{
				return $key;
			}
		}
	}
