	public function save($session, $config)
	{
		if ( ! headers_sent())
		{
			extract($config);

			$payload = $this->crypter->encrypt(serialize($session));

			$this->cookie->put('session_payload', $payload, $lifetime, $path, $domain);
		}
	}
