	public static function load($file)
	{
		// Bail out if already loaded or doesn't exist.
		if (array_key_exists($file, static::$items) or ! file_exists($path = APP_PATH.'config/'.$file.EXT))
		{
			return;
		}

		static::$items[$file] = require $path;
	}
