	public static function update($query, $values)
	{
		$sql = 'UPDATE '.$query->wrap($query->table).' SET ';

		$sets = array();

		foreach (array_keys($values) as $column)
		{
			$sets[] = $query->wrap($column).' = ?';
		}

		return $sql .= implode(', ', $sets).' '.$query->where;
	}
