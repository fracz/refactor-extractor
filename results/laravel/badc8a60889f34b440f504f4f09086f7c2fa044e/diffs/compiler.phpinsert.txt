	public static function insert($query, $values)
	{
		$sql = 'INSERT INTO '.$query->wrap($query->table);

		$columns = array();

		foreach (array_keys($values) as $column)
		{
			$columns[] = $query->wrap($column);
		}

		return $sql .= ' ('.implode(', ', $columns).') VALUES ('.$query->parameterize($values).')';
	}
