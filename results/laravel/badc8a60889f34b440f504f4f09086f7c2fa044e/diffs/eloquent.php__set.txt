	public function __set($key, $value)
	{
		// If the key is a relationship, add it to the ignored attributes.
		if (method_exists($this, $key))
		{
			$this->ignore[$key] = $value;
		}
		else
		{
			$this->attributes[$key] = $value;
			$this->dirty[$key] = $value;
		}
	}
