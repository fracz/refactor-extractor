    private function getProtection()
    {
        $protection = $this->getParentWriter()->getPhpWord()->getProtection();
        if ($protection->getEditing() !== null) {
            $this->settings['w:documentProtection'] = array(
                '@attributes' => array(
                    'w:enforcement' => 1,
                    'w:edit' => $protection->getEditing(),
                )
            );
        }
    }
