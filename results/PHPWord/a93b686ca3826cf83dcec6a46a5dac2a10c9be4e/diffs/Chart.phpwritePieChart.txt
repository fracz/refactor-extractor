    private function writePieChart(XMLWriter $xmlWriter)
    {
        $xmlWriter->startElement('c:pieChart');

        $xmlWriter->writeBlock('c:varyColors', 'val', '1');

        $this->writeSeries($xmlWriter);

        $xmlWriter->endElement(); // c:pie3DChart
    }
