    private function populate_buffer()
    {
        if (feof($this->sock)) {
            $this->reset("");

            return false;
        }

        $block = fread($this->sock, $this->block_size);
        if ($block !== false) {
            $this->reset($block);

            return true;
        } else {
            return $block;
        }
    }
