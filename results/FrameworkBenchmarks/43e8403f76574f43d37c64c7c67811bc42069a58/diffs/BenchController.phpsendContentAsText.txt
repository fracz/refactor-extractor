    private function sendContentAsText($content) {
        $response = new Phalcon\Http\Response();
        $response->setStatusCode(200, "OK");
        $response->setHeader("Content-Type", "text/html; charset=utf-8");
        $response->setContent($content);
        return $response;
    }
