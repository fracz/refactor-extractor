    public function dbAction() {
        $queries = $this->getQueryOrDefault('queries', 1);
        $worlds = array();

        for ($i = 0; $i < $queries; ++$i) {
            $worlds[] = Worlds::findFirst(mt_rand(1, 10000));
        }

        return $this->sendContentAsJson($worlds);
    }
