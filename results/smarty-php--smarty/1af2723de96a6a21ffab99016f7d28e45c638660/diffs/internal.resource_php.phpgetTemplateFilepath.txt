    public function getTemplateFilepath($_template)
    {
        $_filepath = $_template->buildTemplateFilepath ();

        if ($_template->security) {
            $_template->smarty->security_handler->isTrustedResourceDir($_filepath);
        }

        return $_filepath;
    }
