    public function setSource(Smarty_Internal_Template $tpl, $uid = null)

        $tpl->source = null;
        unset($tpl->compiled);
        if (!isset($tpl->source)) {
            $tpl->source = Smarty_Template_Source::load($tpl);
        }
    }
