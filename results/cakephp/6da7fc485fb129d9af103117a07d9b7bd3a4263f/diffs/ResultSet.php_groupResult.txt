	protected function _groupResult() {
		$results = [];
		foreach ($this->_current as $key => $value) {
			$parts = explode('__', $key);
			$table = $this->_defaultTable;
			$field = $key;
			if (count($parts) > 1) {
				list($table, $field) = $parts;
				$value = $this->_castValue($table, $field, $value);
			}
			$results[$table][$field] = $value;
		}
		return $results;
	}
