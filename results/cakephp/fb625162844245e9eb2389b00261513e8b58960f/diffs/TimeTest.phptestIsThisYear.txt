    public function testIsThisYear()
    {
        $time = new Time();
        $this->assertTrue($time->isThisYear());

        $time->year = $time->year + 1;
        $this->assertFalse($time->isThisYear());

        $thisYear = date('Y');
        $time = new Time("$thisYear-01-01 00:00", 'Australia/Sydney');

        $now = clone $time;
        $now->timezone('UTC');
        Time::setTestNow($now);
        $this->assertFalse($time->isThisYear());
    }
