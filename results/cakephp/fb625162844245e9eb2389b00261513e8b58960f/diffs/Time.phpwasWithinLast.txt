    public function wasWithinLast($timeInterval)
    {
        $tmp = str_replace(' ', '', $timeInterval);
        if (is_numeric($tmp)) {
            $timeInterval = $tmp . ' days';
        }

        $interval = new static('-' . $timeInterval);
        $now = new static();

        return $this >= $interval && $this <= $now;
    }
