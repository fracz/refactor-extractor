	function getCrumbs () {

		if (count($this->_crumbs)) {

			$out = array("<a href=\"{$this->base}\">START</a>");
			foreach ($this->_crumbs as $crumb) {
				$out[] = "<a href=\"{$this->base}{$crumb[1]}\">{$crumb[0]}</a>";
			}

			return join(' &raquo; ', $out);
		}
		else
			return null;
	}
