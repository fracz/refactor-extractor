		$result = $table->find('all')->count();
		$this->assertEquals(3, $result);

		$query = $table->find('all')
			->where(['id >' => 1]);
		$result = $query->count();
		$this->assertEquals(2, $result);

		$result = $query->execute();
		$this->assertEquals(['count' => 2], $result->one());
	}
