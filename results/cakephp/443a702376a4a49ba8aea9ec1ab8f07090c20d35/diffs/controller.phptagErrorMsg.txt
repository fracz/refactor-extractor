	function tagErrorMsg ($field, $text) {
		if ($error = $this->tagIsInvalid($field)) {
			return sprintf(SHORT_ERROR_MESSAGE, is_array($text)? (empty($text[$error-1])? 'Error in field': $text[$error-1]): $text);
		}
		else {
			return null;
		}
	}
