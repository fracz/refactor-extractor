        $out = '';
        $out .= $options['before'];
        for ($i = 1; $i <= $params['pageCount']; $i++) {
            if ($i == $params['page']) {
                $out .= $templater->format('current', [
                    'text' => $params['page'],
                    'url' => $this->generateUrl(['page' => $params['page']], $options['model']),
                ]);
            } else {
                $vars = [
                    'text' => $i,
                    'url' => $this->generateUrl(['page' => $i], $options['model']),
                ];
                $out .= $templater->format('number', $vars);
            }
        }
        $out .= $options['after'];
        return $out;
    }
