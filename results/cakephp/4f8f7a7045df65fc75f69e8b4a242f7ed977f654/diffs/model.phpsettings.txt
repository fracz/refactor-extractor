	function settings() {
		$class = null;
		if (is_a($this->_Model, 'Model')) {
			$class = get_class($this->_Model);
		}
		return array('class' => get_class($this),
						'modelName' => $class,
						'dataField' => $this->_dataField,
						'expiryField' => $this->_expiryField);
	}
}
