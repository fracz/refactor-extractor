	public static function user($key = null) {
		if (!empty(static::$_user)) {
			$user = static::$_user;
		} elseif (static::$sessionKey && Session::check(static::$sessionKey)) {
			$user = Session::read(static::$sessionKey);
		} else {
			return null;
		}
		if ($key === null) {
			return $user;
		}
		return Hash::get($user, $key);
	}
