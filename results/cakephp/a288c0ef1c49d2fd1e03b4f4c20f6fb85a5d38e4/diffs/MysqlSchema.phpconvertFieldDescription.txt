	public function convertFieldDescription(Table $table, $row) {
		$field = $this->_convertColumn($row['Type']);
		$field += [
			'null' => $row['Null'] === 'YES' ? true : false,
			'default' => $row['Default'],
			'collate' => $row['Collation'],
			'comment' => $row['Comment'],
		];
		$table->addColumn($row['Field'], $field);
	}
