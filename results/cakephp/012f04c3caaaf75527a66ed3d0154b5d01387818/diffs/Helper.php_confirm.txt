    protected function _confirm($message, $okCode, $cancelCode = '', $options = [])
    {
        $message = json_encode($message);
        $confirm = "if (confirm({$message})) { {$okCode} } {$cancelCode}";
        if (isset($options['escape']) && $options['escape'] === false) {
            $confirm = h($confirm);
        }
        return $confirm;
    }
