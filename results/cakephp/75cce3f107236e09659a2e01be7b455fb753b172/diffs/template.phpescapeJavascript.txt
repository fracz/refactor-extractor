	function escapeJavascript($javascript)
	{
		$javascript = str_replace(array("\r\n","\n","\r"),'\n', $javascript);
		$javascript = str_replace(array('"', "'"), array('\"', "\\'"), $javascript);
		return $javascript;
	}
