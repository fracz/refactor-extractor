				return array_merge((array)$assoc, $query);
			} else {
				if (!empty($query)) {
					$query = array($query);
					if (is_array($assoc)) {
						$query = array_merge($query, $assoc);
					} else {
						$query[] = $assoc;
					}
					return $query;
				} else {
					return $assoc;
				}
			}
		}
		return $query;
	}
