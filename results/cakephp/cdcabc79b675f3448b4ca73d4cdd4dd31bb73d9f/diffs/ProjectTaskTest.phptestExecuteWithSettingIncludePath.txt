	public function testExecuteWithSettingIncludePath() {

		$this->assertTrue(is_dir($this->Task->args[0]), 'No project dir');

		$contents = $file->read();

		$contents = $file->read();
		$this->assertNoPattern('/define\(\'CAKE_CORE_INCLUDE_PATH\', ROOT/', $contents);
	}
