	public function testFormSecurityFields() {

		$fields = array('Model.password', 'Model.username', 'Model.valid' => '0');

		$this->Form->request['_Token'] = array('key' => $key);
		$result = $this->Form->secure($fields);

		$hash = urlencode($hash);
