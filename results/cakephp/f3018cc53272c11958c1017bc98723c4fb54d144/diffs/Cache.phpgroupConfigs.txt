	public static function groupConfigs($group = null) {
		if ($group == null) {
			return self::$_groups;
		}
		if (isset(self::$_groups[$group])) {
			return array($group => self::$_groups[$group]);
		} else {
			throw new CacheException(__d('cake_dev', 'Invalid cache group %s', $group));
		}
	}
