	function _scrapeSQL($sql) {
		if (strpos($sql, 'SELECT') === false) {
			return false;
		}
		$sql = str_replace("\"", '', $sql);
		$preFrom = explode('FROM', $sql);
		$preFrom = $preFrom[0];
		$find	 = array('SELECT');
		$replace = array('');
		$fieldList = trim(str_replace($find, $replace, $preFrom));
		$fields = explode(', ', $fieldList);
		$this->_map = array();

		foreach ($fields as $f) {
			$e = explode('.', $f);
			if (count($e) > 1) {
				$table = $e[0];
				$field = strtolower($e[1]);
			} else {
				$table = 0;
				$field = $e[0];
			}
			$this->_map[] = array($table, $field);
		}
	}
