			$checkbox = Html::activeCheckbox($this->model, $this->attribute, $options);
			return $this->begin() . strtr($this->template, array(
				'{input}' => $checkbox,
				'{label}' => '',
				'{error}' => $this->error(),
			)) . "\n" . $this->end();
		} else {
			return $this->render(Html::activeCheckbox($this->model, $this->attribute, $options));
		}
	}
