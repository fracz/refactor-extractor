	public function addRelatedRecord($relation, $record)
	{
		if ($relation->hasMany) {
			if ($relation->indexBy !== null) {
				$this->_related[$relation->name][$record->{$relation->indexBy}] = $record;
			} else {
				$this->_related[$relation->name][] = $record;
			}
		} else {
			$this->_related[$relation->name] = $record;
		}
	}
