	public function __isset($name)
	{
		if (isset($this->_attributes[$name]) || isset($this->_related[$name])) {
			return true;
		} elseif (isset($this->getTableSchema()->columns[$name]) || method_exists($this, $name)) {
			return false;
		} else {
			return parent::__isset($name);
		}
	}
