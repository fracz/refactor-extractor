    public function ufBoolean($value){
        if (Yii::t('yii', 'No')){
            return false;
        } elseif (Yii::t('yii', 'Yes')){
            return true;
        } else {
            throw new InvalidParamException('Value :' . $value . ' isn\'t a boolean yes or no value.');
        }
    }
