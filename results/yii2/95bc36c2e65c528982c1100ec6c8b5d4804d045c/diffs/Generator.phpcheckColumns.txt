	public function checkColumns($table)
	{
		foreach ($table->columns as $column) {
			if (!preg_match('/^[a-zA-Z_\x7f-\xff][a-zA-Z0-9_\x7f-\xff]*$/', $column->name)) {
				return $table->name . '.' . $column->name;
			}
		}
	}
