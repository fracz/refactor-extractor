	protected function renderExtensions()
	{
		if (empty($this->data['extensions'])) {
			return '';
		}
		$data = [];
		foreach ($this->data['extensions'] as $extension) {
			$data[$extension['name']] = $extension['version'];
		}
		return $this->renderData('Installed Extensions', $data) . "\n";
	}
