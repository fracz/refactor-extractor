	public function color($foreground, $background=null)
	{
		if ($foreground === null && $background === null) {
			return '';
		}

		$codes = array();
		if ($foreground !== null) {
			$codes[] = static::FOREGROUND_COLOR + $foreground;
		}
		if ($background !== null) {
			$codes[] = static::BACKGROUND_COLOR + $background;
		}

		return "\033[" . implode(';', $codes) . "m";
	}
