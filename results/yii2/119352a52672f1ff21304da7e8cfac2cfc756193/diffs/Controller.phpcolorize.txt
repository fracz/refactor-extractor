	public function colorize($string, $foreground = null, $background = null, $style = null)
	{
		$codes = array();
		if ($foreground !== null) {
			$codes[] = static::FOREGROUND_COLOR + $foreground;
		}
		if ($background !== null) {
			$codes[] = static::BACKGROUND_COLOR + $background;
		}
		if ($style !== null) {
			$codes[] = $style;
		}

		$code = implode(';', $codes);
		return "\033[0m" . ($code !== '' ? "\033[" . $code . "m" : '') . $string . "\033[0m";
	}
