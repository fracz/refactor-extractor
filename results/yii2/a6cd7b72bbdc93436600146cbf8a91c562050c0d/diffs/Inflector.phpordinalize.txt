	public static function ordinalize($number)
	{
		if (in_array(($number % 100), range(11, 13))) {
			return $number . 'th';
		}
		switch ($number % 10) {
			case 1: return $number . 'st';
			case 2: return $number . 'nd';
			case 3: return $number . 'rd';
			default: return $number . 'th';
		}
	}
