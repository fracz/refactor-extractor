	public static function camel2words($name, $ucwords = true)
	{
		$label = trim(strtolower(str_replace(array(
			'-',
			'_',
			'.'
		), ' ', preg_replace('/(?<![A-Z])[A-Z]/', ' \0', $name))));
		return $ucwords ? ucwords($label) : $label;
	}
