	public static function addCssClass(&$options, $class)
	{
		if (isset($options['class'])) {
			$classes = ' ' . $options['class'] . ' ';
			if (($pos = strpos($classes, ' ' . $class . ' ')) === false) {
				$options['class'] .= ' ' . $class;
			}
		} else {
			$options['class'] = $class;
		}
	}
