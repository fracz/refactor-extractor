	public function getDb()
	{
		if ($this->_db === null) {
			$db = \Yii::$app->getComponent($this->connectionID);
			if ($db instanceof Connection) {
				$this->_db = $db;
			} else {
				throw new InvalidConfigException("DbCache::connectionID must refer to the ID of a DB application component.");
			}
		}
		return $this->_db;
	}
