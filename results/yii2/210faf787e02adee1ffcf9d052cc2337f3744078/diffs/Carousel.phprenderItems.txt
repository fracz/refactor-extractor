	public function renderItems()
	{
		ob_start();
		echo Html::beginTag('div', array('class' => 'carousel-inner')) . "\n";
		for ($i = 0, $count = count($this->items); $i < $count; $i++) {
			$this->renderItem($this->items[$i], $i);
		}
		echo Html::endTag('div') . "\n";
		return ob_get_clean();
	}
