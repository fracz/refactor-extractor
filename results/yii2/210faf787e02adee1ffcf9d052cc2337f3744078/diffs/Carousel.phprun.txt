	public function run()
	{
		if (empty($this->items)) {
			return;
		}

		echo Html::beginTag('div', $this->options) . "\n";
		echo $this->renderIndicators() . "\n";
		echo $this->renderItems() . "\n";
		echo $this->renderPreviousAndNext() . "\n";
		echo Html::endTag('div') . "\n";

		$this->registerPlugin('carousel');
	}
