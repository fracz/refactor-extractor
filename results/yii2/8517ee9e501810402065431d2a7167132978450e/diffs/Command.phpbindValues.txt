	public function bindValues($values)
	{
		if (!empty($values)) {
			$this->prepare();
			foreach ($values as $name => $value) {
				if (is_array($value)) {
					$type = $value[1];
					$value = $value[0];
				} else {
					$type = $this->connection->getPdoType(gettype($value));
				}
				$this->pdoStatement->bindValue($name, $value, $type);
				$this->_params[$name] = $value;
			}
		}
		return $this;
	}
