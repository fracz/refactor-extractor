	public function getPdoType($type)
	{
		static $typeMap = array(
			'boolean' => \PDO::PARAM_BOOL,
			'integer' => \PDO::PARAM_INT,
			'string' => \PDO::PARAM_STR,
			'NULL' => \PDO::PARAM_NULL,
		);
		return isset($typeMap[$type]) ? $typeMap[$type] : \PDO::PARAM_STR;
	}
