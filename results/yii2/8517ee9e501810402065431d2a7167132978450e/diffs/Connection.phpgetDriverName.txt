	public function getDriverName()
	{
		if (($pos = strpos($this->dsn, ':')) !== false) {
			return strtolower(substr($this->dsn, 0, $pos));
		} else {
			return strtolower($this->getAttribute(\PDO::ATTR_DRIVER_NAME));
		}
	}
