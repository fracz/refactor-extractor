	public function saveAs($file, $deleteTempFile = true)
	{
		if ($this->_error == UPLOAD_ERR_OK) {
			if ($deleteTempFile) {
				return move_uploaded_file($this->_tempName, $file);
			} elseif (is_uploaded_file($this->_tempName)) {
				return copy($this->_tempName, $file);
			}
		}
		return false;
	}
