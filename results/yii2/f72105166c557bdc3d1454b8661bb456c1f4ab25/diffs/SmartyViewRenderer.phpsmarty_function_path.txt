	public function smarty_function_path($params, \Smarty_Internal_Template $template)
	{
		if (!isset($params['route'])) {
			trigger_error("path: missing 'route' parameter");
		}

		array_unshift($params, $params['route']) ;
		unset($params['route']);

		return Html::url($params);
	}
