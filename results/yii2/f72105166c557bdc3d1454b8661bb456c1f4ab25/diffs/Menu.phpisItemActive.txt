	protected function isItemActive($item)
	{
		if (isset($item['url']) && is_array($item['url']) && trim($item['url'][0], '/') === $this->route) {
			unset($item['url']['#']);
			if (count($item['url']) > 1) {
				foreach (array_splice($item['url'], 1) as $name => $value) {
					if (!isset($this->params[$name]) || $this->params[$name] != $value) {
						return false;
					}
				}
			}
			return true;
		}
		return false;
	}
