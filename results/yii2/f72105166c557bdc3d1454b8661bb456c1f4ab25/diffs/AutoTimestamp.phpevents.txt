	public function events()
	{
		$events = array();
		$behavior = $this;
		foreach ($this->attributes as $event => $attributes) {
			if (!is_array($attributes)) {
				$attributes = array($attributes);
			}
			$events[$event] = function () use ($behavior, $attributes) {
				$behavior->updateTimestamp($attributes);
			};
		}
		return $events;
	}
