	public function scenarios()
	{
		$attributes = array();
		foreach ($this->getActiveValidators() as $validator) {
			if ($validator->isActive('default')) {
				foreach ($validator->attributes as $name) {
					$attributes[$name] = true;
				}
			}
		}
		return array(
			'default' => array_keys($attributes),
		);
	}
