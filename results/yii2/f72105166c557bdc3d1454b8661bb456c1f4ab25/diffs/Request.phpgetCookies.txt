	public function getCookies()
	{
		if ($this->_cookies === null) {
			$this->_cookies = new CookieCollection(array(
				'enableValidation' => $this->enableCookieValidation,
				'validationKey' => $this->cookieValidationKey,
			));
		}
		return $this->_cookies;
	}
