	public function run()
	{
		if ($this->route === null && Yii::$app->controller !== null) {
			$this->route = Yii::$app->controller->getRoute();
		}
		if ($this->params === null) {
			$this->params = $_GET;
		}
		$items = $this->normalizeItems($this->items, $hasActiveChild);
		echo Html::tag('ul', $this->renderItems($items), $this->options);
	}
