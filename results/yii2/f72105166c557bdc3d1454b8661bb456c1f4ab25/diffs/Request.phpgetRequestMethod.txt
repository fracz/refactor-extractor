	public function getRequestMethod()
	{
		if (isset($_POST[$this->restVar])) {
			return strtoupper($_POST[$this->restVar]);
		} else {
			return isset($_SERVER['REQUEST_METHOD']) ? strtoupper($_SERVER['REQUEST_METHOD']) : 'GET';
		}
	}
