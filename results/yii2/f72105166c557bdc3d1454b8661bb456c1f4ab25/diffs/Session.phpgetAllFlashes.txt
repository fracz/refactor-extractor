	public function getAllFlashes()
	{
		$counters = $this->get($this->flashVar, array());
		$flashes = array();
		foreach (array_keys($counters) as $key) {
			if (isset($_SESSION[$key])) {
				$flashes[$key] = $_SESSION[$key];
			}
		}
		return $flashes;
	}
