	public function isAssigned($itemName, $userId)
	{
		$query = new Query;
		return $query->select(array('item_name'))
			->from($this->assignmentTable)
			->where(array('user_id' => $userId,	'item_name' => $itemName))
			->createCommand($this->db)
			->queryScalar() !== false;
	}
