	public function removeAll()
	{
		foreach ($this->_cookies as $cookie) {
			setcookie($cookie->name, '', 0, $cookie->path, $cookie->domain, $cookie->secure, $cookie->httponly);
		}
		$this->_cookies = array();
	}
