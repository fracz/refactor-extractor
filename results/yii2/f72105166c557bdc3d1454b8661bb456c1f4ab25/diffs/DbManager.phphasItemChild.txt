	public function hasItemChild($itemName, $childName)
	{
		$query = new Query;
		return $query->select(array('parent'))
			->from($this->itemChildTable)
			->where(array('parent' => $itemName, 'child' => $childName))
			->createCommand($this->db)
			->queryScalar() !== false;
	}
