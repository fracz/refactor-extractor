	public function removeFlash($key)
	{
		$counters = $this->get($this->flashVar, array());
		$value = isset($_SESSION[$key], $counters[$key]) ? $_SESSION[$key] : null;
		unset($counters[$key], $_SESSION[$key]);
		$_SESSION[$this->flashVar] = $counters;
		return $value;
	}
