	public function getPageCount()
	{
		if ($this->pageSize < 1) {
			return $this->itemCount > 0 ? 1 : 0;

			$itemCount = $this->itemCount < 0 ? 0 : (int)$this->itemCount;
			return (int)(($itemCount + $this->pageSize - 1) / $this->pageSize);
		}
	}
