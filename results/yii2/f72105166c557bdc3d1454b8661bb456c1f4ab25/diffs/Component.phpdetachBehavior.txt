	public function detachBehavior($name)
	{
		$this->ensureBehaviors();
		if (isset($this->_behaviors[$name])) {
			$behavior = $this->_behaviors[$name];
			unset($this->_behaviors[$name]);
			$behavior->detach();
			return $behavior;
		} else {
			return null;
		}
	}
