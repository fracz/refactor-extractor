	public function getItemChildren($names)
	{
		if (is_string($names)) {
			return isset($this->_children[$names]) ? $this->_children[$names] : array();
		}

		$children = array();
		foreach ($names as $name) {
			if (isset($this->_children[$name])) {
				$children = array_merge($children, $this->_children[$name]);
			}
		}
		return $children;
	}
