	public function readSession($id)
	{
		$query = new Query;
		$data = $query->select(array('data'))
			->from($this->sessionTable)
			->where('[[expire]]>:expire AND [[id]]=:id', array(':expire' => time(), ':id' => $id))
			->createCommand($this->db)
			->queryScalar();
		return $data === false ? '' : $data;
	}
