	public function beginCache($id, $properties = array())
	{
		$properties['id'] = $id;
		$properties['view'] = $this;
		/** @var $cache FragmentCache */
		$cache = FragmentCache::begin($properties);
		if ($cache->getCachedContent() !== false) {
			$this->endCache();
			return false;
		} else {
			return true;
		}
	}
