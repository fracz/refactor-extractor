	public static function t($category, $message, $params = array(), $language = null)
	{
		if (self::$app !== null) {
			return self::$app->getI18N()->translate($category, $message, $params, $language);
		} else {
			return is_array($params) ? strtr($message, $params) : $message;
		}
	}
