	protected function beforeLogin($identity, $cookieBased)
	{
		$event = new UserEvent(array(
			'identity' => $identity,
			'cookieBased' => $cookieBased,
		));
		$this->trigger(self::EVENT_BEFORE_LOGIN, $event);
		return $event->isValid;
	}
