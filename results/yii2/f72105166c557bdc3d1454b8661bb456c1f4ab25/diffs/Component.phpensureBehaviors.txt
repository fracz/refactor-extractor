	public function ensureBehaviors()
	{
		if ($this->_behaviors === null) {
			$this->_behaviors = array();
			foreach ($this->behaviors() as $name => $behavior) {
				$this->attachBehaviorInternal($name, $behavior);
			}
		}
	}
