	public function quoteSql($sql)
	{
		$db = $this;
		return preg_replace_callback('/(\\{\\{([%\w\-\. ]+)\\}\\}|\\[\\[([\w\-\. ]+)\\]\\])/',
			function($matches) use($db) {
				if (isset($matches[3])) {
					return $db->quoteColumnName($matches[3]);
				} else {
					return str_replace('%', $db->tablePrefix, $db->quoteTableName($matches[2]));
				}
			}, $sql);
	}
