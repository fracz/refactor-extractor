	protected function addValue($key, $value, $expire)
	{
		$this->gc();

		if ($expire > 0) {
			$expire += time();
		} else {
			$expire = 0;
		}

		try {
			$this->db->createCommand()
				->insert($this->cacheTable, array(
					'id' => $key,
					'expire' => $expire,
					'data' => array($value, \PDO::PARAM_LOB),
				))->execute();
			return true;
		} catch (\Exception $e) {
			return false;
		}
	}
