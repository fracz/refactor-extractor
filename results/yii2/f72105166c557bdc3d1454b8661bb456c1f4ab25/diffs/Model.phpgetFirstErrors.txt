	public function getFirstErrors()
	{
		if (empty($this->_errors)) {
			return array();
		} else {
			$errors = array();
			foreach ($this->_errors as $attributeErrors) {
				if (isset($attributeErrors[0])) {
					$errors[] = $attributeErrors[0];
				}
			}
		}
		return $errors;
	}
