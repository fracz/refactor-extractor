	protected function renderHeader()
	{
		$button = $this->renderCloseButton();
		if ($button !== null) {
			$this->header = $button . "\n" . $this->header;
		}
		if ($this->header !== null) {
			return Html::tag('div', "\n" . $this->header . "\n", array('class' => 'modal-header'));
		} else {
			return null;
		}
	}
