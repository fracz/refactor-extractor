	public static function button($content = 'Button', $name = null, $value = null, $options = array())
	{
		$options['name'] = $name;
		$options['value'] = $value;
		if (!isset($options['type'])) {
			$options['type'] = 'button';
		}
		return static::tag('button', $content, $options);
	}
