	public function detachBehaviors()
	{
		$this->ensureBehaviors();
		if ($this->_behaviors !== null) {
			foreach ($this->_behaviors as $name => $behavior) {
				$this->detachBehavior($name);
			}
		}
		$this->_behaviors = array();
	}
