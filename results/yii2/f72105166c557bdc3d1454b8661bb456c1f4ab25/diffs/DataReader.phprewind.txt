	public function rewind()
	{
		if ($this->_index < 0) {
			$this->_row = $this->_statement->fetch();
			$this->_index = 0;
		} else {
			throw new InvalidCallException('DataReader cannot rewind. It is a forward-only reader.');
		}
	}
