	public function getKeys()
	{
		if ($this->_keys === null) {
			$this->_keys = array();
			$models = $this->getModels();
			if ($this->key !== null) {
				foreach ($models as $model) {
					if (is_string($this->key)) {
						$this->_keys[] = $model[$this->key];
					} else {
						$this->_keys[] = call_user_func($this->key, $model);
					}
				}
			} else {
				$this->_keys = array_keys($models);
			}
		}
		return $this->_keys;
	}
