	protected function validateValue($file)
	{
		if ($this->skipOnEmpty && (!$file instanceof UploadedFile || $file->error == UPLOAD_ERR_NO_FILE)) {
			return null;
		} else {
			$result = parent::validateValue($file);
			return empty($result) ? $this->validateImage($file) : $result;
		}
	}
