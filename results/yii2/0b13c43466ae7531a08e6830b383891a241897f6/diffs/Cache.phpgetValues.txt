	protected function getValues($keys)
	{
		/** @var Connection $connection */
		$connection = \Yii::$app->getComponent($this->connectionId);
		$response = $connection->executeCommand('MGET', $keys);
		$result = [];
		$i = 0;
		foreach ($keys as $key) {
			$result[$key] = $response[$i++];
		}
		return $result;
	}
