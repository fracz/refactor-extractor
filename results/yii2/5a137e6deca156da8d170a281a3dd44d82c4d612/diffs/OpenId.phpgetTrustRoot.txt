	public function getTrustRoot()
	{
		if ($this->_trustRoot === null) {
			$this->_trustRoot = (!empty($_SERVER['HTTPS']) ? 'https' : 'http') . '://' . $_SERVER['HTTP_HOST'];
		}
		return $this->_trustRoot;
	}
