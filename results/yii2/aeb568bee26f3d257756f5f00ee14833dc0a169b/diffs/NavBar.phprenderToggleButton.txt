	protected function renderToggleButton()
	{
		$items = array();
		for ($i = 0; $i < 3; $i++) {
			$items[] = Html::tag('span', '', array('class' => 'icon-bar'));
		}
		return Html::a(implode("\n", $items), null, array(
			'class' => 'btn btn-navbar',
			'data-toggle' => 'collapse',
			'data-target' => 'div.navbar-collapse',
		));
	}
