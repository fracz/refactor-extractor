	public function render($view, $params = array())
	{
		$output = Yii::$app->getView()->render($view, $params, $this);
		$layoutFile = $this->findLayoutFile();
		if ($layoutFile !== false) {
			return Yii::$app->getView()->renderFile($layoutFile, array('content' => $output), $this);
		} else {
			return $output;
		}
	}
