	public function init()
	{
		parent::init();

		if ($this->view === null) {
			$this->view = Yii::$app->getView();
		}

		if (!$this->enabled) {
			$this->cache = null;
		} elseif (is_string($this->cache)) {
			$this->cache = Yii::$app->getComponent($this->cache);
		}

		if ($this->getCachedContent() === false) {
			$this->view->cacheStack[] = $this;
			ob_start();
			ob_implicit_flush(false);
		}
	}
