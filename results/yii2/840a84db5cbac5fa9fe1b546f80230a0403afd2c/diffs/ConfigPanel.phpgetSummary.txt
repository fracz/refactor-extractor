	public function getSummary()
	{
		$link = Html::a('more details', array('index', 'tag' => $this->data['tag']));
		return <<<EOD
<div class="yii-debug-toolbar-block">
	PHP: {$this->data['phpVersion']},
	Yii: {$this->data['phpVersion']},
	$link
</div>
EOD;
	}
