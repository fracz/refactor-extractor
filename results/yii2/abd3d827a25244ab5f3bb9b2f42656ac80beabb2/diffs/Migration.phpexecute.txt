	public function execute($sql, $params = array())
	{
		echo "    > execute SQL: $sql ...";
		$time = microtime(true);
		$this->getDbConnection()->createCommand($sql)->execute($params);
		echo " done (time: " . sprintf('%.3f', microtime(true) - $time) . "s)\n";
	}
