            function () {

                $value    = 'buenos días123καλημέρα!@#$%^&*早安()_ `~=+<>';
                $expected = 'buenos días123καλημέρα!@#$%^&*早安()_ `~=+';
                $this->sanitizer('string', $expected, $value);

                $value    = '{[<buenos días 123 καλημέρα! 早安>]}';
                $expected = '{[]}';
                $this->sanitizer('string', $expected, $value);

            }
        );
    }
