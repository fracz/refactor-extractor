    public function testUrlForExternalSite()
    {
        $this->specify(
            "URL for a different language not correct",
            function () {

                $di       = $this->setupDI();
                $url      = new PhTUrl();
                $url->setDI($di);
                $params   = [
                    'for'     => 'wikipedia',
                    'article' => 'Television_news',
                ];
                $expected = '/wiki/Television_news';
                $actual   = $url->get($params);

                expect($actual)->equals($expected);
            }
        );
    }
