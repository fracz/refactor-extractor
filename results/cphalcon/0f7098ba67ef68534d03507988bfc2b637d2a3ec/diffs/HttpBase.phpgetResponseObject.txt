    protected function getResponseObject()
    {
        PhDI::reset();
        $di = new PhDI();

        $di->set(
            'url',
            function () {
                $url = new PhUrl();
                $url->setBaseUri('/');

                return $url;
            }
        );

        $response = new PhTResponse();
        $response->setDI($di);

        return $response;
    }
