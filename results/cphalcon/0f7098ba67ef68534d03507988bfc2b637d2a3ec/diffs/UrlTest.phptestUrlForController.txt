    public function testUrlForController()
    {
        $this->specify(
            "URL for controller not correct",
            function () {

                $di       = $this->setupDI();
                $url      = new PhTUrl();
                $url->setDI($di);
                $params   = [
                    'for'   => 'classApi',
                    'class' => 'Some',
                ];
                $expected = '/api/classes/Some';
                $actual   = $url->get($params);

                expect($actual)->equals($expected);
            }
        );
    }
