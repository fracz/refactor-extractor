    public function testEscaperHtml()
    {
        $this->specify(
            "The escaper does not escape string correctly",
            function () {

                $escaper  = new PhTEscaper();
                $source   = "<h1></h1>";
                $expected = '&lt;h1&gt;&lt;/h1&gt;';
                $actual   = $escaper->escapeHtml($source);

                expect($actual)->equals($expected);
            }
        );
    }
