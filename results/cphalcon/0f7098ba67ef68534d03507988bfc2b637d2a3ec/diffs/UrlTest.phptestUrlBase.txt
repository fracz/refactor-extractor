    public function testUrlBase()
    {
        $this->specify(
            "The base url is not correct",
            function () {

                $_SERVER['PHP_SELF'] = '/index.php';

                $di       = $this->setupDI();
                $url      = new PhTUrl();
                $url->setDI($di);
                $expected = '/';
                $actual   = $url->get();

                expect($actual)->equals($expected);
            }
        );
    }
