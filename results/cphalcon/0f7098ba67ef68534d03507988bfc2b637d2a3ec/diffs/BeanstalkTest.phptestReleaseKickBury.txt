                // kick-job is supported since 1.8
                if (false !== $job->kick()) {
                    $job = $this->client->peekReady();

                    $this->assertTrue($job !== false);
                    $this->assertEquals($task, $job->getBody());
                }

                $this->assertTrue($job->delete());
            }
