    public function testUrlForBlogMixedParameters()
    {
        $this->specify(
            "URL with year/month/title not correct",
            function () {

                $di       = $this->setupDI();
                $url      = new PhTUrl();
                $url->setDI($di);
                $params   = [
                    'for'        => 'lang-controller',
                    'lang'       => 'de',
                    'controller' => 'index',
                ];
                $expected = '/de/index';
                $actual   = $url->get($params);

                expect($actual)->equals($expected);
            }
        );
    }
