    public function testCryptEncryptBase64()

            function () {

                $expected = 'https://github.com/phalcon/cphalcon/issues?state=open';

                expect($actual)->equals($expected);

                $encrypted = $crypt->encryptBase64($expected, $key, true);

                expect($actual)->equals($expected);
