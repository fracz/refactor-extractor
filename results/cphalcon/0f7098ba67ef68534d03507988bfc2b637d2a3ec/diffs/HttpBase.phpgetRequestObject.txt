    protected function getRequestObject()
    {
        PhDI::reset();
        $di = new PhDI();

        $di->set(
            'filter',
            function () {
                return new PhTFilter();
            }
        );

        $request = new PhTRequest();
        $request->setDI($di);

        return $request;
    }
