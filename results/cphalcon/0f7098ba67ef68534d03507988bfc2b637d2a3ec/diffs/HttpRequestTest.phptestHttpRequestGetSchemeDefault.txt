    public function testHttpRequestGetSchemeDefault()
    {
        $this->specify(
            "Default scheme is not http",
            function () {

                $request = $this->getRequestObject();

                $expected = 'http';
                $actual   = $request->getScheme();

                expect($actual)->equals($expected);
            }
        );
    }
