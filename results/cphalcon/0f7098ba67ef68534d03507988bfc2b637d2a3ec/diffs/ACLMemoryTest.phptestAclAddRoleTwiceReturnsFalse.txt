    public function testAclAddRoleTwiceReturnsFalse()
    {
        $this->specify(
            'Acl\Role added twice returns true',
            function () {

                $acl     = new PhTAclMem();
                $aclRole = new PhTAclRole('Administrators', 'Super User access');

                $acl->addRole($aclRole);
                $actual = $acl->addRole($aclRole);

                expect($actual)->false();
            }
        );
    }
