    public function testTagHtmlNameSelfCloseXhtml(CodeGuy $I)

        $actual   = PhTag::tagHtml($name, null, true);
        PhTag::setDoctype('');

        $I->assertEquals(
            $expected,
            $actual,
            sprintf($this->message, 'tagHtml with self close XHTML')
        );
    }
