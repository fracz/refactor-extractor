	public function assertEquals($a, $b)
	{
		if ($a != $b) {
			if(is_array($a) && is_array($b)){
				foreach($a as $k => $v){
					if ($v != $b[$k]) {
						print_r($v);
						print_r($b[$k]);
					}
				}
				throw new Exception('Not equals');
			} else {
				throw new Exception('Not equals '.print_r($a, true).' != '.print_r($b, true));
			}
		}
	}
