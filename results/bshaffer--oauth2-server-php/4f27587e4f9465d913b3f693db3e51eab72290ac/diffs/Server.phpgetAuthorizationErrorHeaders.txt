    private function getAuthorizationErrorHeaders($scope = null)
    {
        $header = sprintf('WWW-Authenticate: %s realm=%s', $this->config['token_type'], $this->config['www_realm'], $scope);
        if ($scope) {
            $header = sprintf('%s, scope=%s', $header, $scope);
        }
        return array($header);
    }
