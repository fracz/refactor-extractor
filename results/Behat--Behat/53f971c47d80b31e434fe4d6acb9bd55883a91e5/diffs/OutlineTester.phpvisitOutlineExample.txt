        // Visit & test steps

            $stResult = $this->visitStep($step, $environment, $tokens, $skip);
            if (0 !== $stResult) {
                $skip = true;
            }
            $itResult = max($itResult, $stResult);
        }

        $this->dispatcher->notify(new Event($outline, 'outline.example.after', array(
            'iteration'     => $row,
            'result'        => $itResult,
            'skipped'       => $skip,
            'environment'   => $environment
        )));

        return $itResult;
    }
