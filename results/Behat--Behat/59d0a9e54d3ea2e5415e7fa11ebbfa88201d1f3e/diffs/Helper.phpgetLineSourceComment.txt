    public function getLineSourceComment($lineLength, $stepsMaxLength, $file, $line)
    {
        $indent = $lineLength > $stepsMaxLength ? 0 : $stepsMaxLength - $lineLength;

        return sprintf("%s <comment># %s:%d</comment>",
            str_repeat(' ', $indent), $file, $line
        );
    }
