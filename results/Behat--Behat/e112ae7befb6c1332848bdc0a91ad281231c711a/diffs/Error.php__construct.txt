    public function __construct($code, $message, $file, $line)

        parent::__construct();

            case E_USER_WARNING:

            case E_USER_NOTICE:
                $type = 'Notice';

            case E_STRICT:
                $type = 'Strict';
                break;
            case E_DEPRECATED:
                $type = 'Deprecated';
                break;
            case E_ERROR:
            case E_USER_ERROR:
            default:
                $type = 'Error';
        }

        $this->code = $code;
        $this->message = $type . ': ' . $message;
        $this->file = $file;
        $this->line = $line;
    }
