    public function getStatus()
    {
        $statuses = array('passed', 'pending', 'undefined', 'failed');
        $code = -1;

        foreach ($this->getStepRunners() as $stepRunner) {
            if (($current = array_search($stepRunner->getStatus(), $statuses)) > $code) {
                $code = $current;
            }
        }

        return $statuses[$code];
    }
