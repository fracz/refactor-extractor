    protected function initLang()
    {
        if (preg_match('#^\#\s*language\:\s*(?P<lang>[\w]+?)\s*$#', $this->lines[0], $values)) {
            $this->i18n->loadLang($values['lang']);
        } else {
            $this->i18n->loadLang('en');
        }
        $this->lexer = new Lexer($this->i18n);
    }
