    public static function codeToString($resultCode)
    {
        switch ($resultCode) {
            case self::SKIPPED:
                return 'skipped';
            case self::PENDING:
                return 'pending';
            case self::FAILED:
                return 'failed';
        }

        return 'passed';
    }
