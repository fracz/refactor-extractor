    private function locateFilesInPath(array $filesPath, $checkExistense = true)
    {
        $paths = array();
        foreach ($filesPath as $path) {
            $path = $this->preparePath($path);
            if (!$checkExistense || is_file($path)) {
                $paths[] = $path;
            }
        }

        return $paths;
    }
