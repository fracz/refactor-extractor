    public function run(RunnerInterface $caller = null)
    {
        $this->setCaller($caller);
        $this->dispatcher->notify(new Event($this, 'step.pre_test'));

        $this->findDefinition();

        if (null === $this->status) {
            try {
                try {
                    $this->definition->run();
                    $this->status = 'passed';
                } catch (Pending $e) {
                    $this->status = 'pending';
                }
            } catch (\Exception $e) {
                $this->status    = 'failed';
                $this->exception = $e;
            }
        }

        $this->dispatcher->notify(new Event($this, 'step.post_test'));

        return $this->status;
    }
