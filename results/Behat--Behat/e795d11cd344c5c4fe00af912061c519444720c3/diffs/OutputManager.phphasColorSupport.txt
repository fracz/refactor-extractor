    protected function hasColorSupport()
    {
        if ($this->colors) {
            if ('\\' == DIRECTORY_SEPARATOR) {
                return false !== getenv('ANSICON');
            } else {
                return function_exists('posix_isatty') && @posix_isatty($this->output->getStream());
            }
        } else {
            return false;
        }
    }
