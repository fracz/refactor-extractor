    protected function getTagsHtml($node)
    {
        $html = '';

        if ($node->hasTags()) {
            $html .= '<ul class="tags">';
            foreach ($node->getTags() as $tag) {
                $html .= '<li>@' . $tag . '</li>';
            }
            $html .= '</ul>';
        }

        return $html;
    }
