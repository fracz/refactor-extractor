    protected function getTableRow(array $row, $class = null)
    {
        if (null === $class) {
            $html = '<tr>';
        } else {
            $html = '<tr class="' . $class . '">';
        }

        foreach ($row as $col) {
            $html .= '<td>' . $col . '</td>';
        }
        $html .= '</tr>';

        return $html;
    }
