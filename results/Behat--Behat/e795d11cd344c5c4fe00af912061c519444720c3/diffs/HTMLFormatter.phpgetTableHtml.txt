    protected function getTableHtml(TableNode $table, $class = null)
    {
        if (null === $class) {
            $html = '<table><tbody>';
        } else {
            $html = '<table class="' . $class . '"><tbody>';
        }

        foreach ($table->getHash() as $row) {
            $html .= $this->getTableRow($row);
        }

        $html .= '</tbody></table>';

        return $html;
    }
