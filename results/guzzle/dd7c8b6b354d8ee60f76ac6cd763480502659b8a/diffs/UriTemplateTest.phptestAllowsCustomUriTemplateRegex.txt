    public function testAllowsCustomUriTemplateRegex()
    {
        $template = new UriTemplate('abc_<$var>');
        $template->setRegex('/\<\$(.+)\>/');
        $this->assertEquals('abc_hi', $template->expand(array(
            'var' => 'hi'
        )));
    }
