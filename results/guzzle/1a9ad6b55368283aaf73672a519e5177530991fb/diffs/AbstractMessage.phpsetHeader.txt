    public function setHeader($header, $value)
    {
        $header = trim($header);
        $name = strtolower($header);
        $this->headerNames[$name] = $header;

        if (is_array($value)) {
            foreach ($value as &$v) {
                $v = trim($v);
            }
            $this->headers[$name] = $value;
        } else {
            $this->headers[$name] = [trim($value)];
        }
    }
