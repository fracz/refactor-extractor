    public function testExceptionInToStringTriggersError()

            throw new \Exception('foo');
        });
        $err = '';
        set_error_handler(function () use (&$err) {
            $err = func_get_args()[1];
        });
        echo $future;
        restore_error_handler();
        $this->assertContains('foo', $err);
    }
