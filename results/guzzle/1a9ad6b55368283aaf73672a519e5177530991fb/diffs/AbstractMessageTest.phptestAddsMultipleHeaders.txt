    public function testAddsMultipleHeaders()

            'foo' => ' bar',

        ]);
        $this->assertEquals([
            'foo' => ['bar'],
            'baz' => ['bam', 'boo'],
            'Host' => ['foo.com']
        ], $h->getHeaders());
    }
