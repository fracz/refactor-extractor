    public function testHandlesMergingInDisparateDataSources()
    {
        $params = array(
            'test' => 'value1',
            'test2' => 'value2',
            'test3' => array('value3', 'value4')
        );
        $this->coll->merge($params);
        $this->assertEquals($this->coll->toArray(), $params);
        $this->coll->merge(new Collection(['test4' => 'hi']));
        $this->assertEquals(
            $this->coll->toArray(),
            $params + ['test4' => 'hi']
        );
    }
