    private static function convertResponse(Response $response)
    {
        $headers = array_map(function ($h) {
            return implode(', ', $h);
        }, $response->getHeaders());

        return [
            'status'  => $response->getStatusCode(),
            'reason'  => $response->getReasonPhrase(),
            'headers' => $headers,
            'body'    => base64_encode((string) $response->getBody())
        ];
    }
