    public function addHeader($header, $value)
    {
        if (is_array($value)) {
            $current = array_merge($this->getHeaderAsArray($header), $value);
        } else {
            $current = $this->getHeaderAsArray($header);
            $current[] = (string) $value;
        }

        $this->setHeader($header, $current);
    }
