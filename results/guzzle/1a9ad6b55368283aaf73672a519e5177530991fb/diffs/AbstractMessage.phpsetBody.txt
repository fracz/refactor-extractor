    public function setBody(StreamInterface $body = null)
    {
        if ($body === null) {
            // Setting a null body will remove the body of the request
            $this->removeHeader('Content-Length');
            $this->removeHeader('Transfer-Encoding');
        }

        $this->body = $body;
    }
