    public function setScheme($scheme)
    {
        // Remove the default port if one is specified
        if ($this->port
            && isset(self::$defaultPorts[$this->scheme])
            && self::$defaultPorts[$this->scheme] == $this->port
        ) {
            $this->port = null;
        }

        $this->scheme = $scheme;
    }
