    public function testNestedFutureErrorsAreResolvedWhenSending()
    {
        $c = new Client();
        $total = 3;
        Server::enqueue([
            new Response(500),
            new Response(501),
            new Response(502)
        ]);
        $c->getEmitter()->on(
            'error',
            function (ErrorEvent $e) use (&$total) {
                if (--$total) {
                    $e->retry();
                }
            }
        );
        try {
            $c->get(Server::$url);
            $this->fail('Did not throw!');
        } catch (RequestException $e) {
            $this->assertEquals(502, $e->getResponse()->getStatusCode());
        }
    }
