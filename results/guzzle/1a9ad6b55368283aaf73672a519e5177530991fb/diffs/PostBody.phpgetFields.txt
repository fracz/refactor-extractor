    public function getFields($asString = false)
    {
        if (!$asString) {
            return $this->fields;
        }

        $query = new Query($this->fields);
        $query->setEncodingType(Query::RFC1738);
        $query->setAggregator($this->getAggregator());

        return (string) $query;
    }
