    public function __toString()
    {
        $lines = array();
        foreach ($this->transactions as $entry) {
            $response = isset($entry['response']) ? $entry['response'] : '';
            $lines[] = '> ' . trim($entry['sent_request'])
                . "\n\n< " . trim($response) . "\n";
        }

        return implode("\n", $lines);
    }
