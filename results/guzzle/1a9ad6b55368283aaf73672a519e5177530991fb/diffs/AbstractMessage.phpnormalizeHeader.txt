    public static function normalizeHeader(MessageInterface $message, $header)
    {
        $h = $message->getHeaderAsArray($header);
        for ($i = 0, $total = count($h); $i < $total; $i++) {
            if (strpos($h[$i], ',') === false) {
                continue;
            }
            foreach (preg_split('/,(?=([^"]*"[^"]*")*[^"]*$)/', $h[$i]) as $v) {
                $h[] = trim($v);
            }
            unset($h[$i]);
        }

        return $h;
    }
