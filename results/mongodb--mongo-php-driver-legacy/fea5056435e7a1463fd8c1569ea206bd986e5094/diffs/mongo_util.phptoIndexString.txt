    public static function toIndexString($keys)
    {
        if (is_string($keys)) {
            $name = str_replace(".", "_", $keys) + "_1";
        } else if (is_array($keys) || is_object($keys)) {
            $key_list = array();
            foreach ($keys as $k=>$v) {
                array_push($key_list, str_replace(".", "_", $k) . "_1");
            }
            $name = implode("_", $key_list);
        }
        return $name;
    }
