    function ensureIndex($keys)
    {
        $ns = (string)$this;
        if (is_string($keys)) {
            $keys = array($keys => 1);
        }
        $name = MongoUtil::toIndexString($keys);
        $coll = $this->parent->selectCollection("system.indexes");
        $coll->insert(array("ns" => $ns, "key" => $keys, "name" => $name));
    }
