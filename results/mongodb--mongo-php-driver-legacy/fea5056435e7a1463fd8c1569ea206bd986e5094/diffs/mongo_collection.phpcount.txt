    function count()
    {
        $result = MongoUtil::dbCommand($this->connection,
                                       array("count" => $this->name),
                                       $this->db);
        if ($result) {
            return (int)$result[ "n" ];
        }
        trigger_error("count failed", E_USER_ERROR);
    }
