    function getIndexInfo()
    {
        $ns           = (string)$this;
        $nscollection = $this->parent->selectCollection("system.indexes");
        $cursor       = $nscollection->find(array("ns" => $ns));
        $a            = array();
        while ($cursor->hasNext()) {
            $obj = $cursor->next();
            unset($obj[ "id" ]);
            unset($obj[ "ns" ]);
            array_push($a, $cursor->next());
        }
        return $a;
    }
