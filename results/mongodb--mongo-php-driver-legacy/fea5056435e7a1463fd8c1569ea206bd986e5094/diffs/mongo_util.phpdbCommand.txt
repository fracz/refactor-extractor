    public static function dbCommand($conn, $data, $db)
    {
        $cmd_collection = $db . MongoUtil::$_CMD;
        $obj            = mongo_find_one($conn, $cmd_collection, $data);

        if ($obj) {
            return $obj;
        } else {
            trigger_error("no db response", E_USER_WARNING);
            return false;
        }
    }
