    private function getCompressionLabel()
    {
        $compression = $this->compression;

        switch ($compression) {
            case Phar::GZ:
                return "gzip";

            case Phar::BZ2:
                return "bzip2";

            default:
                return sprintf("int(%d)", $compression);
        }
    }
