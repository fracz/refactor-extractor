    public function format_grade($value, $keepnull = false) {
        if (is_null($value)) {
            if ($keepnull) {
                return null;
            } else {
                return get_string('null', 'workshop');
            }
        }
        $decimalpoints  = 1;   // todo make the precision configurable
        $localized      = true;

        return format_float($value, $decimalpoints, $localized);
    }
