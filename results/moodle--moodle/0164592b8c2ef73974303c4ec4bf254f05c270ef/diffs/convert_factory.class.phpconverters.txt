    public static function converters($tempdir) {
        global $CFG;

        $converters = array();
        $plugins    = get_list_of_plugins('backup/converter');
        foreach ($plugins as $name) {
            $converters[$name] = self::converter($name, $tempdir);
        }
        return $converters;
    }
