    public function get_file($url, $title) {
        global $CFG;
        $path = $this->prepare_file($title);
        $buffer = '';
        if (!$this->wd->open()) {
            return false;
        }
        $this->wd->get($url, $buffer);
        $fp = fopen($path, 'wb');
        fwrite($fp, $buffer);
        return $path;
    }
