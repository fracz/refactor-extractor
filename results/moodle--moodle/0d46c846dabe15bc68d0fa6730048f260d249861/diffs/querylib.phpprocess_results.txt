      $stripped_term = str_replace($strip_arr, '', $term);

      //--

      $this->total_results = $hitcount;

        $end = $hitcount;

        $hit = $hits[$i];

          $resultdoc->title   = $hit->title;

          $resultdoc->doctype = $hit->doctype;
          $resultdoc->author  = $hit->author;

          $resultdocs[] = clone($resultdoc);
        } //if
      } //foreach
