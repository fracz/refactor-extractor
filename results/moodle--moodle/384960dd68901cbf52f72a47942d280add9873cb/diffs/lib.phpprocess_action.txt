            $collapsed = unserialize($collapsed);
        } else {
            $collapsed = array();
        }

        switch ($action) {
            case 'switch_minus':
                if (!in_array($targetid, $collapsed)) {
                    $collapsed[] = $targetid;
                    set_user_preference('grade_report_grader_collapsed_categories', serialize($collapsed));
                }
                break;

            case 'switch_plus':
                $key = array_search($targetid, $collapsed);
                if ($key !== false) {
                    unset($collapsed[$key]);
                    set_user_preference('grade_report_grader_collapsed_categories', serialize($collapsed));
                }
                break;

            default:
                break;
        }

        return true;
    }
