    function load_scale() {
        if (!empty($this->scaleid)) {
            $this->scale = grade_scale::fetch('id', $this->scaleid);
            if (method_exists($this->scale, 'load_items')) {
                $this->scale->load_items();
            } else {
                $this->scale = null;
            }
        }
        return $this->scale;
    }
