    function test_ilike() {
        $DB = $this->tdb;

        $sql = "SELECT 'SuperDuperRecord' " . $DB->sql_ilike() . " '%per%' AS result";
        $record = $DB->get_record_sql($sql);
        $this->assertEqual('t', $record->result);

        $sql = "SELECT 'SuperDuperRecord' " . $DB->sql_ilike() . " 'per' AS result";
        $record = $DB->get_record_sql($sql);
        $this->assertEqual('f',$record->result);
    }
