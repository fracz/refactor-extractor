    protected static function is_developer() {
        global $CFG;

        if (!isset($CFG->config_php_settings['debug'])) {
            return false;
        }

        $debug = (int)$CFG->config_php_settings['debug'];
        if ($debug & E_ALL and $debug & E_STRICT) {
            return true;
        }

        return false;
    }
