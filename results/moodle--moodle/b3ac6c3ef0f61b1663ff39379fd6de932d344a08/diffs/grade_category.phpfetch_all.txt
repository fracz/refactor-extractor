    function fetch_all($params) {
        if ($categories = grade_object::fetch_all_helper('grade_categories', 'grade_category', $params)) {
            foreach ($categories as $key=>$value) {
                $categories[$key]->path = grade_category::build_path($categories[$key]);
            }
            return $categories;

        } else {
            return false;
        }
    }
