    function tearDown() {
        global $CFG;
        foreach ($this->tables as $table) {
            delete_records($table);
            if (count($this->$table) > 0) {
                unset ($this->$table);
            }
        }
        $CFG->prefix = $CFG->old_prefix;
    }
