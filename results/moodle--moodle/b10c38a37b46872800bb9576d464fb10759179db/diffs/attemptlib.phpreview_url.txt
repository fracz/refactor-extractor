    public function review_url($page = 0, $question = false, $showall = false) {
        global $CFG;
        $fragment = '';
        if ($question) {
            $fragment = '#q' . $question;
        }
        $param = '';
        if ($showall) {
            $param = '$amp;showall=1';
        } else if ($page) {
            $param = '$amp;page=' . $page;
        }
        return $CFG->wwwroot . '/mod/quiz/review.php?attempt=' .
                $this->attempt->id . $param . $fragment;
    }
