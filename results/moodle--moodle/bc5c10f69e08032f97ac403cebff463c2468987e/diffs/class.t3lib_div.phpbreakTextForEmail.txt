	public static function breakTextForEmail($str, $implChar = LF, $charWidth = 76) {
		self::logDeprecatedFunction();

		$lines = explode(LF, $str);
		$outArr = array();
		foreach ($lines as $lStr) {

		}
		return implode(LF, $outArr);
