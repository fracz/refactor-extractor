	public static function danish_strtoupper($string) {
		self::logDeprecatedFunction();

		$value = strtoupper($string);
		return strtr($value, array(
			chr(225) => chr(193),
			chr(233) => chr(201),
			chr(250) => chr(218),
			chr(237) => chr(205),
			chr(226) => chr(196),
			chr(234) => chr(203),
			chr(251) => chr(220),
			chr(244) => chr(214),
			chr(238) => chr(207),
			chr(230) => chr(198),
			chr(248) => chr(216),
			chr(229) => chr(197),
			chr(228) => chr(196),
			chr(246) => chr(214),
			chr(252) => chr(220),
		));
	}
