    function ArrayTagToString($aTag)
    {
        if (!is_array($aTag)) {
            return $aTag;
        }
        $out = '';
        foreach($aTag as $key => $value) {
            if (is_array($value)) {
                $out = $out.$this->ArrayTagToString($value);
            } else {
                $out = $value;
            }
        }
        return $out;
    }
