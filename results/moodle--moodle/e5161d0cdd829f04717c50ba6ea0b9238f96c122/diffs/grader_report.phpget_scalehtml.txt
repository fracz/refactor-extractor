    function get_scalehtml() {
        $scalehtml = '';
        if ($this->showscales) {
            $scalehtml = '<tr><td>'.get_string('range','grades').'</td>';
            foreach ($this->items as $item) {
                $scalehtml .= '<td>'. get_grade_clean($item->grademin).'-'. get_grade_clean($item->grademax).'</td>';
            }
            $scalehtml .= '</tr>';
        }
        return $scalehtml;
    }
