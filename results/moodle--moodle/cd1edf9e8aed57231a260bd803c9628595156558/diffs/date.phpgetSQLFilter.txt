    function getSQLFilter() {
        if(empty($this->_value) && empty($this->_value2)) {
            return null;
        }
        $res = $this->_field . '>0' ;
        if($this->_value) {
            $res .= ' AND ' . $this->_field . '>=' . $this->_value;
        }
        if($this->_value2) {
            $res .= ' AND ' . $this->_field . '<=' . $this->_value2;
        }
        return $res;
    }
