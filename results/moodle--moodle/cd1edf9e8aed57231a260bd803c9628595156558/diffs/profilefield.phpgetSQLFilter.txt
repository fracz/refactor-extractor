    function getSQLFilter() {
        global $CFG;
        $where = '';
        $op = ' IN ';
        switch($this->_operator) {
        case 0: // contains
            if(empty($this->_value)) {
                return null;
            }
            $where = 'data ' . sql_ilike(). ' "%' . $this->_value . '%"';
            break;
        case 1: // does not contain
            if(empty($this->_value)) {
                return null;
            }
            $where = 'data NOT ' . sql_ilike(). ' "%' . $this->_value . '%"';
            break;
        case 2: // equal to
            if(empty($this->_value)) {
                return null;
            }
            $where = 'data="' . $this->_value . '"';
            break;
        case 3: // starts with
            if(empty($this->_value)) {
                return null;
            }
            $where = 'data ' . sql_ilike(). ' "' . $this->_value . '%"';
            break;
        case 4: // ends with
            if(empty($this->_value)) {
                return null;
            }
            $where = 'data ' . sql_ilike(). ' "%' . $this->_value . '"';
            break;
        case 5: // empty
            $where = 'data=""';
            break;
        case 6: // is not defined
            $op = ' NOT IN ';
            break;
        case 7: // is defined
            break;
        }
        if(!empty($this->_profile_field)) {
            if(!empty($where)) {
                $where = ' AND ' . $where;
            }
            $where = 'fieldid=' . $this->_profile_field . $where;
        }
        if(!empty($where)) {
            $where = ' WHERE ' . $where;
        }
        return $this->_field . $op . "(SELECT userid FROM {$CFG->prefix}user_info_data" . $where . ')';
    }
