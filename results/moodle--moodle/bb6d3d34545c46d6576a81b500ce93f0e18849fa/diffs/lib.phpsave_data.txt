    function save_data ($data) {
        if ($this->_is_visible() and !$this->_is_locked()) { /// check that we have permission

            $data = $this->save_data_preprocess($data);

            unset($datarecord);
            $datarecord->userid  = $this->userid;
            $datarecord->fieldid = $this->fieldid;
            $datarecord->data    = $data;

            if ($this->dataid == 0) { /// inserting a new record
                if ($ret = insert_record('user_info_data', $datarecord)) {
                    $this->dataid = $ret;
                }
            } else {
                $datarecord->id = $this->dataid;
                if (update_record('user_info_data', $datarecord)) {
                    $ret = $this->dataid;
                } else {
                    $ret = false;
                }
            }

        } else {
            $ret = 0;
        }

        return $ret;
    }

    /**
     * Hook for child classess to process the data before it gets saved in database
     * @param   mixed
     * @return  mixed
     */
||||||||    function save_data($usernew) {

        $usernew = $this->save_data_preprocess($usernew);

        if (!isset($usernew->{$this->inputname})) {
            // field not present in form, probably locked and incisible - skip it!
            return;
        }

        $data = new object();
        $data->userid  = $usernew->id;
        $data->fieldid = $this->field->id;
        $data->data    = $usernew->{$this->inputname};

        if ($dataid = get_field('user_info_data', 'id', 'userid', $data->userid, 'fieldid', $data->fieldid)) {
            $data->id = $dataid;
            if (!update_record('user_info_data', $data)) {
                error('Error updating custom profile field!');
            }
        } else {
            insert_record('user_info_data', $data);
        }
    }
