    protected function make_theme($name) {
        global $CFG;
        $theme = new stdClass;
        $theme->name = $name;
        $theme->dir = $CFG->themedir . '/' . $name;
        make_upload_directory($this->workspace . '/' . $name);
        return $theme;
    }
