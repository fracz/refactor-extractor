    protected function configure_dbconnection() {
        $this->adodb->SetFetchMode(ADODB_FETCH_ASSOC);

        $sql = "SET NAMES 'utf8'";
        $this->query_start($sql, null, SQL_QUERY_AUX);
        $result = $this->adodb->Execute($sql);
        $this->query_end($result);

        return true;
    }
