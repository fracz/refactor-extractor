    protected function js_url($url) {
        global $CFG;

        if ($url instanceof moodle_url) {
            return $url;
        } else {
            //return new moodle_url($CFG->httpswwwroot.'/lib/javascript.php', array('file'=>$url, 'rev'=>$CFG->jsrev));
            return new moodle_url($CFG->httpswwwroot.$url);
        }
    }
