    function get_options() {
        if (empty($this->type) or empty($this->plugin)) {
            return array();
        }

        $params = array();

        $params['plugin'] = $this->plugin;

        if (!empty($this->courseid)) {
            $params['id'] = $this->courseid;
        }

        if (!empty($this->userid)) {
            $params['userid'] = $this->userid;
        }

        if (!empty($this->page)) {
            $params['page'] = $this->page;
        }

        return $params;
    }
