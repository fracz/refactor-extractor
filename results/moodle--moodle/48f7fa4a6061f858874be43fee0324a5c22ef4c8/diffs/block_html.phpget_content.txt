    function get_content() {
        global $CFG;
        require_once($CFG->dirroot .'/lib/weblib.php');
        if ($this->content !== NULL) {
            return $this->content;
        }

        $this->content = new stdClass;
        $this->content->text = isset($this->config->text) ? $this->config->text : '';
        $this->content->footer = '';

        return $this->content;
    }
}
