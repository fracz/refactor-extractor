    function get_element_type($element) {
        if ($element['object'] == 'filler') {
            return 'filler';
        }

        if (get_class($element['object']) == 'grade_category') {
            if ($element['object']->depth == 2) {
                return 'topcat';
            } else {
                return 'subcat';
            }
        }

        return 'item';
    }
