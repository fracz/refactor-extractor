    public function change_database_structure($sql) {
        $result = true;
        $this->query_start($sql, null, SQL_QUERY_STRUCTURE);

        try {
            $this->pdb->exec($sql);
            $this->reset_caches();
        } catch (PDOException $ex) {
            $this->lastError = $ex->getMessage();
            $result = false;
        }
        $this->query_end($result);
        return $result;
    }
