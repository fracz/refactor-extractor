    public static function get_moodle_cookie() {
        global $CFG;

        $cookiename = 'MOODLEID_'.$CFG->sessioncookie;

        if (empty($_COOKIE[$cookiename])) {
            return '';
        } else {
            $thing = rc4decrypt($_COOKIE[$cookiename]);
            return ($thing == 'guest') ? '': $thing;  // Ignore guest account
        }
    }
