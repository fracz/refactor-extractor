    public function get($name){
        $content = '';
        $fp = fopen($this->dir.'/'.$name.'.data', 'r');
        if ($fp){
            while(!feof($fp))
                $content .= fread($fp, 4096);
            fclose($fp);
            return unserialize($content);
        } else {
            return false;
        }
    }
