    private function write($name, $content){
        $name = $this->dir.'/'.$name.'.data';
        if (file_exists($name)) {
            unlink($name);
        }

        $fp = fopen($name, 'w');
        if($fp) {
            fputs($fp, $content);
            fclose($fp);
        }
    }
