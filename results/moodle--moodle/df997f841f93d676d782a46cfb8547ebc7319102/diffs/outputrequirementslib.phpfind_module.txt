                case 'core_filetree':
                    $module = array('name'     => 'core_filetree',
                                    'fullpath' => '/files/module.js',
                                    'requires' => array('node', 'event', 'overlay', 'io', 'json', 'yui2-treeview'));
                    break;

            if ($dir = get_component_directory($component)) {
                if (file_exists("$dir/module.js")) {

                }
            }
        }

        return $module;
    }
