    function moodle_url($url = null, $params = array()){
        global $FULLME;
        if ($url !== ''){
            if ($url === null){
                $url = strip_querystring($FULLME);
            }
            $parts = parse_url($url);
            if ($parts === FALSE){
                print_error('invalidurl');
            }
            if (isset($parts['query'])){
                parse_str(str_replace('&amp;', '&', $parts['query']), $this->params);
            }
            unset($parts['query']);
            foreach ($parts as $key => $value){
                $this->$key = $value;
            }
            $this->params($params);
        }
    }
