    function get_category() {
        $category = null;

        if (!empty($this->categoryid)) {
            $category = grade_category::fetch('id', $this->categoryid);
        } elseif (!empty($this->iteminstance) && $this->itemtype == 'category') {
            $category = grade_category::fetch('id', $this->iteminstance);
        }

        return $category;
