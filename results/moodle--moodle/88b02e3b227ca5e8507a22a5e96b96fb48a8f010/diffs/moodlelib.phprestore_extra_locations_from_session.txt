    protected function restore_extra_locations_from_session() {
        global $SESSION;
        if (!empty($SESSION->extralangsearchlocations)) {
            foreach ($SESSION->extralangsearchlocations as $plugintype => $path) {
                $this->register_plugin_type($plugintype, $path);
            }
        }
    }
