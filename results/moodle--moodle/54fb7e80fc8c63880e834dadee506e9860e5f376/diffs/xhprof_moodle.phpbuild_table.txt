    function build_table(){
        if ($this->rawdata){
            foreach($this->rawdata as $row){
                $formattedrow = $this->format_row($row);
                // reference row, add 'referencerun' class
                $classname = $row->runreference ? 'referencerun' : '';
                $this->add_data_keyed($formattedrow, $classname);
            }
        }
    }
