    protected static function get_all() {
        $all = self::get_all_ids();
        $rv = array();
        foreach ($all as $id => $unused) {
            if ($coursecat = self::get($id, IGNORE_MISSING, true)) {
                $rv[$id] = $coursecat;
            }
        }
        unset($rv[0]);
        return $rv;
    }
