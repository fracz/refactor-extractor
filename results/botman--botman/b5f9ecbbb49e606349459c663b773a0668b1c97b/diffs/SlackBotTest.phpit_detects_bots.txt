    public function it_detects_bots()
    {
        $slackbot = $this->getBot([
            'event' => [
                'text' => 'Hi Julia',
            ],
        ]);
        $this->assertFalse($slackbot->isBot());

        $slackbot = $this->getBot([
            'event' => [
                'bot_id' => 'foo',
                'text' => 'Hi Julia',
            ],
        ]);
        $this->assertTrue($slackbot->isBot());
    }
