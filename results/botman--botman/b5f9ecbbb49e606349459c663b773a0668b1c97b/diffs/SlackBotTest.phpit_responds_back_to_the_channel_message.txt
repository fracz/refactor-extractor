    public function it_responds_back_to_the_channel_message()
    {
        $slackbot = $this->getBot([
            'token' => 'foo',
            'event' => [
                'channel' => 'general',
            ],
        ]);
        $this->commander
            ->shouldReceive('execute')
            ->once()
            ->with('chat.postMessage', [
                'token' => 'foo',
                'channel' => 'general',
                'text' => 'This is my response',
            ]);

        $slackbot->reply('This is my response');
    }
