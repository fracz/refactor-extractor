	public function delete($id, $call_event = true) {
		if ($result = parent::delete($id, $call_event)) {
			unset($this->mocks[$id]);
			if (!empty($this->query_specs[$id])) {
				foreach ($this->query_specs[$id] as $spec) {
					$this->db->removeQuerySpec($spec);
				}
			}
		}
		return $result;
	}
