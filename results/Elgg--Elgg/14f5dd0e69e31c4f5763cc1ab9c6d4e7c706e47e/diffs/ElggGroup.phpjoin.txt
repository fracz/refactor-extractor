	public function join(\ElggUser $user) {
		$result = add_entity_relationship($user->guid, 'member', $this->guid);

		if ($result) {
			$params = ['group' => $this, 'user' => $user];
			_elgg_services()->events->trigger('join', 'group', $params);
		}

		return $result;
	}
