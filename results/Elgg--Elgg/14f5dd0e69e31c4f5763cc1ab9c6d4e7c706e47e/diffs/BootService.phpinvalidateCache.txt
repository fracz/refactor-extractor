	public function invalidateCache() {
		$CONFIG = _elgg_services()->config->getStorageObject();

		// this gets called a lot on plugins page, avoid thrashing cache
		static $cleared = false;
		if (!$cleared) {
			$this->getStashItem($CONFIG)->clear();
			$cleared = true;
		}
	}
