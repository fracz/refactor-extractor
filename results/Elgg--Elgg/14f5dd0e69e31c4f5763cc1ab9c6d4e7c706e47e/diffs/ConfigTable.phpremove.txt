	function remove($name) {
		$name = trim($name);

		$query = "
			DELETE FROM {$this->db->prefix}config
			WHERE name = :name
		";

		$params = [
			':name' => $name,
		];

		$this->boot->invalidateCache();

		return $this->db->deleteData($query, $params) !== false;
	}
