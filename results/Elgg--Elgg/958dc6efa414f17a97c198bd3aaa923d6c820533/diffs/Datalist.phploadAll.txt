	function loadAll() {
		$result = $this->db->getData("SELECT * FROM {$this->table}");
		$map = array();
		if (is_array($result)) {
			foreach ($result as $row) {
				$map[$row->name] = $row->value;
				$this->cache->put($row->name, $row->value);
			}
		}

		return $map;
	}
