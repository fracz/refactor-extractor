	public function get($key, callable $callback) {
		assert(is_string($key) || is_int($key));

		$item = $this->stash->getItem((string)$key);

		$result = $item->get();

		if ($item->isMiss()) {
			$item->lock();

			$result = call_user_func($callback);

			$this->stash->save($item->set($result));
		}

		return $result;
	}
