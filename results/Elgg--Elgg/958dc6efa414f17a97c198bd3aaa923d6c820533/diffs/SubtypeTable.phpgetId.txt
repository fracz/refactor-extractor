	function getId($type, $subtype) {
		global $SUBTYPE_CACHE;

		if (!$subtype) {
			return false;
		}

		if ($SUBTYPE_CACHE === null) {
			$this->populateCache();
		}

		// use the cache before hitting database
		$result = $this->retrieveFromCache($type, $subtype);
		if ($result !== null) {
			return $result->id;
		}

		return false;
	}
