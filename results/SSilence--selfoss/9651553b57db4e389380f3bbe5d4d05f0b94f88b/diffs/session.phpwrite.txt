	function write($id,$data) {
		$fw=Base::instance();
		$headers=$fw->get('HEADERS');
		$jar=session_get_cookie_params();
		Cache::instance()->set($id.'.@',
			array(
				'data'=>$data,
				'ip'=>$fw->get('IP'),
				'agent'=>isset($headers['User-Agent'])?
					$headers['User-Agent']:'',
				'stamp'=>time()
			),
			$jar['lifetime']
		);
		return TRUE;
	}
