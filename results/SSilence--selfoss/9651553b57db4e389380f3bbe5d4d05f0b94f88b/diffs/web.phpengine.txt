	function engine($arg='socket') {
		$arg=strtolower($arg);
		if ($arg=='curl' && ($curl=extension_loaded('curl')) ||
			$arg=='stream' && ($stream=ini_get('allow_url_fopen')) ||
			$arg=='socket' && ($socket=function_exists('fsockopen')))
			$this->wrapper=$arg;
		elseif ($socket)
			$this->wrapper='socket';
		elseif ($stream)
			$this->wrapper='stream';
		elseif ($curl)
			$this->wrapper='curl';
		else
			user_error(E_Request);
	}
