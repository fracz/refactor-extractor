	static function rle_decode($data, $datalen)
	{
		$len = strlen($data);

		$out = '';

		$i = 0;
		$k = 0;
		while ($i<$len)
		{
			self::dec_bits(ord($data[$i]), $type, $value);
			if ($k >= $datalen)
			{
				break;
			}

			$i++;

			if ($type == 0) //raw
			{
				for ($j=0; $j<3*$value; $j++)
				{
					$out .= $data[$j+$i];
					$k++;
				}
				$i += $value*3;
			}
			else //rle
			{
				for ($j=0; $j<$value; $j++)
				{
					$out .= $data[$i] . $data[$i+1] . $data[$i+2];
					$k++;
				}
				$i += 3;
			}
		}
		return $out;
	}
