	function substr($str,$start,$len=0) {
		if ($start<0) {
			$len=-$start;
			$start=$this->strlen($str)+$start;
		}
		if (!$len)
			$len=$this->strlen($str)-$start;
		return preg_match('/^.{'.$start.'}(.{0,'.$len.'})/us',$str,$match)?
			$match[1]:FALSE;
	}
