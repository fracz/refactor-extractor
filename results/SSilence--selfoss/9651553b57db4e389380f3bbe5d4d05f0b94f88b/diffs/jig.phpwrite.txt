	function write($file,array $data=NULL,$frame=NULL) {
		$fw=\Base::instance();
		switch ($this->format) {
			case self::FORMAT_JSON:
				$out=json_encode($data,@constant('JSON_PRETTY_PRINT'));
				break;
			case self::FORMAT_Serialized:
				$out=$fw->serialize($data);
				break;
		}
		$out=$fw->write($this->dir.$file,$out);
		return $out;
	}
