	public function setDate($date)
	{
		if(! is_numeric($date))
		{
			$date = strtotime($date);
		}

		if($this->version == ATOM)
		{
			$tag    = 'updated';
			$value  = date(DATE_ATOM, $date);
		}
		elseif($this->version == RSS2)
		{
			$tag    = 'pubDate';
			$value  = date(DATE_RSS, $date);
		}
		else
		{
			$tag    = 'dc:date';
			$value  = date("Y-m-d", $date);
		}

		$this->addElement($tag, $value);
	}
