	protected function _mongo($id,$pw,$realm) {
		return (bool)
			$this->mapper->findone(
				array(
					$this->args['id']=>$id,
					$this->args['pw']=>$pw
				)+
				(isset($this->args['realm'])?
					array($this->args['realm']=>$realm):array())
			);
	}
