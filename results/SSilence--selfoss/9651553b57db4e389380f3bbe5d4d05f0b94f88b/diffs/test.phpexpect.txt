	function expect($cond,$text=NULL) {
		$out=(bool)$cond;
		foreach (debug_backtrace() as $frame)
			if (isset($frame['file'])) {
				$this->data[]=array(
					'status'=>$out,
					'text'=>$text,
					'source'=>Base::instance()->
						fixslashes($frame['file']).':'.$frame['line']
				);
				break;
			}
	}
