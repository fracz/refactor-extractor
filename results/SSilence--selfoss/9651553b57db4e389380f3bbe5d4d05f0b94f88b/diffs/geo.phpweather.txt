	function weather($latitude,$longitude) {
		$fw=\Base::instance();
		$web=\Web::instance();
		$query=array(
			'lat'=>$latitude,
			'lng'=>$longitude,
			'username'=>$fw->hash($fw->get('IP'))
		);
		if (($req=$web->request(
			'http://ws.geonames.org/findNearByWeatherJSON?'.
				http_build_query($query))) &&
			$data=json_decode($req['body'],TRUE))
			return $data['weatherObservation'];
		return FALSE;
	}
