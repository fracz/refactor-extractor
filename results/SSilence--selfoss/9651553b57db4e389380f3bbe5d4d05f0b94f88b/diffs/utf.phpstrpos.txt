	function strpos($stack,$needle,$ofs=0,$case=FALSE) {
		preg_match('/^(.*?)'.preg_quote($needle,'/').'/us'.($case?'i':''),
			$this->substr($stack,$ofs),$match);
		return isset($match[1])?$this->strlen($match[1]):FALSE;
	}
