	function paginate($pos=0,$size=10,$filter=NULL,array $options=NULL) {
		return array(
			'subset'=>$this->find($filter,
				array_merge(
					$options?:array(),
					array('limit'=>$size,'offset'=>$pos*$size)
				)
			),
			'count'=>($count=ceil($this->count($filter,$options)/$size)),
			'pos'=>($pos && $pos<$count?$pos:NULL)
		);
	}
