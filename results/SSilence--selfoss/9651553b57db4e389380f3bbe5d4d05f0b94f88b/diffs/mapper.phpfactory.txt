	protected function factory($row) {
		$mapper=clone($this);
		$mapper->reset();
		foreach ($row as $key=>$val)
			$mapper->{array_key_exists($key,$this->fields)?'fields':'adhoc'}
				[$key]['value']=$val;
		return $mapper;
	}
