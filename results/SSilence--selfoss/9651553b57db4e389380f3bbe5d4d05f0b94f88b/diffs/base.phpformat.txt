	function format() {
		$args=func_get_args();
		$val=array_shift($args);
		setlocale(LC_ALL,$this->locales);
		// Get formatting rules
		$conv=localeconv();
		$out=preg_replace_callback(
			'/{(\d+)(?:,(\w+)(?:,(\w+))?)?}/',
			function($expr) use($args,$conv) {
				if (empty($args[$expr[1]]))
					return $expr[0];
				if (isset($expr[2]))
					switch ($expr[2]) {
						case 'number':
							if (isset($expr[3]))
								switch ($expr[3]) {
									case 'integer':
										return
											number_format(
												$args[$expr[1]],0,'',
												$conv['thousands_sep']);
									case 'currency':
										return
											$conv['currency_symbol'].
											number_format(
												$args[$expr[1]],
												$conv['frac_digits'],
												$conv['decimal_point'],
												$conv['thousands_sep']);
									case 'percent':
										return
											number_format(
												$args[$expr[1]]*100,0,
												$conv['decimal_point'],
												$conv['thousands_sep']).'%';
								}
							break;
						case 'date':
							return strftime(empty($expr[3]) ||
								$expr[3]=='short'?'%x':'%A, %d %B %Y',
								$args[$expr[1]]);
						case 'time':
							return strftime('%X',$args[$expr[1]]);
					}
				return $args[$expr[1]];
			},
			$val
		);
		return preg_match('/^win/i',PHP_OS)?
			iconv('Windows-1252',$this->hive['ENCODING'],$out):$out;
	}
