	function reroute($uri) {
		if (PHP_SAPI!='cli') {
			@session_commit();
			header('Location: '.(preg_match('/^https?:\/\//',$uri)?
				$uri:($this->hive['BASE'].$uri)));
			$this->status($this->hive['VERB']=='GET'?301:303);
			die;
		}
		$this->mock('GET '.$uri);
	}
