	protected function getMigrationStub()
	{
		$stub = file_get_contents(__DIR__.'/stubs/migration.stub');

		return str_replace(
			array('sluggable_table', 'sluggable_column'),
			array($this->argument('table'), $this->argument('column')),
			$stub
		);
	}
