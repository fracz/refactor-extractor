	protected function needsSlugging()
	{
		$config = $this->getSluggableConfig();
		$save_to = $config['save_to'];
		$on_update = $config['on_update'];

		if (empty($this->{$save_to})) {
			return true;
		}

		if ($this->isDirty($save_to)) {
			return false;
		}

		return ( !$this->exists || $on_update );
	}
