    private function isAllowed($action, $object, $om)
    {
        $config = $this->getConfiguration($om, get_class($object));

        if (!isset($config['actions'])) {
            return false;
        }

        if (!in_array($action, $config['actions'])) {
            return false;
        }

        return true;
    }
