    private function createLog($action, $object, $om)
    {
        $class = $this->getObjectClass();
        $log = new $class();

        $log->setAction($action);
        $log->setUser(self::$user);
        $log->setObject($object);

        $this->insertLogRecord($om, $log);

        return $log;
    }
