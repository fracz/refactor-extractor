	public function bindScm()
	{
		// Currently only one
		$scm = $this->app['rocketeer.rocketeer']->getOption('scm.scm');
		$scm = 'Rocketeer\Scm\\'.ucfirst($scm);

		$this->app->bind('rocketeer.scm', function ($app) use ($scm) {
			return new $scm($app);
		});
	}
