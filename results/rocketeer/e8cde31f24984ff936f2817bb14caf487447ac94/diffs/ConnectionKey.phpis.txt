    public function is($connection, $server = null)
    {
        if (is_string($connection)) {
            $name = $connection;
            $server = $server ?: $this->server;
        } else {
            $name = $connection->name;
            $server = $connection->server;
        }

        return $this->name === $name && $this->server === $server;
    }
