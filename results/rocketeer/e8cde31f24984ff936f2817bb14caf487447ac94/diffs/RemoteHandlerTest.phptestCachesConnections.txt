    public function testCachesConnections()

                'password' => 'foobar',
            ],
        ]);

        $connection = $this->handler->connection();
        $this->assertInstanceOf(Connection::class, $connection);
        $this->assertEquals('production', $connection->getName());

            'production' => [],
        ]);

        $connection = $this->handler->connection();
        $this->assertInstanceOf(Connection::class, $connection);
        $this->assertEquals('production', $connection->getName());
    }
