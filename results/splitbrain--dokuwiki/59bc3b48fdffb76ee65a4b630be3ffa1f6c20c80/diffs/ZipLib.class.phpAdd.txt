    function Add($files,$compact) {
        if(!is_array($files[0])) $files=Array($files);

        for($i=0;$files[$i];$i++){
            $fn = $files[$i];
            if(!in_Array(dirname($fn[0]),$this->dirs))
                $this->add_Dir(dirname($fn[0]));
            if(utf8_basename($fn[0]))
                $ret[utf8_basename($fn[0])]=$this->add_File($fn[1],$fn[0],$compact);
        }
        return $ret;
    }
