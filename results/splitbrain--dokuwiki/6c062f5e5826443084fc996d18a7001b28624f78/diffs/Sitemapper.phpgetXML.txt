    private function getXML($items) {
        ob_start();
        print '<?xml version="1.0" encoding="UTF-8"?>'.NL;
        print '<urlset xmlns="http://www.sitemaps.org/schemas/sitemap/0.9">'.NL;
        foreach ($items as $item) {
            print $item->toXML();
        }
        print '</urlset>'.NL;
        $result = ob_get_contents();
        ob_end_clean();
        return $result;
    }
