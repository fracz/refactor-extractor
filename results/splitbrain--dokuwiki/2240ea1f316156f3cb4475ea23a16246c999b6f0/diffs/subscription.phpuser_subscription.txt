        global $ID;
        global $conf;

        if(!$id)   $id = $ID;
        if(!$user) $user = $_SERVER['REMOTE_USER'];


        $subs = $this->subscribers($id, $user);
        if(!count($subs)) return false;

        $result = array();
        foreach($subs as $target => $data) {

            );
        }

        return $result;
