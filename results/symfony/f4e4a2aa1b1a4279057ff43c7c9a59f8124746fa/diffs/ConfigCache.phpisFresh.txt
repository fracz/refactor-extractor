    public function isFresh()
    {
        $file = $this->getCacheFile();
        if (!file_exists($file)) {
            return false;
        }

        if (!$this->debug) {
            return true;
        }

        $metadata = $this->getCacheFile('meta');
        if (!file_exists($metadata)) {
            return false;
        }

        $time = filemtime($file);
        $meta = unserialize(file_get_contents($metadata));
        foreach ($meta as $resource) {
            if (!$resource->isFresh($time)) {
                return false;
            }
        }

        return true;
    }
