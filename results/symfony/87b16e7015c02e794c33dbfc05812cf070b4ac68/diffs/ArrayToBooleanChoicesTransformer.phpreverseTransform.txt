    public function reverseTransform($value)
    {
        if (!is_array($value)) {
            throw new UnexpectedTypeException($value, 'array');
        }

        $choices = array();

        foreach ($value as $choice => $selected) {
            if ($selected) {
                $choices[] = $choice;
            }
        }

        return $choices;
    }
