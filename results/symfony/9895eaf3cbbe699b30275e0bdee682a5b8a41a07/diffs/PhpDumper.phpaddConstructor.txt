    protected function addConstructor()
    {
        if (!$this->container->getParameters()) {
            return '';
        }

        return <<<EOF

    /**
     * Constructor.
     */
    public function __construct()
    {
        parent::__construct();

        \$this->parameters = \$this->getDefaultParameters();
    }

EOF;
    }
