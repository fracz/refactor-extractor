    public function isEnabled()
    {
        if (version_compare(phpversion(), '5.4.0', '<') || defined('HHVM_VERSION')) {
            return false;
        }

        if (!extension_loaded('pcntl')) {
            return false;
        }

        return parent::isEnabled();
    }
