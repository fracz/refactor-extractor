    public function get($option)
    {
        $this->reading = true;

        if (!array_key_exists($option, $this->options)) {
            throw new \OutOfBoundsException('The option "' . $option . '" does not exist.');
        }

        if (isset($this->lazy[$option])) {
            $this->resolve($option);
        }

        return $this->options[$option];
    }
