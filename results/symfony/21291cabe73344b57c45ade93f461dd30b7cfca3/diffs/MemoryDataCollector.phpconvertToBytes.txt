    private function convertToBytes($memoryLimit)
    {
        if (preg_match('#^(\d+)([bkmgt])#i', $memoryLimit, $match)) {
            $shift = array('b' => 0, 'k' => 10, 'm' => 20, 'g' => 30, 't' => 40);
            $memoryLimit = ($match[1] * (1 << $shift[strtolower($match[2])]));
        }

        return (int) $memoryLimit;
    }
