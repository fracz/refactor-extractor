    public function getPostMaxSize()
    {
        $iniMax = $this->getNormalizedIniPostMaxSize();

        if ('' === $iniMax) {
            return null;
        }

        if (preg_match('#^(\d+)([bkmgt])#i', $iniMax, $match)) {
            $shift = array('b' => 0, 'k' => 10, 'm' => 20, 'g' => 30, 't' => 40);
            $iniMax = ($match[1] * (1 << $shift[strtolower($match[2])]));
        }

        return (int) $iniMax;
    }
