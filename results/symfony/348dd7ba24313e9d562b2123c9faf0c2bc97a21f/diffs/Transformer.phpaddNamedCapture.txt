    public function addNamedCapture($regExp, $lenght = 1)
    {
        $namedCapture = $this->getNamedCapture();
        $namedCapture = str_repeat($namedCapture, $lenght);
        return '?P<'.$namedCapture.'>' . $regExp;
    }
