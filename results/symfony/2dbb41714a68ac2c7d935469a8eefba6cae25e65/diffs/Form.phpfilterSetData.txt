    public function filterSetData(FilterDataEvent $event)
    {
        $field = $event->getField();

        if (null === $field->getClientTransformer() && null === $field->getNormTransformer()) {
            $data = $event->getData();

            if (empty($data)) {
                $event->setData($this->dataMapper->createEmptyData());
            }
        }
    }
