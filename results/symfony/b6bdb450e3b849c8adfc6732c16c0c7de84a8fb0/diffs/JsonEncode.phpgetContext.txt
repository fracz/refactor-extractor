    private function getContext()
    {
        if (!$this->serializer) {
            return 0;
        }

        $context = $this->serializer->getContext();

        if (empty($context)) {
            $context = array(0);
        }

        if (!is_array($context)) {
            $context = array($context);
        }

        return array_sum($context);
    }
