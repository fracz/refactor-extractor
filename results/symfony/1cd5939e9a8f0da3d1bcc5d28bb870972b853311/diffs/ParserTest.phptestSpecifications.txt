                if (!$yaml) {
                    continue;
                }

                $test = $this->parser->parse($yaml);
                if (isset($test['todo']) && $test['todo']) {
                    // TODO

                    $expected = var_export(eval('return '.trim($test['php']).';'), true);

                    $this->assertEquals($expected, var_export($this->parser->parse($test['yaml']), true), $test['test']);
                }
            }
        }
    }
