    private function replaceBeginStyle($match)
    {
        if (!$this->isDecorated()) {
            return '';
        }

        if (isset($this->styles[strtolower($match[1])])) {
            $style = $this->styles[strtolower($match[1])];
        } else {
            $style = $this->createStyleFromString($match[1]);

            if (false === $style) {
                return $match[0];
            }
        }

        return $style->getBeginStyle();
    }
