    public function getMetadata($key = '', $domain = 'messages')
    {
        if (empty($domain)) {
            return $this->metadata;
        }

        if (isset($this->metadata[$domain])) {
            if (!empty($key)) {
                if (isset($this->metadata[$domain][$key])) {
                    return $this->metadata[$domain][$key];
                }
            } else {
                return $this->metadata[$domain];
            }
        }
    }
