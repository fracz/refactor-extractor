    protected function getSecondsOffset($timezone)
    {
        preg_match('/GMT(?P<signal>[+-])(?P<hours>\d{2}):(?P<minutes>\d{2})/', $timezone, $matches);
        $seconds = ($matches['hours'] * 60 * 60) + ($matches['minutes'] * 60);
        $seconds *= $matches['signal'] == '-' ? -1 : 1;
        return $seconds;
    }
