    private function parseUri($uri)
    {
        if ('http' !== substr($uri, 0, 4)) {
            return array(null, '/');
        }

        $path = parse_url($uri, PHP_URL_PATH);

        return array(preg_replace('#^(.*?//[^/]+)\/.*$#', '$1', $uri), $path);
    }
