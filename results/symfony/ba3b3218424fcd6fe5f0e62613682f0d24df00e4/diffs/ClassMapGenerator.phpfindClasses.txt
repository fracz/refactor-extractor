    {
        $contents = file_get_contents($path);
        $tokens   = token_get_all($contents);

        $classes = array();

        $namespace = '';
        for ($i = 0, $max = count($tokens); $i < $max; $i++) {
            $token = $tokens[$i];

            if (is_string($token)) {
                continue;
            }

            $class = '';

            switch ($token[0]) {
                case T_NAMESPACE:
                    $namespace = '';
                    // If there is a namespace, extract it
                    while (($t = $tokens[++$i]) && is_array($t)) {
                        if (in_array($t[0], array(T_STRING, T_NS_SEPARATOR))) {
                            $namespace .= $t[1];
                        }
                    }
                    $namespace .= '\\';
                    break;
                case T_CLASS:
                case T_INTERFACE:
                    // Find the classname
                    while (($t = $tokens[++$i]) && is_array($t)) {
                        if (T_STRING === $t[0]) {
                            $class .= $t[1];
                        } else if ($class !== '' && T_WHITESPACE == $t[0]) {
                            break;
                        }
                    }

                    if (empty($namespace)) {
                        $classes[] = $class;
                    } else {
                        $classes[] = $namespace . $class;
                    }
                    break;
                default:
                    break;
            }
        }

        return $classes;
    }
