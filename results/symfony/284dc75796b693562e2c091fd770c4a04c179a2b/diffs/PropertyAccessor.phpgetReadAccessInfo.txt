                $access[self::ACCESS_TYPE] = self::ACCESS_TYPE_PROPERTY;
                $access[self::ACCESS_NAME] = $property;

                $access[self::ACCESS_NAME] = $getter;

                    $reflClass->name

            $this->readPropertyCache[$key] = $access;
        }
