    public function write($data)
    {
        $unpack = unpack('H*', serialize($data));
        $data = $unpack[1];

        $db = $this->initDb(false);
        $args = array(
            ':token' => $this->token,
            ':data' => (string) $data,
            ':time' => time()
        );
        $this->exec($db, 'INSERT INTO data (token, data, created_at) VALUES (:token, :data, :time)', $args);
        $this->close($db);
    }
