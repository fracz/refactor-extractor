    private function resolveValue($value)
    {
        if (is_array($value)) {
            $resolved = array();
            foreach ($value as $k => $v) {
                $resolved[$this->resolveValue($k)] = $this->resolveValue($v);
            }

            return $resolved;
        } else if (is_string($value)) {
            return $this->resolveString($value);
        }

        return $value;
    }
