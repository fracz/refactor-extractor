    public function getChildren()
    {
        if (null === $this->children) {
            $this->children = array();
            foreach ($this->storage->findChildren($this->token) as $token) {
                $this->children[] = $this->loadFromToken($token['token']);
            }
        }

        return $this->children;
    }
