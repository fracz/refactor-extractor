    public function setToken($token)
    {
        $this->token = $token;

        if (false !== $items = $this->storage->read($token)) {
            list($data, $this->parent, $this->ip, $this->url, $this->time) = $items;
            $this->set(unserialize(base64_decode($data)));

            $this->empty = false;
        } else {
            $this->empty = true;
        }
    }
