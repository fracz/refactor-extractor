    private function extractReturns($n, TokenCollection $tokens) {
        $returns =array();
        $len = sizeof($tokens, COUNT_NORMAL);
        for($i = $n; $i < $len; $i++) {
            $token = $tokens[$i];
            if(T_RETURN === $token->getType()) {
                array_push($returns, null);
            }
        }
        return $returns;
    }
