    public function select($column, $table, array &$criteria) {
        $where = $criteria ? "where %s" : '';
        $query = "select %s from %s $where";
        $params = array();
        foreach ($criteria as $k => $v) {
            if ($v === null) {
                $params[] = "$k IS NULL ";
                unset($criteria[$k]);
            } else {
                $params[] = "$k = ? ";
            }
        }
        $params = implode('AND ', $params);

        return sprintf($query, $column, $table, $params);
    }
