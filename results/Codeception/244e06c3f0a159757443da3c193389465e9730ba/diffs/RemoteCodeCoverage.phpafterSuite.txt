    public function afterSuite(SuiteEvent $e)
    {
        if (!$this->module or !$this->remote) {
            return;
        }

        $suite = $e->getSuite()->getName();
        if ($this->options['xml']) {
            $this->retrieveAndPrintXml($suite);
        }
        if ($this->options['html']) {
            $this->retrieveAndPrintHtml($suite);
        }
    }
