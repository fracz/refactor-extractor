    protected function buildPath($basePath, $testName)
    {
        $testName = str_replace('/',DIRECTORY_SEPARATOR, $testName);
        $dirs = explode(DIRECTORY_SEPARATOR, $testName);
        array_pop($dirs);

        $path = $basePath;
        foreach ($dirs as $dir) {
            $path .= DIRECTORY_SEPARATOR.$dir;
            @mkdir($path);
        }
        return $path;
    }
