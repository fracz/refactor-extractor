    protected static function regex($namespace, $suffix)
    {
        $namespace = str_replace("\\", '\\\\', $namespace);
        if ($namespace) {
            return sprintf('~\\\\?%s\\\\\w*?%s$~', $namespace, $suffix);
        } else {
            return sprintf('~\w*?%s$~', $suffix);
        }
    }
