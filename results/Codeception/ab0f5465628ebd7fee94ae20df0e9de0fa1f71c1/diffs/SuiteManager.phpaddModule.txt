    public static function addModule($modulename, $path = null) {
        if ($path) require_once $path;
        if (!($modulename instanceof \Codeception\Module)) {
            $module = new $modulename;
        } else {
            $module = $modulename;
            $modulename = get_class($module);
        }
        self::$modules[$modulename] = $module;
        return $module;
    }
