    private function mergeOptions($options) {

        foreach ($this->options as $option => $default) {
            if (!isset($options[$option])) {
                $options[$option] = $default;
            }
            if (isset($this->config['settings'][$option])) {
                $options[$option] = $this->config['settings'][$option];
            }
        }
        return $options;
    }
