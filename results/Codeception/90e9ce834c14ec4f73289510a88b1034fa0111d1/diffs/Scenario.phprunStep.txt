    public function runStep()
    {
        if (empty($this->steps)) return;
        $this->currentStep++;
        $step = $this->lastStep();
        if (!$step->executed) {
            $result = $this->test->runStep($step);
            $step->executed = true;
            return $result;
        }
    }
