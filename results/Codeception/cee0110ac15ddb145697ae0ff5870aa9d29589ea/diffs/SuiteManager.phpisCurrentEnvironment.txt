    protected function isCurrentEnvironment($envs)
    {
        if (empty($envs)) {
            return true;
        }
        if (!$this->env) {
            return false;
        }

        $currentEnvironments = explode(',', $this->env);
        foreach ($envs as $envList) {
            $envList = explode(',', $envList);
            if (count($envList) == count(array_intersect($currentEnvironments, $envList))) {
                return true;
            }
        }
        return false;
    }
