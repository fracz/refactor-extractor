	public function checkAuthUser($authkey) {
		$result = false;
		$user = ClassRegistry::init('User')->findByAuthkey($authkey);
		if (isset($user['User'])) {
			$role = ClassRegistry::init('Role')->findById($user['User']['role_id']);
			if ($role['Role']['perm_auth']) {
				$result = true;
			}
		}
		return $result;
	}
