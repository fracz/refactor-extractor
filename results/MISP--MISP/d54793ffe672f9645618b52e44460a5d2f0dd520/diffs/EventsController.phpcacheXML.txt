	public function cacheXML() {
		if ($this->_isSiteAdmin()) $isSiteAdmin = '1';
		else $isSiteAdmin = '0';
		CakeResque::enqueue(
			'default',
			'EventShell',
			array('cachexml', $this->Auth->user('org'), $this->_isSiteAdmin())
		);

	}
