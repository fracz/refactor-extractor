    public function setFiles(array $files)
    {
        $this->files = array();
        foreach ($files as $key => $file)
        {
            if (!is_array($file)) {
                $file = array('content' => $file);
            }

            $file = array_merge(array(
                'content'   => null,
                'mtime'     => null,
                'checksum'  => null
            ), $file);

            $this->setFile($key, $file['content'], $file['mtime'], $file['checksum']);
        }
    }
