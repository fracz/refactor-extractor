    public function listDirectory($directory)
    {
        $keys = array();
        $files = $this->parseRawlist(
            ftp_rawlist($this->getConnection(), $directory) ? : array()
        );

        foreach ($files as $file) {
            if ('-' === substr($file['perms'], 0, 1)) {
                $keys[] = trim($directory . '/' . $file['name'], '/');
            }
        }

        return $keys;
    }
