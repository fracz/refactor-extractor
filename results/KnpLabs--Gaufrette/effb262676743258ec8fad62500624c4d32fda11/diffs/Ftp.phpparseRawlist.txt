    public function parseRawlist(array $rawlist)
    {
        $parsed = array();
        foreach ($rawlist as $line) {
            $infos = preg_split("/[\s]+/", $line, 9);
            if ('total' !== $infos[0]) {
                $parsed[] = array(
                    'perms' => $infos[0],
                    'num'   => $infos[1],
                    'owner' => $infos[2],
                    'group' => $infos[3],
                    'size'  => $infos[4],
                    'month' => $infos[5],
                    'day'   => $infos[6],
                    'time'  => $infos[7],
                    'name'  => $infos[8]
                );
            }
        }

        return $parsed;
    }
