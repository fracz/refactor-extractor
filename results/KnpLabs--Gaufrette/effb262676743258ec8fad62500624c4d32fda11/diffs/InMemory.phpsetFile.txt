    public function setFile($key, $content = null, $mtime = null, $checksum = null)
    {
        if (null === $mtime) {
            $mtime = time();
        }

        if (null === $checksum) {
            $checksum = Checksum::fromContent($content);
        }

        $this->files[$key] = array(
            'content'   => (string)  $content,
            'mtime'     => (integer) $mtime,
            'checksum'  => (string)  $checksum
        );
    }
