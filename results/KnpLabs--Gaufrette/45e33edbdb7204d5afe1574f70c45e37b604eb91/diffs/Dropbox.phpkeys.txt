    public function keys()
    {
        $metadata = $this->client->getMetaData('/', true);
        $files    = isset($metadata['contents']) ? $metadata['contents'] : array();

        $files = array_map(
            function($value) {
                return ltrim($value['path'], '/');
            },
            $files
        );

        $dirs = array();
        foreach ($files as $file)
        {
            if ('.' !== dirname($file)) {
                $dirs[] = dirname($file);
            }
        }

        $keys = array_merge($files, $dirs);
        sort($keys);

        return $keys;
    }
