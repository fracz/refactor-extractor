    public function directoryExists($directory)
    {
        if (!$this->isConnected()) {
            $this->connect();
        }

        if (!@ftp_chdir($this->connection, $directory)) {
            return false;
        }

        // change directory again to return in the base directory
        @ftp_chdir($this->connection, $this->directory);

        return true;
    }
