    public function keys()
    {
        $metadata = $this->client->getMetaData('/', true);
        $contents    = isset($metadata['contents']) ? $metadata['contents'] : array();

        $keys = array();
        foreach ($contents as $value) {
            $file = ltrim($value['path'], '/');
            $keys[] = $file;
            if ('.' !== dirname($file)) {
                $keys[] = dirname($file);
            }
        }
        sort($keys);

        return $keys;
    }
