  public static function loadActiveEngines() {
    $engines = self::loadAllEngines();

    $active = array();
    foreach ($engines as $key => $engine) {
      if (!$engine->isEnabled()) {
        continue;
      }

      $active[$key] = $engine;
    }

    return $active;
  }
