    AphrontRequest $request) {

    $object = self::newFromCallsign($callsign, $request->getUser());

    $use_branches = $object->supportsBranches();
    $parsed = self::parseRequestBlob(idx($data, 'dblob'), $use_branches);

    $object->setUser($request->getUser());
