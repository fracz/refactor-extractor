  final public function run() {
    while (true) {
      // Reload the repository every time to pick up changes from the web
      // console.
      $this->repository = $this->loadRepository();
      $this->discoverCommits();

      $sleep = max(2, $this->getRepository()->getDetail('pull-frequency'));
      $this->sleep($sleep);
    }
