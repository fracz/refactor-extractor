  protected function loadRepository() {
    $argv = $this->getArgv();
    if (count($argv) !== 1) {
      throw new Exception("No repository PHID provided!");
    }

    $repository = id(new PhabricatorRepository())->loadOneWhere(
      'phid = %s',
      $argv[0]);

    if (!$repository) {
      throw new Exception("No such repository exists!");
    }

    return $repository;
  }
