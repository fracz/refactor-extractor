  public function getFileIcon($filename) {
    $path_info = pathinfo($filename);
    $extension = $path_info['extension'];
    switch ($extension) {
      case 'psd':
      case 'ai':
        $icon = 'preview';
        break;
      case 'conf':
        $icon = 'wrench';
        break;
      case 'wav':
      case 'mp3':
      case 'aiff':
        $icon = 'music';
        break;
      case 'm4v':
      case 'mov':
        $icon = 'film';
        break;
      case 'sql';
      case 'db':
      case 'csv':
        $icon = 'data';
        break;
      case 'ics':
        $icon = 'calendar';
        break;
      case 'zip':
      case 'tar':
      case 'bz':
      case 'tgz':
      case 'gz':
        $icon = 'zip';
        break;
      case 'png':
      case 'jpg':
      case 'bmp':
      case 'gif':
        $icon = 'image';
        break;
      default:
        $icon = 'file';
        break;
    }
    return $icon;
  }
