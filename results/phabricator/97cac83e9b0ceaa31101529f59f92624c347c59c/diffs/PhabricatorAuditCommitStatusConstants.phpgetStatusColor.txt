  public static function getStatusColor($code) {
    switch ($code) {
      case self::CONCERN_RAISED:
        $color = 'red';
        break;
      case self::NEEDS_AUDIT:
        $color = 'orange';
        break;
      case self::PARTIALLY_AUDITED:
        $color = 'yellow';
        break;
      case self::FULLY_AUDITED:
        $color = 'green';
        break;
      case self::NONE:
        $color = 'bluegrey';
        break;
      default:
        $color = null;
        break;
    }
    return $color;
  }
