  private function filterSendable($value, $phids) {
    $result = array();
    foreach ($value as $phid) {
      if (isset($phids[$phid]) && $phids[$phid]['mailable']) {
        $result[$phid] = $phids[$phid]['email'];
      }
    }
    return $result;
  }
