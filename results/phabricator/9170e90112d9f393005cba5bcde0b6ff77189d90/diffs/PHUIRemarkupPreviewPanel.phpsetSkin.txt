  public function setSkin($skin) {
    static $skins = array(
      'default' => true,
      'document' => true,
    );

    if (empty($skins[$skin])) {
      throw new Exception(
        pht(
          "Invalid skin '%s'. Valid skins are: %s.",
          $skin,
          implode(', ', array_keys($skins))));
    }

    $this->skin = $skin;
    return $this;
  }
