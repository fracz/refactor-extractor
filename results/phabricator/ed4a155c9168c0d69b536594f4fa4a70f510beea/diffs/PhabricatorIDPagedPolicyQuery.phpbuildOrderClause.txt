  final protected function buildOrderClause(AphrontDatabaseConnection $conn_r) {
    if ($this->beforeID) {
      return qsprintf(
        $conn_r,
        'ORDER BY %Q ASC',
        $this->getPagingColumn());
    } else {
      return qsprintf(
        $conn_r,
        'ORDER BY %Q DESC',
        $this->getPagingColumn());
    }
  }
