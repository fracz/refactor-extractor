  public function getHash() {
    if ($this->hash) {
      return $this->hash;
    }

    $parts = array(
      $this->getType(),
      $this->getName(),
      $this->getFile(),
      $this->getLine(),
      $this->getLength(),
      $this->getLanguage(),
      $this->getContentRaw(),
      $this->getDocblockRaw(),
      mpull($this->extends, 'toHash'),
      mpull($this->links, 'toHash'),
    );

    return md5(serialize($parts)).'N';
  }
