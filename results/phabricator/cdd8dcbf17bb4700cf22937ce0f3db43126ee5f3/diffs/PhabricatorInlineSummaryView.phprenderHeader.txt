  private function renderHeader() {
    $icon = id(new PHUIIconView())
      ->setIconFont('fa-comment bluegrey msr');

    $header = phutil_tag_div(
      'phabricator-inline-summary',
      array(
        $icon,
        pht('Inline Comments'),
      ));
    return $header;
  }
