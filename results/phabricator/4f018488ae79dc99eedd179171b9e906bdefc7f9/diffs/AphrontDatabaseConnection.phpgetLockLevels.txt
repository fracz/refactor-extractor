  protected function &getLockLevels() {
    static $levels = array();
    $key = $this->getTransactionKey();
    if (!isset($levels[$key])) {
      $levels[$key] = array(
        'read'  => 0,
        'write' => 0,
      );
    }

    return $levels[$key];
  }
