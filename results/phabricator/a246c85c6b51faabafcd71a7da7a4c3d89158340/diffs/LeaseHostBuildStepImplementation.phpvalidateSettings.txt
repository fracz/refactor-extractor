  public function validateSettings() {
    $settings = $this->getSettings();

    if ($settings['name'] === null || !is_string($settings['name'])) {
      return false;
    }
    if ($settings['platform'] === null || !is_string($settings['platform'])) {
      return false;
    }

    return true;
  }
