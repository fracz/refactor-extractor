  public function getMessageText() {
    switch ($this->getCommand()) {
      case 'PRIVMSG':
        $matches = null;
        $raw = $this->getRawData();
        if (preg_match('/^\S+\s+:?(.*)$/', $raw, $matches)) {
          return rtrim($matches[1], "\r\n");
        }
        break;
    }
    return null;
  }
