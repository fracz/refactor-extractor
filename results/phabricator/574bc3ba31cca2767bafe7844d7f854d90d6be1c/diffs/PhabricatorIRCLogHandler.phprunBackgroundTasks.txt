  public function runBackgroundTasks() {
    foreach ($this->futures as $key => $future) {
      try {
        if ($future->isReady()) {
          unset($this->futures[$key]);
        }
      } catch (Exception $ex) {
        unset($this->futures[$key]);
        phlog($ex);
      }
    }
  }
