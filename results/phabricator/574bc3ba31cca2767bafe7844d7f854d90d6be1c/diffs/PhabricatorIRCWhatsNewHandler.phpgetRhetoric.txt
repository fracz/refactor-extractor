  public function getRhetoric($input) {
    switch ($input) {
      case 'comment':
      case 'none':
        return 'commented on';
        break;
      case 'update':
        return 'updated';
        break;
      case 'commit':
        return 'closed';
        break;
      case 'create':
        return 'created';
        break;
      case 'concern':
        return 'raised concern for';
        break;
      case 'abandon':
        return 'abandonned';
        break;
      case 'accept':
        return 'accepted';
        break;
      default:
        return $input;
        break;
    }
  }
