  private function isCommentField(PhabricatorEditField $field) {
    // TODO: This is a little bit hacky.
    if ($field->getKey() == 'comment') {
      return true;
    }

    if ($field->getIsLocked()) {
      return false;
    }

    if ($field->getIsHidden()) {
      return false;
    }

    return true;
  }
