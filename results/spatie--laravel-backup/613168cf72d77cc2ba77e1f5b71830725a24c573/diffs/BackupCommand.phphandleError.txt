    protected function handleError(Throwable $error)
    {
        $backupHasFailedEvent = new BackupHasFailed($error);

        $this->getNotificationHandler()->whenBackupHasFailed($backupHasFailedEvent);

        event($backupHasFailedEvent);
    }
