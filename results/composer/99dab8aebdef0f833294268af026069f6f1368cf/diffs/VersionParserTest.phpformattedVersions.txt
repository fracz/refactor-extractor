                'sourceReference' => 'v1.0.0',
                'truncate' => false,

            ),
            array(

                'truncate' => false,
                'expected' => 'PrettyVersion bbf527a27356414bfa9bf520f018c5cb7af67c77'
            ),
        );

        $self = $this;
