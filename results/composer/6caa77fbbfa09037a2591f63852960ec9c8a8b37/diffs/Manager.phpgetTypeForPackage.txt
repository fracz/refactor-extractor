    private function getTypeForPackage(PackageInterface $package)
    {
        if ($package->getDistType()) {
            $type = 'dist';
        } elseif ($package->getSourceType()) {
            $type = 'source';
        } else {
            throw new \UnexpectedValueException(
                'Package '.$package->getName().' has no source or dist URL.'
            );
        }

        return $type;
    }
