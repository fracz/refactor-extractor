    public function equals(Rule $rule)
    {
        if ($this->ruleHash !== $rule->ruleHash) {
            return false;
        }

        if (count($this->literals) != count($rule->literals)) {
            return false;
        }

        for ($i = 0, $n = count($this->literals); $i < $n; $i++) {
            if (!$this->literals[$i]->getId() === $rule->literals[$i]->getId()) {
                return false;
            }
        }

        return true;
    }
