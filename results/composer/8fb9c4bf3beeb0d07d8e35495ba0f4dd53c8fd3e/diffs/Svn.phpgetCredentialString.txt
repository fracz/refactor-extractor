    public function getCredentialString()
    {
        if ($this->hasAuth === null) {
            $this->hasAuth();
        }
        if (!$this->hasAuth) {
            return '';
        }
        return sprintf(
            ' %s--username %s --password %s ',
            $this->getAuthCache(),
            escapeshellarg($this->getUsername()),
            escapeshellarg($this->getPassword())
        );
    }
