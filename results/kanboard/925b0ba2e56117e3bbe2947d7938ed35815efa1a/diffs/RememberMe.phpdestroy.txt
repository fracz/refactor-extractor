    public function destroy($user_id)
    {
        $credentials = $this->readCookie();

        if ($credentials !== false) {

            $this->deleteCookie();

            $this->db
                 ->table(self::TABLE)
                 ->eq('user_id', $user_id)
                 ->eq('token', $credentials['token'])
                 ->remove();
        }
    }
