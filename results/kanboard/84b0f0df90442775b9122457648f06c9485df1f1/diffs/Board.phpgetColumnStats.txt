    public function getColumnStats($project_id)
    {
        return $this->db
                    ->table(Task::TABLE)
                    ->eq('project_id', $project_id)
                    ->eq('is_active', 1)
                    ->groupBy('column_id')
                    ->listing('column_id', 'COUNT(*) AS total');
