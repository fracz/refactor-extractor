    public function __get($name)
    {
        if (isset($this->container[$name])) {

            if (is_callable($this->container[$name])) {
                return $this->container[$name]();
            }
            else {
                return $this->container[$name];
            }
        }

        throw new \RuntimeException('Identifier not found in the registry: '.$name);
    }
