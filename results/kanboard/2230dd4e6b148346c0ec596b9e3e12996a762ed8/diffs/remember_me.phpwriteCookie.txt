    public function writeCookie($token, $sequence, $expiration)
    {
        setcookie(
            self::COOKIE_NAME,
            $this->encodeCookie($token, $sequence),
            $expiration,
            BASE_URL_DIRECTORY,
            null,
            ! empty($_SERVER['HTTPS']),
            true
        );
    }
