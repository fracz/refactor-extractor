    public function getAllowedUsers($project_id)
    {
        return $this->db
            ->table(self::TABLE_USERS)
            ->join(User::TABLE, 'id', 'user_id')
            ->eq('project_id', $project_id)
            ->asc('username')
            ->listing('user_id', 'username');
    }
