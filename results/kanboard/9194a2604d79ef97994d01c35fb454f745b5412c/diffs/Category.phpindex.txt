    public function index()
    {
        $project = $this->getProject();

        $this->response->html($this->template->layout('category_index', array(
            'categories' => $this->category->getList($project['id'], false),
            'values' => array('project_id' => $project['id']),
            'errors' => array(),
            'project' => $project,
            'menu' => 'projects',
            'title' => t('Categories')
        )));
    }
