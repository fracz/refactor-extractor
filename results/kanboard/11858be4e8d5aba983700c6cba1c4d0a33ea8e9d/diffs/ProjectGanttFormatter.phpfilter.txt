    public function filter(array $project_ids)
    {
        if (empty($project_ids)) {
            $this->projects = array();
        } else {
            $this->projects = $this->db
                ->table(self::TABLE)
                ->asc('start_date')
                ->in('id', $project_ids)
                ->eq('is_active', self::ACTIVE)
                ->eq('is_private', 0)
                ->findAll();
        }

        return $this;
    }
