    public function formRadios($name, array $options, array $values = array())
    {
        $html = '';

        foreach ($options as $value => $label) {
            $html .= $this->formRadio($name, $label, $value, isset($values[$name]) && $values[$name] == $value);
        }

        return $html;
    }
