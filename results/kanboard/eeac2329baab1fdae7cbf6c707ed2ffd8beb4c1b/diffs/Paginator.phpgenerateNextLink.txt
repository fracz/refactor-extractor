    public function generateNextLink()
    {
        $html = '<span class="pagination-next">';

        if (($this->total - $this->offset) > $this->limit) {
            $html .= $this->container['helper']->a(
                t('Next').' &rarr;',
                $this->controller,
                $this->action,
                $this->getUrlParams($this->page + 1, $this->order, $this->direction)
            );
        }
        else {
            $html .= t('Next').' &rarr;';
        }

        $html .= '</span>';

        return $html;
    }
