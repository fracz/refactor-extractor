    public function paginate($offset = 0, $limit = 25, $column = 'username', $direction = 'ASC')
    {
        return $this->db
                    ->table(self::TABLE)
                    ->columns(
                        'id',
                        'username',
                        'name',
                        'email',
                        'is_admin',
                        'default_project_id',
                        'is_ldap_user',
                        'notifications_enabled',
                        'google_id',
                        'github_id'
                    )
                    ->offset($offset)
                    ->limit($limit)
                    ->orderBy($column, $direction)
                    ->findAll();
    }
