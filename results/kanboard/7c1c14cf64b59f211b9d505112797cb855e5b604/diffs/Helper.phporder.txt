    public function order($label, $column, array $pagination)
    {
        extract($pagination);

        $prefix = '';

        if ($order === $column) {
            $prefix = $direction === 'DESC' ? '&#9660; ' : '&#9650; ';
            $direction = $direction === 'DESC' ? 'ASC' : 'DESC';
        }

        $order = $column;

        return $prefix.$this->a($label, $controller, $action, $params + compact('offset', 'order', 'direction'));
    }
