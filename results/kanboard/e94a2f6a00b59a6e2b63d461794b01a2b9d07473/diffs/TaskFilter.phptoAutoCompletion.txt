    public function toAutoCompletion()
    {
        return $this->query->columns('id', 'title')->filter(function(array $results) {

            foreach ($results as &$result) {
                $result['value'] = $result['title'];
                $result['label'] = '#'.$result['id'].' - '.$result['title'];
            }

            return $results;

        })->findAll();
    }
