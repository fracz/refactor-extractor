    public static function fetchAll($limit = 8)
    {
        $topic_ids = self::orderBy('weight', 'DESC')
                         ->limit($limit)
                         ->lists('topic_id')
                         ->toArray();
        return Topic::whereIn('id', $topic_ids)
                    ->with('user')
                    ->orderBy(\DB::raw('FIELD(id, '.implode(',', $topic_ids).')'))
                    ->get();
    }
