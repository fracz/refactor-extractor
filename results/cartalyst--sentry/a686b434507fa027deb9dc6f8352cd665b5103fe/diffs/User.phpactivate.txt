	public function activate($login, $activationCode)
	{
		$user = $this->findByLogin($login);

		if ($user and $activationCode === $user->activation_hash)
		{
			$user->activation_hash = null;
			$user->activated = 1;
			$user->save();

			return true;
		}

		return false;
	}
