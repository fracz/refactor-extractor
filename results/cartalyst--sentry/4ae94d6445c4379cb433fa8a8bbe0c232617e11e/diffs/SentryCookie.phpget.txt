    public function get($key, $default = null)
    {
        if (isset($_COOKIE[$key])) {
            return $_COOKIE[$key];
        } elseif(!is_null($default)) {
            return $default;
        } else {
            throw new SentryCookieException("Requested cookie doesn't exist!");
        }
    }
