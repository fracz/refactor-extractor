    private function parseResponse(
        CommandInterface $command,
        ResponseInterface $response,
        $uri
    ) {
        $parser = $this->parser;
        $result = $parser($command, $response);
        $result['@effectiveUri'] = (string) $uri;
        $result['@status'] = $response->getStatusCode();

        return $result;
    }
