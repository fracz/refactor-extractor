    public function generatePart()
    {
        if (!$this->uploadPart) {
            $this->uploadPart = UploadPart::fromArray(array(
                'partNumber'  => (int) ($this->offset / $this->maxSize + 1),
                'checksum'    => $this->treeHash->getHash(),
                'contentHash' => $this->chunkHash->getHash(),
                'size'        => $this->size,
                'offset'      => $this->offset
            ));
        }

        return $this->uploadPart;
    }
