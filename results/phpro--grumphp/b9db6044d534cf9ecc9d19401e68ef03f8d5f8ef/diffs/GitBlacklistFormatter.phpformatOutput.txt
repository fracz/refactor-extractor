    private function formatOutput($output)
    {
        $result = "\033[m";
        foreach (array_filter(explode(PHP_EOL, $output)) as $lineNumber => $line) {
            if (preg_match('/^[0-9]+/', $line)) {
                $result .= $this->trimOutputLine($line, $lineNumber) . PHP_EOL;
            } else {
                $result .= $line . PHP_EOL;
            }
        }
        return trim($result);
    }
