	protected function _init(Elastica_Response $response) {
		$this->_response = $response;
		$result = $response->getResponse();

		if (isset($result['hits']['hits'])) {
			foreach ($result['hits']['hits'] as $hit) {
				$this->_results[] = new Elastica_Result($hit);
			}
		}
	}
