	public function _addQuery($type, $args) {
		if ($args instanceof Elastica_Query_Abstract) {
			$args = $args->toArray();
		}

		if (!is_array($args)) {
			throw new Elastica_Exception_Invalid('Invalid parameter. Has to be array or instance of Elastica_Query');
		}

		$varName = '_' . $type;
		$this->{$varName}[] = $args;
		return $this;
	}
