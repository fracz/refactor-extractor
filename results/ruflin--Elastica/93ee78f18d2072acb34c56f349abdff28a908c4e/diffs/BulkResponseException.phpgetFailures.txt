    public function getFailures()
    {
        $errors = array();
        foreach ($this->getResponseSet()->getBulkResponses() as $bulkResponse) {
            if ($bulkResponse->hasError()) {
                $error = array(
                    'action' => $bulkResponse->getOpType(),
                ) + $bulkResponse->getData();
                $errors[] = $error;
            }
        }

        return $errors;
    }
