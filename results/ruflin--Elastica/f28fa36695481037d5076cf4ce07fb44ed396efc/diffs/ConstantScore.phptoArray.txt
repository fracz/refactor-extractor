	public function toArray() {
		$ret = array(
			'constant_score' => array(
				'filter' => (
					$this->_filter instanceof Elastica_Filter_Abstract
					? $this->_filter->toArray()
					: $this->_filter
				),
			)
		);

		if (! is_null($this->_boost)) {
			$ret['boost'] = $this->_boost;
		}

		return $ret;
	}
