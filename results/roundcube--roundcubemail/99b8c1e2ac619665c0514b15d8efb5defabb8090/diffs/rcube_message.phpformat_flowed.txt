    public static function format_flowed($text, $length = 72)
    {
        $out = '';

        foreach (preg_split('/\r?\n/', trim($text)) as $line) {
            // don't wrap quoted lines (to avoid wrapping problems)
            if ($line[0] != '>')
                $line = rc_wordwrap(rtrim($line, "\r\n"), $length - 1, " \r\n");

            $out .= $line . "\r\n";
        }

        return $out;
    }
