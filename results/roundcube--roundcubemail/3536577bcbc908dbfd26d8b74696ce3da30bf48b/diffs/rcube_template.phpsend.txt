    public function send($templ = null, $exit = true)
    {
        if ($templ != 'iframe') {
            $this->parse($templ, false);
        }
        else {
            $this->framed = $templ == 'iframe' ? true : $this->framed;
            $this->write();
        }

        if ($exit) {
            exit;
        }
    }
