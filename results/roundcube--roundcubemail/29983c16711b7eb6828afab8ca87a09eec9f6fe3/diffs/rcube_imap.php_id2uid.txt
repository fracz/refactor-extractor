    private function _id2uid($id, $mbox_name=NULL)
    {
        if (!$mbox_name)
            $mbox_name = $this->mailbox;

        if ($uid = array_search($id, (array)$this->uid_id_map[$mbox_name]))
            return $uid;

        $uid = $this->conn->ID2UID($mbox_name, $id);
        $this->uid_id_map[$mbox_name][$uid] = $id;

        return $uid;
    }
