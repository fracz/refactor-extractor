    private function putLine($string, $endln=true)
    {
        if (!$this->fp)
            return false;

		if (!empty($this->prefs['debug_mode'])) {
    		write_log('imap', 'C: '. rtrim($string));
	    }

        return fputs($this->fp, $string . ($endln ? "\r\n" : ''));
    }
