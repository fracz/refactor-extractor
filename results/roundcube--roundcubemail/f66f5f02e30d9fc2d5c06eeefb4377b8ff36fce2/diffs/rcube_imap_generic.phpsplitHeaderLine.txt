    private function splitHeaderLine($string)
    {
        $pos = strpos($string, ':');
        if ($pos>0) {
            $res[0] = substr($string, 0, $pos);
            $res[1] = trim(substr($string, $pos+1));
            return $res;
        }
        return $string;
    }
