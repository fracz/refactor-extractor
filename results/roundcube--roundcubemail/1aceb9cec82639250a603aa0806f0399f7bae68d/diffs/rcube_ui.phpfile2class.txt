    public static function file2class($mimetype, $filename)
    {
        list($primary, $secondary) = explode('/', $mimetype);

        $classes = array($primary ? $primary : 'unknown');
        if ($secondary) {
            $classes[] = $secondary;
        }
        if (preg_match('/\.([a-z0-9]+)$/i', $filename, $m)) {
            $classes[] = $m[1];
        }

        return strtolower(join(" ", $classes));
    }
