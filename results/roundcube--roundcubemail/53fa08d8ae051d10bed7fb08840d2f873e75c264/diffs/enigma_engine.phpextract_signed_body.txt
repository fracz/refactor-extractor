    private function extract_signed_body($body, $boundary)
    {
        $boundary     = '--' . $boundary;
        $boundary_len = strlen($boundary) + 2;
        $start        = strpos($body, $boundary) + $boundary_len;
        $end          = strpos($body, $boundary, $start);
        $body         = substr($body, $start, $end - $start - 2);

        return $body;
    }
