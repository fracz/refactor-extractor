    private function parseNamespace($str, &$i, $len=0, $l)
    {
	    if (!$l) {
	        $str = str_replace('NIL', '()', $str);
	    }
	    if (!$len) {
	        $len = strlen($str);
	    }
	    $data      = array();
	    $in_quotes = false;
	    $elem      = 0;

        for ($i; $i<$len; $i++) {
		    $c = (string)$str[$i];
		    if ($c == '(' && !$in_quotes) {
			    $i++;
			    $data[$elem] = $this->parseNamespace($str, $i, $len, $l++);
			    $elem++;
		    } else if ($c == ')' && !$in_quotes) {
			    return $data;
    		} else if ($c == '\\') {
			    $i++;
			    if ($in_quotes) {
				    $data[$elem] .= $str[$i];
        		}
		    } else if ($c == '"') {
			    $in_quotes = !$in_quotes;
			    if (!$in_quotes) {
				    $elem++;
        		}
		    } else if ($in_quotes) {
			    $data[$elem].=$c;
		    }
	    }

        return $data;
    }
