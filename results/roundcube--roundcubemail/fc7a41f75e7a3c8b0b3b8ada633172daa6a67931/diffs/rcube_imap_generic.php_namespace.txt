    function _namespace()
    {
        if (!$this->getCapability('NAMESPACE')) {
	        return false;
	    }

	    list($code, $response) = $this->execute('NAMESPACE');

		if ($code == self::ERROR_OK && preg_match('/^\* NAMESPACE /', $response)) {
	        $data = $this->parseNamespace(substr($response, 11), $i, 0, 0);
		}

	    if (!is_array($data)) {
	        return false;
	    }

        return $data;
    }
