    function search_once($mailbox='', $str=NULL, $ret_uid=false)
    {
        if (!$str)
            return false;

        if (!strlen($mailbox)) {
            $mailbox = $this->mailbox;
        }

        return $this->conn->search($mailbox, $str, $ret_uid);
    }
