  function remove_message_cache($key, $uid)
    {
    if (!$this->caching_enabled)
      return;

    $this->db->query(
      "DELETE FROM ".get_table_name('messages')."
       WHERE  user_id=?
       AND    cache_key=?
       AND    uid=?",
      $_SESSION['user_id'],
      $key,
      $uid);
    }
