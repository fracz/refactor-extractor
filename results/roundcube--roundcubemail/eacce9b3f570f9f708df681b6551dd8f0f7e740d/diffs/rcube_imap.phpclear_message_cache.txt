    private function clear_message_cache($key, $start_index=1)
    {
        if (!$this->caching_enabled)
            return;

        $this->db->query(
            "DELETE FROM ".get_table_name('messages').
            " WHERE user_id=?".
            " AND cache_key=?".
            " AND idx>=?",
            $_SESSION['user_id'], $key, $start_index);
    }
