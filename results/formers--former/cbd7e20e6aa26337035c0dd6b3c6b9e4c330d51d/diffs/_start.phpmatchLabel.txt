  protected function matchLabel($name = 'foo', $field = 'foo', $required = false)
  {
    $text = str_replace('[]', null, $name);
    if ($required) {
      $text .= '*';
    }

    return array(
      'tag'     => 'label',
      'content' => $text,
      'attributes' => array(
        'for'   => $field,
        'class' => 'control-label',
      ),
    );
  }
