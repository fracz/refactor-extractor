    public function batch( $type, $ids ) {
        $collection = array();
		$rows = $this->writer->selectRecord($type,$ids);
		$this->stash = array();
		foreach($rows as $row) {
			$this->stash[$row["id"]] = $row;
		}
        foreach($ids as $id) {
            $collection[ $id ] = $this->load( $type, $id );
        }
		$this->stash = NULL;
        return $collection;
    }
