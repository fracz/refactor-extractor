	public static function untag($bean,$tagList) {
		if ($tagList!==false && !is_array($tagList)) $tags = explode( ",", (string)$tagList); else $tags=$tagList;
		foreach($tags as $tag) {
			$t = R::findOne("tag"," title = ? ",array($tag));
			if ($t) {
				R::unassociate( $bean, $t );
			}

		}

	}
