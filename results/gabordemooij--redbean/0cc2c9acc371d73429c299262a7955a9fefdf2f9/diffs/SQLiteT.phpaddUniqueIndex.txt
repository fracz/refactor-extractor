	public function addUniqueIndex( $table,$columns ) {
		$table = $this->safeTable($table);
		$name = 'UQ_'.sha1(implode(',',$columns));
		$sql = "CREATE UNIQUE INDEX IF NOT EXISTS $name ON $table (".implode(',',$columns).")";
		$this->adapter->exec($sql);
	}
