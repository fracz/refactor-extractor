	public function wrap( $bean, $prefix=false, $suffix=false ) {

		if (!$prefix) {
			$prefix = RedBean_Setup_Namespace_PRFX;
		}

		if (!$suffix) {
			$suffix = RedBean_Setup_Namespace_SFFX;
		}
		$dclass = $prefix.$this->type.$suffix;
		$deco = new $dclass( floatval( $bean->id ) );
		$deco->setData( $bean );
		return $deco;

	}
