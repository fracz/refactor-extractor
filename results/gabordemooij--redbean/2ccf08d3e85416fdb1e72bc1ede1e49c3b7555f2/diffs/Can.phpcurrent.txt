	public function current() {
		if (isset($this->collectionIDs[$this->pointer])) {
			$id = $this->collectionIDs[$this->pointer];
			return $this->wrap( $this->provider->getBeanStore()->get( $this->type, $id ) );
		}
		else {
			return null;
		}
	}
