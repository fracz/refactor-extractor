     public function setLockingTime( $timeInSecs ) {

        if (is_int($timeInSecs) && $timeInSecs >= 0) {
            $this->locktime = $timeInSecs;
        }
        else {
            throw new RedBean_Exception_InvalidArgument( "time must be integer >= 0" );
        }
    }
