	public function logSCQuery( $sql, $db )

		$line = "\n".date("H:i:s")."|".$_SERVER["REMOTE_ADDR"]."|UID=".$this->userid."|".$sql;
		file_put_contents( $this->getFilename(), $line, FILE_APPEND );
		return null;
	}
