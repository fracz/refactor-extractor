	public static function set($item, $value)
	{
		$parts = explode('.', $item);

		$item =& static::$items;
		foreach ($parts as $part)
		{
			// if it's not an array it can't have a subvalue
			if ( ! is_array($item))
			{
				return false;
			}

			// if the part didn't exist yet: add it
			if ( ! isset($item[$part]))
			{
				$item[$part] = array();
			}

			$item =& $item[$part];
		}
		$item = $value;

		return true;
	}
