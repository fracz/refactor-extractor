  private function nextSeason()
  {
    $nowKey = (int)array_search($this->_currentSeason, $this->_season);
    if ($nowKey < 3) {
      $nextSeason = $this->_season[$nowKey+1];
    }else{
      $nextSeason = 'spring';
    }
    $this->_currentSeason = $nextSeason;
    $this->setState($this->_currentSeason);
  }
