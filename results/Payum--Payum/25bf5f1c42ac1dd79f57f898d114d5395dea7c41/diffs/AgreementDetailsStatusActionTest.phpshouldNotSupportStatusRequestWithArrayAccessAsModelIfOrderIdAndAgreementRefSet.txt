    public function shouldNotSupportStatusRequestWithArrayAccessAsModelIfOrderIdAndAgreementRefSet()

        $array

            ->method('offsetExists')
            ->with('agreementRef')

        ;
        $array
            ->expects($this->at(1))
            ->method('offsetExists')
            ->with('orderId')

        ;

        $this->assertFalse($action->supports(new BinaryMaskStatusRequest($array)));
    }
