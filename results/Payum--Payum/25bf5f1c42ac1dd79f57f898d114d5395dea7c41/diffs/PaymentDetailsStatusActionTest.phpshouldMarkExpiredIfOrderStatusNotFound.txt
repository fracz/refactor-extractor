    public function shouldMarkExpiredIfOrderStatusNotFound()

        $status = new BinaryMaskStatusRequest(array(
            'orderStatus' => OrderApi::ORDERSTATUS_NOT_FOUND,
            'orderId' => 'anId',
            'autoPay' => false,
        ));

        //guard

        $action->execute($status);
