    public function setField($name, $value)
    {
        if (is_array($value)) {
            $this->addFields(array($name => $value));
            return;
        }

        if ('[]' == substr($name, -2)) {
            $this->fields[substr($name, 0, -2)][] = $value;
        } else {
            $this->fields[$name] = $value;
        }
    }
