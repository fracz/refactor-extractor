  protected function setDataStore(array $data)
  {
    $this->store = $data;

    // cache sequence and sequence lookup hashes
    $this->pointer_index = array_keys($this->store);
    $this->key_index = array_flip($this->pointer_index);

    $this->rewind();
  }
