    public function getInterfacesNames()
    {
        $result = array();
        $interfaces = $this->node->getElementsByTagName('implements');

        /** @var $interface \DOMElement */
        foreach ($interfaces as $interface) {
            $result[] = $interface->nodeValue;
        }

        return $result;
    }
