    protected function getDirectElementsByTagName($node, $element_name)
    {
        $result = array();

        /** @var \DOMElement $element */
        foreach ($node->childNodes as $element) {
            if ($element->nodeName != $element_name) {
                continue;
            }

            $result[] = $element;
        }

        return $result;
    }
