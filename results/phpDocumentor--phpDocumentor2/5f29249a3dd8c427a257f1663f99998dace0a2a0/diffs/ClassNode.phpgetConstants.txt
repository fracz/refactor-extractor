    public function getConstants()
    {
        $result = array();
        $nodes = $this->getDirectElementsByTagName($this->node, 'constant');
        foreach ($nodes as $node) {
            $node = new ConstantNode($node, $this->nodes, $this);
            $result[$node->getName()] = $node;
        }

        return $result;
    }
