    private function resolve(Thenable $result)
    {
        if (null !== $this->result) {
            return;
        }

        $this->result = $result;
        $this->result->done($this->onFulfilled, $this->onRejected ?: new ThenQueue());

        $this->onFulfilled = null;
        $this->onRejected = null;
        $this->onCancelled = null;
    }
