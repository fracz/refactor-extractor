    protected function parameterize()
    {
        $args       = func_get_args();
        $parameters = [];

        if (count($args) > 2) {
            $parameters[] = $args[0];
            $keyword      = $args[2];

            foreach ($args[1] as $param) {
                $parameters[] = $param;
            }
        } else {
            $keyword = $args[1];
            foreach ($args[0] as $param) {
                $parameters[] = $param;
            }
        }
        $parameters = $this->replaceWithKeyword($parameters, $keyword);

        return $parameters;
    }
