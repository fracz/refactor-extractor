    private function getResult()
    {
        $this->result_object = $this->query->get();
        if ($this->query_type == 'builder') {
            $this->result_array = array_map(function ($object) {
                return (array) $object;
            }, $this->result_object);
        } else {
            $this->result_array = array_map(function ($object) {
                return (array) $object;
            }, $this->result_object->toArray());
        }
    }
