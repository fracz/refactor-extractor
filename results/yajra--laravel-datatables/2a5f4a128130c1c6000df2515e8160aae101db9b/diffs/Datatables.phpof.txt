	public static function of($query)
	{
		$ins = new static;
		$ins->saveQuery($query);

		//Get columns to temp var.
		if($ins->query_type == 'eloquent') {
			$ins->connection = $ins->query->getModel()->getConnection();
		}
		else {
			$ins->connection = $query->getConnection();
		}

		return $ins;
	}
