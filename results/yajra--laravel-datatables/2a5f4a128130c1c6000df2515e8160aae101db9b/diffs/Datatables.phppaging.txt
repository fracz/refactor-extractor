	private function paging()
	{
		if(!is_null($this->input->get('iDisplayStart')) && $this->input->get('iDisplayLength') != -1)
		{
			$this->query->skip($this->input->get('iDisplayStart'))->take($this->input->get('iDisplayLength',10));
		}
	}
