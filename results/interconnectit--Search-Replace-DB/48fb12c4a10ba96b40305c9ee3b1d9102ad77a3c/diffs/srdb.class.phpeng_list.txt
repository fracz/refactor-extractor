	public function eng_list( $input_arr = array( ), $sep = ', ', $before = '"', $after = '"' ) {
		if ( ! is_array( $input_arr ) )
			return false;

		$_tmp = $input_arr;

		if ( count( $_tmp ) >= 2 ) {
			$end2 = array_pop( $_tmp );
			$end1 = array_pop( $_tmp );
			array_push( $_tmp, $end1 . $after . ' or ' . $before . $end2 );
		}

		return $before . implode( $before . $sep . $after, $_tmp ) . $after;
	}
