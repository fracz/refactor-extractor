	private static function getDefaultStreamInfo($streams)
	{
		foreach (array_reverse($streams) as $stream) {
			if ($stream['default']) {
				break;
			}
		}
		unset($stream['default']);
		if (isset($stream['name'])) {
			unset($stream['name']);
		}

		$info = $stream;
		$info['streams'] = $streams;

		return $info;
	}
