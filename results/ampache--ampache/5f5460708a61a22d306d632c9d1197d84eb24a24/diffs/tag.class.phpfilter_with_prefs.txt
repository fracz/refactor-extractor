    public static function filter_with_prefs($l)
    {
        $colors = array('#0000FF','#00FF00', '#FFFF00', '#00FFFF','#FF00FF','#FF0000');
        $prefs = 'tag company';

        $ulist = explode(' ', $prefs);
        $req = '';

        foreach ($ulist as $i) {
            $req .= "'" . Dba::escape($i) . "',";
        }
        $req = rtrim($req, ',');

        $sql = 'SELECT `id`,`username` FROM `user` WHERE ';

        if ($prefs=='all') {
            $sql .= '1';
        } else {
            $sql .= 'username in ('.$req.')';
        }

        $db_results = Dba::read($sql);

        $uids=array();
        $usernames = array();
        $p = 0;
        while ($r = Dba::fetch_assoc($db_results)) {
            $usernames[$r['id']] = $r['username'];
            $uids[$r['id']] = $colors[$p];
            $p++;
            if ($p == sizeof($colors)) {
                $p = 0;
            }
        }

        $res = array();

        foreach ($l as $i) {
            if ($GLOBALS['user']->id == $i['user']) {
                $res[] = $i;
            } elseif (isset($uids[$i['user']])) {
                $i['color'] = $uids[$i['user']];
                $i['username'] = $usernames[$i['user']];
                $res[] = $i;
            }
        }

        return $res;

    } // filter_with_prefs
