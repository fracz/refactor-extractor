	function deleteMatch($key)
	{
		$key = strtolower( $key );

		if( isset( $_SESSION["HA::STORE"] ) && count( $_SESSION["HA::STORE"] ) ) {
		    $f = $_SESSION['HA::STORE'];
		    foreach( $f as $k => $v ){
				if( strstr( $k, $key ) ){
					unset( $f[ $k ] );
				}
			}
			$_SESSION["HA::STORE"] = $f;

		}
	}
