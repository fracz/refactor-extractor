    public function getUrlPath()
    {
        // TODO: chybi v interface
        $path = strtok($_SERVER['REQUEST_URI'], '?');
        $path = rawurldecode(preg_replace('#%(40|3[ABDF]|2[46BCF])#i', '%25$1', $path)); // decode %XX (@see RFC 2616 - 3.2.3 URI Comparison
        return $path;
    }
