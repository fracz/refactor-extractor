    public function setRawUrl($value)
    {
        if (!is_string($value)) {
            return;
        }

        $this->rawUrl = '/' . ltrim($value, '/');
        $this->normalizedUrl = NULL;

        $_GET = array();
        if (($pos = strpos($value, '?')) !== FALSE) {
            parse_str(substr($values, $pos + 1), $_GET);
            self::fuckingQuotes(array(&$_GET));
        }
    }
