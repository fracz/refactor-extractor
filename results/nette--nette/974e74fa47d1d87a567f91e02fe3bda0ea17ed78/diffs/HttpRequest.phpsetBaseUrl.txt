    public function setBaseUrl($value)
    {
        // TODO: predelat
        if (!is_string($value)) {
            return;
        }
        $value = '/' . ltrim($value, '/');
        $this->baseScript = $value;
        // do not use dirinfo!
        $this->basePath = substr($value, 0, strrpos($value, '/') + 1);
    }
