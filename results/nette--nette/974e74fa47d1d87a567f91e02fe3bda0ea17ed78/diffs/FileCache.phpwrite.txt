    public function write($key, $data, $tags, $lifeTime, $priority)
    {
        if ($data === NULL) {
            $this->remove($key);
            return;
        }

        $this->key = $key;
        $this->data = $data;

        $s = serialize(array(
            'data' => $data
        ));
        $file = $this->getFile($key);
        $len = file_put_contents($file, $s); // intentionally @
        if ($len !== strlen($s)) {
            unlink($file);
            $this->data = NULL;
        }
    }
