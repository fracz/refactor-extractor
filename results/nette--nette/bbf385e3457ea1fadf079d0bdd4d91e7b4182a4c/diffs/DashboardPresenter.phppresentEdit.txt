	public function presentEdit($id = 0)
	{
		if (!$this->request->isPost()) return; // continue to renderEdit()...

		$artist = trim($this->request->post['artist']);
		$title = trim($this->request->post['title']);

		if ($id !== 0) {
			if ($artist != '' && $title != '') {
				$data = array(
					'artist' => $artist,
					'title'  => $title,
				);
				$album = new Albums();
				$album->update($id, $data);

				$this->redirect('default');

			}
		}
	}
