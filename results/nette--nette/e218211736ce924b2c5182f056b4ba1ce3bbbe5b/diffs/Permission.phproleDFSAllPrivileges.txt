	private function roleDFSAllPrivileges($role, $resource)
	{
		$dfs = array(
			'visited' => array(),
			'stack'   => array($role),
		);

		while (NULL !== ($role = array_pop($dfs['stack']))) {
			if (!isset($dfs['visited'][$role])) {
				if (NULL !== ($result = $this->roleDFSVisitAllPrivileges($role, $resource, $dfs))) {
					return $result;
				}
			}
		}

		return NULL;
	}
