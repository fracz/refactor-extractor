	public function addHeader($name, $value)
	{
		if ($value == '') { // intentionally ==
			unset($this->headers[$name]);

		} elseif (is_array($value)) {
			$this->headers[$name][key($value)] = current($value);

		} else {
			$this->headers[$name] = $value;
		}
	}
