	private function addRecipient($header, $email, $name)
	{
		if (!$name && preg_match('#^(.+) +<(.*)>$#', $email, $matches)) {
			list(, $name, $email) = $matches;
		}

		if (!$name) {
			$name = $email;
		} elseif ($this->encodeHeader($name) === $name && strpos($name, ',') !== FALSE) {
			$name = "\"$name\" <$email>";
		} else {
			$name = "$name <$email>";
		}

		$this->addHeader($header, array($email => $name));
	}
