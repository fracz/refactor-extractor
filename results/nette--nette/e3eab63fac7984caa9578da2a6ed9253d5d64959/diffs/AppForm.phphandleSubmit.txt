	public function handleSubmit()
	{
		if (!$this->isSubmitted() || ($this->onlyValid && !$this->isValid())) {
			return;
		}

		if ($this->submittedBy instanceof FormControl) {
			$this->submittedBy->Click();
		}

		$this->onSubmit($this);
	}
