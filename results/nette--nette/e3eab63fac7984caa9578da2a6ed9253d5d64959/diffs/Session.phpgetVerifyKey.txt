	private static function getVerifyKey()
	{
		$list = array(
			'HTTP_ACCEPT_CHARSET', 'HTTP_ACCEPT_ENCODING',
			'HTTP_ACCEPT_LANGUAGE', 'HTTP_USER_AGENT',
		);

		$key = array();
		foreach ($list as $item) {
			if (isset($_SERVER[$item])) $key[] = $_SERVER[$item];
		}
		return md5(implode("\0", $key));
	}
