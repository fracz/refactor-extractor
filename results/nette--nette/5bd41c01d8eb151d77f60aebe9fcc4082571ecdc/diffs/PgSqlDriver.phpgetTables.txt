	public function getTables()
	{
		return $this->connection->query("
			SELECT table_name as name, CAST(table_type = 'VIEW' AS INTEGER) as view
			FROM information_schema.tables
			WHERE table_schema = current_schema()
		")->fetchAll();
	}
