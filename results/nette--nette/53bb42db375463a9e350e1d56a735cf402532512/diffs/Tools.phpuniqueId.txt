	public static function uniqueId()
	{
		static $entropy = 0;
		$entropy++;
		$id = isset($_SERVER['REMOTE_ADDR']) ? $_SERVER['REMOTE_ADDR'] : '';
		$id = md5(uniqid($id . $entropy, TRUE));
		$id = base_convert($id, 16, 36);
		return $id;
	}
