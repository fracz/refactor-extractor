	protected function getContainerRegistry()
	{
		$ref = $this->getContainerReflection()->getProperty('registry');
		$ref->setAccessible(TRUE);
		$registry = $ref->getValue($this->container);
		$ref->setAccessible(FALSE);
		return $registry;
	}
