	private function refreshCache()
	{
		$this->lookupCache = array();

		if ($this instanceof IComponentContainer) {
			foreach ($this->getComponents() as $component) {
				if ($component instanceof self) {
					$component->refreshCache();
				}
			}
		}
	}
