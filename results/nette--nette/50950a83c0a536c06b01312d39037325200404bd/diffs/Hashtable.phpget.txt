	public function get($key, $default = NULL)
	{
		if (!is_scalar($key)) {
			throw new /*\*/InvalidArgumentException('Key must be either a string or an integer.');
		}

		if (parent::offsetExists($key)) {
			return parent::offsetGet($key);

		} else {
			return $default;
		}
	}
