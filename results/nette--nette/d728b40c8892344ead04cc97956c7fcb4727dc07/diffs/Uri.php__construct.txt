	public function __construct($uri = NULL)
	{
		if ($uri !== NULL) {
			foreach (parse_url($uri) as $key => $val) {
				$this->$key = $val;
			}

			if (!$this->port) {
				if ($this->scheme === 'http') {
					$this->port = 80;
				} if ($this->scheme === 'https') {
					$this->port = 443;
				}
			}
		}
	}
