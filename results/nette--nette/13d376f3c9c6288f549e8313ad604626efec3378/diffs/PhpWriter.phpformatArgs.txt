	public function formatArgs($input)
	{
		$out = '';
		$tokenizer = $this->preprocess(new MacroTokenizer($input));
		while ($token = $tokenizer->fetchToken()) {
			$out .= $this->canQuote($tokenizer) ? "'$token[value]'" : $token['value'];
		}
		return $out;
	}
