	public function macroCapture(MacroNode $node, $writer)
	{
		$name = $writer->fetchWord($node->args); // $variable

		if (substr($name, 0, 1) !== '$') {
			throw new ParseException("Invalid capture block parameter '$name'");
		}
		$node->data->name = $name;
		return 'ob_start()';
	}
