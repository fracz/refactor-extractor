	protected static function formatTemplateFiles($presenter, $view)
	{
		if (!preg_match("#^[a-zA-Z_\x7f-\xff][a-zA-Z0-9_\x7f-\xff]*$#", $view)) {
			return array();
		}
		$presenter = str_replace(':', 'Module/', $presenter);
		return array(
			"$presenter/$view.phtml",
			"$presenter.$view.phtml",
		);
	}
