	public function constructUrl(PresenterRequest $request)
	{
		$uri = $this->getRouter()->constructUrl($request, Environment::getHttpRequest());
		if ($uri === NULL) {
			throw new ApplicationException('No route.');
		}
		return $uri;
	}
