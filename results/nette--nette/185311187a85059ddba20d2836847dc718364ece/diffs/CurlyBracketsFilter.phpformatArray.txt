	public static function formatArray($s, $prefix = '')
	{
		$s = preg_replace_callback(
			'/(?:
				\'(?:\\\\.|[^\'\\\\])*\'|      ## single quoted string
				"(?:\\\\.|[^"\\\\])*"|         ## double quoted string
				(?<=[,=(]|=>|^)\s*([a-z\d_]+)(?=\s*[,=)]|$)|   ## 1) symbol
				(?<![=><!])(=)(?![=><!])       ## 2) equal sign
			)/xi',
			array(__CLASS__, 'cbArgs'),
			trim($s)
		);
		return $s === '' ? '' : $prefix . "array($s)";
	}
