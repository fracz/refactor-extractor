	public static function formatModifiers($var, $modifiers)
	{
		if (!$modifiers) return $var;
		preg_match_all(
			'/
				[^\'"}\s|:]+|               ## symbol
				[|:]|                       ## separator
				\'(?:\\\\.|[^\'\\\\])*\'|   ## single quoted string
				"(?:\\\\.|[^"\\\\])*"       ## double quoted string
			/xs',
			$modifiers . '|',
			$tokens
		);
		$inside = FALSE;
		foreach ($tokens[0] as $token) {
			if ($token === ':' || $token === '|') {
				if (!isset($prev)) {

				} elseif (!$inside) {
					$var = "\$template->$prev($var";
					unset($prev);
					$inside = TRUE;

				} else {
					$var .= ', ' . self::formatString($prev);
					unset($prev);
				}

				if ($token === '|' && $inside) {
					$var .= ')';
					$inside = FALSE;
				}
			} else {
				$prev = $token;
			}
		}
		return $var;
	}
