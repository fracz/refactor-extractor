	private function readFile($file, & $contentType)
	{
		if (!is_file($file)) {
			throw new \FileNotFoundException("File '$file' not found.");
		}
		if (!$contentType && $info = getimagesize($file)) {
			$contentType = $info['mime'];
		}
		return file_get_contents($file);
	}
