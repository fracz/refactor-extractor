    public function getNumExecutedLines()
    {
        if ($this->numExecutedLines == -1) {
            $this->numExecutedLines = 0;

            foreach ($this->executedLines as $line) {
                if (count($line) > 0) {
                    $this->numExecutedLines++;
                }
            }
        }

        return $this->numExecutedLines;
    }
