    public static function bitStringToCodeCoverage($strings, $status)
    {
        $result = array();

        foreach ($strings as $file => $string) {
            if (is_array($string)) {
                return $strings;
            }

            $data   = array();
            $length = strlen($string);

            for ($i = 0; $i < $length; $i++) {
                $ord = ord($string{$i});

                for ($j = 0; $j < 8; $j++) {
                    if ($ord & (0x01 << $j)) {
                        $data[$i * 8 + $j + 1] = $status;
                    }
                }
            }

            $result[$file] = $data;
        }

        return $result;
    }
}
