	function randomise($scanline, $width)
	{
		$new_line = '';
		$start = floor($width/2);
		$end = strlen($scanline) - ceil($width/2);

		for ($i = $start; $i < $end; $i++)
		{
			$pixel = ord($scanline{$i});

			if ($pixel < 190)
			{
				$new_line .= chr(mt_rand(0, 205));
			}
			else if ($pixel > 190)
			{
				$new_line .= chr(mt_rand(145, 255));
			}
			else
			{
				$new_line .= $scanline{$i};
			}
		}

		return $new_line;
	}
