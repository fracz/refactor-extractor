	function _read($var)
	{
		$result = eaccelerator_get($this->key_prefix . $var);

		if ($result === null)
		{
			return false;
		}

		// Handle serialized objects
		if (is_string($result) && strpos($result, $this->serialize_header . 'O:') === 0)
		{
			$result = unserialize(substr($result, strlen($this->serialize_header)));
		}

		return $result;
	}
