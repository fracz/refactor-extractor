	public function set_cache($cache = null)
	{
		if ($cache)
		{
			$this->cached_paths = $cache->get('class_loader');

			if ($this->cached_paths === false)
			{
				$this->cached_paths = array();
			}
		}

		$this->cache = $cache;
	}
