  public function testIdIsIgnoredForOtherHeaders()
  {
    foreach (array('References', 'In-Reply-To') as $name)
    {
      $header = $this->_getHeader($name);
      $header->fieldChanged('id', 'foo@bar');
      $this->assertNotEqual('foo@bar', $header->getId());
    }
  }
