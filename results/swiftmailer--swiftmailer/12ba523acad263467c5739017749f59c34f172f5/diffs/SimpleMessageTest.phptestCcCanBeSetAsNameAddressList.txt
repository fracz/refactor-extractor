  public function testCcCanBeSetAsNameAddressList()
  {
    $message = $this->_createMessage(array(), $this->_encoder, $this->_cache);
    $message->setCc(array(
      'chris.corbyn@swiftmailer.org'=>'Chris',
      'john@site.tld' => 'John'
      ));
    $this->assertEqual(array(
      'chris.corbyn@swiftmailer.org'=>'Chris',
      'john@site.tld' => 'John'),
      $message->getCc()
      );
  }
