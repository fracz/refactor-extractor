  public function _outputBytesInRange($output)
  {
    $outputBytes = unpack('C*', $output);

    foreach ($outputBytes as $byte)
    {
      if (!in_array($byte, $this->_allowedOutputBytes))
      {
        return false;
      }
    }

    return true;
  }
