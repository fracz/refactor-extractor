  protected function _getHeaderModel($field)
  {
    if ($header = $this->getHeader($field))
    {
      return $header->getFieldBodyModel();
    }
    else
    {
      $field = strtolower($field);
      return array_key_exists($field, $this->_fieldModels)
        ? $this->_fieldModels[$field]
        : null;
    }
  }
