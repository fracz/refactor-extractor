	public function __construct($title, $id, $size = '')
	{
		switch ($size) {
			case 'small':
				$this->_modal_class_size = 'modal-sm';
				break;
			case 'large':
				$this->_modal_class_size = 'modal-lg';
				break;
			case '':
				break;
			default:
				throw new Exception('Incorrect size, pass either large or small');
		}
		$this->_id = $id;
		$this->_title = $title;

		$this->addClass('modal', 'fade');

		return $this;
	}
