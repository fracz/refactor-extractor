    protected function makeCustomerForm()

            $this->context->smarty,
            $this->context,
            $this->getTranslator(),
            $this->makeCustomerFormatter(),
            new CustomerPersister(
                $this->context,
                new Hashing(),
                $this->getTranslator(),
