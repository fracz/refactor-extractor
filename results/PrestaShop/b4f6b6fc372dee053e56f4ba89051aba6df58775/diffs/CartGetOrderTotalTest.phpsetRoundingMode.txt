    {
        $mode = null;

        switch ($modeStr)
        {
            case 'up':

                break;
            case 'down':
                $mode = PS_ROUND_DOWN;
                break;

                break;
            default:
                throw new Exception(sprintf('Unknown rounding mode `%s`.', $modeStr));
        }

        Configuration::set('PS_PRICE_ROUND_MODE', $mode);

        return $mode;
    }
