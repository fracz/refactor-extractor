	public function checkedLimit($limit)
	{
		if (!preg_match('#^[0-9]+$#', trim($limit['start'])) || !preg_match('#^[0-9]+$#', trim($limit['end'])))
		{
			$this->error_sql['checkedLimit'] = false;
			return false;
		}
		return true;
	}
}
