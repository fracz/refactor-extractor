	public function set($key, $value, $expire = 0)
	{
		$path = $this->getPath();
		for ($i = 0; $i < $this->_depth; $i++)
			$path .= $key[$i].'/';
		if (@file_put_contents($path.$key, serialize($value)))
		{
			$this->_keysCached[$key] = true;
			return $key;
		}
		return false;
	}
