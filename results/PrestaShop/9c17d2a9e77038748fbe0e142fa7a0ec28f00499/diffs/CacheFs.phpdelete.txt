	public function delete($key, $timeout = 0)
	{
		$path = $this->getPath();
		if (!isset($this->_keysCached[$key]))
			return;
		for ($i = 0; $i < $this->_depth; $i++)
			$path.=$key[$i].'/';
		if (!file_exists($path.$key))
			return true;
		if (!unlink($path.$key))
			return false;
		unset($this->_keysCached[$key]);
		return true;
	}
