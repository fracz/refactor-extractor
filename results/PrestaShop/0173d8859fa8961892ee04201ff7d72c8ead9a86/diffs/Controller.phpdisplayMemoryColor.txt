	private function displayMemoryColor($n)
	{
		$n /= 1048576;
		if ($n > 3)
			return '<span style="color:red">'.sprintf('%0.2f', $n).'</span>';
		if ($n > 1)
			return '<span style="color:orange">'.sprintf('%0.2f', $n).'</span>';
		return '<span style="color:green">'.sprintf('%0.2f', $n).'</span>';
	}
