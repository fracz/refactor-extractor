    public function generate(Request $request)
    {
        if ($this->cache->has($request->getHash())) {
            return;
        }

        if (!$this->source->has($request->getFilename())) {
            throw new ImageNotFoundException('Could not find the file: ' . $request->getFilename());
        }

        $this->cache->write(
            $request->getHash(),
            $this->api->run(
                $request,
                $this->source->read(
                    $request->getFilename()
                )
            )
        );
    }
