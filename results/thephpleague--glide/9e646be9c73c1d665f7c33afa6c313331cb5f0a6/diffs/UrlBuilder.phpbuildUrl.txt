    protected function buildUrl($parts, $params)
    {
        $url = '';

        if (isset($parts['scheme']) and isset($parts['host'])) {
            $url .= $parts['scheme'].'://'.$parts['host'];

            if (isset($parts['port'])) {
                $url .= ':'.$parts['port'];
            }
        }

        $url .= $parts['path'];

        if (count($params)) {
            $url .= '?'.http_build_query($params);
        }

        return $url;
    }
