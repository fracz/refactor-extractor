	public function setPassword( $uid, $password ){
		if( $this->userExists($uid) ){
			$query = OC_DB::prepare( "UPDATE *PREFIX*users SET password = ? WHERE uid = ?" );
			$result = $query->execute( array( sha1( $password ), $uid ));

			return true;
		}
		else{
			return false;
		}
	}
