	public function checkPassword( $uid, $password ){
		$query = OC_DB::prepare( "SELECT uid FROM *PREFIX*users WHERE uid LIKE ? AND password = ?" );
		$result = $query->execute( array( $uid, sha1( $password )));

		$row=$result->fetchRow();
		if($row){
			return $row['uid'];
		}else{
			return false;
		}
	}
