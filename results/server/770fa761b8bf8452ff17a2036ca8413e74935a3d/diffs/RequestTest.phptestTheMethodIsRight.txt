	public function testTheMethodIsRight() {

			'get' => array('name' => 'John Q. Public', 'nickname' => 'Joey'),
			'method' => 'GET',
		);

		$request = new Request($vars, $this->stream);
		$this->assertEquals('GET', $request->method);
