	function addFile($path,$source=''){
		if($this->fileExists($path)){
			$this->remove($path);
		}
		if(file_exists($source)){
			$header=array();
			$dummy='';
			$this->tar->_openAppend();
			$result=$this->tar->_addfile($source,$header,$dummy,$dummy,$path);
		}else{
			$result=$this->tar->addString($path,$source);
		}
		return $result;
	}
