	public function renderHTML() {
		if (!$this->footerAdded) {
			$this->footerAdded = true;
			if ($this->bodyOpened) {
				$this->htmlBody .= $this->bodyEnd;
			}
			$this->htmlBody .= $this->tail;
		}
		return $this->htmlBody;
	}
