    function _removedDependency($p1, $p2)
    {
        if (empty($this->_packageSortTree[$p2])) {
            return false;
        }
        if (!in_array($p1, $this->_packageSortTree[$p2])) {
            foreach ($this->_packageSortTree[$p2] as $potential) {
                if ($this->_removedDependency($p1, $potential)) {
                    return true;
                }
            }
            return false;
        }
        return true;
    }
