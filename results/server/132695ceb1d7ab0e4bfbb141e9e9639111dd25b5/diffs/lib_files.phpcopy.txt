	public static function copy($sourceDir,$source,$targetDir,$target){
		if(OC_USER::isLoggedIn()){
			$targetFile=$targetDir.'/'.$target;
			$sourceFile=$sourceDir.'/'.$source;
			return OC_FILESYSTEM::copy($sourceFile,$targetFile);
		}
	}
