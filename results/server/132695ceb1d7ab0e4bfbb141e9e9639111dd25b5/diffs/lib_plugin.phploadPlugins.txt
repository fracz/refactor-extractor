	public static function loadPlugins() {
		global $CONFIG_INSTALLED;
		if($CONFIG_INSTALLED){
			global $SERVERROOT;
			$plugins = self::listPlugins();
			$blacklist=self::loadBlacklist();
			foreach($plugins as $plugin){
				if (array_search($plugin,$blacklist)===false) {
					self::load($plugin);
				}
			}
		}
	}
