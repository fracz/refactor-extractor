	public function toTmpFile($path){
		$this->connect();
		$parent=dirname($path);
		$name=substr($path,strlen($parent)+1);
		if(substr($name,0,1)=='/'){
			$name=substr($name,1);
		}
		$filename=$this->remote->getFile($parent,$name);
		if($filename){
			$this->notifyObservers($path,OC_FILEACTION_READ);
			return $filename;
		}else{
			return false;
		}
	}
