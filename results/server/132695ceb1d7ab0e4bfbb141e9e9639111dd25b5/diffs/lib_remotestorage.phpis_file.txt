	public function is_file($path){
		$this->connect();
		$parent=dirname($path);
		$name=substr($path,strlen($parent)+1);
		$this->cacheDir($parent);
		if(!isset($this->statCache[$name])){
			return false;
		}
		return ($this->statCache[$name]['type'!='dir']);
	}
