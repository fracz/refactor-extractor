    function _getRevPkgDeps($p)
    {
        if (!isset($p['info']['release_deps'])) {
            return array();
        }
        $ret = array();
        foreach($p['info']['release_deps'] as $dep) {
            if ($dep['type'] == 'pkg') {
                $ret[] = $dep['name'];
            }
        }
        return $ret;
    }
