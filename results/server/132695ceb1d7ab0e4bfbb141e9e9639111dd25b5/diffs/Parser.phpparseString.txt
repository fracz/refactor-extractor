    function parseString($data, $eof = false)
    {
        if (!isset($this->parser) || !is_resource($this->parser)) {
            $this->reset();
        }

        if (!$this->_parseString($data, $eof)) {
           $error = &$this->raiseError();
           $this->free();
           return $error;
        }

        if ($eof === true) {
            $this->free();
        }
        return true;
    }
