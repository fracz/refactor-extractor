	public function opendir($path){
		$this->connect();
		$this->cacheDir($path);
		$dirs=array_keys($this->statCache);
		$id=uniqid();
		global $FAKEDIRS;
		$FAKEDIRS[$id]=$dirs;
		if($return=opendir("fakedir://$id")){
			$this->notifyObservers($path,OC_FILEACTION_READ);
		}
		return $return;
	}
