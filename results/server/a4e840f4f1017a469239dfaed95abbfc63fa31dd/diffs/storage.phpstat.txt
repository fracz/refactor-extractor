	public function stat($path){
		$ctime=-1;
		$path=$this->stripPath($path);
		if($path==''){
			$stat=stat($this->path);
		}else{
			if($this->is_dir($path)){
				$stat=array('size'=>0);
				$stat['mtime']=filemtime($this->path);
			}else{
				$stat=array();
				$stat['mtime']=$this->archive->mtime($path);
				$stat['size']=$this->archive->filesize($path);
			}
		}
		$stat['ctime']=$ctime;
		return $stat;
	}
