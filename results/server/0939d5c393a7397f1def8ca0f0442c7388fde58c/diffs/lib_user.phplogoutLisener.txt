	public static function logoutlisener(){
		if(isset($_GET['logoutbutton']) && isset($_SESSION['username'])){
			OC_LOG::event($_SESSION['username'],2,'');
			$_SESSION['user_id']=false;
			$_SESSION['username']='';
			$_SESSION['username_clean']='';
		}
	}
