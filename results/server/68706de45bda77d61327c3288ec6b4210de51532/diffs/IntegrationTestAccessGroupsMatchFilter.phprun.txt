	public function run() {
		$cases = ['case1', 'case2', 'case3'];

		foreach ($cases as $case) {
			print("running $case " . PHP_EOL);
			if (!$this->$case()) {
				print(PHP_EOL . '>>> !!! Test ' . $case . ' FAILED !!! <<<' . PHP_EOL . PHP_EOL);
				exit(1);
			}
		}

		print('Tests succeeded' . PHP_EOL);
	}
