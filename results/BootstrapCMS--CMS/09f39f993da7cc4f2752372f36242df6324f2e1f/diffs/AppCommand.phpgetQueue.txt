    protected function getQueue($type) {
        if ($this->laravel['config']['queue.default'] == 'sync') {
            return $type;
        } else {
            return $this->laravel['config']['queue.connections.'.$this->laravel['config']['queue.default'].'.'.$type];
        }
    }
