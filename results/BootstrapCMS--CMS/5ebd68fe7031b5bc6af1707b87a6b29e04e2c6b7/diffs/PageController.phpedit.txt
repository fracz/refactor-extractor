    public function edit($slug) {
        $page = $this->page->findBySlug($slug);

        if (!$page) {
            if ($slug == 'home') {
                App::abort(500, 'The Homepage Is Missing');
            } else {
                App::abort(404, 'Page Not Found');
            }
        }

        return View::make('pages.edit', array('page' => $page));
    }
