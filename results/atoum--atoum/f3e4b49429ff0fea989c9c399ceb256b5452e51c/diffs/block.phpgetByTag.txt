	public function getByTag($tag)
	{
		$tags = array();

		foreach ($this as $child)
		{
			if ($child->getTag() == $tag)
			{
				$tags[] = $child;
			}

			$tags = array_merge($tags, $child->getByTag($tag));
		}

		return $tags;
	}
