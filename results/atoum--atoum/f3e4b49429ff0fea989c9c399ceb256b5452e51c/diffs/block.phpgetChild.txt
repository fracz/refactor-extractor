	public function getChild($rank)
	{
		if (is_int($rank) === false || $rank < 0)
		{
			trigger_error('Rank must be an integer greater than or equal to zero', E_USER_ERROR);
		}
		else
		{
			return (isset($this->children[$rank]) === false ? null : $this->children[$rank]);
		}
	}
