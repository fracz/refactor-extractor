    protected function get()
    {
        $c = $this->lookAhead;
        $this->lookAhead = null;
        if ($c === null) {
            if ($this->inputIndex < $this->inputLength) {
                $c = $this->input[$this->inputIndex];
                $this->inputIndex += 1;
            } else {
                $c = null;
            }
        }
        return ($c === "\r")
            ? "\n"
            : ($c === null || $c === "\n" || ord($c) >= self::ORD_SPACE
                ? $c
                : ' ');
    }
