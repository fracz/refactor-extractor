    private function _escape($string, $escapeChar) {
        if ($escapeChar) {
            $this->buffer = $escapeChar;
            return preg_replace_callback(
                '/\\' . $escapeChar . '(.)' .'/',
                array(&$this, '_escapeBis'),
                $string
            );

        } else {
            return $string;
        }
    }
