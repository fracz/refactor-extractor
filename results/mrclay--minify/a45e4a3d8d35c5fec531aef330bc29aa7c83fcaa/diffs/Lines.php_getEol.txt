    private static function _getEol($str)
    {
        $r = strpos($str, "\r");
        $n = strpos($str, "\n");
        if (false === $r && false === $n) {
            return '';
        }
        return ($r !== false)
            ? ($n == ($r + 1)
                ? "\r\n"
                : "\r")
            : "\n";
    }
