    public function hmget($key) {

        $set     = $this->get($key, array());
        $fields  = func_get_args();
        $values  = array();

        for ($i=1; $i<count($fields); $i++){
            $field = $fields[$i];
            $values[] = isset($set[$field]) ? $set[$field]:null;
        }

        return $values;
    }
