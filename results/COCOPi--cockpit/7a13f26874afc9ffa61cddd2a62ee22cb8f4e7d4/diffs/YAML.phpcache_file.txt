    protected function cache_file($file, $cachedfile, $filemtime = null) {

        if (!$filemtime){
            $filemtime = filemtime($file);
        }

        $data = var_export(Spyc::YAMLLoad($file), true);

        if (file_put_contents($cachedfile, "<?php return {$data};")) {
            touch($cachedfile,  $filemtime);
            return $cachedfile;
        }

        return false;
    }
