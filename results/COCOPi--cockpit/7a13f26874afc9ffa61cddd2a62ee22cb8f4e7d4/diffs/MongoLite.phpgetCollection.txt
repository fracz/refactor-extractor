    public function getCollection($name, $db = null){

        if(strpos($name, '/') !== false) {
            list($db, $name) = explode('/', $name, 2);
        }

        if(!$db) {
            $db = $this->db;
        }

        $name = str_replace('/', '_', $name);

        return $this->client->selectCollection($db, $name);
    }
