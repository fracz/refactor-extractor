    public function save($collection, &$data) {

        if(isset($data["_id"]) && is_string($data["_id"])) $data["_id"] = new \MongoId($data["_id"]);

        $ref = $data;

        $return = $this->getCollection($collection)->save($ref);

        if(isset($ref["_id"])) $ref["_id"] = (string) $ref["_id"];

        $data = $ref;

        return $return;
    }
