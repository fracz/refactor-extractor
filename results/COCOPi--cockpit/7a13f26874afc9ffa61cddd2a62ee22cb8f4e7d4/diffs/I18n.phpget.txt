    public function get($key, $alternative=null, $lang=null){

        if(!$lang) {
            $lang = $this->locale;
        }

        if(!$alternative){
            $alternative = $key;
        }

        return isset($this->_languages[$lang][$key]) ? $this->_languages[$lang][$key]:$alternative;
    }
