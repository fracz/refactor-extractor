    public function mkdir($path, $mode = 0755) {

        if (strpos($path, ':') !== false) {
            list($namespace, $additional) = explode(":", $path, 2);
            $dir = $this->app->path("{$namespace}:").$additional;
        } else {
            $dir = $path;
        }

        if (!is_dir($dir) && !@mkdir($dir, $mode, true)) {
            return false;
        }

        return true;
    }
