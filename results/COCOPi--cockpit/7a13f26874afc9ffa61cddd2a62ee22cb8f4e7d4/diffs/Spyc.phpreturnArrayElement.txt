    private function returnArrayElement ($line) {
        if (strlen($line) <= 1) return array(array()); // Weird %)
        $array = array();
        $value   = trim(substr($line,1));
        $value   = $this->_toType($value);
        if ($this->isArrayElement($value)) {
            $value = $this->returnArrayElement($value);
        }
        $array[] = $value;
        return $array;
    }
