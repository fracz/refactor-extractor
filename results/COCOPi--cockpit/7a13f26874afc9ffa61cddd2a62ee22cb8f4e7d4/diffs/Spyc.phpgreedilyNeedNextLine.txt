    private static function greedilyNeedNextLine($line) {
        $line = trim ($line);
        if (!strlen($line)) return false;
        if (substr ($line, -1, 1) == ']') return false;
        if ($line[0] == '[') return true;
        if (preg_match ('#^[^:]+?:\s*\[#', $line)) return true;
        return false;
    }
