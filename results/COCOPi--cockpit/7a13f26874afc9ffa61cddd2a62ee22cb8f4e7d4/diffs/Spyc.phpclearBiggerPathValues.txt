    private function clearBiggerPathValues ($indent) {


        if ($indent == 0) $this->path = array();
        if (empty ($this->path)) return true;

        foreach ($this->path as $k => $_) {
            if ($k > $indent) unset ($this->path[$k]);
        }

        return true;
    }
