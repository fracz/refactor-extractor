    public function get($key, $default = null) {

        $stmt = $this->connection->query("SELECT * FROM ".$this->table." WHERE `key`='{$key}';");
        $res  = $stmt->fetch(\PDO::FETCH_ASSOC);

        return  isset($res["key"]) ? json_decode($res["keyval"], true) : $default;
    }
