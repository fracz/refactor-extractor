  public function allow($group, $resource, $actions = array()){

    $actions = (array)$actions;

    if(!count($actions)){
        $actions = $this->resources[$resource];
    }

    foreach($actions as &$action){
      $this->rights[$group][$resource][$action] = true;
    }

  }
