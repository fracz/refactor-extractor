    public function get($name) {
        if (isset($this->_deleted_cookies[$name])) {
            return null;
        }

        if (array_key_exists($name, $this->_cookies)) {
            return $this->_cookies[$name];
        }

        $value = isset($_COOKIE[$name]) ? $_COOKIE[$name] : null;
        $this->_cookies[$name] = $value;

        return $value;
    }
