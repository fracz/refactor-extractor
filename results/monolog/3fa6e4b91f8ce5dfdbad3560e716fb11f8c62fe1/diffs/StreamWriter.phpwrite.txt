    public function write($log, $message)
    {
        if (null === $this->stream) {
            $this->stream = fopen($this->url, 'a');
        }
        if ($this->formatter) {
            $message = $this->formatter->format($log, $message);
        }
        fwrite($this->stream, (string) $message['message']);
    }
