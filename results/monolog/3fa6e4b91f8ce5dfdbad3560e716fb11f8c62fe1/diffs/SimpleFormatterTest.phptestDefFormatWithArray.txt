    public function testDefFormatWithArray()
    {

            'level' => Logger::ERROR,
            'message' => array(
                'log' => 'log',
                'level' => 'WARNING',
                'message' => 'foo',
            )
        ));
        $this->assertEquals('['.date('Y-m-d').'] log.WARNING: foo'."\n", $message);
    }
