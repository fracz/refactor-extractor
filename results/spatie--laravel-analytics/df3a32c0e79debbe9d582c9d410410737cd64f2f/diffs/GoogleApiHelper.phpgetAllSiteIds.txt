    public function getAllSiteIds()
    {
        if (empty($this->siteIds)) {
            $sites = $this->service->management_profiles->listManagementProfiles("~all", "~all");

            foreach ($sites['items'] as $site) {
                $this->siteIds[$site['websiteUrl']] = 'ga:'.$site['id'];
            }
        }

        return $this->siteIds;
    }
