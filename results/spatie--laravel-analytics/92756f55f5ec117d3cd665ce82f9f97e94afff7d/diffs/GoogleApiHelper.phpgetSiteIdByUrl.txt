    public function getSiteIdByUrl($url)
    {
        static $siteIds = null;

        if (is_null($siteIds))
        {
            $siteIds = $this->getSiteIds();
        }


        if (isset($this->siteIds[$url])) {
            return $this->siteIds[$url];
        }

        throw new \Exception("Site $url is not present in your Analytics account.");
    }
