	function isUrlFound($url)
	{
		$menu = Piwik_Menu::getInstance()->get();
		foreach($menu as $mainMenuName => $subMenus)
		{
			foreach($subMenus as $subMenuName => $menuUrl)
			{
				if($menuUrl == $url)
				{
					return true;
				}
			}
		}
		return false;
	}
