	static public function getCurrency($idSite)
	{
		static $symbols = null;
		if(is_null($symbols))
		{
			$symbols = Piwik_SitesManager_API::getInstance()->getCurrencySymbols();
		}
		$site = new Piwik_Site($idSite);
		return $symbols[$site->getCurrency()];
	}
