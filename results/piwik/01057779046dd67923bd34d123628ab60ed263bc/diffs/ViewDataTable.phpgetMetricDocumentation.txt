    public function getMetricDocumentation($columnName)
    {
        if ($this->metricsDocumentation === false) {
            $this->loadDocumentation();
        }

        if (!empty($this->metricsDocumentation[$columnName])) {
            return $this->metricsDocumentation[$columnName];
        }

        return false;
    }
