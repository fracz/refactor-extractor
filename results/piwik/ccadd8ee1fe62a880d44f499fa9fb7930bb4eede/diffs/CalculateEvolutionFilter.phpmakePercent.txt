	private static function makePercent($value, $divisor, $quotientPrecision)
	{
		if($value == 0)
		{
			$evolution = 0;
		}
		elseif($divisor == 0)
		{
			$evolution = 100;
		}
		else
		{
			$evolution = ($value / $divisor) * 100;
		}

		return round($evolution, $quotientPrecision).'%';
	}
