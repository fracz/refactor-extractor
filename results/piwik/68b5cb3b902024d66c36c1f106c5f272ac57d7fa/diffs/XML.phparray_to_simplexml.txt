	function array_to_simplexml($array, $name="result" ,&$xml=null )
	{
	    if(is_null($xml))
	    {
	        $xml = new SimpleXMLElement("<{$name}/>");
	    }

	    foreach($array as $key => $value)
	    {
	        if(is_array($value))
	        {
	            $xml->addChild($key);
	            $this->array_to_simplexml($value, $name, $xml->$key);
	        }
	        else
	        {
	            $xml->addChild($key, $value);
	        }
	    }
	    return $xml;
	}
