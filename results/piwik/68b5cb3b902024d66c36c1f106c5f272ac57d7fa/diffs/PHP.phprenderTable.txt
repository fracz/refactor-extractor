	function renderTable($table)
	{
		$array = array();

		foreach($table->getRows() as $row)
		{
			$newRow = array(
				'columns' => $row->getColumns(),
				'details' => $row->getDetails(),
				'idsubdatatable' => $row->getIdSubDataTable()
				);
			$array[] = $newRow;
		}

		if($this->serialize)
		{
			serialize($array);
		}
		else
		{
			return $array;
		}
	}
