	private function initProcess()
	{

		$saveStats = Piwik_LogStats_Config::getInstance()->LogStats['record_statistics'];
		if($saveStats == 0)
		{
			$this->setState(self::LOGGING_DISABLE);
		}

		if( count($_GET) == 0)
		{
			$this->setState(self::NO_GET_VARIABLE);
		}
	}
