	function __construct($table = null, $renderSubTables = null)
	{
		if(!is_null($table))
		{
			$this->setTable($table);
		}
		if(is_null($renderSubTables))
		{
			$this->renderSubTables = (bool)Piwik_Common::getRequestVar('expanded', false);
		}
		else
		{
			$this->renderSubTables = $renderSubTables;
		}
	}
