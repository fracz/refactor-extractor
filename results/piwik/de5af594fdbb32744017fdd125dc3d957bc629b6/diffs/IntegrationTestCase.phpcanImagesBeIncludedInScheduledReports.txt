	private static function canImagesBeIncludedInScheduledReports()
	{
		$gdInfo = gd_info();
		return
			(stristr(php_uname(), 'precise32') || stristr(php_uname(), 'ubuntu')) &&
			stristr(phpversion(), '5.3.10') &&
			$gdInfo['GD Version'] == '2.0';
	}
