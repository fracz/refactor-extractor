	public function __set($name, $value)
	{
		$this->checkWritePermissionOnFile();
		if(is_null($this->userConfig))
		{
			$this->userConfig = new Zend_Config(array(), true);
		}
		if(is_array($value)
			|| $this->userConfig->$name != $value)
		{
			$this->configFileUpdated = true;
		}
		$this->userConfig->$name = $value;
	}
