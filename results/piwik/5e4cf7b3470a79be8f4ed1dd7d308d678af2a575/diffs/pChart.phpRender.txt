   function Render($FileName)
    {
     if ( $this->ErrorReporting )
      $this->printErrors($this->ErrorInterface);

     /* Save image map if requested */
     if ( $this->BuildMap )
      $this->SaveImageMap();

     imagepng($this->Picture,$FileName);
    }
