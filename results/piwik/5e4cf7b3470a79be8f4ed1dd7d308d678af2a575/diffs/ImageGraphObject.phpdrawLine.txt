	function drawLine($X1,$Y1,$X2,$Y2,$R,$G,$B,$GraphFunction=FALSE)
    {
    	if(!$this->aliasedGraph)
    	{
    		imageline($this->Picture, $X1, $Y1, $X2, $Y2, imagecolorallocate($this->Picture, $R, $G, $B));
    		return;
    	}

    	parent::drawLine($X1,$Y1,$X2,$Y2,$R,$G,$B,$GraphFunction);
    }
