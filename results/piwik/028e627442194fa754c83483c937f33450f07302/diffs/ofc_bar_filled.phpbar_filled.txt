	function bar_filled( $colour=null, $outline_colour=null )
	{
		$this->type      = "bar_filled";
		parent::bar_base();

		if( isset( $colour ) )
			$this->set_colour( $colour );

		if( isset( $outline_colour ) )
			$this->set_outline_colour( $outline_colour );
	}
