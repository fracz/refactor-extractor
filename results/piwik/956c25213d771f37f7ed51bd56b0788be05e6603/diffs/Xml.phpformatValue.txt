	protected function formatValue($value)
	{
		if(is_string($value)
			&& !is_numeric($value))
		{
			$value = html_entity_decode($value, ENT_COMPAT, 'UTF-8');
			$value = htmlspecialchars($value);
		}
		elseif($value===false)
		{
			$value = 0;
		}
		return $value;
	}
