    function sortString($rowA, $rowB)
    {
        $valA = $rowA[0];
        $valB = $rowB[0];

        return !isset($valA)
        && !isset($valB)
            ? 0
            : (!isset($valA)
                ? 1
                : (!isset($valB)
                    ? -1
                    : $this->sign *
                    strcasecmp($valA,
                        $valB
                    )
                )
            );
    }
