    public function getMetricsForLabel($label)
    {
        $query = $this->queryVisitsByDimension($label);
        $metrics = array();
        while ($row = $query->fetch()) {
            if (!isset($metrics[$row['label']])) {
                $metrics[$row['label']] = $this->makeEmptyRow();
            }
            $this->sumMetrics($row, $metrics[$row['label']]);
        }
        return $metrics;
    }
