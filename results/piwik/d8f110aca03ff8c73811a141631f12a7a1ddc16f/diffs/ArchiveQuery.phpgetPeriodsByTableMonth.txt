    private function getPeriodsByTableMonth($periods)
    {
        $result = array();
        foreach ($periods as $period) {
            $tableMonth = $period->getDateStart()->toString('Y_m');
            $result[$tableMonth][] = $period;
        }
        return $result;
    }
