    private static function getChmodForPath($path)
    {
        $pathIsTmp = self::getPathToPiwikRoot() . '/tmp';
        if (strpos($path, $pathIsTmp) === 0) {
            // tmp/* folder
            return 0750;
        }
        // plugins/* and all others
        return 0755;
    }
