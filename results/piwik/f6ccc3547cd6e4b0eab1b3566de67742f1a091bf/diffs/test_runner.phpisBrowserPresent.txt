	public function isBrowserPresent()
	{
		return isset($_SERVER['HTTP_HOST'])
			   && (isset($_SERVER['PATH_INFO'])
				   || isset($_SERVER['REQUEST_URI'])
				   || isset($_SERVER['SCRIPT_NAME']));
	}
