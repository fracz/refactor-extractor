    public function test_isNumericNotValid()
    {
    	$notvalid = array(
    			'-1.0.0', '1e3','1,2',  '0x123', '--1', '-.',  "-1e-2",  '- 1', '1-',
    		);
    	foreach($notvalid as $toTest)
    	{
    		$this->assertFalse(Piwik::isNumeric($toTest), $toTest." valid but shouldn't!");
    	}
    }
