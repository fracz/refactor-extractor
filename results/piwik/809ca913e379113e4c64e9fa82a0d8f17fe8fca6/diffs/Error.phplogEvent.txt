    public function logEvent($errno, $errstr, $errfile, $errline, $backtrace)
    {
        $event = array();
        $event['errno'] = $errno;
        $event['message'] = $errstr;
        $event['errfile'] = $errfile;
        $event['errline'] = $errline;
        $event['backtrace'] = $backtrace;

        parent::log($event, Log::ERR, null);
    }
