    private function formatValue($value)
    {
        if (is_string($value)) {
            $value = "'$value'";
        }
        if (is_null($value)) {
            $value = 'null';
        }
        if (is_array($value)) {
            $value = "array( " . implode(", ", $value) . ")";
        }
        return $value;
    }
