    public function format($event)
    {
        foreach ($event as &$value) {
            $value = str_replace("\n", '\n', $value);
            $value = '"' . $value . '"';
        }
        $ts = $event['timestamp'];
        unset($event['timestamp']);
        return $ts . ' ' . implode(" ", $event) . "\n";
    }
