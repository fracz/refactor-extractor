	static public function getCurrentHost()
	{
		$url = self::getCurrentScheme();
		$url .= '://';

		if(isset($_SERVER['HTTP_HOST']))
		{
			$url .= $_SERVER['HTTP_HOST'];
		}
		else
		{
			$url .= 'unknown';
		}
		return $url;
	}
