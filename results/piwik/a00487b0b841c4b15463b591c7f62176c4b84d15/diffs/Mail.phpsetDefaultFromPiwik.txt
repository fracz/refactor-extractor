        $customLogo = new CustomLogo();

            ? Piwik::translate('CoreHome_WebAnalyticsReports')
            : Piwik::translate('ScheduledReports_PiwikReports');
        $fromEmailAddress = Config::getInstance()->General['noreply_email_address'];
