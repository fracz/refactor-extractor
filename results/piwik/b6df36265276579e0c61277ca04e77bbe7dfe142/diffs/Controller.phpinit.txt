	protected function init()
	{
		$plugins = Piwik_PluginsManager::getInstance()->getLoadedPluginsName();

		$loaded = 0;
		foreach( $plugins as $plugin )
		{
			$plugin = Piwik::unprefixClass($plugin);

			try {
				Piwik_API_Proxy::getInstance()->registerClass($plugin);
				$loaded++;
			}
			catch(Exception $e){
			}
		}
		return $loaded;
	}
}
