	static public function dropTables()
	{
		$tablesAlreadyInstalled = self::getTablesInstalled();
		$db = Zend_Registry::get('db');

		foreach($tablesAlreadyInstalled as $tableName)
		{
			$db->query("DROP TABLE $tableName");
		}
	}
