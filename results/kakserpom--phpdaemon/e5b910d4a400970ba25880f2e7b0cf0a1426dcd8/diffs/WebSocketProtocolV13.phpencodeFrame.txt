    protected function encodeFrame($data, $type = 'STRING')
    {
        $frames = array() ;
        $mask =	chr(rand(0, 0xFF)) .
 						chr(rand(0, 0xFF)) .
						chr(rand(0, 0xFF)) .
						chr(rand(0, 0xFF)) ;
		$packet = chr(bindec($a = '1000'.str_pad(decbin($this->getFrameType($type)), 4, '0', STR_PAD_LEFT)));
        $dataLength = strlen($data);
        if ($dataLength <= 125)
        {
            $packet .= chr($dataLength + 128);
        }
        elseif ($dataLength <= 65535)
        {
            $packet .=	chr(254) . // 126 + 128
            			chr($dataLength >> 8) .
            			chr($dataLength & 0xFF);
        }
        else {
            $packet .=	chr(255) . // 127 + 128
             			chr($dataLength >> 56) .
           				chr($dataLength >> 48) .
            			chr($dataLength >> 40) .
             			chr($dataLength >> 32) .
             			chr($dataLength >> 24) .
             			chr($dataLength >> 16) .
               			chr($dataLength >>  8) .
             			chr($dataLength & 0xFF);
        }

        $packet .=	$mask .
 					$this->mask($data, $mask);
		return $packet;
    }
