	public function ping($to = null, $cb = null) {
		if (!isset($this->xml)) {
			return false;
		}
		if ($to === null) {
			$to = $this->host;
		}
		//Daemon::log('Sending ping to '.$to);
		return $this->iqGetTo($to, '<ping xmlns="urn:xmpp:ping"/>', $cb);
	}
