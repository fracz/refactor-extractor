	public function onFrame($data, $type)
	{
		if (Daemon::$config->logevents->value)
		{
			Daemon::$process->log(get_class($this) . '::' . __METHOD__ . ' invoked');
		}

		if (!isset($this->upstream))
		{
			return FALSE;
		}

		$this->upstream->onFrame($data, $type);

		return TRUE;
	}
