	public function HumanToPlain($value) {
		$value = preg_replace_callback('~(".*?")|(\'.*?\')|(\s*,\s*)~s', function ($m) {
			if (!empty($m[3])) {
				return "\x00";
			}
			if (!empty($m[2])) {
				return substr($m[2], 1, -1);
			}
			if (!empty($m[1])) {
				return substr($m[1], 1, -1);
			}
		}, $value);
		return explode("\x00", $value);
	}
