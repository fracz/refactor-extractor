	public static function removeFromArray(&$arr, $cb) {
		if ($arr === null) {
			$arr = [];
		}
		$e = static::extractCb($cb);
		foreach ($arr as $k => $item) {
			if (static::extractCb($item) === $e) {
				unset($arr[$k]);
				return true;
			}
		}
		return false;
	}
