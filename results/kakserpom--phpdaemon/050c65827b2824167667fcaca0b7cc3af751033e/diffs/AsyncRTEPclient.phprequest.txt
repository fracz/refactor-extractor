	public function request($req) {
		if (!$this->connect()) {
			Daemon::log('RTEPClient: Couldn\'t send request.');
		}

		return $this->write(json_encode(array(
			'id'      => ++$this->seq,
			'type'    => 'request',
			'request' => $req
		)) . "\n");
	}
