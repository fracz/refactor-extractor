	public function connect() {
		if ($this->connected) {
			return TRUE;
		}

		foreach ($this->servers as &$addr) {
			try {
				if ($this->initStream($addr)) {
					$this->enable();

					return $this->connected = TRUE;
				}
			} catch (BadStreamDescriptorException $e) {
				Daemon::log(get_class($e) . ': Couldn\'t connect to ' . $addr . '.');
			}
		}

		return FALSE;
	}
