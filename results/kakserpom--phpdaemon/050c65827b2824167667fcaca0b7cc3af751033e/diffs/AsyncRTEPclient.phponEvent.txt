	public function onEvent($packet) {
		if (Daemon::$config->logevents->value) {
			Daemon::log(__METHOD__ . ': ' . Debug::dump($packet));
		}

		$event = $packet['event'];

		if (!isset($this->callbacks[$event['name']])) {
			return FALSE;
		}

		foreach ($this->callbacks[$event['name']] as $cb) {
			call_user_func($cb, $event);
		}

		return TRUE;
	}
