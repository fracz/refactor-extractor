	public function onPacket($packet) {
		if ($packet['type'] === 'response') {
			$this->onResponse($packet);
		}
		elseif ($packet['type'] === 'request') {
			$this->onRequest($packet);
		}
		elseif ($packet['type'] === 'event') {
			$this->onEvent($packet);
		}
	}
