	public function file($path) {
		$req = $this;
		$job = $this->job;
		$job('readfile', function ($name, $job) use ($req, $path) {
			$req->sendfile($path, function($file, $success) use ($job, $name) {
				$job->setResult($name);
			});
		});
	}
