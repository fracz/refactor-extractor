	public function getConnectionByKey($key, $cb = null) {
		if (
			($this->dtags_enabled)
			&& (($sp = strpos($key, '[')) !== FALSE)
			&& (($ep = strpos($key, ']')) !== FALSE)
			&& ($ep > $sp)
		) {
			$key = substr($key, $sp + 1, $ep - $sp - 1);
		}

		srand(crc32($key));
		$addr = array_rand($this->servers);
		srand();
		return $this->getConnection($addr, $cb);
	}
