	public function appInstancesReloadReady() {
		$ready = TRUE;

		foreach (Daemon::$appInstances as $k => $app) {
			foreach ($app as $appInstance) {
				if (!$appInstance->handleStatus($this->currentStatus)) {
					$this->log(__METHOD__ . ': waiting for ' . $k);

					$ready = FALSE;
				}
			}
		}
		return $ready;
	}
