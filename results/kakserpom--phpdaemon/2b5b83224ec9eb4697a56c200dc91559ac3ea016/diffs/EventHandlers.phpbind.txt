	public function bind($event, $cb) {
		if ($cb !== null) {
			$cb = CallbackWrapper::wrap($cb);
		}
		is_array($event) or $event = [$event];
		foreach ($event as $e) {
			if (!isset($this->eventHandlers[$e])) {
				$this->eventHandlers[$e] = [];
			}
			$this->eventHandlers[$e][] = $cb;
		}
		return true;
	}
