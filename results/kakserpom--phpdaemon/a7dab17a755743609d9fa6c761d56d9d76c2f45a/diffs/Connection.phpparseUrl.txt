	public function parseUrl($url) {
		$u = Daemon_Config::parseCfgUri($url);
		if (!$u) {
			return false;
		}
		if (!isset($u['port']) && isset($this->pool->config->port->value)) {
			$u['port'] = $this->pool->config->port->value;
		}
		return $u;
	}
