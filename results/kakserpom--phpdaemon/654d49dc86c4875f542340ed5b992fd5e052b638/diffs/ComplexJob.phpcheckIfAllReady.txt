	public function checkIfAllReady() {
		if ($this->resultsNum >= $this->jobsNum) {
			$this->jobs = array();
			$this->state = self::STATE_DONE;
			while ($cb = array_pop($this->listeners)) {
				$cb($this);
			}
		}
	}
