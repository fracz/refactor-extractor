	public function init() {
		if ($this->config->enable->value) {
			if ($this->config->allowedclients->value !== '') {
					$this->allowedClients = explode(',', $this->config->allowedclients->value);
			}

			$this->bindSockets(
				$this->config->listen->value,
				$this->config->listenport->value,
				TRUE
			);
		}
	}
