	public function init() {
		$this->prefix = $this->config->prefix->value;
		$servers = explode(',', $this->config->servers->value);

		foreach ($servers as $s) {
			$e = explode(':',$s);
			$this->addServer($e[0], isset($e[1]) ? $e[1] : NULL);
		}
	}
