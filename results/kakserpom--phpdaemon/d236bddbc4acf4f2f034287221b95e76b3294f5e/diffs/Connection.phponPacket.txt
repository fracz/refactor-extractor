	protected function onPacket() {
		$this->result = $this->ptr;
		if (($mtype = $this->isSubMessage()) !== false) { // sub callback
			$chan = $this->result[1];
			if ($mtype === 'pmessage') {
				$t = $this->psubscribeCb;
			} else {
				$t = $this->subscribeCb;
			}
			if (isset($t[$chan])) {
				foreach ($t[$chan] as $cb) {
					if (is_callable($cb)) {
						call_user_func($cb, $this);
					}
				}
			}
		}
		else { // request callback
			$this->onResponse->executeOne($this);
		}
		$this->checkFree();
		$this->result       = null;
		$this->error        = false;
	}
