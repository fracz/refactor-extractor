	public function parseEncodedBinary(&$s, &$p) {
		$f = ord(binarySubstr($s, $p, 1));
		++$p;

		if ($f <= 250) {
			return $f;
		}

		if ($f === 251) {
			return NULL;
		}

		if ($f === 255) {
			return FALSE;
		}

		if ($f === 252) {
			$o = $p;
			$p += 2;

			return $this->bytes2int(binarySubstr($s, $o, 2));
		}

		if ($f === 253) {
			$o = $p;
			$p += 3;

			return $this->bytes2int(binarySubstr($s, $o, 3));
		}

		$o = $p;
		$p =+ 8;

		return $this->bytes2int(binarySubstr($s, $o, 8));
	}
