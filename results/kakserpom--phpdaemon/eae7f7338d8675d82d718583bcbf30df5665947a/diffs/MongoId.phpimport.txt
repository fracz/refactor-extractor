	public static function import($id, $tolerance = true) {
		if ($id instanceof static) {
			return $id;
		}
		elseif ($id instanceof \MongoId) {
			$id = (string) $id;
		}
		elseif (!is_string($id)) {
			return false;
		}
		elseif (strlen($id) === 24) {
			 if (!ctype_xdigit($id)) {
				return false;
			}
		} elseif (ctype_alnum($id)) {
			$id = gmp_strval(gmp_init($id, 62), 16);
			if (strlen($id) > 24) {
				return false;
			}
			if (strlen($id) < 24) {
				$id = str_pad($id, 24, '0', STR_PAD_LEFT);
			}
		} else {
			return false;
		}
		return new static($id);
	}
