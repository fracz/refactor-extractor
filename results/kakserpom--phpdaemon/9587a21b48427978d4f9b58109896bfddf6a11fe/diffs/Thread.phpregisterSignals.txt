	protected function registerSignals()
	{
		foreach (self::$signals as $no => $name) {
			if (
				($name === 'SIGKILL')
				|| ($name == 'SIGSTOP')
			) {
				continue;
			}

			if (!pcntl_signal($no, array($this, 'sighandler'), TRUE)) {
				$this->log('Cannot assign ' . $name . ' signal');
			}
		}
	}
