	public function onError() {
		$this->instate = 0;
		$this->onResponse->executeOne($this, false);
		$this->resultRows = [];
		$this->resultFields = [];

		if ($this->state === self::STATE_AUTH_PACKET_SENT) {
			// in case of auth error
			$this->state = self::STATE_AUTH_ERROR;
			$this->finish();
		}

		Daemon::log(__METHOD__ . ' #' . $this->errno . ': ' . $this->errmsg);
	}
}
