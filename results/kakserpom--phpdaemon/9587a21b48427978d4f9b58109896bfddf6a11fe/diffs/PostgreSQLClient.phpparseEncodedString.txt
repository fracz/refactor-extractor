	public function parseEncodedString(&$s, &$p) {
		$l = $this->parseEncodedBinary($s, $p);

		if (
			($l === NULL)
			|| ($l === FALSE)
		) {
			return $l;
		}

		$o = $p;
		$p += $l;

		return binarySubstr($s, $o, $l);
	}
