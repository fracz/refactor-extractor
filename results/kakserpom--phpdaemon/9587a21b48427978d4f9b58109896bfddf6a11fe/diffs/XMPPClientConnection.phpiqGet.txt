	public function iqGet($xml, $cb) {
		if (!isset($this->xml)) {
			return false;
		}
		$id = $this->getId();
		$this->xml->addIdHandler($id, $cb);
		$this->sendXML('<iq xmlns="jabber:client" type="get" id="'.$id.'">'.$xml.'</iq>');
		return true;
	}
