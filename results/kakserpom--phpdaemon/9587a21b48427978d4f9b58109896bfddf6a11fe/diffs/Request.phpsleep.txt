	public function sleep($time = 0, $set = false) {
		if ($this->state === Request::STATE_FINISHED) {
			return;
		}
		if ($this->state !== Request::STATE_RUNNING) {
			$set = true;
		}

		$this->sleepTime = $time;

		if (!$set) {
			throw new RequestSleepException;
		}
		else {
			$this->ev->del();
			$this->ev->add($this->sleepTime);
		}

		$this->state = Request::STATE_WAITING;
	}
