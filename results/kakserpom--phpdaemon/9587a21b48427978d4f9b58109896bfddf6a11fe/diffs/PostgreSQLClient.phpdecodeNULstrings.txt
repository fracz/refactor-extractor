	public function decodeNULstrings($data, $limit = 1, &$p = 0) {
		$r = [];

		for ($i = 0; $i < $limit; ++$i) {
			$pos = strpos($data, "\x00", $p);

			if ($pos === FALSE) {
				break;
			}

			$r[] = binarySubstr($data, $p, $pos - $p);

			$p = $pos + 1;
		}

		return $r;
	}
