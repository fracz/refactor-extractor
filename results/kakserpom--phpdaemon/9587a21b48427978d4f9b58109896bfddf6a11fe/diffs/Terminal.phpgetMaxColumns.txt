	protected function getMaxColumns() {
		if (
			preg_match_all("/columns.([0-9]+);/", strtolower(@exec('stty -a | grep columns')), $output)
			&& 2 == sizeof($output)
		) {
			return $output[1][0];
		}

		return 80;
	}
