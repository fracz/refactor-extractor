	public function buildLenEncodedBinary($s) {
		if ($s === NULL) {
			return "\251";
		}

		$l = strlen($s);

		if ($l <= 250) {
			return chr($l) . $s;
		}

		if ($l <= 0xFFFF) {
			return "\252" . $this->int2bytes(2, $l) . $s;
		}

		if ($l <= 0xFFFFFF) {
			return "\254" . $this->int2bytes(3, $l) . $s;
		}

		return $this->int2bytes(8, $l) . $s;
	}
