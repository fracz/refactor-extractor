	public function command($cmd, $q = '', $cb = NULL) {
		if ($this->state !== self::STATE_AUTH_OK) {
			return FALSE;
		}

		$this->onResponse->push($cb);
		$this->sendPacket($cmd, $q);

		return TRUE;
	}
