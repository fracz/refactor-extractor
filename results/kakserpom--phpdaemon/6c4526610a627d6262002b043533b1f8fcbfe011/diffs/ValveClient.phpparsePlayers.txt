	public function parsePlayers(&$st) {
		$playersn = Binary::getByte($st);
		for ($i = 1; $i < $playersn; ++$i) {
			$n = Binary::getByte($st);
			$name = Binary::getString($st);
			$score = Binary::getDWord($st,TRUE);
			if (strlen($st) === 0) {
				break;
			}
			$u = unpack('f', binarySubstr($st, 0, 4));
			$st = binarySubstr($st, 4);
			$time = $u[1];
			if ($time == -1) {
				continue;
			}
			$players[] = array(
				'name' => $name,
				'score' => $score,
				'time' => $time
			);
		}
		return $players;
	}
