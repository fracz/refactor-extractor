 public function onFinish()
 {
  if (Daemon::$settings['logevents']) {Daemon::log(get_class($this).'::'.__METHOD__.' invoked');}
  if ($this->upstream) {$this->upstream->onFinish();}
  unset($this->upstream);
  unset($this->appInstance->sessions[$this->connId]);
 }
