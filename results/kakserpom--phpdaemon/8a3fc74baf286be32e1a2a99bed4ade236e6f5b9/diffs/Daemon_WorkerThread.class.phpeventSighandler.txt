	public function eventSighandler($fd,$events,$arg) {
		$signo = $arg[0];
		if (is_callable($c = array($this, strtolower(self::$signals[$signo])))) {
			call_user_func($c);
		}
		elseif (is_callable($c = array($this,'sigunknown'))) {
			call_user_func($c, $signo);
		}
	}
