	public function stdin($buf) {
		// from mysqld to client.
		if ($this->appInstance->config->protologging->value) {
			Daemon::log('MysqlProxy: Server --> Client: ' . Debug::exportBytes($buf) . "\n\n");
		}

		$this->downstream->write($buf);
	}
