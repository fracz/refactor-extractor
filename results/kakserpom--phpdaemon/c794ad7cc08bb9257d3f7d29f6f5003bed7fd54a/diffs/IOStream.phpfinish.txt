	public function finish() {
		if ($this->finished) {
			return;
		}
		$this->finished = true;
		$this->onFinish();
		if ($this->pool) {
			$this->pool->detach($this);
			$this->pool = null;
		}
		if (!$this->sending) {
			$this->close();
		}
		return true;
	}
