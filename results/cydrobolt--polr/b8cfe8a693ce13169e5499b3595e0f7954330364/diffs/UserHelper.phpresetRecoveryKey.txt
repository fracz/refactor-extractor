    public static function resetRecoveryKey($username) {
        $recovery_key = CryptoHelper::generateRandomHex(50);

        $user = self::getUserByUsername($username);

        if (!$user) {
            return false;
        }

        $user->recovery_key = $recovery_key;
        $user->save();

        return true;
    }
