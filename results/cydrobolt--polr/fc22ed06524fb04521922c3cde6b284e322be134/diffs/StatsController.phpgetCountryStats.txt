    private function getCountryStats($link_id) {
        $stats = $this->getBaseRows($link_id)
            ->select(DB::raw("country AS label, count(*) AS clicks"))
            ->groupBy('country')
            ->orderBy('clicks', 'desc')
            ->get();

        return $stats;
    }
