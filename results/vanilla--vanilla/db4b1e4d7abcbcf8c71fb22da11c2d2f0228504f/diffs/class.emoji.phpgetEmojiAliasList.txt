   public function getEmojiAliasList($emojiAlias = '') {
      $emojiAliasList = array(
         ':)'          => 'smile',
         ':D'          => 'smiley',
         ':('          => 'disappointed',
         ';)'          => 'wink',
         ':\\'         => 'confused',
         ':o'          => 'open_mouth',
         ':s'          => 'confounded',
         ':p'          => 'stuck_out_tongue',
         ":'("         => 'cry',
         ':|'          => 'neutral_face',
       //'D:'          => 'anguished',
         'B)'          => 'sunglasses',
         ':#'          => 'grin',
         'o:)'         => 'innocent',
         '<3'          => 'heart',
         '(*)'         => 'star',
         '>:)'         => 'smiling_imp'
       );

      return (!$emojiAlias)
         ? $emojiAliasList
         : $emojiAliasList[$emojiAlias];
   }
