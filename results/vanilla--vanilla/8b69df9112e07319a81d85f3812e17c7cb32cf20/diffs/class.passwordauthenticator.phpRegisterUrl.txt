   public function RegisterUrl($Redirect = '/') {
      $Return = sprintf('/entry/register?Target=%s', $Redirect);
      if ($this->_Protocol == 'https')
         $Return = str_replace('http:', 'https:', Url($Return, TRUE));

      return $Return;
	}
