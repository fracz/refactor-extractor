   public function RemoteSignOutUrl() {
      $Session = Gdn::Session();
      $Url = sprintf($this->_SignOutUrl, urlencode(Gdn_Url::Request()));
      $Url = str_replace('{Session_TransientKey}', $Session->TransientKey(), $Url);
      return $Url;
   }
