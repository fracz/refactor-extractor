    public function getDefaultEmailLogo() {
	$logo = array();
	if ($logo['source'] = c('Garden.Logo', '')) {
	    $logo['source'] = Gdn_Upload::url(c('Garden.Logo', ''));
	}
	$logo['link'] = url('/', true);
	$logo['alt'] = c('Garden.LogoTitle', c('Garden.Title', ''));
	return $logo;
    }
