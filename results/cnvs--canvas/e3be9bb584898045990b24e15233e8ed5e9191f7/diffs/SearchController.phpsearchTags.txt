    public function searchTags()
    {
        $tnt = new TNTSearch;
        $tnt->loadConfig(config('services.tntsearch'));
        $tnt->selectIndex("tags.index");
        $res = $tnt->search(request('search'), 12);
        return Tag::whereIn('id', $res['ids'])->get();
    }
