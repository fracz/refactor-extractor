    public static function findBySlug($slug)
    {
        $instance = new static;
        $builder = $instance->newQuery();
        $post = $builder->where('post_name', $slug)->first();

        if ($post == null) {
            throw new \Exception("Post not found with slug [{$slug}] and postType [{$instance->postType}]");
        }

        return $post;
    }
