	public function getProviderIcon () {
		$icon = $this->Url->getAbsolute('/favicon.png');

		if (@getimagesize($icon)) {
			return $icon;
		}

		$icon = $this->Url->getAbsolute('/favicon.ico');

		if (@getimagesize($icon)) {
			return $icon;
		}
	}
