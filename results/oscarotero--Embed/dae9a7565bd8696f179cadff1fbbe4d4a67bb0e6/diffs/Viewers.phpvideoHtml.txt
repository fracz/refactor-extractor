    public static function videoHtml($poster, $sources, $width = 0, $height = 0)
    {
        $code = self::element('video', array(
            'poster' => ($poster ?: null),
            'width' => ($width ?: null),
            'height' => ($height ?: null),
            'controls' => true,
        ));

        foreach ((array) $sources as $source) {
            $code .= self::element('source', array('src' => $source));
        }

        return $code.'</video>';
    }
