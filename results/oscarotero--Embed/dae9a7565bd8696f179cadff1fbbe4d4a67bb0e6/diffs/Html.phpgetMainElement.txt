        }

        // Popular ids: #main, #content, #page

            if ($article->hasAttribute('id') && (strpos($article->getAttribute('id'), 'post-') === 0)) {
                return $article;
            }
        }

        // Returns <body>
        return $html->getElementsByTagName('body')->item(0);
    }
