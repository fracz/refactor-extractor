    public function getImages()
    {
        $images = array();

        foreach ($this->providers as $provider) {
            $src = $provider->getImage();

            if (empty($src)) {
                continue;
            }

            if (!is_array($src)) {
                $images[] = $this->request->url->getAbsolute($src);
                continue;
            }

            foreach ($src as $src) {
                if (!empty($src)) {
                    $images[] = $this->request->url->getAbsolute($src);
                }
            }
        }

        return array_unique($images);
    }
