	public function saveIcon ($directory, $replace = false) {
		$file = parse_url($this->getUrl(), PHP_URL_HOST).'.ico';
		$file = self::saveFile("http://icons.duckduckgo.com/i/$file", $directory, $file, $replace);

		if ($file !== false) {
			$this->Provider->set('icon', $file);

			return true;
		}

		return false;
	}
