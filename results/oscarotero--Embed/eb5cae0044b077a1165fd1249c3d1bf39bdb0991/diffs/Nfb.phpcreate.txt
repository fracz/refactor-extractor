	static public function create (Url $Url) {
		if (!$Url->match('http://www.nfb.ca/film/*')) {
			return false;
		}

		return new static(new OEmbed('http://www.nfb.ca/remote/services/oembed/', $Url->getUrl(), 'xml'));
	}
