	public function getImage () {
		if ($this->Provider->has('image')) {
			return $this->Provider->get('image');
		}

		if ($this->getType() === 'photo') {
			return $this->Provider->get('url') ?: $this->Provider->get('thumbnail_url');
		}

		return $this->Provider->get('thumbnail_url');
	}
