	public function __construct (Url $Url) {
		$Xml = $Url->getXMLContent();

		if (!$Xml) {
			throw new \Exception('The xml is not valid');
		}

		$data = self::parseRss($Xml) ?: self::parseAtom($Xml);

		if ($data) {
			$this->url = $data['url'];
			$this->urls = $data['urls'];
		}
	}
