    public function getImagesUrls()
    {
        $images = [];

        if (($image = $this->api->get('misc', 'image'))) {
            $images[] = $image;
        }

        if (is_array($files = $this->api->get('files'))) {
            foreach ($files as $url => $info) {
                if ($info['format'] === 'Thumbnail') {
                    $images[] = $url;
                }
            }
        }

        return $images;
    }
