    public static function getAllData(array $providers, $name)
    {
        $method = 'get'.$name;
        $values = [];

        foreach ($providers as $provider) {
            $value = $provider->$method();

            if (!empty($value)) {
                if (is_array($value)) {
                    $values = array_merge($values, $value);
                } else {
                    $values[] = $value;
                }
            }
        }

        return array_unique($values);
    }
