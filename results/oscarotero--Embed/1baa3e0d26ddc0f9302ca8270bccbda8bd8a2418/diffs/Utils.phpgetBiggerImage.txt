    public static function getBiggerImage(array $images)
    {
        $biggerArea = 0;
        $biggerSrc = null;

        foreach ($images as $src => $image) {
            $area = $image[0] * $image[1];

            if ($area > $biggerArea) {
                $biggerArea = $area;
                $biggerSrc = $src;
            }
        }

        return $biggerSrc;
    }
