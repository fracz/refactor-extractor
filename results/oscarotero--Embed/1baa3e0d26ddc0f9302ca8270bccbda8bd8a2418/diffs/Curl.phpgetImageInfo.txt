    public static function getImageInfo($url, array $config = null)
    {
        $finfo = finfo_open(FILEINFO_MIME_TYPE);
        $image = new static($url, $finfo, $config);

        $curl = $image->getConnection();
        curl_exec($curl);
        curl_close($curl);

        $info = $image->getInfo();

        finfo_close($finfo);

        return $info;
    }
