            $v = $argv[$ii];

            if (substr($v,0,2)==='--') {
                // This is a long-form option.

                // This is a short-form option.
                foreach(str_split(substr($v,1)) as $option) {
                    $options[$option] = true;
                }

            } else {

                $positional[] = $v;

            }

        }

        return array($options, $positional);
