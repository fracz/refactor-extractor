    public function getDigestHash($sRealm, $sUserName)
	{
		$sHash = '';
		try
		{
			$sHash = $this->oPersonalAddressBook->GetUserHashByEmail($sUserName, true);
		}
		catch (Exception $oException) {}

//		var_dump($sHash);
//		exit();

		if (!empty($sHash))
		{
			$this->currentUser = $sUserName;
			return \md5($sUserName.':'.$sRealm.':'.$sHash);
		}

		return null;
	}
