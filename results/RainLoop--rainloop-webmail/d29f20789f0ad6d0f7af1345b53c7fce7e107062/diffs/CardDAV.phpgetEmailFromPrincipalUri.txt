	private function getEmailFromPrincipalUri($sPrincipalUri)
	{
		$sEmail = '';
		$aMatch = array();
		if (\preg_match('/\/?principals\/([^@\/]+@[^@\/]+)/i', $sPrincipalUri, $aMatch) && !empty($aMatch[1]))
		{
			$sEmail = \trim($aMatch[1]);
		}

		return $sEmail;
	}
