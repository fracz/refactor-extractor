    public function addLead($key, ListLead $lead)
    {
        $action = ($this->leads->contains($lead)) ? 'updated' : 'added';

        $leadEntity = $lead->getLead();
        $this->changes['leads'][$action][$leadEntity->getId()] = $leadEntity->getPrimaryIdentifier();
        $this->leads[$key] = $lead;

        return $this;
    }
