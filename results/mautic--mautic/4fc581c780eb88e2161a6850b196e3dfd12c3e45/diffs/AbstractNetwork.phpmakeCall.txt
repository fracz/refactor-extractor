    public function makeCall($url) {
        $referer  = $this->getRefererUrl();

        $ch = curl_init();
        curl_setopt($ch, CURLOPT_RETURNTRANSFER, 1);
        curl_setopt($ch, CURLOPT_URL, $url);
        curl_setopt($ch, CURLOPT_REFERER, $referer);
        $data = @curl_exec($ch);
        curl_close($ch);

        $data = json_decode($data);
        return $data;
    }
