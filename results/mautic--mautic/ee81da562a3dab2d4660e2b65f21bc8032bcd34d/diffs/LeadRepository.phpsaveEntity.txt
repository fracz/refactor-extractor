    public function saveEntity($entity, $flush = true)
    {
        $this->_em->persist($entity);

        if ($flush) {
            $this->_em->flush($entity);
        }

        $fields = $entity->getUpdatedFields();
        if (!empty($fields)) {
            $this->_em->getConnection()->update(MAUTIC_TABLE_PREFIX.'leads', $fields, ['id' => $entity->getId()]);
        }
    }
