        $node
            ->children()

                    ->children()
                        ->arrayNode('orm')
                            ->children()
                                ->scalarNode('class')

                                ->end()
                                ->scalarNode('manager_name')
                                    ->defaultNull()
                                ->end()
                                ->arrayNode('properties')

                                    ->end()
                                ->end()
                            ->end()
                        ->end()
