	protected function _library($path) {
		foreach (Libraries::get() as $name => $library) {
			if (strpos($path, $library['path']) !== 0) {
				continue;
			}
			$path = str_replace(array($library['path'], '.php'), null, $path);
			return '\\' . $name . str_replace('/', '\\', $path);
		}
	}
