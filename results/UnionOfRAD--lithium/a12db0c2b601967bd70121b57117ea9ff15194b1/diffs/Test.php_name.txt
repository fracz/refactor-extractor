	protected function _name($request) {
		$type = $request->action;
		$name = $request->args();

		if ($command = $this->_instance($type)) {
			$request->params['action'] = $name;
			$name = $command->invokeMethod('_class', array($request));
		}
		$request->params['action'] = $type;
		return $name;
	}
