	public static function toCommand($changes) {
		$map = array(
			'create'    => null,
			'update'    => '$set',
			'increment' => '$inc',
			'remove'    => '$unset',
			'rename'    => '$rename'
		);
		$result = array();

		foreach ($map as $from => $to) {
			if (!isset($changes[$from])) {
				continue;
			}
			if (!$to) {
				$result = array_merge($result, $changes[$from]);
			}
			$result[$to] = $changes[$from];
		}
		unset($result['$set']['_id']);
		return $result;
	}
