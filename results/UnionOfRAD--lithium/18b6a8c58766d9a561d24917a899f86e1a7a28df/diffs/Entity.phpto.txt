	public function to($format, array $options = array()) {
		switch ($format) {
			case 'array':
				$data = $this->_updated + $this->_data;
				$rel = array_map(function($obj) { return $obj->data(); }, $this->_relationships);
				$data = array_merge($data, $rel);
				$result = Collection::toArray($data, $options);
			break;
			default:
				$result = $this;
			break;
		}
		return $result;
	}
}
