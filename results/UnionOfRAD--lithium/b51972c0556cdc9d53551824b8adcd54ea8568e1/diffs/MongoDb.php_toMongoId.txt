	protected function _toMongoId($data) {
		if (isset($data['_id']) && !is_object($data['_id'])) {
			if (preg_match('/^[0-9a-f]{24}$/', (string) $data['_id'])) {
				$data['_id'] = new MongoId($data['_id']);
			}
		}
		return $data;
	}
