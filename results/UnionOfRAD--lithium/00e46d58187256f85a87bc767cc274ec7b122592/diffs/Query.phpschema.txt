	public function schema($field = null) {
		if (is_array($field)) {
			$this->_config['schema'] = $field;
			return $this;
		}

		if (isset($this->_config['schema'])) {
			$schema = $this->_config['schema'];

			if ($field) {
				return isset($schema[$field]) ? $schema[$field] : null;
			}
			return $schema;
		}

		if ($model = $this->model()) {
			return $model::schema($field);
		}
	}
