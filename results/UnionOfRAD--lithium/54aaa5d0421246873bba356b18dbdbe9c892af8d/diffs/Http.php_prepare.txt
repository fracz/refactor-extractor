	protected function _prepare($data = array()) {
		if (empty($data)) {
			return null;
		}
		if (is_array($data)) {
			$this->request->headers(array(
				'Content-Type' => 'application/x-www-form-urlencoded',
			));
			$data = substr($this->request->queryString($data), 1);
		}
		return $this->request->body($data);
	}
