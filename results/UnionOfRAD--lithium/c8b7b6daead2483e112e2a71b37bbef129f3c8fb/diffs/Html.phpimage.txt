	public function image($path, $options = array()) {
		$defaults = array('alt' => '');
		$options += $defaults;

		if (is_array($path)) {
			$path = $this->_context->url($path);
		}
		return $this->_render(__METHOD__, 'image', compact('path', 'options'));
	}
