	public function skip() {
		$message = 'MongoDb Extension is not loaded';
		$this->skipIf(!MongoDb::enabled(), $message);

		$mongodb = new MongoDb($this->_testConfig);
		$this->skipIf(
			!$mongodb->isConnected(),
			"`{$this->_testConfig['database']}` database or connection unavailable"
		);
	}
