	public function fields($fields = null) {
		if ($fields === false) {
			$this->_fields = array();
			return;
		}
		if (empty($fields)) {
			return $this->_fields;
		}

		if (is_array($fields)) {
			$this->_fields = array_merge($this->_fields, $fields);
		} else {
			if (!isset($this->_fields[$fields])) {
				$this->_fields[] = $fields;
			}
		}
	}
