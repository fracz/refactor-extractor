	public static function pushDiff($array, $array2) {
		if (empty($array) && !empty($array2)) {
			return $array2;
		}
		if (!empty($array) && !empty($array2)) {
			foreach ($array2 as $key => $value) {
				if (!isset($array[$key])) {
					$array[$key] = $value;
				} else {
					if (is_array($value)) {
						$array[$key] = static::pushDiff($array[$key], $array2[$key]);
					}
				}
			}
		}
		return $array;
	}
