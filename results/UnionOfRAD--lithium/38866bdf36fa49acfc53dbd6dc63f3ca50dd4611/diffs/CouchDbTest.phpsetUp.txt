	public function setUp() {
		$this->db = new CouchDb(array('socket' => false));
		$model = $this->_model;
		$model::resetSchema();
		$model::$connection = $this->db;

		$entity = new Document(compact('model'));
		$type = 'create';
		$this->query = new Query(compact('model', 'entity', 'type'));
	}
