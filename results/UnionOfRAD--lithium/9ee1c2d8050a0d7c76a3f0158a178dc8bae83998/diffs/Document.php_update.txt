	protected function _update($id = null) {
		if ($id) {
			$id = (array) $id;
			$model = $this->_model;
			foreach ((array) $model::meta('key') as $i => $key) {
				$this->__set($key, $id[$i]);
			}
		}
		$this->_exists = true;
	}
