	protected function _populate() {
		if ($this->closed() || !$this->_result->valid()) {
			return;
		}

		$data = $this->_result->current();
		if ($this->_query) {
			$data = $this->_mapRecord($data);
		}
		$result = $this->_set($data, null, array('exists' => true));
		$this->_result->next();

		return $result;
	}
