	public function reset() {
		if (is_object($this->_result) && method_exists($this->_result, 'rewind')) {
			$this->_closed = false;
			$this->_init = false;
			$this->_started = false;
			$this->_valid = false;
			$this->_data = array();
			$this->_index = array();
			$this->_result->rewind();
			$this->_columns = $this->_columnMap();
			return true;
		}
		return false;
	}
