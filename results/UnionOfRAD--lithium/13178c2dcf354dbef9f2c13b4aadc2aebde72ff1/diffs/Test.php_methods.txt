	protected function _methods() {
		$use = $this->_use();
		$path = Libraries::path($use);

		if (!file_exists($path)) {
			return "";
		}
		$methods = Inspector::methods($use, 'extents');
		$testMethods = array();

		foreach (array_keys($methods) as $method) {
			$testMethods[] = "\tpublic function test" . ucwords($method) . "() {}";
		}
		return join("\n", $testMethods);
	}
}
