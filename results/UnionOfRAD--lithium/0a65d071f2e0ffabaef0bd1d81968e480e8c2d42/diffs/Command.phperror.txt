	public function error($str = null, $newlines = 1) {
		if (is_array($str)) {
			foreach ($str as $string) {
				$this->error($string, $newlines);
			}
			return;
		}
		if ($newlines) {
			$str = $str . str_pad("\n", $newlines, "\n");
		}
		return $this->response->error($str);
	}
