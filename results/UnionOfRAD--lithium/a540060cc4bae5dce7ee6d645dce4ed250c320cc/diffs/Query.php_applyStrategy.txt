	protected function _applyStrategy(Source $source) {
		if ($this->_built) {
			return;
		}
		$this->_built = true;
		if (!$this->_config['with']) {
			return;
		}
		$options = array();
		if (isset($this->_config['mode'])) {
			$options = array('mode' => $this->_config['mode']);
		}
		$source->applyStrategy($options, $this);
	}
