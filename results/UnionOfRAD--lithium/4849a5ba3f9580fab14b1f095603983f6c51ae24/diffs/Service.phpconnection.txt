	public function &connection($config = array()) {
		$config += $this->_config;

		try {
			$this->connection = Libraries::instance('socket', $config['socket'], $config);
		} catch (ClassNotFoundException $e) {
			$this->connection = null;
		}
		return $this->connection;
	}
