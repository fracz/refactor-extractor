	public function testToContext() {
		$expected = array('http' => array(
			'method' => 'GET',
			'content' => '',
			'header' => array(
				'Host: localhost',
				'Connection: Close',
				'User-Agent: Mozilla/5.0'
			)
		));
		$result = $this->request->to('context');
		$this->assertEqual($expected, $result);
	}
}
