	public static function isInRange($value, $lower = null, $upper = null) {
		if (!is_numeric($value)) {
			return false;
		}
		if (isset($lower) && isset($upper)) {
			return ($value > $lower && $value < $upper);
		}
		return is_finite($value);
	}
