    public function overlay($message = null, $title = 'Notice', $level = 'info')
    {
        if ($message) {
            $this->message($message, $level);
        }

        $overlay = true;

        $this->updateLastMessage(compact('overlay', 'title', 'level'));

        return $this;
    }
