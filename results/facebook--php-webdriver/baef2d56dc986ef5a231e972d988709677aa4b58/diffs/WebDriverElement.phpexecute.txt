  private function execute($name, array $params = array()) {
    $params[':id'] = $this->id;
    $command = array(
      'sessionId' => $this->sessionID,
      'name' => $name,
      'parameters' => $params,
    );
    $raw = $this->executor->execute($command);
    return $raw['value'];
  }
