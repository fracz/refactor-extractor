    public function getFilteredLog($level, $message = '', array $context = array())
    {
        $result = $message;
        if ($this->checkLevel($level)) {
            $data = $this->format($context);
            $result = sprintf("%s\t%s", $message, $data);
        }
        return $result;
    }
