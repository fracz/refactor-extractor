    private static function getLoggerInstance($config)
    {
        $key = $config['key'];
        if (isset(Log::$instances[$key])) {
            return Log::$instances[$key];
        }
        $logger = self::initLoggerInstance($config);
        Log::$instances[$key] = $logger;
        return Log::$instances[$key];
    }
