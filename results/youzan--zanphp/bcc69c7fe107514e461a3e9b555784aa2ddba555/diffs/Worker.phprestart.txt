    public function restart()
    {
        $time = isset($this->config['max_live_time'])?$this->config['max_live_time']:1800000;
        $time += $this->workerId * self::GAP_TIME;

        Timer::after($time, $this->classHash.'_restart',[$this,'closePre']);
    }
