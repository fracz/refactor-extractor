    private function initConfig($config)
    {
        if (is_array($config)) {
            foreach ($config as $cf) {
                if (!isset($cf['engine'])) {
                    if (is_array($config)) {
                        $this->initConfig($cf);
                    }
                } else {
                    if (empty($cf['pool'])) {
                        continue;
                    }
                    //åˆ›å»ºè¿žæŽ¥æ±
                    $factoryType = $cf['engine'];
                    if (in_array($factoryType, $this->engineMap)) {
                        $factoryType = ucfirst($factoryType);
                        $this->initPool($factoryType, $cf['pool']);
                    }
                }

            }
        }
    }
