    public function getTplFullPath($path)
    {
        if(false !== strpos($path, '.html')) {
            return $path;
        }
        if(!preg_match('/^static./i', $path)){
            $path = explode('/', $path);
            $mod = array_shift($path);
            return APP_PATH . $mod . '/views/' . join('/', $path) . '.html';
        }
        $path = substr($path,7);
        $path = COMMON_STATIC_PATH  .  $path . '.html';
        return $path;
    }
