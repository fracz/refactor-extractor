    public static function decode($url, $rules = [])
    {
        if (!$rules) return false;

        foreach ($rules as $regex => $route) {
            if (preg_match($regex, $url, $matching)) {
                $parameter = self::getParameter($matching);
                $route['parameter'] = $parameter;
                return $route;
            }
        }
        return false;
    }
