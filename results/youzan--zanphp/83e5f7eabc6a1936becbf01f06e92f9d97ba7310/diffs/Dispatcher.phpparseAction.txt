    private function parseAction($action)
    {
        $arr = explode('.', $action);
        $ret = [
            'action'    => $arr[0],
        ];

        if(isset($arr[1])){
            $ret['format']  = $arr[1];
        } else {
            $ret['format']  = 'html';
        }

        return $ret;
    }
