	public function directDataPacket($id, $data = array(), $pid = 0x00, $recover = true){
		if($this->connected === false){
			return false;
		}
		$data["id"] = $id;
		$data["pid"] = $pid;
		$data["sendtime"] = microtime(true);
		$count = $this->counter[0]++;
		if($recover !== false){
			$this->recoveryQueue[$count] = $data;
		}
		$this->send(0x80, array(
			$count,
			$pid,
			$data,
		));
		++$this->packetStats[0];
	}
