		$this->putFloat($this->speedZ);
		$this->putFloat($this->yaw);
		$this->putFloat($this->pitch);
		$this->put(Binary::writeMetadata($this->metadata));
		$this->putShort(count($this->links));
		foreach($this->links as $link){

			$this->putLong($link[1]);
			$this->putByte($link[2]);
		}
	}
