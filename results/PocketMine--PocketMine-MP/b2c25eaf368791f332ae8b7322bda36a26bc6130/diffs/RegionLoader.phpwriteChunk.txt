	public function writeChunk(FullChunk $chunk){
		$this->lastUsed = time();
		$chunkData = $chunk->toBinary();
		if($chunkData !== false){
			$this->saveChunk($chunk->getX() - ($this->getX() * 32), $chunk->getZ() - ($this->getZ() * 32), $chunkData);
		}
	}
