	public function getDefaultPermSubscriptions($op){
		$subs = [];

		if($op === true){
			foreach($this->defSubsOp as $k => $perm){
				/** @var \WeakRef $perm */
				if($perm->valid()){
					$subs[] = $perm->get();
				}else{
					unset($this->defSubsOp[$k]);
				}
			}
		}else{
			foreach($this->defSubs as $k => $perm){
				/** @var \WeakRef $perm */
				if($perm->valid()){
					$subs[] = $perm->get();
				}else{
					unset($this->defSubs[$k]);
				}
			}
		}

		return $subs;
	}
