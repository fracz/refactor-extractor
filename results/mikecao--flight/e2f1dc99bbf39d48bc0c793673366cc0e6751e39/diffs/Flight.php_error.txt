    public static function _error(Exception $e) {
        error_log($e->getMessage());
        self::response(false)
            ->status(500)
            ->write(
                '<h1>500 Internal Server Error</h1>'.
                '<h3>'.$e->getMessage().'</h3>'.
                '<pre>'.$e->getTraceAsString().'</pre>'
            )
            ->send();
    }
