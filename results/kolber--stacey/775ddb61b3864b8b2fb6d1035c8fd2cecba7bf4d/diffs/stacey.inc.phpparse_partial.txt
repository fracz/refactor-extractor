	function parse_partial($file) {
		$file = file_get_contents($file);
		preg_match('/([\S\s]*)foreach[\S\s]*{([\S\s]*)}([\S\s]*)/', $file, $matches);
		return array(preg_replace('/\t/', '', $matches[1]), $matches[2], preg_replace('/\t/', '', $matches[3]));
	}
