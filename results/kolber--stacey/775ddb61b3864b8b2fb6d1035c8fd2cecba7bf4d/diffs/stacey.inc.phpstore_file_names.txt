	function store_file_names() {
		$dir = "../content/projects/";
		// read contents of projects directory
		if(is_dir($dir)) {
			if($dh = opendir($dir)) {
				while (($file = readdir($dh)) !== false) {
					if(!is_dir($file)) $this->stored_file_names[] = $file;
				}
				closedir($dh);
			}
		}
	}
