	public function __construct( $query = '' ) {
		$this->db = $GLOBALS['wpdb'];

		$this->query_var_defaults = array(
			'fields'            => '',
			'ID'                => '',
			'site__in'          => '',
			'site__not_in'      => '',
			'number'            => 100,
			'offset'            => '',
			'no_found_rows'     => true,
			'orderby'           => 'id',
			'order'             => 'ASC',
			'network_id'        => 0,
			'network__in'       => '',
			'network__not_in'   => '',
			'domain'            => '',
			'domain__in'        => '',
			'domain__not_in'    => '',
			'path'              => '',
			'path__in'          => '',
			'path__not_in'      => '',
			'public'            => null,
			'archived'          => null,
			'mature'            => null,
			'spam'              => null,
			'deleted'           => null,
			'search'            => '',
			'search_columns'    => array(),
			'count'             => false,
			'date_query'        => null, // See WP_Date_Query
			'update_site_cache' => true,
		);

		if ( ! empty( $query ) ) {
			$this->query( $query );
		}
	}
