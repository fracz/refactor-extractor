    public function isFirst()
    {
        /** @var Pages $pages */
        $pages = self::$grav['pages'];
        $parent = $pages->get($this->parent);

        if ($this->path() == array_values($parent->items)[0]) {
            return true;
        } else {
            return false;
        }
    }
