    public function enabled($state = null)
    {
        if (isset($state)) {
            $this->enabled = $state;
        } else {
            if (!isset($this->enabled)) {
                $this->enabled = $this->grav['config']->get('system.debugger.enabled');
            }
        }
        return $this->enabled;
    }
