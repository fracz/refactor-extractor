    public function link($reset = true)
    {
        if ($this->mode !== 'source') {
            $this->display('source');
        }

        !empty($this->width) && $this->linkAttributes['data-width'] = $this->width;
        !empty($this->height) && $this->linkAttributes['data-height'] = $this->height;

        return parent::link($reset);
    }
