    public function getUpdatable()
    {
        $plugins = $this->getUpdatablePlugins();
        $themes  = $this->getUpdatableThemes();

        $items = [
            'total'   => count($plugins)+count($themes),
            'plugins' => $plugins,
            'themes'  => $themes
        ];

        return $items;
    }
