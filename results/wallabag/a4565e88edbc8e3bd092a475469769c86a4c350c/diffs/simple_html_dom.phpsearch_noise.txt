	function search_noise($text)
	{
		global $debug_object;
		if (is_object($debug_object)) { $debug_object->debugLogEntry(1); }

		foreach($this->noise as $noiseElement)
		{
			if (strpos($noiseElement, $text)!==false)
			{
				return $noiseElement;
			}
		}
	}
