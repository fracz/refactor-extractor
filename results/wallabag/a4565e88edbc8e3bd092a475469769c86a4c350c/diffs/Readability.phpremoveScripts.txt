    public function removeScripts($doc) {
        $scripts = $doc->getElementsByTagName('script');
        for($i = $scripts->length-1; $i >= 0; $i--)
        {
            $scripts->item($i)->parentNode->removeChild($scripts->item($i));
        }
    }
