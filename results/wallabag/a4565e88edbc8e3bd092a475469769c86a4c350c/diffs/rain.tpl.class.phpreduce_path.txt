	protected function reduce_path( $path ){
		$path = str_replace( "://", "@not_replace@", $path );
		$path = str_replace( "//", "/", $path );
		$path = str_replace( "@not_replace@", "://", $path );
		return preg_replace('/\w+\/\.\.\//', '', $path );
	}
