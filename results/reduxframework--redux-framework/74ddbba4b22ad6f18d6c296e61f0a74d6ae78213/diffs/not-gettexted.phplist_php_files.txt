	function list_php_files($dir) {
		$files = array();
		$items = scandir( $dir );
		foreach ( (array) $items as $item ) {
			$full_item = $dir . '/' . $item;
			if ('.' == $item || '..' == $item)
				continue;
			if ('.php' == substr($item, -4))
				$files[] = $full_item;
			if (is_dir($full_item))
				$files += array_merge($files, NotGettexted::list_php_files($full_item, $files));
		}
		return $files;
	}
