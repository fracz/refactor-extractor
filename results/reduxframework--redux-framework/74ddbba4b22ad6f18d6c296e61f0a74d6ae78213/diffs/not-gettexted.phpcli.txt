	function cli() {
		global $argv, $commands;
		if (count($argv) < 4 || !in_array($argv[1], array_keys($this->commands))) {
			$this->usage();
			exit(1);
		}
		call_user_func_array(array(&$this, $this->commands[$argv[1]]), array_slice($argv, 2));
	}
