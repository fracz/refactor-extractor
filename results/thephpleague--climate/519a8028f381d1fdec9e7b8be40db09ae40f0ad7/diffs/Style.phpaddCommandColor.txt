    public function addCommandColor($command, $color)
    {
        if (array_key_exists($color, $this->colors)) {
            $this->command_colors[$command] = $color;

            $this->buildTags();
        } else {
            throw new \Exception("The color '{$color}' for command '{$command}' is not defined.");
        }
    }
