    protected function currentStyleCode()
    {
        $formats   = $this->current['formatting']['code'];

        $colors    = [
                        $this->current['foreground']['code'],
                        $this->current['background']['code'],
                    ];

        $style     = array_filter(array_merge($formats, $colors));

        return implode(';', $style);
    }
