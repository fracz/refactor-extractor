    protected function makeUrlRelativeTo($url, $root)
    {
        if (substr($url, 0, 1) === '/' || preg_match('/^[a-z]+:\/\//', $url)) {
            return $url;
        }
        return rtrim($root, '/') . "/$url";
    }
