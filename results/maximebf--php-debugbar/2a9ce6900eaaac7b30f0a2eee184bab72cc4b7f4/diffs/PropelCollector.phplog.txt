        if (strpos($message, 'DebugPDOStatement::execute') !== false) {

            if (!$this->logQueriesToLogger) {
                return;

            $message = "$sql ($duration_str)";
        }

        if ($this->logger !== null) {
            $this->logger->log($this->convertLogLevel($severity), $message);
