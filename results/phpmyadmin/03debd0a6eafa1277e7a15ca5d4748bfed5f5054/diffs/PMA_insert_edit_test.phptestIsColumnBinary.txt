    public function testIsColumnBinary()
    {
        $column = array();
        $column['Type'] = 'binaryfoo';
        $this->assertEquals('binaryfoo', PMA_isColumnBinary($column));

        $column['Type'] = 'Binaryfoo';
        $this->assertEquals('Binaryfoo', PMA_isColumnBinary($column));

        $column['Type'] = 'varbinaryfoo';
        $this->assertEquals('binaryfoo', PMA_isColumnBinary($column));

        $column['Type'] = 'barbinaryfoo';
        $this->assertFalse(PMA_isColumnBinary($column));
    }
