    {

            && is_array($analyzed_sql[0]['where_clause_identifiers'])

                $highlight_columns[$wci] = 'true';
            }
        }

        $this->__set('highlight_columns', $highlight_columns);

    } // end of the '_setHighlightedColumnGlobalField()' function
