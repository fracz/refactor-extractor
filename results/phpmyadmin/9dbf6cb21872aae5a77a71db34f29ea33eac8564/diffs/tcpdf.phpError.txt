	public function Error($msg) {
		// unset all class variables
		$this->_destroy(true);
		// exit program and print error
		if (!K_TCPDF_THROW_EXCEPTION_ERROR) {
			die('<strong>TCPDF ERROR: </strong>'.$msg);
		} else {
			throw new Exception('TCPDF ERROR: '.$msg);
		}
	}
