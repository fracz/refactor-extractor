    private function _readFormPathsCallback($value, $key, $prefix)
    {
        if (is_array($value)) {
            $prefix .= (empty($prefix) ? '' : '/') . $key;
            array_walk($value, array($this, '_readFormPathsCallback'), $prefix);
        } else {
            if (!is_int($key)) {
                $this->default[$prefix . '/' . $key] = $value;
                $value = $key;
            }
            $this->fields[] = $prefix . '/' . $value;
        }
    }
