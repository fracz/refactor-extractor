    {
        $this->startElement($name);
        $this->writeAttribute('x1',$x1);
        $this->writeAttribute('y1',$y1);
        $this->writeAttribute('x2', $x2);

        $this->writeAttribute('style', $styles);
        $this->endElement();
    }
