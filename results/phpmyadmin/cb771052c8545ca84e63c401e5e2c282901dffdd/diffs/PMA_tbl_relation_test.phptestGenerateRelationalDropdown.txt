        $this->assertStringStartsWith(

            '</select>',
            PMA_generateRelationalDropdown('name')
        );

            '<select name="name"',
            PMA_generateRelationalDropdown('name')
        );

        // test for title
        $this->assertStringStartsWith(
            '<select name="name" title="title"',
            PMA_generateRelationalDropdown('name', array(), false, 'title')
        );

            PMA_generateRelationalDropdown('name', $values)
