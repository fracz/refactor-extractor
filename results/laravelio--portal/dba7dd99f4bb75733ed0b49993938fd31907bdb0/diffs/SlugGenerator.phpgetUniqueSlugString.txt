    protected function getUniqueSlugString($slugString)
    {
        $padding = '';
        $slug = '';

        do {
            $slug = $slugString . $padding;

            $slugFound = Slug::where('primary', '=', 1)->where('slug', '=', $slug)->first();

            if ($slugFound) {
                if (empty($padding)) {
                    $padding = 1;
                }

                $padding++;
            }
        } while($slugFound);

        return $slug;
    }
