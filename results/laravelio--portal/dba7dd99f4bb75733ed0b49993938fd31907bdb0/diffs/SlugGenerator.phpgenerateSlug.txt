    protected function generateSlug()
    {
        $slugString = $this->getUniqueSlugString($this->model->getSlugString());

        $this->removeHistoricalSlug($slugString);

        return Slug::create([
            'slug'       => $slugString,
            'owner_id'   => $this->model->id,
            'owner_type' => get_class($this->model),
            'primary'    => 1,
        ]);
    }
