    public function forumThreads()
    {
        return $this->hasMany('Lio\Comments\Comment', 'author_id')->whereNull('parent_id')->where('type', '=', \Lio\Comments\Comment::TYPE_FORUM)->orderBy('created_at', 'desc');
    }
