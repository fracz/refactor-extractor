    public function getArticleCommentsPaginated(Article $article, $perPage = 20)
    {
    	return $this->model
            ->where('owner_id', '=', $article->id)
            ->where('type', '=', Comment::TYPE_ARTICLE)
    		->orderBy('created_at', 'asc')
    		->paginate($perPage);
    }
