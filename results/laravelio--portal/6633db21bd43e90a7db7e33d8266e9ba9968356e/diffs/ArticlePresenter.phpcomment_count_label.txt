    public function comment_count_label()
    {
        if ($this->resource->comment_count == 0) {
            return '0 Comments';
        } elseif($this->resource->comment_count == 1) {
            return '1 Comment';
        }

        return $this->resource->comment_count . ' Comments';
    }
