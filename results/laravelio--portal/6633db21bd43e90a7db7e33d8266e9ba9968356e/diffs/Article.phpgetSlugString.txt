    public function getSlugString()
    {
        if ( ! $this->status == static::STATUS_PUBLISHED) {
            return '';
        }

        $authorName = $this->author->name;
        $date       = date("m-d-Y", strtotime($this->published_at));
        $title      = $this->title;

        return \Str::slug("{$date} {$title} {$authorName}");
    }
