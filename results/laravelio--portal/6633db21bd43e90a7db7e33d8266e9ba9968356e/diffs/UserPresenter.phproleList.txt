    public function roleList()
    {
        $roles = $this->getRoles();

        if ( ! $roles->count()) {
            return "none";
        }

        $roleArray = [];

        foreach ($roles as $role) {
            $roleArray[] = $role->name;
        }

        return implode(', ', $roleArray);
    }
