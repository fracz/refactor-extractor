    public function getTagList()
    {
        $tagLinks = [];

        foreach ($this->items as $item) {
            $tagLinks[] = HTML::link(action('ForumController@getIndex') . '?tags=' . $item->slug, $item->name);
        }

        return implode(', ', $tagLinks);
    }
