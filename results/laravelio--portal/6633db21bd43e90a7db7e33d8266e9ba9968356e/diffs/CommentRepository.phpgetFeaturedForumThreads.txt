    public function getFeaturedForumThreads($count = 3)
    {
        return $this->model->with(['slug', 'tags'])
                   ->where('owner_type', '=', 'Lio\Forum\ForumCategory')
                   ->orderBy('created_at', 'desc')
                   ->take($count)
                   ->get();
    }
