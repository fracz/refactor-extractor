    private function updateRecord($reply, $observer, $data)
    {
        $reply->fill($data);

        // check the model validation
        if ( ! $this->replies->save($reply)) {
            return $observer->replyUpdateError($reply->getErrors());
        }

        return $observer->replyUpdated($reply);
    }
