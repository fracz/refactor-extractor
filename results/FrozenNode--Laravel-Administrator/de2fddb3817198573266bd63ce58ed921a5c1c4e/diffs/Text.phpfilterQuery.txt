	public function filterQuery(&$query, $model, &$selects)
	{
		//run the parent method
		parent::filterQuery($query, $model, $selects);

		//if there is no value, return
		if (!$this->value)
		{
			return;
		}

		$query->where($model->getTable().'.'.$this->field, 'LIKE', '%' . $this->value . '%');
	}
