	public function formatOptions($model, Field $field, array $results)
	{
		$model = $this->config->getDataModel();

		return array_map(function($m) use ($field, $model)
		{
			return array(
				'id' => $m->{$model->getKeyName()},
				'text' => $m->{$field->getOption('name_field')},
			);
		}, $results);
	}
