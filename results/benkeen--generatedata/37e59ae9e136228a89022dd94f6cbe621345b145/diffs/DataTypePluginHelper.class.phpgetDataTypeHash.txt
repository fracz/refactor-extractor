	public function getDataTypeHash($groupedDataTypes) {
		$hash = array();
		while (list($group_name, $dataTypes) = each($groupedDataTypes)) {
			foreach ($dataTypes as $dataType) {
				$hash[$dataType->getName()] = $dataType;
			}
		}
		return $hash;
	}
