	function gd_get_cities()
	{
	  global $g_table_prefix;

	  $query = mysql_query("
	    SELECT city
	    FROM   {$g_table_prefix}cities
	      ");

	  $cities = array();
	  while ($city_info = mysql_fetch_assoc($query))
	    $cities[] = $city_info['city'];

	  return $cities;
	}
