	function gd_extract_data_type_info($folder)
	{
	  global $L;

	  if (!is_file("$folder/ui.php"))
	    return array();

	  @include("$folder/ui.php");

	  $info = get_defined_vars();
	  unset($info["L"]);
	  $info["data_type_name"] = $L["{$info["data_folder_name"]}_name"];

	  return $info;
	}
