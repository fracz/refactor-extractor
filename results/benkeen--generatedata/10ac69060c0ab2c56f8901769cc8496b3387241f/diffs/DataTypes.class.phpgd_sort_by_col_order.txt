	function gd_sort_by_col_order($template)
	{
		$ordered = array();
	  while (list($order, $data_types) = each($template))
	  {
	    foreach ($data_types as $data_type)
	    {
	    	$order = $data_type["column_num"];
	    	$ordered["order$order"] = $data_type;
	    }
	  }
	  asort($ordered);

		return array_values($ordered);
	}
