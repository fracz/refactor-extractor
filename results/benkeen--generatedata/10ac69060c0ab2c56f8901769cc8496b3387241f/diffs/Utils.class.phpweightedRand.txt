	public static function weightedRand($weights)
	{
	  $r = mt_rand(1, 1000);
	  $offset = 0;
	  foreach ($weights as $k => $w)
	  {
	    $offset += $w * 1000;
	    if ($r <= $offset)
	      return $k;
	  }
	}
