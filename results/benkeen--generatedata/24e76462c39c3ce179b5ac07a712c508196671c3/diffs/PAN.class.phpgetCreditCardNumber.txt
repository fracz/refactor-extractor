	private static function getCreditCardNumber($prefixList, $length, $howMany) {
		$result = array();
		for ($i=0; $i<$howMany; $i++) {
			$ccnumber = $prefixList[ array_rand($prefixList) ];
			$result[] = pan_completed_number($ccnumber, $length);
		}
		return $result;
	}
