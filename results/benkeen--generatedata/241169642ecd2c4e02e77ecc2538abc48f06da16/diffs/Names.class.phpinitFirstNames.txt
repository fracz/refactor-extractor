	private function initFirstNames() {
	  $query = mysql_query("
	    SELECT first_name
	    FROM   {$g_table_prefix}first_names
	      ");
	  $names = array();
	  while ($name = mysql_fetch_assoc($query))
	    $names[] = $name['first_name'];
	}
