	public static function returnRandomSubset($set, $num) {
		if (!is_array($set) || !is_numeric($num)) {
			throw new Exception(ErrorCodes::INVALID_PARAMS);
			return;
		}
		// check $num is no greater than the total set
		if ($num > count($set)) {
			$num = count($set);
		}
		shuffle($set);
		return array_slice($set, 0, $num);
	}
