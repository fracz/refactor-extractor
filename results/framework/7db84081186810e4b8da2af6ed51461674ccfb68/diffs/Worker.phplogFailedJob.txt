    protected function logFailedJob($connection, Job $job)
    {
        if (! $this->failer) {
            return;
        }

        $failedId = $this->failer->log($connection, $job->getQueue(), $job->getRawBody());

        $job->delete();

        $job->failed();

        $this->raiseFailedJobEvent($connection, $job, $failedId);
    }
