	protected function registerShutdownHandler()
	{
		$app = $this->app;

		register_shutdown_function(function() use ($app)
		{
			set_exception_handler(array(new StubShutdownHandler($app), 'handle'));

			$app['kernel.error']->handleFatal();
		});
	}
