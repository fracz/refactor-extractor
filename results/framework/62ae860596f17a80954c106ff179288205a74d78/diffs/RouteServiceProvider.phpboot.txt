	public function boot()
	{
		$this->app['url']->setRootControllerNamespace($this->rootUrlNamespace);

		$this->app->call([$this, 'before']);

		if ($this->app->routesAreCached())
		{
			$this->loadCachedRoutes();
		}
		else
		{
			$this->loadRoutes();
		}
	}
