    protected function replaceParameterFields(array $fields, $attribute)
    {
        if ($keys = $this->getAttributeKeys($attribute)) {
            $fields = array_map(function ($field) use ($keys) {
                return $this->replaceAsterisksWithKeys($field, $keys);
            }, $fields);
        }

        return $fields;
    }
