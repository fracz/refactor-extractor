    public static function increasePlayCount($songId, User $user)
    {
        $interaction = self::firstOrCreate([
            'song_id' => $songId,
            'user_id' => $user->id,
        ]);

        if (!$interaction->exists) {
            $interaction->liked = false;
        }

        ++$interaction->play_count;
        $interaction->save();

        return $interaction;
    }
