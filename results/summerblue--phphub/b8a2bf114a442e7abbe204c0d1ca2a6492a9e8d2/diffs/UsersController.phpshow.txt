	public function show($id)
	{
		$user = User::findOrFail($id);
		$topics = Topic::userRecentTopics($user->id);
		$replies = Reply::userRecentReplies($user->id);
		return View::make('users.show', compact('user', 'topics', 'replies'));
	}
