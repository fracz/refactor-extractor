	public function show($id)
	{
		$topic = Topic::findOrFail($id);
		$replies = $topic->getRepliesWithLimit();

		$topic->view_count++;
		$topic->save();

		$nodeTopics = $topic->getSameNodeTopics();

		$node = $topic->node;

		return View::make('topics.show', compact('topic', 'replies', 'nodeTopics', 'node'));
	}
