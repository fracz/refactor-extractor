    public function getFilters()
    {

        if (!$this->filters) {
            foreach ($this->filterFields as $name => $description) {
                $filter = $this->getFilterInstance($this->filterFields[$name]);

                if ($filter) {
                    $this->filters[$name] = $filter;
                }
            }
        }

        return $this->filters;
    }
