    public function getQueryBuilder($values = array())
    {

        $queryBuidler = $this->getBaseQueryBuilder();

        foreach ($this->getFilters() as $name => $filter) {

            $value = isset($values[$name]) ? $values[$name] : null;

            $filter->apply($queryBuidler, $value);
        }

        return $queryBuidler;
    }
