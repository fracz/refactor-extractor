    public static function cleanFileName($filename)
    {
        $regex = array(
          '/[\?\[\]\/\\\=\<\>\:\;\,\'\"\&\$\#\*\(\)\|\~\`\!\{\}]/',
          '/\.$/',
          '/^\./',
        );
        $replace = array('-', '', '');
        return trim(preg_replace($regex, $replace, trim($filename)),'-');
    }
