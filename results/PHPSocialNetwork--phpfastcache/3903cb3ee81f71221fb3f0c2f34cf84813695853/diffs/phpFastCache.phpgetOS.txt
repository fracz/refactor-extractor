    protected static function getOS()
    {
        $os = array(
          'os' => PHP_OS,
          'php' => PHP_SAPI,
          'system' => php_uname(),
          'unique' => md5(php_uname() . PHP_OS . PHP_SAPI),
        );
        return $os;
    }
