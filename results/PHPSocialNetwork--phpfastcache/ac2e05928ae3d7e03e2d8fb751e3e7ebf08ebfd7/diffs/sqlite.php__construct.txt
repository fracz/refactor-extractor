    function __construct($config = array()) {
        /*
         * init the path
         */
        $this->setup($config);
        if(!$this->checkdriver() && !isset($config['skipError'])) {
	        $this->fallback = true;
        }

        if(!@file_exists($this->getPath()."/sqlite")) {
            if(!@mkdir($this->getPath()."/sqlite",$this->__setChmodAuto())) {
	            $this->fallback = true;
            }
        }
        $this->path = $this->getPath()."/sqlite";
    }
