    public function touchTags($tags = array(), $time = 300, $options = array())
    {
        if (!is_array($tags)) {
            $tags = array($tags);
        }
        $lists = $this->getTags($tags);
        foreach ($lists as $tag => $keywords) {
            foreach ($keywords as $keyword => $time) {
                $this->touch($keyword, $time, $options);
            }
        }
        return true;
    }
