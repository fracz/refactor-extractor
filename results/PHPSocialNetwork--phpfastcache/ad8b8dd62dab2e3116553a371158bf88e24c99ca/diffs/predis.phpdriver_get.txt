    public function driver_get($keyword, $option = array())
    {
        if ($this->connectServer()) {
            // return null if no caching
            // return value if in caching'
            $x = $this->instant->get($keyword);
            if ($x == false) {
                return null;
            } else {

                return $this->decode($x);
            }
        } else {
            $this->backup()->get($keyword, $option);
        }
    }
