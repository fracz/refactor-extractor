    protected function _handleTags($keyword, $time, $tags)
    {
        foreach ($tags as $tag) {
            $list = $this->_tagCaching()->get($this->_getTagName($tag));
            if (is_null($list)) {
                $list = array();
            }
            $list[ $keyword ] = time() + $time;
            $this->_tagCaching()->set($this->_getTagName($tag), $list, 3600 * 24 * 30);
        }
    }
