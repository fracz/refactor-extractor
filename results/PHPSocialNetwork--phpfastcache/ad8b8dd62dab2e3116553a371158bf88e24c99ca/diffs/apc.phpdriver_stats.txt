    public function driver_stats($option = array())
    {
        $res = array(
          'info' => '',
          'size' => '',
          'data' => '',
        );

        try {
            $res[ 'data' ] = apc_cache_info('user');
        } catch (\Exception $e) {
            $res[ 'data' ] = array();
        }

        return $res;
    }
