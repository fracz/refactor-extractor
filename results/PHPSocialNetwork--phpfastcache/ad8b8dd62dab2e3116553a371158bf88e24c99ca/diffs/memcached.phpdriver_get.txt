    public function driver_get($keyword, $option = array())
    {
        // return null if no caching
        // return value if in caching
        $this->connectServer();
        $x = @$this->instant->get($keyword);// Prevent memcached to return a warning for long keywords
        if ($x == false) {
            return null;
        } else {
            return $x;
        }
    }
