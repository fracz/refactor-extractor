    public function driver_set(
      $keyword,
      $value = '',
      $time = 300,
      $option = array()
    ) {
        if (isset($option[ 'skipExisting' ]) && $option[ 'skipExisting' ] == true) {
            return apc_add($keyword, $value, $time);
        } else {
            return apc_store($keyword, $value, $time);
        }
    }
