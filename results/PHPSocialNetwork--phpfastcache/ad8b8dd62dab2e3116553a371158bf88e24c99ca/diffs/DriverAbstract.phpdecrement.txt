    public function decrement($keyword, $step = 1, array $option = array())
    {
        $object = $this->get($keyword, array('all_keys' => true));
        if ($object == null) {
            return false;
        } else {
            $value = (int)$object[ 'value' ] - (int)$step;
            $time = $object[ 'expired_time' ] - time();
            $this->set($keyword, $value, $time, $option);
            return true;
        }
    }
