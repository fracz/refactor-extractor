    public function decrementTags($tags = array(), $step = 1, $option = array())
    {
        if (!is_array($tags)) {
            $tags = array($tags);
        }
        $lists = $this->getTags($tags);
        foreach ($lists as $tag => $keywords) {
            foreach ($keywords as $keyword => $time) {
                $this->decrement($keyword, $step, $option);
            }
        }
        return true;
    }
