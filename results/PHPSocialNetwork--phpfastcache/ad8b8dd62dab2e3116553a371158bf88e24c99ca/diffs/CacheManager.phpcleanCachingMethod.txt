    public static function cleanCachingMethod($instance = null)
    {
        if (is_null($instance)) {
            foreach (self::$instances as $instance => $data) {
                self::__CleanCachingMethod($instance);
                unset($data);
            }
        } else {
            self::__CleanCachingMethod($instance);
        }
    }
