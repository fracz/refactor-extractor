    public function driver_isExisting($keyword)
    {
        $file_path = $this->getFilePath($keyword, true);
        if (!file_exists($file_path)) {
            return false;
        } else {
            // check expired or not
            $value = $this->get($keyword);

            return !($value == null);
        }
    }
