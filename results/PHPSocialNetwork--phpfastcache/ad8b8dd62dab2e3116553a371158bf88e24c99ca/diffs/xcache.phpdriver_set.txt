    public function driver_set($keyword, $value = "", $time = 300, $option = array())
    {

        if (isset($option[ 'skipExisting' ]) && $option[ 'skipExisting' ] == true) {
            if (!$this->isExisting($keyword)) {
                return xcache_set($keyword, serialize($value), $time);
            }
        } else {
            return xcache_set($keyword, serialize($value), $time);
        }
        return false;
    }
