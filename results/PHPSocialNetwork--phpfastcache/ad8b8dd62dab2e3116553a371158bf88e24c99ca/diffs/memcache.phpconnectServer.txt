    public function connectServer()
    {
        $server = $this->config[ 'memcache' ];
        if (count($server) < 1) {
            $server = array(
              array('127.0.0.1', 11211),
            );
        }

        foreach ($server as $s) {
            $name = $s[ 0 ] . "_" . $s[ 1 ];
            if (!isset($this->checked[ $name ])) {
                try {
                    if (!$this->instant->addserver($s[ 0 ], $s[ 1 ])) {
                        $this->fallback = true;
                    }

                    $this->checked[ $name ] = 1;
                } catch (\Exception $e) {
                    $this->fallback = true;
                }


            }

        }
    }
