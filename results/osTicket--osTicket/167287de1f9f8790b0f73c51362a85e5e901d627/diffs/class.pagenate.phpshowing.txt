    function showing() {
        $html = '';
        $start = $this->getStart() + 1;
        $end = min($start + $this->limit + $this->slack - 1, $this->total);
        if ($end < $this->total) {
            $to= $end;
        } else {
            $to= $this->total;
        }
        $html="&nbsp;".__('Showing')."&nbsp;&nbsp;";
        if ($this->total > 0) {
            $html .= sprintf(__('%1$d - %2$d of %3$d' /* Used in pagination output */),
               $start, $end, $this->total);
        }else{
            $html .= " 0 ";
        }
        return $html;
    }
