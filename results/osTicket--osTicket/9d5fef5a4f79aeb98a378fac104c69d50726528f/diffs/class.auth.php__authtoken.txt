    static private function __authtoken($token) {

        switch ($token[0]) {
            case 'c': //Collaborator c+[token]
                if (($c = Collaborator::lookupByAuthToken($token)))
                    return new ClientSession($c); //Decorator
                break;
            case 'o': //Ticket owner  o+[token]
                break;
        }
    }
