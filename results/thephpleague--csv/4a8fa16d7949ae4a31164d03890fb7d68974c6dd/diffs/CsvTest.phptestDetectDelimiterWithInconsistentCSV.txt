    public function testDetectDelimiterWithInconsistentCSV()
    {
        $csv = Writer::createFromFileObject(new SplTempFileObject);
        $csv->setDelimiter(';');
        $csv->insertOne(['toto', 'tata', 'tutu']);
        $csv->setDelimiter('|');
        $csv->insertAll([
            ['toto', 'tata', 'tutu'],
            ['toto', 'tata', 'tutu'],
            ['toto', 'tata', 'tutu']
        ]);

        $csv->detectDelimiter(5, ['toto', '|']);
    }
