    public function getInputBOM()
    {
        $bom = [
            self::BOM_UTF8,
            self::BOM_UTF16_BE,
            self::BOM_UTF16_LE,
            self::BOM_UTF32_BE,
            self::BOM_UTF32_LE
        ];
        $csv = $this->getIterator();
        $csv->rewind();
        $line = $csv->fgets();

        $res = array_filter($bom, function ($sequence) use ($line) {
            return strpos($line, $sequence) === 0;
        });

        return array_shift($res);
    }
