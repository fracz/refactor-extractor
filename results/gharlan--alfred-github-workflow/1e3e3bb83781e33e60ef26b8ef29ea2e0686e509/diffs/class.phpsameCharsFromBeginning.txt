  static public function sameCharsFromBeginning($str1, $str2)
  {
    $str1 = strtolower($str1);
    $str2 = strtolower($str2);
    $end = min(strlen($str1), strlen($str2));
    for ($i = 0; $i < $end && $str1[$i] === $str2[$i]; ++$i);
    return $i;
  }
