                'date'          => $info[$this->info_positions['DATE']],

                'owner'         => $info[$this->info_positions['OWNER']],
                'group'         => $info[$this->info_positions['GROUP']],
                'size'          => $info[$this->info_positions['SIZE']],
                'name'          => $info[$this->info_positions['NAME']]
            );
        }
