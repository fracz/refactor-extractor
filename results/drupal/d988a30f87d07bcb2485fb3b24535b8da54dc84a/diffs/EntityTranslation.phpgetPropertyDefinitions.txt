  public function getPropertyDefinitions() {
    $definitions = array();
    foreach ($this->parent->getPropertyDefinitions() as $name => $definition) {
      if (!empty($definition['translatable']) || !$this->strict) {
        $definitions[$name] = $definition;
      }
    }
    return $definitions;
  }
