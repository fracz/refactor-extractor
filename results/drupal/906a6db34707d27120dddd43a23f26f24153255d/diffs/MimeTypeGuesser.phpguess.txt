    public function guess($path)
    {
        if (!is_file($path)) {
            throw new FileNotFoundException($path);
        }

        if (!is_readable($path)) {
            throw new AccessDeniedException($path);
        }

        $mimeType = null;

        foreach ($this->guessers as $guesser) {
            $mimeType = $guesser->guess($path);

            if (null !== $mimeType) {
                break;
            }
        }

        return $mimeType;
    }
