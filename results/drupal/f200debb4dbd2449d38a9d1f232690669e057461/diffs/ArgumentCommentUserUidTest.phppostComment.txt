  function postComment($node, $comment = array()) {
    $comment += array(
      'uid' => $this->loggedInUser->uid,
      'nid' => $node->nid,
      'cid' => '',
      'pid' => '',
    );
    return entity_create('comment', $comment)->save();
  }
