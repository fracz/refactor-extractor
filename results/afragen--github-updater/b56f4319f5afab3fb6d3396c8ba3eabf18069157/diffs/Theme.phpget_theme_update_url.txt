	protected function get_theme_update_url( $action, $theme ) {
		$update_url = esc_attr(
			add_query_arg(
				array(
					'action' => $action,
					'theme'  => urlencode( $theme ),
				),
				self_admin_url( 'update.php' )
			) );

		return $update_url;
	}
