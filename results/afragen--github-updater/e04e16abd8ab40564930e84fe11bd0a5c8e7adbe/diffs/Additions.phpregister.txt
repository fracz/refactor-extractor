	public function register( $config, $repos, $type ) {
		if ( empty( $config ) ) {
			return false;
		}
		if ( null === ( $config = json_decode( $config, true ) ) ) {
			return false;
		}
		if ( 'plugin' === $type ) {
			$this->add_plugin_headers( $config, $repos );
		}
		if ( 'theme' === $type ) {
			$this->add_theme_headers( $config, $repos );
		}
	}
