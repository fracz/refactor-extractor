	private function get_gitlab_webhook_data() {
		$request_body = file_get_contents('php://input');
		$request_data = json_decode($request_body, TRUE);

		if ( !$request_data ) {
			return NULL;
		}

		$res = array();
		$res["webhook"] = "gitlab";
		$res["hash"] = $request_data["after"];
		$response['branch']  = array_pop( explode( '/', $request_data['ref'] ) );

		return $res;
	}
