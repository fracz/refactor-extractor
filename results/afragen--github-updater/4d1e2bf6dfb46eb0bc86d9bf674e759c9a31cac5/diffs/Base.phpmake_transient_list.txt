	protected function make_transient_list( $type ) {
		$transients = get_site_transient( 'ghu-' . $type );
		if ( $transients ) {
			return false;
		}
		set_site_transient( 'ghu-' . $type, self::$transients, ( self::$hours * HOUR_IN_SECONDS ) );
		self::$transients = array();

		return true;
	}
