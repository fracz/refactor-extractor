	protected function get_remote_version() {
		$response = $this->get_remote_info();
		if ( ! $response )
			return false;

		preg_match( '/^[ \t\/*#@]*Version\:\s*(.*)$/im', base64_decode( $response->content ), $matches );

		if ( ! empty( $matches[1] ) )
			return $matches[1];

		return false;
	}
