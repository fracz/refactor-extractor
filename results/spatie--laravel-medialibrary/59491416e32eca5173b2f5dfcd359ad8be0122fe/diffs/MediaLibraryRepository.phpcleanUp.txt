    public function cleanUp()
    {
        $media = Media::where('temp', 1)
            ->where('created_at', '<=', Carbon::now()->subDay(1)->toDateTimeString())
            ->get();

        foreach ($media as $mediaItem) {
            $this->remove($mediaItem->id);
        }

        return count($media);
    }
