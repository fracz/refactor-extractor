    protected static function isAValidUrlGeneratorClass($customUrlClass) : bool
    {
        if (!$customUrlClass) {
            return false;
        }

        if (!class_exists($customUrlClass)) {
            return false;
        }

        if (!is_subclass_of($customUrlClass, UrlGenerator::class)) {
            return false;
        }

        return true;
    }
