    protected function scope($scopes, $methods = null)
    {
        if (is_null($methods)) {
            $this->scopedMethods['*'] = (array) $scopes;
        } else {
            foreach ((array) $methods as $method) {
                $this->scopedMethods[$method] = (array) $scopes;
            }
        }

        return $this;
    }
