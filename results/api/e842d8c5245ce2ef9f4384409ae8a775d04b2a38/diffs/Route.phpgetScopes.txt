    public function getScopes()
    {
        if (is_null($this->scopes)) {
            $this->scopes = Arr::pull($this->action, 'scopes', []);

            $this->findControllerOptions('scopes', function ($value) {
                $this->scopes = array_merge($this->scopes, $value['scopes']);
            });
        }

        return $this->scopes;
    }
