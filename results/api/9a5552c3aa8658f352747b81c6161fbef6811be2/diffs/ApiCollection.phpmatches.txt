	public function matches($request)
	{
		if ($this->matchDomain($request))
		{
			return true;
		}
		elseif ($this->matchPrefix($request))
		{
			return true;
		}
		elseif ( ! $this->option('prefix') and ! $this->option('domain'))
		{
			return true;
		}

		return false;
	}
