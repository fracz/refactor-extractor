    protected function addApiRoute($route)
    {
        $versions = array_get(last($this->groupStack), 'version', []);

        foreach ($versions as $version) {
            if ($collection = $this->getApiRouteCollection($version)) {
                $collection->add($route);
            }
        }

        return $route;
    }
