	public function addSureNotType(Node $expr, string $exprString, Type $type): self
	{
		$types = $this->sureNotTypes;
		if (isset($types[$exprString])) {
			$type = $types[$exprString][1]->combineWith($type);
		}

		$types[$exprString] = [
			$expr,
			$type,
		];

		return new self(
			$this->sureTypes,
			$types
		);
	}
