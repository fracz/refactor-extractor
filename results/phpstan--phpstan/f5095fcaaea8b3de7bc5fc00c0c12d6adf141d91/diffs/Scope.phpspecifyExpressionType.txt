			$variableTypes = $this->getVariableTypes();

			return new self(
				$this->broker,
				$this->printer,
				$this->getFile(),
				$this->isDeclareStrictTypes(),
				$this->getClass(),
				$this->getFunction(),

				$variableTypes,
				$this->inClosureBindScopeClass,

				$this->getInFunctionCall(),
				$this->isNegated(),
				$this->moreSpecificTypes
			);
