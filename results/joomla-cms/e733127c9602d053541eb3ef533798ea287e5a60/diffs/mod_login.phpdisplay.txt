	function display(&$params)
	{
		$greeting	= $params->def('greeting', 1);

		$type 	= modLoginHelper::getType();
		$return	= modLoginHelper::getReturnURL();

		switch($type)
		{
			case 'login' 	:
				modLogin::showLogin($params, $return);
				break;
			case 'logout'	:
				modLogin::showLogout($params, $return);
				break;
		}
	}
