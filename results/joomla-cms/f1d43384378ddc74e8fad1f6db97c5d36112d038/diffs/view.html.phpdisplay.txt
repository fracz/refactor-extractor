	public function display($tpl = null)
	{
		$state		= $this->get('State');
		//$item		= $this->get('Item');

		// Check for errors.
		if (count($errors = $this->get('Errors'))) {
			JError::raiseError(500, implode("\n", $errors));
			return false;
		}

		$this->assignRef('state',		$state);
		//$this->assignRef('item',		$item);

		$this->_setToolbar();
		parent::display($tpl);
	}
