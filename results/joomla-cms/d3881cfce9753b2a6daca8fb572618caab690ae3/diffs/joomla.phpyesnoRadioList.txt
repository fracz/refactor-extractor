	function yesnoRadioList( $tag_name, $tag_attribs, $selected, $yes='yes', $no='no' ) {
		global $_LANG;

		$arr = array(
			mosHTML::makeOption( '0', $_LANG->_( $no ) ),
			mosHTML::makeOption( '1', $_LANG->_( $yes ) )
		);
		return mosHTML::radioList( $arr, $tag_name, $tag_attribs, $selected );
	}
