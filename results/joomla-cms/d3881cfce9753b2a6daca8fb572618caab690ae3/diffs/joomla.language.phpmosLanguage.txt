	function mosLanguage( $userLang='' ) {
		$this->_strings = array();

		if( isset( $this->_locale ) ) {
			setlocale (LC_TIME, $this->_locale);
		}
		$this->_defaultLang = 'english';
		$this->_userLang = $userLang;
	}
