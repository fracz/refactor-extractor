	function remember( $username, $password ) {
		switch ($this->_sessionType) {
			case 'php':
				// not recommended
				break;

			case 'cookie':
			default:
				$lifetime = time() + 365*24*60*60;
				setcookie( 'usercookie[username]', $user->username, $lifetime, '/' );
				setcookie( 'usercookie[password]', $user->password, $lifetime, '/' );
				break;
		}
	}
