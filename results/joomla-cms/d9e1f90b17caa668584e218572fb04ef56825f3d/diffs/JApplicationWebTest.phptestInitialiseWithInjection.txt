			->method('test')

		$mockDocument = $this->getMock('JDocument', array('test'), array(), '', false);

			->expects($this->any())
			->method('test')
			->willReturnSelf();

		$mockLanguage
			->expects($this->any())
			->method('test')
			->willReturnSelf();

		$mockDispatcher = $this->getMock('JEventDispatcher', array('test'), array(), '', false);
		$mockDispatcher
			->expects($this->any())
			->method('test')
			->willReturnSelf();
