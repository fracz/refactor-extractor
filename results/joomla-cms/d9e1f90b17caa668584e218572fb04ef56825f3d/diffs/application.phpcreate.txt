			'get',

			'getIdentity',
			'getRouter',
			'getTemplate',
			'getMenu',
			'getLanguage'
		);

		// Create the mock.
		$mockObject = $test->getMock(
			'JApplication',
			$methods,
			// Constructor arguments.
			array(),
			// Mock class name.
			'',
			// Call original constructor.
			false
		);

		$menu = TestMockMenu::create($test);
