	public function check()
	{
		$this->checkStatus = -1;
		if ($this->checkQuery) {
			$this->db->setQuery($this->checkQuery);
			if ($rows = $this->db->loadObjectList()) {
				if (count($rows) == 1) {
					$this->checkStatus = 1;
				}
			}
			else {
				$this->checkStatus = -2;
			}
		}
		return $this->checkStatus;
	}
