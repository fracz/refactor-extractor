	function display($tpl = null)
	{
		$data = $this->getData();
		if ($data === false) {
			return false;
		}

		// Menu Parameters
		$params = &JSiteHelper::getMenuParams();

		$this->set('params', $params);
		$this->set('data'  , $data);

		parent::display($tpl);
	}
