	public function countMenuChildren()
	{
		static $children;

		if (!isset($children))
		{
			$db = JFactory::getDbo();
			$app = JFactory::getApplication();
			$menu = $app->getMenu();
			$active = $menu->getActive();
			if ($active)
			{
				$query = $db->getQuery(true)
					->select('COUNT(*)')
					->from('#__menu')
					->where('parent_id = ' . $active->id)
					->where('published = 1');
				$db->setQuery($query);
				$children = $db->loadResult();
			}
			else
			{
				$children = 0;
			}
		}

		return $children;
	}
