	function getRows( $start, $limit )
	{
        if ( (!is_array($this->_items)) || (count($this->_items) <= 0) ) {
            return null;
        }

        $i = 0;
        $rows = null;
		foreach ($this->_items as $item)
		{
			if ( (($i >= $start) && ($i < $start+$limit)) || ($limit == 0) ) {
				$rows[] = $item;
			}
			$i++;
		}
		return $rows;
	}
