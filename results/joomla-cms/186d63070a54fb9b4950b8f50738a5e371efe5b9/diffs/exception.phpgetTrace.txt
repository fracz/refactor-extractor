	function getTrace()
	{
		if (isset( $this ) && isset( $this->backtrace )) {
			$trace = &$this->backtrace;
		} else {
			$trace = function_exists( 'debug_backtrace' ) ? debug_backtrace() : null;
		}

		return $trace;
	}
