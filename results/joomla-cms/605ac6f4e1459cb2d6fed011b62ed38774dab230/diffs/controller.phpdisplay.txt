	public function display()
	{
		if ($this->_task == 'config') {
			JRequest::setVar('view', 'config');
		} else if ($this->_task == 'view') {
			JRequest::setVar('view', 'message');
		} else if ($this->_task == 'reply') {
			JRequest::setVar('view', 'message');
			JRequest::setVar('layout', 'edit');
		}

		parent::display();
	}
