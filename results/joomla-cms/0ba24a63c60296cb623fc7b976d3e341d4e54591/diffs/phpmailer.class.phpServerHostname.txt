	function ServerHostname() {
		if ($this->Hostname != "")
			$result = $this->Hostname;
		elseif ($this->ServerVar('SERVER_NAME') != "")
			$result = $this->ServerVar('SERVER_NAME');
		else
			$result = "localhost.localdomain";

		return $result;
	}
