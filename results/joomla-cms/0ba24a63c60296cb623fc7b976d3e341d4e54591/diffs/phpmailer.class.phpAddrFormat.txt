	function AddrFormat($addr) {
		if(empty($addr[1]))
			$formatted = $addr[0];
		else
		{
			$formatted = $this->EncodeHeader($addr[1], 'phrase') . " <" .
						 $addr[0] . ">";
		}

		return $formatted;
	}
