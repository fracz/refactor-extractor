	function &getGroups()
	{
		// Get a database connector object
		$db = &$this->getDbo();

		// get list of Positions for dropdown filter
		$query = 'SELECT folder AS value, folder AS text' .
				' FROM #__extensions' .
				' WHERE type = "plugin"' .
				' GROUP BY folder' .
				' ORDER BY folder';
		$db->setQuery($query);

		$types[] = JHtml::_('select.option',  '', JText::_('All'));
		$types = array_merge($types, $db->loadObjectList());

		return $types;
	}
