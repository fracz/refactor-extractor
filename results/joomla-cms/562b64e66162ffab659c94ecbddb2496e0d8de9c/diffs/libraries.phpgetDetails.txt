	function &getDetails($file) {
		$library = new JLibraryManifest();
		$retval = false;
		$library->manifest_filename = $file;
		if ($library->loadManifestFromXML(JPATH_MANIFESTS . DS . 'libraries' . DS . $file . '.xml'))
			return $library;
		else
			return $retval;
    }
