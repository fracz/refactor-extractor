	function discover_install()
	{
		$model	= &$this->getModel('discover');
		$view	= &$this->getView('discover');
		$model->discover_install();
		$ftp	= &JClientHelper::setCredentialsFromRequest('ftp');
		$view->assignRef('ftp', $ftp);
		$view->setModel($model, true);
		$view->display();
	}
