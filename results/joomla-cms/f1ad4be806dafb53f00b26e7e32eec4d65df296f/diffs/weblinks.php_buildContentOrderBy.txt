	function _buildContentOrderBy()
	{
		if ($this->_filter->order == 'a.ordering'){
			$orderby 	= ' ORDER BY category, a.ordering '.$this->_filter->order_Dir;
		} else {
			$orderby 	= ' ORDER BY '.$this->_filter->order.' '.$this->_filter->order_Dir.' , category, a.ordering ';
		}

		return $orderby;
	}
