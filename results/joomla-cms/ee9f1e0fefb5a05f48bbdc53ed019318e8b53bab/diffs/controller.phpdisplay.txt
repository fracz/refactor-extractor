	function display()
	{
		// Intercept URL's using the old task format
		$task	= $this->getTask();

		switch ($task)
		{
			case 'sysinfo':
			case 'help':
			case 'changelog':
				JRequest::setVar('view', $task);
				break;
		}

		if (JRequest::getVar('view')) {
			parent::display();
		}
	}
