	public function testGetAuthorisedViewLevels()
	{
		usleep(100);

		$access = new JAccess;
		$array1 = array(0 => 1, 1 => 1, 2 => 2, 3 => 3);

		$this->assertThat(
			$access->getAuthorisedViewLevels(42),

		);
	}
