	function buildLanguageList( $client=2, $actualLanguage ) {
		global $_LANG, $mainframe;

		$list = array();

		if( !isset( $client ) ) {
			$client = $mainframe->getClient( );
		}

		// cache activation
		if( class_exists( 'mosCache' ) ) {
			$cache =& mosCache::getCache( 'JLanguage' );
			$langs = $cache->call( 'JLanguage::getKnownLanguages', $client );
		} else {
			$langs = JLanguage::getKnownLanguages( $client );
		}

		foreach ($langs as $lang=>$name) {
			$option = array();

			$option['text'] = $_LANG->_( $name );
			$option['value'] = $lang;
			if( $lang == $actualLanguage ) {
				$option['selected'] = 'selected="true"';
			}
			$list[] = $option;
		}

		return $list;
	}
}
