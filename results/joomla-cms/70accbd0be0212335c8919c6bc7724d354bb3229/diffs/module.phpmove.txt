	public function move($direction)
	{
		$row = &JTable::getInstance('module');
		if (!$row->load($this->_id)) {
			$this->setError($this->_db->getErrorMsg());
			return false;
		}

		if (!$row->move($direction, 'position = '.$this->_db->Quote($row->position).' AND client_id='.(int) $client->id)) {
			$this->setError($this->_db->getErrorMsg());
			return false;
		}

		return true;
	}
