	function edit()
	{
		JRequest::setVar('hidemainmenu', 1);
		JRequest::setVar('view', 'module');
		JRequest::setVar('edit', true);

		// Checkout the module
		$model = $this->getModel('module');
		$model->checkout();

		parent::display();
	}
