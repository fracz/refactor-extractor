	public function &getParams()
	{
		// Get the state parameters
		$module	= &$this->getData();
		$params	= new JParameter($module->params);

		if ($xml = &$this->_getXML())
		{
			if ($ps = & $xml->document->params) {
				foreach ($ps as $p)
				{
					$params->setXML($p);
				}
			}
		}
		return $params;
	}
