	public function isCheckedOut($uid=0)
	{
		if ($this->_id)
		{
			$module = &JTable::getInstance('module');
			if (!$module->load($this->_id)) {
				$this->setError($this->_db->getErrorMsg());
				return false;
			}
			return $module->isCheckedOut($uid);
		}
	}
