	function add()
	{
		JRequest::setVar('hidemainmenu', 1);
		JRequest::setVar('view', 'selecttype');
		JRequest::setVar('edit', false);

		// Checkout the module
		$model = $this->getModel('module');
		$model->checkout();

		parent::display();
		return;
	}
