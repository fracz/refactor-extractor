    function _detectID()
    {
        if (JSession::useCookies()) {
            if (isset($_COOKIE[JSession::name()])) {
                return $_COOKIE[JSession::name()];
            }
        } else {
            if (isset($_GET[JSession::name()])) {
                return $_GET[JSession::name()];
            }
            if (isset($_POST[JSession::name()])) {
                return $_POST[JSession::name()];
            }
        }
        return null;
    }
