    function isExpired()
    {
        if (isset($_SESSION['__HTTP_SESSION_EXPIRE_TS']) && $_SESSION['__HTTP_SESSION_EXPIRE_TS'] < time()) {
            return true;
        } else {
            return false;
        }
    }
