    function isIdle()
    {
		if (isset($_SESSION['__HTTP_SESSION_IDLE_TS']) && (($_SESSION['__HTTP_SESSION_IDLE_TS'] + $_SESSION['__HTTP_SESSION_IDLE']) < time())) {
            return true;
        } else {
            return false;
        }
    }
