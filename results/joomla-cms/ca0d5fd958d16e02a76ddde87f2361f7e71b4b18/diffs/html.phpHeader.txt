	function Header($text, $iconClass)
	{
		$iconClass	= JFile::stripExt($iconClass);
		$html		= null;

		$html	.= "<div class=\"header icon-48-$iconClass\">\n";
		$html	.= "$text\n";
		$html	.= "</div>\n";

		return $html;
	}
