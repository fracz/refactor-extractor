	function importGroup( $group )
	{
		static $bots;

		if (!isset($bots)) {
			$bots = JPluginHelper::_load();
		}

		$n = count( $bots);
		for ($i = 0; $i < $n; $i++) {
			if($bots[$i]->folder == $group) {
				JPluginHelper::import( $bots[$i]->folder, $bots[$i]->element, $bots[$i]->published, $bots[$i]->params );
			}
		}
		return true;
	}
