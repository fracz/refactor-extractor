	function report($cid = array(), $report = -1)
	{
		$user 	=& JFactory::getUser();

		if (count($cid))
		{
			JArrayHelper::toInteger($cid);
			$cids = implode(',', $cid);

			$query = 'UPDATE #__weblinks'
				. ' SET state = '.(int) $report
				. ' WHERE id IN ('.$cids.')'
			;
			$this->_db->setQuery($query);
			if (!$this->_db->query()) {
				$this->setError($this->_db->getErrorMsg());
				return false;
			}
		}

		return true;
	}
