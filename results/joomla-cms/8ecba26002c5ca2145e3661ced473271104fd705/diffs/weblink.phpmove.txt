	function move($direction)
	{
		$table =& $this->getTable();
		if (!$table->load($this->_id)) {
			$this->setError($this->_db->getErrorMsg());
			return false;
		}

		if (!$table->move($direction, ' catid = '.(int) $table->catid.' AND published >= 0 ')) {
			$this->setError($this->_db->getErrorMsg());
			return false;
		}

		return true;
	}
