	function _render(&$params)
	{
		foreach($this->_renderers as $type => $renderers)
		{
			foreach($renderers as $renderer)
			{
				if($html = $this->get($type, $renderer->name, $params)) {
					$this->_engine->addVar('document', $type.'_'.$renderer->name, $html);
				}
			}
		}
	}
