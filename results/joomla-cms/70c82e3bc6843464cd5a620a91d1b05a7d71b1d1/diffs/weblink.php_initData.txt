	function _initData()
	{
		// Lets load the content if it doesn't already exist
		if (empty($this->_data))
		{
			$weblink = new stdClass();
			$weblink->id				= 0;
			$weblink->catid				= 0;
			$weblink->sid				= 0;
			$weblink->title				= null;
			$weblink->alias				= null;
			$weblink->url				= null;
			$weblink->description		= null;
			$weblink->date				= null;
			$weblink->hits				= 0;
			$weblink->state				= 0;
			$weblink->checked_out		= 0;
			$weblink->checked_out_time	= 0;
			$weblink->ordering			= 0;
			$weblink->archived			= 0;
			$weblink->approved			= 0;
			$weblink->params			= null;
			$weblink->category			= null;
			$this->_data				= $weblink;
			return (boolean) $this->_data;
		}
		return true;
	}
