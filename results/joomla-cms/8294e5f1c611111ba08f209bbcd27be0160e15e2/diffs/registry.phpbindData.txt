	protected function bindData(& $parent, $data)
	{
		// Ensure the input data is an array.
		$data = (array) $data;

		foreach ($data as $k => $v) {
			if ($k[0] != '_') {
				if ((is_array($v) && JArrayHelper::isAssociative($v)) || is_object($v)) {
					$parent->$k = new stdClass();
					$this->bindData($parent->$k, $v);
				}
				else {
					$parent->$k = $v;
				}
			}
		}
	}
