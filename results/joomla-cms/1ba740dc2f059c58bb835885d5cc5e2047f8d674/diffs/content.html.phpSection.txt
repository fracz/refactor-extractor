	function Section($row, $params)
	{

		if ($params->get('section'))
		{
?>
				<span>
				<?php

			echo $row->section;
			// writes dash between section & Category Name when both are active
			if ($params->get('category'))
			{
				echo ' - ';
			}
?>
				</span>
			<?php

		}
	}
