	{
		$rehash = false;
		$match = false;

		if ($hash[0] == '$')
		{
			// JCrypt::hasStrongPasswordSupport() includes a fallback for us in the worst case
			JCrypt::hasStrongPasswordSupport();
			$match = password_verify($password, $hash);

			// Uncomment this line if we actually move to bcrypt.

			$rehash = true;
		}
		elseif (substr($hash,0,8) == '{SHA256}')
		{
			// Check the password
			$parts     = explode(':', $hash);
			$crypt     = $parts[0];
			$salt      = @$parts[1];
			$testcrypt = JUserHelper::getCryptedPassword($password, $salt, 'sha256', true);

			if ($hash == $testcrypt)
			{
				$match = true;
			}

			$rehash = true;
		}
		else
		{
			// Check the password
			$parts = explode(':', $hash);
			$crypt = $parts[0];
			$salt  = @$parts[1];

			if (!$salt)

			}

			{
				$match = true;
			}
		}

		{
			$user = new JUser($user_id);
			$user->password = static::hashPassword($password);
			$user->save();
		}
