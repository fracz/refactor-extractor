	function loadByOption($option)
	{
		$db = &$this->getDbo();
		$query = 'SELECT id' .
				' FROM #__components' .
				' WHERE ' . $db->nameQuote('option') . '=' . $db->Quote($option) .
				' AND parent = 0';
		$db->setQuery($query, 0, 1);
		$id = $db->loadResult();

		if ($id === null) {
			return false;
		} else {
			return $this->load($id);
		}
	}
