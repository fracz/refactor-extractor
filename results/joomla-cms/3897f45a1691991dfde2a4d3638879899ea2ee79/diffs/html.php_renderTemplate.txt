	protected function _renderTemplate() {
		$replace = array();
		$with = array();
		foreach($this->_template_tags AS $jdoc => $args) {
			$replace[] = $jdoc;
			$with[] = $this->getBuffer($args['type'], $args['name']);
		}
		return str_replace($replace, $with, $this->_template);
	}
