	public function setQuery($query)
	{
		if (!is_array($query))
		{
			if (strpos($query, '&amp;') !== false)
			{
			   $query = str_replace('&amp;','&',$query);
			}
			parse_str($query, $this->_vars);
		}

		if (is_array($query)) {
			$this->_vars = $query;
		}

		//empty the query
		$this->_query = null;
	}
