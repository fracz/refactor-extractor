	public static function getActions()
	{
		$user	= JFactory::getUser();
		$result	= new JObject;

		$actions = JAccess::getActions('com_messages');

		foreach ($actions as $action)
		{
			$result->set($action->name,	$user->authorise($action->name, 'com_messages'));
		}

		return $result;
	}
