	function objectToString( &$object ) {

		// Build the object variables string
		$vars = '';
		foreach (get_object_vars( $object ) as $k => $v) {
			if ($k == "_name") {
				$name = $v;
			} else {
				$vars .= "\tvar $". $k . " = '" . addslashes($v) . "';\n";
			}
		}

		$str = "<?php\nclass $name {\n";
		$str .= $vars;
		$str .= "}\n?>";

		return $str;
	}
