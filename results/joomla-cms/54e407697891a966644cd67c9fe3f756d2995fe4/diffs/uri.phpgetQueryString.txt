	function getQueryString()
	{
		if (!empty ($this->_vars)) {
			$query = array ();
			foreach ($this->_vars as $name => $value) {
				if (is_array($value)) {
					foreach ($value as $k => $v) {
						$query[] = $name.'='.$v;
					}
				}
				elseif (!is_null($value)) {
					$query[] = $name.'='.$value;
				} else {
					$query[] = $name;
				}
			}
			$query = implode('&', $query);
		} else {
			$query = null;
		}

		return $query;
	}
