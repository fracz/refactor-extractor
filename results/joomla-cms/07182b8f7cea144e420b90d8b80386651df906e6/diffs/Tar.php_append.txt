	function _append($p_filelist, $p_add_dir='', $p_remove_dir='')
	{
		if (!$this->_openAppend())
			return false;

		if ($this->_addList($p_filelist, $p_add_dir, $p_remove_dir))
			$this->_writeFooter();

		$this->_close();

		return true;
	}
