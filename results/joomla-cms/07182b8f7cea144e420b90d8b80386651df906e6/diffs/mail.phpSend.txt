	function &Send()
	{
		if ( ( $this->Mailer == 'mail' ) && ! function_exists('mail') )
		{
			return JError::raiseNotice( 500, JText::_('MAIL_FUNCTION_DISABLED') );
		}

		@ $result = parent::Send();

		if ($result == false)
		{
			// TODO: Set an appropriate error number
			$result =& JError::raiseNotice( 500, JText::_($this->ErrorInfo) );
		}
		return $result;
	}
