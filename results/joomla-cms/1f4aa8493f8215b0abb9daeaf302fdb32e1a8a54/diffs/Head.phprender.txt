	function render( $head = null, $params = array() )
	{
		global $mainframe;

		ob_start();

		echo $this->_tmpl->fetchHead();

		if(isset($mainframe) && class_exists('JEditor')) {
			$editor =& JEditor::getInstance();
			echo $editor->init();
		}

		$contents = ob_get_contents();
		ob_end_clean();

	}
