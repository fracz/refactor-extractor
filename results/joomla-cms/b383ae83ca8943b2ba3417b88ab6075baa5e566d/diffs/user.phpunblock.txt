	function unblock($id)
	{
		global $mainframe;

		/*
		 * Initialize variables
		 */
		$db = & $mainframe->getDBO();

		/*
		 * Perform user unblock query
		 */
		$query = "UPDATE `#__users` " .
				"\n SET `block` = '0' " .
				"\n WHERE `id` = '$id'";
		$db->setQuery( $query );
		if (!$db->query())
		{
			return false;
		} else
		{
			/*
			 * Load the user plugins
			 */
			JPluginHelper::importPlugin( 'user' );
			$results = $mainframe->triggerEvent( 'onUnblock', $id );
			return true;
		}
	}
}
