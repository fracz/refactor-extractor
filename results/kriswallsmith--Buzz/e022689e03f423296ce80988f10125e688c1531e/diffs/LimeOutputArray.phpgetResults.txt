  protected function &getResults($file)
  {
    foreach ($this->results as $key => &$fileResults)
    {
      if ($fileResults['file'] == $file)
      {
        return $fileResults;
      }
    }

    $newResults = array(
      'file' => $file,
      'tests' => array(),
      'stats' => array(
        'plan' => 0,
        'total' => 0,
        'failed' => array(),
        'passed' => array(),
        'skipped' => array(),
      ),
    );

    $this->results[] =& $newResults;

    return $newResults;
  }
