  public static function indent($text, $numberOfSpaces)
  {
    $indentation = str_repeat(' ', $numberOfSpaces);
    $lines = explode("\n", $text);

    for ($i = 0, $c = count($lines); $i < $c; ++$i)
    {
      $lines[$i] = $indentation.$lines[$i];
    }

    return implode("\n", $lines);
  }
