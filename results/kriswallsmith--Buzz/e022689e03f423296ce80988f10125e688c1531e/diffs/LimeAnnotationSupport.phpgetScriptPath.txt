  protected static function getScriptPath()
  {
    $traces = debug_backtrace();
    $file = $traces[count($traces)-1]['file'];

    if (!is_file($file))
    {
      throw new RuntimeException('The script name from the traces is not valid: '.$file);
    }

    return $file;
  }
