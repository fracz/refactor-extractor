  protected function indent($lines, $indentation = 2)
  {
    $lines = explode("\n", $lines);

    foreach ($lines as $key => $line)
    {
      $lines[$key] = str_repeat(' ', $indentation).$line;
    }

    return trim(implode("\n", $lines));
  }
