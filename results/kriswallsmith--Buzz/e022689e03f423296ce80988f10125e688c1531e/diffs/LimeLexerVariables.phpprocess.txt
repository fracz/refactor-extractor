  protected function process($text, $id)
  {
    if ($id == T_VARIABLE && !$this->inClass() && !$this->inFunction()
        && (!$this->inAnnotation() || in_array($this->getCurrentAnnotation(), $this->includedAnnotations)))
    {
      $this->variables[] = $text;
    }
  }
