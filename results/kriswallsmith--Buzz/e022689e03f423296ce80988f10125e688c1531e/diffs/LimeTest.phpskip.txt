  public function skip($message = '', $nbTests = 1)
  {
    list ($file, $line) = LimeTrace::findCaller('LimeTest');

    for ($i = 0; $i < $nbTests; $i++)
    {
      $this->output->skip($message, $file, $line);
    }
  }
