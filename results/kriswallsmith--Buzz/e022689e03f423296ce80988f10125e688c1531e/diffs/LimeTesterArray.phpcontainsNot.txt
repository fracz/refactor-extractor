  public function containsNot(LimeTesterInterface $expected)

    foreach ($this as $key => $value)
    {
      $equal = true;

      try
      {
        $value->is($expected);
      }
      catch (LimeAssertionFailedException $e)
      {
        $equal = false;
      }

      if ($equal)
      {
        throw new LimeAssertionFailedException($this->dumpAll(), $expected);
      }
    }
  }
