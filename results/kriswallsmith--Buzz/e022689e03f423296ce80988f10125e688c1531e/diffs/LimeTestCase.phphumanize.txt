  protected function humanize($method)
  {
    if (substr($method, 0, 4) == 'test')
    {
      $method = substr($method, 4);
    }

    $method = preg_replace('/([a-z])([A-Z])/', '$1 $2', $method);

    return ucfirst(strtolower($method));
  }
