  protected function colorize($text, $style)
  {
    if (is_null($this->colorizer))
    {
      return $text;
    }
    else
    {
      if (is_null($style))
      {
        return preg_replace_callback('/("[^"]*"|(?<!\w)\d+(\.\d+)?(?!\w)|true|false'/*|(->|::)?\w+\([^\)]*\)*/.')/', array($this, 'autoColorize'), $text);
      }
      else
      {
        return $this->colorizer->colorize($text, $style);
      }
    }
  }
