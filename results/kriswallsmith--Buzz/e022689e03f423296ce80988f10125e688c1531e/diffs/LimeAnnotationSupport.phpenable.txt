  public static function enable()
  {
    // make sure that annotations are not replaced twice at the same time
    if (!self::$enabled)
    {
      self::$enabled = true;

      $support = new LimeAnnotationSupport(self::getScriptPath());
      $support->execute();

      exit;
    }
  }
