    protected function getParameters(array $parameters)
    {
        $parameterDefinitions = array();

            if ($parameter->name == '...') {
                continue;
            }
            $parameterDefinitions[] = $this->getParameterCode($parameter);
        }

        return $parameterDefinitions;
    }
