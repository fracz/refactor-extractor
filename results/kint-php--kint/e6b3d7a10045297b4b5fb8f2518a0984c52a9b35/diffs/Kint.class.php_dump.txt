	private static function _dump( $var, $name = '' )
	{
		kintParser::reset();
		if ( self::$textOnly ) {
			return Kint_Decorators_Plain::decorate(
				kintParser::factory( $var, $name )
			);
		}
		return Kint_Decorators_Rich::decorate(
			kintParser::factory( $var, $name )
		);
	}
