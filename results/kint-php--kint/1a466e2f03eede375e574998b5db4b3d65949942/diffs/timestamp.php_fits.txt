	private static function _fits( $variable )
	{
		if ( !is_string( $variable ) && !is_int( $variable ) ) return false;

		$len = strlen( (int)$variable );
		return ( $len === 9 || $len === 10 ) // a little naive
			&& ( (string)(int)$variable == $variable );
	}
