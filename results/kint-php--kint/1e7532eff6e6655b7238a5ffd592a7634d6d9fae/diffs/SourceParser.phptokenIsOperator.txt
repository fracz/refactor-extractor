    private static function tokenIsOperator($token)
    {
        if (KINT_PHP56) {
            self::$operator[T_POW] = true;
            self::$operator[T_POW_EQUAL] = true;
        }

        if (KINT_PHP70) {
            self::$operator[T_SPACESHIP] = true;
        }

        return self::tokenIs($token, self::$operator);
    }
