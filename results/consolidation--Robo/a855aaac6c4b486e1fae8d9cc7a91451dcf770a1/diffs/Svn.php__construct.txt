    public function __construct($pathToSvn='svn', $username='', $password='')
    {
        $this->svn = $pathToSvn;
        if (! empty($username)) {
            $this->svn .= " --username $username";
        }
        if (! empty($password)) {
            $this->svn .= " --password $password";
        }
        $this->result = Result::success($this);
    }
