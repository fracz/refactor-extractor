    protected function sshCommand($command)
    {
        $command = $this->retrieveCommand($command);
        $sshOptions = $this->arguments;
        $hostSpec = $this->hostname;
        if ($this->user) {
            $hostSpec = $this->user . '@' . $hostSpec;
        }

        return sprintf("ssh{$sshOptions} {$hostSpec} '{$command}'");
    }
