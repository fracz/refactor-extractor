    public function increment($what = 'patch')
    {
        $types = ['major', 'minor', 'patch'];
        if (!in_array($what, $types)) {
            throw new TaskException(
                $this,
                'Bad argument, only one of the following is allowed: ' .
                implode(', ', $types)
            );
        }

        $this->version[$what]++;
        return $this;
    }
