    protected function getPathSpec($type)
    {
        if ($type !== 'from' && $type !== 'to') {
            throw new TaskException($this, 'Type must be "from" or "to".');
        }
        foreach (['host', 'user', 'path'] as $part) {
            $varName = $type . ucfirst($part);
            $$part = $this->$varName;
        }
        $spec = isset($path) ? $path : '';
        if (!empty($host)) {
            $spec = "{$host}:{$spec}";
        }
        if (!empty($user)) {
            $spec = "{$user}@{$spec}";
        }

        return $spec;
    }
