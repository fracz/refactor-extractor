    public function csrf_verify()
    {
        global $RTR;

        $module = $RTR->fetch_module();
        $controller = $RTR->fetch_class();

        $bypass = FALSE;

        if (in_array($module .'/'. $controller, $this->ignored_controllers))
        {
            $bypass = TRUE;
        }

        if ( ! $bypass) {
            parent::csrf_verify();
        }
    }
