	public function abbrev_name( $name = '')
	{
		Console::log($name);

		if ($name)

		{
			list( $fname, $lname ) = explode( ' ', $name, 2 );
			if ( is_null($lname) ) //Meaning only one name was entered...
			{
				$lastname = ' ';
			}
			else
			{
				$lname = explode( ' ', $lname );
				$size = sizeof($lname);
				$lastname = $lname[$size-1];
			}
		return $fname.' '.$lastname ;

		}

	return $name;
	}
