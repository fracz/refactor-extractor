	public function deactivate($users=false)
	{
		if (!$users)
		{
			return;
		}

		$this->auth->restrict('Bonfire.Users.Manage');

		foreach ($users as $user_id)
		{
			$this->user_status($user_id,0,0);
		}

	}//end deactivate()
