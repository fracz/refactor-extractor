	public function activate($users=false)
	{
		if (!$users)
		{
			return;
		}

		$this->auth->restrict('Bonfire.Users.Manage');
		foreach ($users as $user_id)
		{
			$this->user_status($user_id,1,0);
		}

	}//end activate()
