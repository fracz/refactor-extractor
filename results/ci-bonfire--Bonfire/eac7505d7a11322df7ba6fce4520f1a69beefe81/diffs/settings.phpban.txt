	public function ban($users=array(), $ban_message='')
	{

		if (empty($users))
		{
			return;
		}

		$this->auth->restrict('Bonfire.Users.Manage');

		foreach ($users as $user_id)
		{
			$data = array(
				'banned'		=> 1,
				'ban_message'	=> $ban_message
				);

			$this->user_model->update($user_id, $data);
		}

	}//end ban()
