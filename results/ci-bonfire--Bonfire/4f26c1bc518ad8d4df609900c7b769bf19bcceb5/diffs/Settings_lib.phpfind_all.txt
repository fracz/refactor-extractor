    public function find_all()
    {
        if (self::$cache) {
            return self::$cache;
        }

        self::init();

        if (! self::$settingsModelLoaded) {
            return null;
        }

        $settings = self::$ci->settings_model->find_all();
        foreach ($settings as $setting) {
            self::$cache[$setting->name] = $setting->value;
        }

        return self::$cache;
    }
