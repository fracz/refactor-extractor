	private function get_postgre_connection_string($hostname, $port = '', $username = '', $password = '', $db_name = '')
	{
		$connect_string = '';
		$components = array(
			'host'		=> $hostname,
			'port'		=> $port,
			'dbname'	=> $db_name,
			'user'		=> $username,
			'password'	=> $password,
		);

		foreach ($components as $key => $val) {
			if ( ! empty($val)) {
				$connect_string .= " {$key}={$val}";
			}
		}

		return trim($connect_string);
	}
