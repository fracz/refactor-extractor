    public static function theme_url($resource = '')
	{
		$url = base_url();

        // Add theme path and theme.
        $url .= self::$theme_paths[0] . '/' . self::theme();

        // Cleanup, just to be safe.
        $url = str_replace(array('//', ':/'), array('/', '://'), $url);

		return $url . $resource;
    }
