	public static function content()
	{
        self::debug_message('Current View = ' . self::$current_view);

        $output = '';
        self::load_view(
            self::$current_view,
            null,
            self::$ci->router->class . '/' . self::$ci->router->method,
            false,
            $output
        );

		Events::trigger('after_page_render', $output);

		return $output;
    }
