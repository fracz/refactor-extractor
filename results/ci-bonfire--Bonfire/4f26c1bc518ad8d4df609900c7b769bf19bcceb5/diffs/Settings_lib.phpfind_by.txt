    public function find_by($field = null, $value = null)
    {
        self::init();

        if (! self::$settingsModelLoaded) {
            return null;
        }

        $settings = self::$ci->settings_model->find_by($field, $value);
        foreach ($settings as $setting) {
            self::$cache[$setting['name']] = $setting['value'];
        }

        return $settings;
    }
