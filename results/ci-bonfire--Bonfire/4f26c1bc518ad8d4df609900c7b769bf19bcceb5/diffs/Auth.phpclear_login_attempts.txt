    protected function clear_login_attempts($login, $expires = 86400)
	{
        $this->ci->db->where(array(
                                'ip_address' => $this->ip_address,
                                'login'      => $login,
                            ))
                     ->or_where('time <', $this->getLoginTimestamp(time() - $expires))
                     ->delete('login_attempts');
    }
