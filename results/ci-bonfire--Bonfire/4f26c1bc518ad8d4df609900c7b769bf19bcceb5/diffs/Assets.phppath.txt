    protected static function path()
    {
        $params = func_get_args();
        $path = array();
        $sep = '/';

        foreach ($params as &$param) {
            $param = rtrim($param, $sep);
            if (! empty($param)) {
                $path[] = $param;
            }
        }

        return implode($sep, $path);
    }
