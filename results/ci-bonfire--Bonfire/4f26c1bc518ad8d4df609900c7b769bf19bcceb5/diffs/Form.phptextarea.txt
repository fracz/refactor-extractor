	public static function textarea($options)
	{
		$value = '';
        if (isset($options['value'])) {
			$value = $options['value'];
			unset($options['value']);
		}

        return '<textarea ' . self::attributesToString($options) . '>' .
            self::prepValue($value) . '</textarea>';
    }
