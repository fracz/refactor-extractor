    public function role_name_by_id($role_id)
		{
        if (! is_numeric($role_id)) {
            return '';
		}

        // Retrieve the roles.
        $roles = $this->loadRoles();

        // Try to return the role name.
        if (isset($roles[$role_id])) {
            return $roles[$role_id];
        }

        return '';
    }
