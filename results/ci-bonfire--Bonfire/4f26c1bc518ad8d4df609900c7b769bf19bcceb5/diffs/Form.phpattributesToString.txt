    private static function attributesToString($attr)
		{
        if (! is_array($attr)) {
			$attr = (array) $attr;
		}

        $attributes = array();
        foreach ($attr as $property => $value) {
            if ($property == 'label') {
				continue;
			}

            if ($property == 'value') {
                $value = self::prepValue($value);
			}

            $attributes[] = "{$property}='{$value}'";
		}

        return implode(' ', $attributes);
    }
