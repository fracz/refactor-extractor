    public static function set_message($message = '', $type = 'info')
			{
        if (empty($message)) {
            return;
			}

        if (! self::$ignore_session && isset(self::$ci->session)) {
            self::$ci->session->set_flashdata('message', "{$type}::{$message}");
		}

        self::$message = array('type' => $type, 'message' => $message);
    }
