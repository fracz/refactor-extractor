    public static function set_attrs($attrs = array())
    {
        if (empty($attrs) || ! is_array($attrs)) {
            return null;
        }

        foreach ($attrs as $attr => $value) {
            if (isset(self::$attr)) {
                self::$attr = $value;
            }
        }
    }
