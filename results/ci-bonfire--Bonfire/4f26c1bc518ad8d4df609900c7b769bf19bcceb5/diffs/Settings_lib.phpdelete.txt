    public static function delete($name, $module = 'core')
    {
        self::init();

        if (self::$settingsModelLoaded && isset(self::$cache[$name])) {
            if (self::$ci->settings_model->delete_where(array('name' => $name, 'module' => $module))) {
                unset(self::$cache[$name]);

                return true;
            }
        }

        return false;
    }
