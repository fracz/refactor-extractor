    public static function add_module_css($module, $path = null, $media = 'screen')
    {
        if (empty($path)) {
            return;
        }

        if ($media == '1') {
            $media = 'screen';
        }

        // Add a string
        if (is_string($path)) {
            $path = array(
                'module' => $module,
                'file'   => $path,
                'media'  => $media
            );
        }

        // Add an array
        if (is_array($path) && count($path)) {
            foreach ($path as $file) {
                self::$styles['module'][] = array(
                    'module' => $module,
                    'file'   => $file,
                    'media'  => $media
                );
            }
        }
    }
