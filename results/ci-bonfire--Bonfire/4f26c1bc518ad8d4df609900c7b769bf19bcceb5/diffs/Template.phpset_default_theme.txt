    public static function set_default_theme($theme = null)
		{
        if (empty($theme) || ! is_string($theme)) {
			return;
		}

		// Make sure a trailing slash is there
        if (substr($theme, -1) !== '/') {
			$theme .= '/';
		}

		self::$default_theme = $theme;
    }
