    protected function getLoginTimestamp($time = null)
		{
        if (empty($time)) {
            $time = time();
        }
        return strtolower($this->ci->config->item('time_reference')) == 'gmt' ?
            gmdate($this->loginDateFormat, $time) : date($this->loginDateFormat, $time);
		}
