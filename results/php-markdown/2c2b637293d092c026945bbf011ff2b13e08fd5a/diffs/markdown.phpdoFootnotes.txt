	function doFootnotes($text) {
	#
	# Replace footnote references in $text [^id] with a link to the footnote.
	#
		$text = preg_replace_callback('{\[\^(.+?)\]}',
			array(&$this, '_doFootnotes_callback'), $text);

		return $text;
	}
