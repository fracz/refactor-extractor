    protected function determineFormKey($parameters)
    {
        if (is_array($parameters) && null === $this->formKey) {
            // Assign the formKey
            foreach ($parameters as $key => $parameter) {
                if ($parameter instanceof FormInterface) {
                    $this->setFormKey($key);
                    break;
                }
            }
        }
    }
