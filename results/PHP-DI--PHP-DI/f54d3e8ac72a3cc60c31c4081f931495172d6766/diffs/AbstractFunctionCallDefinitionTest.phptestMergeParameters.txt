    public function testMergeParameters()

            0 => 'a',
            1 => 'b',
        ));

        $definition2->replaceParameters(array(
            1 => 'c',
            2 => 'd',
        ));

        $definition1->merge($definition2);

        $this->assertEquals(array('a', 'b', 'd'), $definition1->getParameters());
    }
