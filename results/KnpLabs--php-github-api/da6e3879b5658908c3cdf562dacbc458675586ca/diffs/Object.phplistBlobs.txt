    public function listBlobs($username, $repo, $treeSHA)
    {
        $tree = $this->showTree($username, $repo, $treeSHA, true);

        if (isset($tree['tree'])) {
            $blobs = array_filter($tree['tree'], function ($element) {
                return 'blob' === $element['type'];
            });

            return $blobs;
        }
    }
