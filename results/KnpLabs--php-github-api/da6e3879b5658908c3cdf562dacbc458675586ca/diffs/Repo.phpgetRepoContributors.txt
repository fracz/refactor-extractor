    public function getRepoContributors($username, $repo, $includingAnonymous = false)
    {
        $url = 'repos/'.urlencode($username).'/'.urlencode($repo).'/contributors';
        if ($includingAnonymous) {
            $url .= '?anon=1';
        }

        return $this->get($url);
    }
