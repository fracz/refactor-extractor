    public function search($keyword, $language = null, $startPage = 1)
    {
        $url = 'legacy/repos/search/'.urlencode($keyword);

        $params = array();
        if (null !== $language) {
            $params['language'] = $language;
        }
        if (1 !== $startPage) {
            $params['start_page'] = $startPage;
        }

        if (!empty($params)) {
            $url .= '?' . http_build_query($params);
        }

        return $this->get($url);
    }
