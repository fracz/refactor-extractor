    private function getBranchSha($username, $repoName, $branchName)
    {
        $info = $this->get('repos/'.urlencode($username).'/'.urlencode($repoName).'/git/refs/'.urlencode($branchName));

        if (!isset($info['ref'])) {
            return null;
        }

        return $info['object']['sha'];
    }
