    public function getBranchCommits($username, $repo, $branch)
    {
        $url = 'repos/'.urlencode($username).'/'.urlencode($repo).'/commits';

        $sha = $this->getBranchSha($username, $repo, $branch);
        if (null !== $sha) {
            $url .= '?sha='.urlencode($sha);
        }

        return $this->get($url);
    }
