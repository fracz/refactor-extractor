    public function getList($username, $repo, $state = null, $parameters = array())
    {
        $url = 'repos/'.urlencode($username).'/'.urlencode($repo).'/issues';
        if ($state) {
            $parameters['state'] = $state;
        }
        if ($parameters) {
            $url .= '?'.http_build_query($parameters);
        }

        return $this->get($url);
    }
