    public function showBlob($username, $repo, $treeSHA, $path)
    {
        $tree = $this->showTree($username, $repo, $treeSHA, true);

        if (isset($tree['tree'])) {
            $blobs = array_filter($tree['tree'], function ($element) use ($path) {
                return $path === $element['path'];
            });

            return reset($blobs);
        }
    }
