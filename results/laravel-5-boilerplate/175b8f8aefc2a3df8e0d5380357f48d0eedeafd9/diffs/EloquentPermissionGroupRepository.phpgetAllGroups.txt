    public function getAllGroups($withChildren = false)
    {
        if ($withChildren) {
            return PermissionGroup::orderBy('name', 'asc')->get();
        }

        return PermissionGroup::with('children', 'permissions')
            ->whereNull('parent_id')
            ->orderBy('sort', 'asc')
            ->get();
    }
