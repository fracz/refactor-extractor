    function save($values) {

        $data = $this->transformToPostField($values);

        $this->postMetaKey = $values['meta_key'];
        $this->postMetaVpId = $values['vp_id'];

        $this->saveEntity($data, array($this, 'notifyChangeListeners'));
    }
