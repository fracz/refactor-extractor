    private static function containsDirtyPatterns($sets) {
        $dirtyPatterns = array(
            "/DATE_ADD[ ]?\\(.*/" //https://regex101.com/r/tS2iC6/1
        );
        foreach ($sets as $set) {
            foreach ($dirtyPatterns as $pattern) {
                if (preg_match($pattern, $set)) {
                    return true;
                }
            }
        }
        return false;
    }
