    private static function setToString($setOperations) {
        $sqlChunk = ' SET';
        foreach ($setOperations as $setOperation) {
            $sqlChunk .= " " . $setOperation->column . "=" . $setOperation->value . ",";
        }

        return rtrim($sqlChunk, ',');
    }
