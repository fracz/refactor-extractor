        $info = array();
        $wpTheme = wp_get_theme();

        $info['stylesheet'] = $wpTheme->get_stylesheet();

        $parent = $wpTheme->parent();
        $parentName = '';
        if ($parent instanceof \WP_Theme) {

        }
        $info['parent'] = $parentName;

        return $info;
