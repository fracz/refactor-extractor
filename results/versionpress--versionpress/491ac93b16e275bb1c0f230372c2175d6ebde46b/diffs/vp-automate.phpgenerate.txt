    public function generate($args, $assoc_args) {
        global $wpdb;
        $this->faker = Faker\Factory::create();

        $this->database = $wpdb;

        $preferedOrder = array(
            'options',
            'users',
            'terms',
            'posts',
            'comments'
        );

        foreach ($preferedOrder as $entity) {
            if (!isset($assoc_args[$entity])) {
                continue;
            }

            $this->generateEntities($entity, $assoc_args[$entity]);
        }
    }
