    public function syncRoles(...$roles)
    {
        $this->roles()->detach();

        collect($roles)
            ->flatten()
            ->map(function ($role) {
                return $this->getStoredRole($role);
            })->each(function (Role $role) {
                $this->roles()->save($role);
            });

        return $this;
    }
