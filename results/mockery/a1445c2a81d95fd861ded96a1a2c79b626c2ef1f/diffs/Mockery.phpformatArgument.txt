    {
        if (is_object($argument)) {
            return 'object(' . get_class($argument) . ')';

        if (is_resource($argument)) {
            return 'resource(...)';
        }

        $argument = (string) $argument;

        return $depth === 0 ? '"' . $argument . '"' : $argument;
